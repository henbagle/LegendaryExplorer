405 {
406 sc.Serialize(ref gsv.UV);
407 }
408
409 if (svb.NumTexCoords > 1)
410 {
411 if (sc.IsLoading)
412 {
413 sc.ms.Skip((svb.NumTexCoords - 1) * (svb.bUseFullPrecisionUVs ? 8 : 4));
414 }
415 else

417 throw new Exception("Should never be saving more than one UV! Num UVs (NumTexCoords): "+svb.NumTexCoords);
418 }
419 }

421 }

423 else
424 {
