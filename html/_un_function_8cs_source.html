271 public static string GetPropertyType(ExportEntry exp)
272 {
273 switch (exp.ClassName)
274 {
275 case "BoolProperty":
276 return "bool";
277 case "NameProperty":
278 return "Name";
279 case "IntProperty":
280 return "int";
281 case "FloatProperty":
282 return "float";
283 case "StrProperty":
284 return "String";
285 case "ByteProperty":
286 return "byte";
287 case "StringRefProperty":
288 return "stringref";
289 case "ClassProperty":
290 return "UClass";
291 case "ComponentProperty":
292 return "Component";
293 case "ArrayProperty":
294 return "ArrayProperty";
295 case "DelegateProperty":
296 return "delegate";
297 case "InterfaceProperty":
298 return "interface";
299 default:
300 Debug.WriteLine("Unknown property type for ME1 Function signature parsing: " + exp.ClassName);
301 return "???";

305 private string GetReturnType()
306 {
307 //var childIdx = EndianReader.ToInt32(Export.DataReadOnly, 0x18, Export.FileRef.Endian);
308 try
309 {
310 var childProbe = Export.Game.IsLEGame() ? 0x14 : 0x18;
311 var childIdx = EndianReader.ToInt32(Export.DataReadOnly, childProbe, Export.FileRef.Endian);
312 while (Export.FileRef.TryGetUExport(childIdx, out var parsingExp))
313 {
314 var data = parsingExp.DataReadOnly;
315 if (parsingExp.ObjectName == "ReturnValue")
316 {
317 if (parsingExp.ClassName == "ObjectProperty" || parsingExp.ClassName == "StructProperty")
318 {
319 var uindexOfOuter = EndianReader.ToInt32(data, parsingExp.DataSize - 4, _self.FileRef.Endian);
320 IEntry entry = parsingExp.FileRef.GetEntry(uindexOfOuter);
321 if (entry != null)
322 {
323 return entry.ObjectName;
324 }
325 }
326
327 return parsingExp.ClassName;
328 }
329
330 var nCdx = EndianReader.ToInt32(data, 0x10, Export.FileRef.Endian);
331 if (nCdx == childIdx || (parsingExp.Parent.UIndex != Export.UIndex)) // not sure what second half of this if statement is for, but i'm not going not modify it
