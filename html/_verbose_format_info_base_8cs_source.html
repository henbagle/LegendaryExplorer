577 internal virtual bool AppendFriendlyTypeName(Type type, object instance, StringBuilder sb, ref int maxLength)
578 {
579 var success = true;
580 var originalLength = sb.Length;
581
582 if (type != null)
583 {
584 var typeName = type.Name;
585 var backQuotePos = typeName.IndexOf('`');
586 if (backQuotePos >= 0)
587 {
588 typeName = typeName.Substring(0, backQuotePos);
589 }
590
591 success = FormatInfoUtils.TryAppendString(this, sb, typeName, success, ref maxLength);
592
593 if (success && type.IsGenericType)
594 {
595 var genericArguments = type.GetGenericArguments();
596
597 success = FormatInfoUtils.TryAppendString(this, sb, "<", success, ref maxLength);
598
599 for (var i = 0; success && i < genericArguments.Length; i++)
600 {
601 if (i > 0)
602 {
603 success = FormatInfoUtils.TryAppendString(this, sb, ", ", success, ref maxLength);
604 }
605 success = success && AppendFriendlyTypeName(genericArguments[i], null, sb, ref maxLength);
606 }
607
608 success = FormatInfoUtils.TryAppendString(this, sb, ">", success, ref maxLength);
609 }
610 }
611
612 if (!success)
613 {
614 sb.Length = originalLength;
615 }
616
617 return success;

711 internal bool FormatLinePrefix(StringBuilder sb, bool firstValueFollows, bool lastValueFollows, bool skipRightMostIndentationString,
712 int padLineToLength, ref int maxLength)
713 {
714 var originalLength = sb.Length;
715 var success = true;
716
717 if (firstValueFollows && _isMultiLinedFormat)
718 {
719 success = FormatInfoUtils.TryAppendString(this, sb, Environment.NewLine, success, ref maxLength);
720 }
721 else if (!firstValueFollows)
722 {
723 success = FormatInfoUtils.TryAppendString(this, sb, _fieldDelimiter, success, ref maxLength);
724 }
725
726 if (_isMultiLinedFormat)
727 {
728 success = FormatInfoUtils.TryAppendString(this, sb, _linePrefix, success, ref maxLength);
729
730 for (var i = 0; success && i < IndentationLevel; i++)
731 {
732 string indentString = null;
733 string lastIndentString = null;
734
735 if (IsIndentationLevelOccupied(i + 1))
736 {
737 indentString = IndentationString;
738 lastIndentString = LastIndentationString;
739 }
740 else
741 {
742 indentString = string.Empty;
743 for (var j = 0; j < IndentationString.Length; j++)
744 {
745 if (IndentationString[j] == '\t')
746 {
747 indentString += '\t';
748 }
749 else
750 {
751 indentString += ' ';
752 }
753 }
754
755 lastIndentString = string.Empty;
756 for (var j = 0; j < LastIndentationString.Length; j++)
757 {
758 if (LastIndentationString[j] == '\t')
759 {
760 lastIndentString += '\t';
761 }
762 else
763 {
764 lastIndentString += ' ';
765 }
766 }
767 }
768
769 var rightMostIndentString = (skipRightMostIndentationString ? indentString : RightMostIndentationString);
770 var lastRightMostIndentString = (skipRightMostIndentationString ? lastIndentString : LastRightMostIndentationString);
771
772 if (lastValueFollows && i == IndentationLevel - 1)
773 {
774 success = FormatInfoUtils.TryAppendString(this, sb,
775 lastRightMostIndentString ?? lastIndentString ?? rightMostIndentString ?? indentString, success, ref maxLength);
776 }
777 else if (lastValueFollows)
778 {
779 success = FormatInfoUtils.TryAppendString(this, sb, lastIndentString ?? indentString, success, ref maxLength);
780 }
781 else if (i == IndentationLevel - 1)
782 {
783 success = FormatInfoUtils.TryAppendString(this, sb, rightMostIndentString ?? indentString, success, ref maxLength);
784 }
785 else
786 {
787 success = FormatInfoUtils.TryAppendString(this, sb, indentString, success, ref maxLength);
788 }
789 }
790
791 if (padLineToLength > 0)
792 {
793 var lineLength = GetCurrentLineLength(sb);
794 while (success && lineLength++ < padLineToLength)
795 {
796 success = FormatInfoUtils.TryAppendChar(this, sb, ' ', success, ref maxLength);
797 }
798 }
799 }
800
801 if (!success)
802 {
803 sb.Length = originalLength;
804 }
805
806 return success;
807 }
