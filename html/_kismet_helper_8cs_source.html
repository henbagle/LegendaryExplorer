163 public static void AddObjectToSequence(ExportEntry newObject, ExportEntry sequenceExport, bool removeLinks = false)
164 {
165 if (sequenceExport.ClassName is not "SequenceReference")
166 {
167 ArrayProperty<ObjectProperty> seqObjs = sequenceExport.GetProperty<ArrayProperty<ObjectProperty>>("SequenceObjects") ?? new ArrayProperty<ObjectProperty>("SequenceObjects");
168 seqObjs.Add(new ObjectProperty(newObject));
169 sequenceExport.WriteProperty(seqObjs);
170 }
171
172 PropertyCollection newObjectProps = newObject.GetProperties();
173 newObjectProps.AddOrReplaceProp(new ObjectProperty(sequenceExport, "ParentSequence"));
174 newObjectProps.RemoveNamedProperty("ObjPosX");
175 newObjectProps.RemoveNamedProperty("ObjPosY");
176 newObject.WriteProperties(newObjectProps);
177 if (removeLinks)
178 {
179 RemoveAllLinks(newObject);
180 }
181 newObject.Parent = sequenceExport;
182 }

191 public static List<List<OutboundLink>> GetOutboundLinksOfNode(ExportEntry node)
192 {
193 var outputLinksMapping = new List<List<OutboundLink>>();
194 var outlinksProp = node.GetProperty<ArrayProperty<StructProperty>>("OutputLinks");
195 if (outlinksProp != null)
196 {
197 int i = 0;
198 foreach (var ol in outlinksProp)
199 {
200 List<OutboundLink> oLinks = new List<OutboundLink>();
201 outputLinksMapping.Add(oLinks);
202
203 var links = ol.GetProp<ArrayProperty<StructProperty>>("Links");
204 if (links != null)
205 {
206 foreach (var l in links)
207 {
208 oLinks.Add(OutboundLink.FromStruct(l, node.FileRef));
209 }
210 }
211
212 i++;
213 }
214 }
215
216 return outputLinksMapping;
217 }
