27 protected StatementList ReadBytecode(out BytecodeReader bcReader)
28 {
29 var s = new MemoryStream(_bytecode);
30 var reader = new EndianReader(s) { Endian = _self.FileRef.Endian };
31 bcReader = new BytecodeReader(_self.FileRef, reader);
32 var statements = new StatementList((Statement)null);
33 bool keepParsing = true;
34 while (s.Position < s.Length && keepParsing)
35 {
36 int startOffset = (int)s.Position;
37 //Debug.WriteLine("Reading token at 0x" + startOffset.ToString("X4"));
38
39 BytecodeToken bc;
40 try
41 {
42 bc = bcReader.ReadNext();
43 }
44 catch (Exception)
45 {
46 //extra bytes at the end can trigger this. They are not used. may be something like byte aligning or something
47 break;
48 }
49 if (bc == null /*|| bc is EndOfScriptToken*/) break;
50 statements.Add(new Statement(startOffset, (int)s.Position, bc, bcReader));
51 if (bc is ErrorBytecodeToken errorToken)
52 {
53 int bytecode = errorToken.UnknownBytecode;
54 if (bytecode >= 0)
55 {
56 ProblemRegistry.RegisterUnknownBytecode((byte)bytecode, this, errorToken.SubsequentBytes);
57 }
58 else
59 {
60 ProblemRegistry.RegisterBytecodeError(this, errorToken.ToString());
61 }
62 break;
63 }
64
65 if (/*bc is LabelTableToken || /*bc is StopToken ||*/ bc is EndOfScriptToken)
66 {
67 keepParsing = false;
68 break; //Nothing else to parse is bytecode
69 }
70 }
71 return statements;
72 }

75 {
76 var labelTableStatement = statements.Find(s => s.Token is LabelTableToken);
