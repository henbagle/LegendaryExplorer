18 public Expression DecompileExpression()
19 {
20 StartPositions.Push((ushort)Position);
21 var token = PeekByte;
22
23 if (token >= extNativeIndex)
24 {
25 PopByte();
26 int nativeIndex;
27 if ((token & 0xF0) == extNativeIndex) //Extended Native
28 {
29 byte b2 = ReadByte();
30 nativeIndex = ((token - extNativeIndex) << 8) + b2;
31 }
32 else
33 {
34 nativeIndex = token; //Native
35 }
36 return DecompileNativeFunction(nativeIndex);

312 public Expression DecompileObjectLookup()
313 {
314 var obj = ReadObject();
315 if (obj == null)
316 return new SymbolReference(null, "ERRORNULL"); // ERROR
317
318 StartPositions.Pop();
319
320 //attempt to resolve Enum references so that byte constants can be converted to enum values
321 ASTNode node = ResolveEnumReference(obj);
322
323 return new SymbolReference(node, obj.ObjectName.Instanced);
324 }

326 private ASTNode ResolveEnumReference(IEntry obj)
327 {
328 ASTNode node = null;
329 if (obj.ClassName == "ByteProperty")
330 {
331 if (LibInitialized)
332 {
333 IEntry typeExp = obj.Parent;
334 string scope = null;
335 while (typeExp.ClassName != "Class" && typeExp.ClassName != "ScriptStruct")
336 {
337 scope = scope is null ? typeExp.ObjectName.Name : $"{typeExp.ObjectName}.{scope}";
338 typeExp = typeExp.Parent;
339 }
340
341 if (ReadOnlySymbolTable.TryGetType(typeExp.ObjectName, out VariableType cls))
342 {
343 scope = scope is null ? cls.GetScope() : $"{cls.GetScope()}.{scope}";
344 if (ReadOnlySymbolTable.TryGetSymbolFromSpecificScope(obj.ObjectName, out ASTNode astNode, scope)
345 && astNode is VariableDeclaration {VarType: Enumeration enumeration})
346 {
347 node = enumeration;
348 }
349 }
350 }
351
352 if (node is null && obj is ExportEntry exp &&
353 Pcc.GetEntry(exp.GetBinaryData<UByteProperty>().Enum) is IEntry enumEntry)
354 {
355 if (enumEntry is ExportEntry enumExp)
356 {
357 node = ScriptObjectToASTConverter.ConvertEnum(enumExp.GetBinaryData<UEnum>());
358 }
359 else if (LibInitialized &&
360 ReadOnlySymbolTable.TryGetType(enumEntry.ObjectName, out Enumeration enumeration))
361 {
362 node = enumeration;
363 }
364 }
365 }
366
367 return node;
368 }

380 public Expression DecompileContext(bool isClass = false)
381 {
382 PopByte();
383
384 var left = DecompileExpression();
385 if (left == null)
386 return null; // ERROR
387
388 ReadInt16(); // discard MemSize value. (memory size of expr-right)
389 if (Game >= MEGame.ME3)
390 {
391 ReadObject(); // discard RetValRef.
392 }
393 var propType = ReadByte(); // discard propType.
394
395 isInContextExpression = true;
396 var right = DecompileExpression();
397 if (right == null)
398 return null; // ERROR
399 isInContextExpression = false;
400
401 //testing code TODO: remove when done testing
402 //if (false)//Game <= MEGame.ME2)
403 //{
404 // if (right is not PrimitiveCast && propType is not (0 or 1 or 4 or 8 or 12 or 0x24))
405 // {
406 // string message = $"proptype: {propType} for expression of type {right.GetType()}";
407 // Debugger.Log(0, "", $"{message}\n");
408 // }
409 //}
410
411 StartPositions.Pop();
412 switch (right)
413 {
414 //A const accessed by an instance of another class can be compiled as EX_Context(instancevar, literal),
415 //which, if decompiled naively, can lead to nonsense like: Weapon.-1
416 //in such a case, the context expression can be safely replaced with the literal
417 case IntegerLiteral _:
418 case FloatLiteral _:
419 case BooleanLiteral _:
420 case StringLiteral _:
421 case NameLiteral _:
422 case StringRefLiteral _:
423 return right;
424 default:
425 return new CompositeSymbolRef(left, right, isClass);
426 }

538 public Expression DecompileNativeFunction(int index)
539 {
540 var parameters = new List<Expression>();
541 while (!CurrentIs(OpCodes.EndFunctionParms))
542 {
543 var param = DecompileExpression();
544 if (param == null)
545 return null; // ERROR
546
547 parameters.Add(param);
548 }
549 PopByte();
550
551 var entry = NativeTable[index];
552 Expression call = null;
553
554 switch (entry.Type)
555 {
556 case NativeType.Function:
557 var func = new SymbolReference(null, entry.Name);
558 call = new FunctionCall(func, parameters, null, null);
559 break;
560
561 case NativeType.Operator:
562 var op = new InOpDeclaration(entry.Name, entry.Precedence, index, null, null, null);
563 var opRef = new InOpReference(op, parameters[0], parameters[1]);
564 DecompileEnumOperatorComparisons(opRef);
565 call = opRef;
566 break;
567
568 case NativeType.PreOperator:
569 var preOp = new PreOpDeclaration(entry.Name, null, index, null);
570 call = new PreOpReference(preOp, parameters[0]);
571 break;
572
573 case NativeType.PostOperator:
574 var postOp = new PostOpDeclaration(entry.Name, null, index, null);
575 call = new PostOpReference(postOp, parameters[0], null, null);
576 break;
577 }
578
579 StartPositions.Pop();
580 return call;
581 }

583 private static void DecompileEnumOperatorComparisons(InOpReference opRef)
584 {
585 switch (opRef.Operator.OperatorKeyword)
586 {
587 case "==":
588 case "!=":
589 case ">":
590 case "<":
591 case ">=":
592 case "<=":
593 if (!ResolveEnumValues(ref opRef.LeftOperand, ref opRef.RightOperand))
594 {
595 ResolveEnumValues(ref opRef.RightOperand, ref opRef.LeftOperand);
596 }
597 break;
598 }
599
600 }
601 static bool ResolveEnumValues(ref Expression a, ref Expression b)
602 {
603 SymbolReference symRef = a as SymbolReference;
604 if (symRef is null && a is CastExpression cast && cast.CastType.Name == INT)
605 {
606 symRef = cast.CastTarget as SymbolReference;
607 }
608
609 if (symRef?.Node is Enumeration enm)
610 {
611 switch (b)
612 {
613 case IntegerLiteral {Value: >= 0} intLit when intLit.Value < enm.Values.Count:
614 a = symRef;
615 b = new SymbolReference(enm.Values[intLit.Value], enm.Values[intLit.Value].Name);
616 return true;
617 case ConditionalExpression {TrueExpression: IntegerLiteral { Value: >= 0 } trueLit, FalseExpression: IntegerLiteral { Value: >= 0 } falseLit} condExpr:
618 a = symRef;
619 condExpr.TrueExpression = new SymbolReference(enm.Values[trueLit.Value], enm.Values[trueLit.Value].Name);
620 condExpr.FalseExpression = new SymbolReference(enm.Values[falseLit.Value], enm.Values[falseLit.Value].Name);
621 return true;
622 }
623 }
624
625 return false;
626 }

675 public Expression DecompileFunctionCall(bool byName = false, bool withFuncListIdx = false, bool global = false)
676 {
677 PopByte();
678 string funcName;
679 bool isSuper = false;
680 VariableType superSpecifier = null;
681 if (byName)
682 {
683 funcName = ReadNameReference();
684 }
685 else
686 {
687 var funcObj = ReadObject();
688 funcName = funcObj.ObjectName.Instanced;
689
690 if (NonNativeOperators.TryGetValue(funcName, out InOpDeclaration opDecl))
691 {
692 Expression parm1 = DecompileExpression();
693 if (parm1 is null)
694 {
695 return null;
696 }
697 Expression parm2 = DecompileExpression();
698 if (parm2 is null)
699 {
700 return null;
701 }
702 PopByte();//EndFunctionParms
703
704 StartPositions.Pop();
705 return new InOpReference(opDecl, parm1, parm2);
706 }
707
708 if (IsSuper(funcObj)) // If we're calling ourself, it's a super call
709 {
710 var classExp = DataContainer.Export.Parent;
711 while (classExp != null && classExp.ClassName != "Class")
712 {
713 classExp = classExp.Parent;
714 }
715 var currentClass = (classExp as ExportEntry).GetBinaryData<UClass>();
716 classExp = funcObj;
717 while (classExp != null && classExp.ClassName != "Class")
718 {
719 classExp = classExp.Parent;
720 }
721 var funcOuterClass = classExp?.ObjectName.Instanced;
722 isSuper = true;
723 if (currentClass == null || currentClass.SuperClass == 0 || currentClass.SuperClass.GetEntry(Pcc).ObjectName.Instanced != funcOuterClass)
724 {
725 superSpecifier = new VariableType(funcOuterClass);
726 }
727 }
728 }
729
730 if (withFuncListIdx)
731 ReadInt16();
732
733 List<Expression> parameters = DecompileArgumentList();
734 if (parameters is null)
735 {
736 return null;
737 }
738
739 StartPositions.Pop();
740 var func = new SymbolReference(null, funcName)
741 {
742 IsGlobal = global,
743 IsSuper = isSuper,
744 SuperSpecifier = superSpecifier
745 };
746 return new FunctionCall(func, parameters, null, null);
747 }

749 private List<Expression> DecompileArgumentList()
750 {
751 var parameters = new List<Expression>();
752 while (!CurrentIs(OpCodes.EndFunctionParms))
753 {
754 if (CurrentIs(OpCodes.Nothing) || CurrentIs(OpCodes.EmptyParmValue))
755 {
756 PopByte(); // TODO: is this reasonable? what does it mean?
757 parameters.Add(null);
758 continue;
759 }
760
761 var param = DecompileExpression();
762 if (param == null)
763 return null;
764
765 parameters.Add(param);
766 }
767
768 PopByte();
769 return parameters;
770 }

772 bool IsSuper(IEntry funcObj)
773 {
774 //can't be a super call if it's being called on a specific object
775 if (isInContextExpression)
776 {
777 return false;
778 }
779
780 string funcName = funcObj.ObjectName.Instanced;
781 ExportEntry parentContextExport = funcObj.Parent as ExportEntry;
782
783 //if the function being called is in the current context, it's not a super call
784 if (parentContextExport == DataContainer.Export.Parent)
785 {
786 return false;
787 }
788
789 //if it's in a parent context, it's a super call if there is a function with the same name in the current context
790 return DataContainer.Export.Parent.GetChildren().Any(child => child.ObjectName.Instanced.CaseInsensitiveEquals(funcName));
791 }
