429 //based on algorithm described here: http://www.codeproject.com/Articles/13525/Fast-memory-efficient-Levenshtein-algorithm
430 public static int LevenshteinDistance(this string a, string b)
431 {
432 int n = a.Length;
433 int m = b.Length;
434 if (n == 0)
435 {
436 return m;
437 }
438 if (m == 0)
439 {
440 return n;
441 }
442
443 var v1 = new int[m + 1];
444 for (int i = 0; i <= m; i++)
445 {
446 v1[i] = i;
447 }
448
449 for (int i = 1; i <= n; i++)
450 {
451 int[] v0 = v1;
452 v1 = new int[m + 1];
453 v1[0] = i;
454 for (int j = 1; j <= m; j++)
455 {
456 int above = v1[j - 1] + 1;
457 int left = v0[j] + 1;
458 int cost;
459 if (j > m || j > n)
460 {
461 cost = 1;
462 }
463 else
464 {
465 cost = a[j - 1] == b[j - 1] ? 0 : 1;
466 }
467 cost += v0[j - 1];
468 v1[j] = Math.Min(above, Math.Min(left, cost));
469
470 }
471 }
472
473 return v1[m];
474 }

757 public static bool IsOfGenericType(this Type typeToCheck, Type genericType, out Type concreteGenericType)
758 {
759 while (true)
760 {
761 concreteGenericType = null;
762
763 if (genericType == null)
764 throw new ArgumentNullException(nameof(genericType));
765
766 if (!genericType.IsGenericTypeDefinition)
767 throw new ArgumentException("The definition needs to be a GenericTypeDefinition", nameof(genericType));
768
769 if (typeToCheck == null || typeToCheck == typeof(object))
770 return false;
771
772 if (typeToCheck == genericType)
773 {
774 concreteGenericType = typeToCheck;
775 return true;
776 }
777
778 if ((typeToCheck.IsGenericType ? typeToCheck.GetGenericTypeDefinition() : typeToCheck) == genericType)
779 {
780 concreteGenericType = typeToCheck;
781 return true;
782 }
783
784 if (genericType.IsInterface)
785 foreach (var i in typeToCheck.GetInterfaces())
786 if (i.IsOfGenericType(genericType, out concreteGenericType))
787 return true;
788
789 typeToCheck = typeToCheck.BaseType;

793 public static float AsFloat16(this ushort float16bits)
794 {
795 int sign = (float16bits >> 15) & 0x00000001;
796 int exp = (float16bits >> 10) & 0x0000001F;
797 int mant = float16bits & 0x000003FF;
798 switch (exp)
799 {
800 case 0:
801 return 0f;
802 case 31:
803 return 65504f;
804 }
805 exp += (127 - 15);
806 int i = (sign << 31) | (exp << 23) | (mant << 13);
807 return BitConverter.ToSingle(BitConverter.GetBytes(i), 0);
808 }

810 public static ushort ToFloat16bits(this float f)
811 {
812 byte[] bytes = BitConverter.GetBytes((double)f);
813 ulong bits = BitConverter.ToUInt64(bytes, 0);
814 ulong exponent = bits & 0x7ff0000000000000L;
815 ulong mantissa = bits & 0x000fffffffffffffL;
816 ulong sign = bits & 0x8000000000000000L;
817 int placement = (int)((exponent >> 52) - 1023);
818 if (placement > 15 || placement < -14)
819 return 0;
820 ushort exponentBits = (ushort)((15 + placement) << 10);
821 ushort mantissaBits = (ushort)(mantissa >> 42);
822 ushort signBits = (ushort)(sign >> 48);
823 return (ushort)(exponentBits | mantissaBits | signBits);
824 }

848 public static Rotator GetRotator(this Matrix4x4 m)
849 {
850 var xAxis = new Vector3(m.M11, m.M12, m.M13);
851 var yAxis = new Vector3(m.M21, m.M22, m.M23);
852 var zAxis = new Vector3(m.M31, m.M32, m.M33);
853
854 var pitch = Math.Atan2(xAxis.Z, Math.Sqrt(Math.Pow(xAxis.X, 2) + Math.Pow(xAxis.Y, 2)));
855 var yaw = Math.Atan2(xAxis.Y, xAxis.X);
856
857 var sy = Math.Sin(yaw);
858 var cy = Math.Cos(yaw);
859
860
861 var syAxis = new Vector3((float)-sy, (float)cy, 0f);
862
863 var roll = Math.Atan2(Vector3.Dot(zAxis, syAxis), Vector3.Dot(yAxis, syAxis));
864
865 return new Rotator(RadToURR(pitch), RadToURR(yaw), RadToURR(roll));
