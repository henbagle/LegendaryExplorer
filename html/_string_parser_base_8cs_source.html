88 public VariableIdentifier ParseVariableIdentifier()
89 {
90 var name = Consume(TokenType.Word);
91 if (name == null) return null;
92 if (name.Value.CaseInsensitiveEquals("None"))
93 {
94 TypeError("'None' cannot be used as a variable name!", name);
95 }
96
97 if (Consume(TokenType.LeftSqrBracket) != null)
98 {
99 var size = Consume(TokenType.IntegerNumber);
100 if (size == null)
101 {
102 throw ParseError("Expected an integer number for size!", CurrentPosition);
103 }
104
105 if (Consume(TokenType.RightSqrBracket) == null)
106 {
107 throw ParseError("Expected ']'!", CurrentPosition);
108 }
109
110 return new VariableIdentifier(name.Value, name.StartPos, name.EndPos, int.Parse(size.Value));
111 }
112
113 return new VariableIdentifier(name.Value, name.StartPos, name.EndPos);
114 }

116 public VariableType ParseTypeRef()
117 {
118 if (Matches(ARRAY, EF.Keyword))
119 {
120 var arrayToken = Tokens.Prev();
121 if (Consume(TokenType.LeftArrow) is null)
122 {
123 throw ParseError("Expected '<' after 'array'!", CurrentPosition);
124 }
125 var elementType = ParseTypeRef();
126 if (elementType == null)
127 {
128 throw ParseError("Expected element type for array!", CurrentPosition);
129 }
130
131 if (elementType is DynamicArrayType)
132 {
133 throw ParseError("Arrays of Arrays are not supported!", elementType.StartPos, elementType.EndPos);
134 }
135 if (Consume(TokenType.RightArrow) is null)
136 {
137 throw ParseError("Expected '>' after array type!", CurrentPosition);
138 }
139 return new DynamicArrayType(elementType, arrayToken.StartPos, CurrentPosition);
140 }
141
142 if (Matches(DELEGATE, EF.Keyword))
143 {
144 var delegateToken = Tokens.Prev();
145 if (Consume(TokenType.LeftArrow) is null)
146 {
147 throw ParseError("Expected '<' after 'delegate'!", CurrentPosition);
148 }
149
150 string functionName = "";
151 do
152 {
153 if (Consume(TokenType.Word) is ScriptToken identifier)
154 {
155 identifier.SyntaxType = EF.Function;
156 if (functionName.Length > 0)
157 {
158 functionName += ".";
159 }
160 functionName += identifier.Value;
161 }
162 else
163 {
164 throw ParseError("Expected function name for delegate!", CurrentPosition);
165 }
166 } while (Matches(TokenType.Dot, EF.Function));
167 if (Consume(TokenType.RightArrow) is null)
168 {
169 throw ParseError("Expected '>' after function name!", CurrentPosition);
170 }
171 return new DelegateType(new Function(functionName, default, null, null, null), delegateToken.StartPos, CurrentPosition);
172 }
173
174 if (Consume(CLASS) is { } classToken)
175 {
176 classToken.SyntaxType = EF.Keyword;
177 if (Consume(TokenType.LeftArrow) is null)
178 {
179 return new ClassType(new VariableType(OBJECT));
180 }
181
182 if (!(Consume(TokenType.Word) is { } classNameToken))
183 {
184 throw ParseError("Expected class name!", CurrentPosition);
185 }
186
187 classNameToken.SyntaxType = EF.TypeName;
188
189 if (Consume(TokenType.RightArrow) is null)
190 {
191 throw ParseError("Expected '>' after class name!", CurrentPosition);
192 }
193 return new ClassType(new VariableType(classNameToken.Value), classToken.StartPos, PrevToken.EndPos);
194 }
195
196 ScriptToken type = Consume(TokenType.Word);
197 if (type == null)
198 {
199 return null;
200 }
201
202 type.SyntaxType = type.Value is INT or FLOAT or BOOL or BYTE or BIOMASK4 or STRING or STRINGREF or NAME ? EF.Keyword : EF.TypeName;
203 return new VariableType(type.Value, type.StartPos, type.EndPos);
204 }

284 public static bool TypeCompatible(VariableType dest, VariableType src, bool coerce = false)
285 {
286 if (dest is DynamicArrayType destArr && src is DynamicArrayType srcArr)
287 {
288 return TypeCompatible(destArr.ElementType, srcArr.ElementType);
289 }
290
291 if (dest is ClassType destClassType && src is ClassType srcClassType)
292 {
293 return destClassType.ClassLimiter == srcClassType.ClassLimiter || ((Class)srcClassType.ClassLimiter).SameAsOrSubClassOf(destClassType.ClassLimiter.Name);
294 }
295
296 if (dest.PropertyType == EPropertyType.Byte && src.PropertyType == EPropertyType.Byte)
297 {
298 return true;
299 }
300
301 if (dest is DelegateType destDel && src is DelegateType srcDel)
302 {
303 return true;
304 // this seems like how it ought to be done, but there is bioware code that would have only compiled if all delegates are considered the same type
305 // maybe log a warning here instead of an error?
306 //return destDel.DefaultFunction.SignatureEquals(srcDel.DefaultFunction);
307 }
308
309 if (dest is Class destClass)
310 {
311 if (src is Class srcClass)
312 {
313 bool sameAsOrSubClassOf = srcClass.SameAsOrSubClassOf(destClass);
314 if (srcClass.IsInterface)
315 {
316 return sameAsOrSubClassOf || destClass.Implements(srcClass);
317 }
318
319 if (destClass.IsInterface)
320 {
321 return sameAsOrSubClassOf || srcClass.Implements(destClass);
322 }
323 return sameAsOrSubClassOf
324 //this seems super wrong obviously. A sane type system would require an explicit downcast.
325 //But to make this work with existing bioware code, it's this, or write a control-flow analyzer that implicitly downcasts based on typecheck conditional gates
326 //I have chosen the lazy path
327 || destClass.SameAsOrSubClassOf(srcClass);
328 }
329
330 if (destClass.Name.CaseInsensitiveEquals("Object") && src is ClassType)
331 {
332 return true;
333 }
334
335 if (src is null)
336 {
337 return true;
338 }
339 }
340
341 if (dest.Name.CaseInsensitiveEquals(src?.Name)) return true;
342 ECast cast = CastHelper.GetConversion(dest, src);
343 if (coerce)
344 {
345 return cast != ECast.Max;
346 }
347 return cast.Has(ECast.AutoConvert);
348 }

352 public Expression ParseLiteral()
353 {
354 ScriptToken token = CurrentToken;
355 if (Matches(TokenType.IntegerNumber))
356 {
357 int val = int.Parse(token.Value, CultureInfo.InvariantCulture);
358 return new IntegerLiteral(val, token.StartPos, token.EndPos) { NumType = INT };
359 }
360
361 if (Matches(TokenType.FloatingNumber))
362 {
363 return new FloatLiteral(float.Parse(token.Value, CultureInfo.InvariantCulture), token.StartPos, token.EndPos);
364 }
365
366 if (Matches(TokenType.StringLiteral))
367 {
368 return new StringLiteral(token.Value, token.StartPos, token.EndPos);
369 }
370
371 if (Matches(TokenType.NameLiteral))
372 {
373 if (token.Value.Length > 63)
374 {
375 Log.LogWarning($"Names should be less than 64 characters! (This name is {token.Value.Length})", token.StartPos, token.EndPos);
376 }
377 return new NameLiteral(token.Value, token.StartPos, token.EndPos);
378 }
379
380 if (Matches(TokenType.StringRefLiteral))
381 {
382 return new StringRefLiteral(int.Parse(token.Value, CultureInfo.InvariantCulture), token.StartPos, token.EndPos);
383 }
384
385 if (Matches(TRUE, FALSE))
386 {
387 token.SyntaxType = EF.Keyword;
388 return new BooleanLiteral(bool.Parse(token.Value), token.StartPos, token.EndPos);
389 }
390
391 if (Matches(NONE, EF.Keyword))
392 {
393 return new NoneLiteral(token.StartPos, token.EndPos);
394 }
395
396 if (CurrentIs(VECT) && Tokens.LookAhead(1).Type == TokenType.LeftParenth)
397 {
398 token.SyntaxType = EF.Keyword;
399 Tokens.Advance();
400 return ParseVectorLiteral();
401 }
402
403 if (CurrentIs(ROT) && Tokens.LookAhead(1).Type == TokenType.LeftParenth)
404 {
405 token.SyntaxType = EF.Keyword;
406 Tokens.Advance();
407 return ParseRotatorLiteral();
408 }
409
410 return null;
411 }

501 protected Expression ParseObjectLiteral(ScriptToken className, ScriptToken objName, bool noActors = true)
502 {
503 className.SyntaxType = EF.TypeName;
504 bool isClassLiteral = className.Value.CaseInsensitiveEquals(CLASS);
505
506 var classType = new VariableType((isClassLiteral ? objName : className).Value);
507 if (!Symbols.TryResolveType(ref classType))
508 {
509 throw ParseError($"No type named '{classType.Name}' exists!", className);
510 }
511
512 if (classType is Class cls)
513 {
514 if (isClassLiteral)
515 {
516 objName.AssociatedNode = classType;
517 classType = new ClassType(classType);
518 }
519 else
520 {
521 if (noActors && cls.SameAsOrSubClassOf("Actor"))
522 {
523 TypeError("Object constants must not be Actors!", className);
524 }
525
526 className.AssociatedNode = classType;
527 }
528
529
530 return new ObjectLiteral(new NameLiteral(objName.Value, objName.StartPos, objName.EndPos), classType, className.StartPos, objName.EndPos);
531 }
532
533 throw ParseError($"'{classType.Name}' is not a class!", className);
534 }

569 public Expression ParseConstValue()
570 {
571 //minus sign is not parsed as part of a literal, so do it manually
572 bool isNegative = Matches(TokenType.MinusSign);
573
574 Expression literal = ParseLiteral();
575 if (literal is null)
576 {
577 TypeError("Expected a literal value for the constant!", CurrentPosition);
578 }
579
580 if (isNegative)
581 {
582 switch (literal)
583 {
584 case FloatLiteral floatLiteral:
585 floatLiteral.Value *= -1;
586 break;
587 case IntegerLiteral integerLiteral:
588 integerLiteral.Value *= -1;
589 break;
590 default:
591 TypeError("Malformed constant value!", CurrentPosition);
592 break;
593 }
594 }
595
596 return literal;
597 }
