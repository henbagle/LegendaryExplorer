279 private void CopyBytes(long value, int bytes, byte[] buffer, int index)
280 {
281 if (buffer == null)
282 {
283 throw new ArgumentNullException(nameof(buffer));
284 }
285
286 if (buffer.Length < index + bytes)
287 {
288 throw new ArgumentOutOfRangeException(nameof(index));
289 }
290
291 for (var i = 0; i < bytes; i++)
292 {
293 if (_byteOrder == ByteOrder.BigEndian)
294 {
295 buffer[(index + bytes - 1) - i] = unchecked((byte) (value & 0xff));
