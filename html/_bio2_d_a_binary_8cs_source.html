103 public static void Serialize(this SerializingContainer2 sc, ref Bio2DACell cell)
104 {
105 if (sc.IsLoading)
106 {
107 cell = new Bio2DACell
108 {
109 Type = (Bio2DACell.Bio2DADataType)sc.ms.ReadByte()
110 };
111 }
112 else
113 {
114 if (cell.Type == Bio2DACell.Bio2DADataType.TYPE_NULL) return; // DO NOT SERIALIZE!
115 sc.ms.Writer.WriteByte((byte)cell.Type);
116 }
117 switch (cell.Type)
118 {
119 case Bio2DACell.Bio2DADataType.TYPE_INT:
120 int intV = cell.IntValue;
121 sc.Serialize(ref intV);
122 if (sc.IsLoading)
123 cell.IntValue = intV;
124 break;
125 case Bio2DACell.Bio2DADataType.TYPE_NAME:
126 NameReference nameV = cell.NameValue;
127 sc.Serialize(ref nameV);
128 if (sc.IsLoading)
129 cell.NameValue = nameV;
130 break;
131 case Bio2DACell.Bio2DADataType.TYPE_FLOAT:
132 float floatV = cell.FloatValue;
133 sc.Serialize(ref floatV);
134 if (sc.IsLoading)
135 cell.FloatValue = floatV;
136 break;
137 case Bio2DACell.Bio2DADataType.TYPE_NULL:
138 // THIS CELL TYPE IS NOT SERIALIZED AND IS LEC INTERNAL ONLY (so it cell can be populated)
139 if (sc.IsLoading)
140 throw new Exception("Malformed 2DA: NULL cell was written at some point in the past!");
141 break;
142 default:
143 throw new ArgumentOutOfRangeException();
144 }
145 }
146 }
