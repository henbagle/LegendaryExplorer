71 public Bio2DA(ExportEntry export) : this()
72 {
73 //Console.WriteLine("Loading " + export.ObjectName);
74 Export = export;
75
76 // GET ROW NAMES
77 if (export.ClassName == "Bio2DA")
78 {
79 const string rowLabelsVar = "m_sRowLabel";
80 var props = export.GetProperty<ArrayProperty<NameProperty>>(rowLabelsVar);
81 if (props != null)
82 {
83 foreach (NameProperty n in props)
84 {
85 _rowNames.Add(n.Value.Instanced);
86 RowNamesUI.Add(n.Value.Instanced.Replace("_", "__"));
87 }
88 }
89 else
90 {
91 Debug.WriteLine("Unable to find row names property!");
92 }
93 }
94 else
95 {
96 var props = export.GetProperty<ArrayProperty<IntProperty>>("m_lstRowNumbers");//Bio2DANumberedRows
97 if (props != null)
98 {
99 foreach (IntProperty n in props)
100 {
101 _rowNames.Add(n.Value.ToString());
102 RowNamesUI.Add(n.Value.ToString());
103 }
104 }
105 else
106 {
107 Debug.WriteLine("Unable to find row names property (m_lstRowNumbers)!");
108 }
109 }
110
111 var binary = export.GetBinaryData<Bio2DABinary>();
112
113 for (int i = 0; i < binary.ColumnNames.Count; i++)
114 {
115 _columnNames.Add(binary.ColumnNames[i]);
116 mappedColumnNames[binary.ColumnNames[i]] = i;
117 }
118
119 for (int i = 0; i < RowCount; i++)
120 {
121 mappedRowNames[RowNames[i]] = i;
122 }
123
124 Cells = new Bio2DACell[RowCount, ColumnCount];
125 if (ColumnCount > 0) // Prevents division by zero
126 {
127 foreach ((int index, Bio2DACell cell) in binary.Cells)
128 {
129 int row = index / ColumnCount;
130 int col = index % ColumnCount;
131 this[row, col] = cell.Type switch
132 {
133 Bio2DACell.Bio2DADataType.TYPE_INT => new Bio2DACell(cell.IntValue) { package = export.FileRef },
134 Bio2DACell.Bio2DADataType.TYPE_NAME => new Bio2DACell(cell.NameValue, export.FileRef) { package = export.FileRef },
135 Bio2DACell.Bio2DADataType.TYPE_FLOAT => new Bio2DACell(cell.FloatValue) { package = export.FileRef },
136 // Null populated later
137 _ => throw new ArgumentOutOfRangeException()
138 };
139 }
140 }
141
142 // Populate null cells with TYPE_NULL cells
143 for (int row = 0; row < RowCount; row++)
144 {
145 for (int col = 0; col < ColumnCount; col++)
146 {
147 if (this[row, col] == null)
148 {
149 this[row, col] = new Bio2DACell() { package = export.FileRef };
150 }
151 }
152 }
153
154 IsIndexed = binary.IsIndexed;
155 }

188 public void Write2DAToExport(ExportEntry export = null)
189 {
190 var binary = new Bio2DABinary
191 {
192 ColumnNames = ColumnNames.Select(s => new NameReference(s)).ToList(),
193 Cells = new OrderedMultiValueDictionary<int, Bio2DACell>(),
194 Export = Export,
195 IsIndexed = IsIndexed
196 };
197
198 for (int rowindex = 0; rowindex < RowCount; rowindex++)
199 {
200 for (int colindex = 0; colindex < ColumnCount; colindex++)
201 {
202 Bio2DACell cell = Cells[rowindex, colindex];
203 if (cell != null && cell.Type != Bio2DACell.Bio2DADataType.TYPE_NULL)
204 {
205 int index = (rowindex * ColumnCount) + colindex;
206 binary.Cells.Add(index, cell.Type switch
207 {
208 Bio2DACell.Bio2DADataType.TYPE_INT => new Bio2DACell { IntValue = cell.IntValue, Type = Bio2DACell.Bio2DADataType.TYPE_INT },
209 Bio2DACell.Bio2DADataType.TYPE_NAME => new Bio2DACell { NameValue = cell.NameValue, Type = Bio2DACell.Bio2DADataType.TYPE_NAME },
210 Bio2DACell.Bio2DADataType.TYPE_FLOAT => new Bio2DACell { FloatValue = cell.FloatValue, Type = Bio2DACell.Bio2DADataType.TYPE_FLOAT },
211 // NULL IS NOT ADDED
212 _ => throw new ArgumentOutOfRangeException()
213 });
214 }
215 }
216 }
217
218 // This is so newly minted 2DA can be installed into an export.
219 export ??= Export;
220
221 if (RowNames.Count > 0)
222 {
223 Property rowsProp = Export.ClassName switch
224 {
225 "Bio2DA" => new ArrayProperty<NameProperty>(RowNames.Select(n => new NameProperty(n)), "m_sRowLabel"),
226 "Bio2DANumberedRows" => new ArrayProperty<IntProperty>(RowNames.Select(n => new IntProperty(int.Parse(n))), "m_lstRowNumbers"),
227 _ => throw new ArgumentOutOfRangeException()
228 };
229 export.WritePropertyAndBinary(rowsProp, binary);
230 }
231 else
232 {
233 export.RemoveProperty("m_sRowLabel"); // No rows.
234 export.WriteBinary(binary);
235 }

321 private void Expand2DA(bool expandVertically)
322 {
323 // Called before column or row was added.
324
325 int newRowCount = RowCount;
326 int newColCount = ColumnCount;
327 if (expandVertically)
328 newRowCount++;
329 else
330 newColCount++;
331
332 var oldCells = Cells;
333 Cells = new Bio2DACell[newRowCount, newColCount];
334 for (int i = 0; i < RowCount; i++)
335 {
336 for (int j = 0; j < ColumnCount; j++)
337 {
338 Cells[i, j] = oldCells[i, j];
339 }
340 }
341
342 if (expandVertically)
343 {
344 // New row. Populate all columns in this row
345 for (int i = 0; i < ColumnCount; i++)
346 {
347 Cells[newRowCount - 1, i] = new Bio2DACell() { package = Export?.FileRef }; // -1 as it's 0 indexed
348 }
349 }
350 else
351 {
352 // New Column. Populate all rows for this column
353 for (int i = 0; i < RowCount; i++)
354 {
355 Cells[i, newColCount - 1] = new Bio2DACell() { package = Export?.FileRef }; ; // -1 as it's 0 indexed
356 }
357 }
