155 public PropertyInfo(PropertyType type, string reference = null, bool transient = false, int staticArrayLength = 1)
156 {
157 Type = type;
158 Reference = reference;
159 Transient = transient;
160 StaticArrayLength = staticArrayLength;
161 }
162
163 public bool IsEnumProp() => Type == PropertyType.ByteProperty && Reference != null && Reference != "Class" && Reference != "Object";
164
165 public bool IsStaticArray() => StaticArrayLength > 1;
166
167 #region IEquatable
168
169 public bool Equals(PropertyInfo other)
170 {
171 if (other is null) return false;
172 if (ReferenceEquals(this, other)) return true;
173 return Type == other.Type && string.Equals(Reference, other.Reference) && Transient == other.Transient;
174 }
175
176 public override bool Equals(object obj)
177 {
178 if (obj is null) return false;
179 if (ReferenceEquals(this, obj)) return true;
180 return obj.GetType() == this.GetType() && Equals((PropertyInfo)obj);
181 }
182
183 public override int GetHashCode()
184 {
185 unchecked
186 {
187 int hashCode = (int)Type;
188 hashCode = (hashCode * 397) ^ (Reference != null ? Reference.GetHashCode() : 0);
189 hashCode = (hashCode * 397) ^ Transient.GetHashCode();
190 return hashCode;
191 }
192 }
193
194 public static bool operator ==(PropertyInfo left, PropertyInfo right)
195 {
196 return Equals(left, right);
