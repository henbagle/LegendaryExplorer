475 public static MemoryStream DecompressFullyCompressedPackage(EndianReader rawInput, ref UnrealPackageFile.CompressionType compressionType)
476 {
477 rawInput.Position = 0;
478 var magic = rawInput.ReadInt32();
479 var blockSize = rawInput.ReadInt32();
480 var compressedSize = rawInput.ReadInt32();
481 var decompressedSize = rawInput.ReadInt32();
482
483 var blockCount = 0;
484 if (decompressedSize < blockSize)
485 {
486 blockCount = 1;
487 }
488 else
489 {
490 // Calculate the number of blocks
491 blockCount = decompressedSize / blockSize;
492 if (decompressedSize % blockSize != 0) blockCount++; //Add one to decompress the final data
493 }
494
495
496 MemoryStream outStream = MemoryManager.GetMemoryStream();
497 List<(int blockCompressedSize, int blockDecompressedSize)> blockTable = new List<(int blockCompressedSize, int blockDecompressedSize)>();
498
499 // Read Block Table
500 int i = 0;
501 while (i < blockCount)
502 {
503 blockTable.Add((rawInput.ReadInt32(), rawInput.ReadInt32()));
504 i++;
505 }
506
507 int index = 0;
508 foreach (var btInfo in blockTable)
509 {
510 // Decompress
511 //Debug.WriteLine($"Decompressing data at 0x{raw.Position:X8}");
512 var datain = rawInput.ReadToBuffer(btInfo.blockCompressedSize); // This is kinda poor performance. But fully compressed packages aren't on PC
513 if (compressionType == UnrealPackageFile.CompressionType.None)
514 {
515 // We have to determine if it's LZMA or LZX based on first few bytes
516 if (datain[0] == 0x5D && BitConverter.ToInt32(datain, 1) == 0x10000)
517 {
518 // This is LZMA header
519 Debug.WriteLine("Fully compressed package: Detected LZMA compression");
520 compressionType = UnrealPackageFile.CompressionType.LZMA;
521 }
522 else
523 {
524 Debug.WriteLine("Fully compressed package: Detected LZX compression");
525 compressionType = UnrealPackageFile.CompressionType.LZX;
526 }
527 }
528
529 var dataout = MemoryManager.GetByteArray(btInfo.blockDecompressedSize);
530 //var dataout = new byte[btInfo.blockDecompressedSize];
531 if (dataout.Length == datain.Length)
532 {
533 // WiiU SFXGame has weird case where one single block has same sizes and does not have LZMA compression flag for some reason
534 dataout = datain;
535 }
536 else
537 {
538 switch (compressionType)
539 {
540 case UnrealPackageFile.CompressionType.LZO:
541 if (LZO2.Decompress(datain, dataout) != btInfo.blockDecompressedSize)
542 throw new Exception("LZO decompression failed!");
543 break;
544 case UnrealPackageFile.CompressionType.Zlib:
545 if (Zlib.Decompress(datain, dataout) != btInfo.blockDecompressedSize)
546 throw new Exception("Zlib decompression failed!");
547 break;
548 case UnrealPackageFile.CompressionType.LZMA:
549 // Todo: This needs to use MemoryManager system. But it's internally different from others so will take
550 // a bit more work to do
551 // Mgamerz 2/23/2021
552 dataout = LZMA.Decompress(datain, (uint)btInfo.blockDecompressedSize);
553 if (dataout.Length != btInfo.blockDecompressedSize)
554 throw new Exception("LZMA decompression failed!");
555 break;
556 case UnrealPackageFile.CompressionType.LZX:
557 if (LZX.Decompress(datain, (uint)datain.Length, dataout, (uint)btInfo.blockDecompressedSize) != 0)
558 throw new Exception("LZX decompression failed!");
559 break;
560 case UnrealPackageFile.CompressionType.OodleLeviathan:
561 OodleHelper.Decompress(datain, dataout);
562 if (dataout.Length != btInfo.blockDecompressedSize)
563 throw new Exception("Oodle-Leviathan decompression failed!");
564 break;
565 default:
566 throw new Exception("Unknown compression type for this package.");
567 }
568 }
569
570 index++;
571 outStream.Write(dataout, 0, btInfo.blockDecompressedSize); // write from start to decomp size as the amount we receive may not be the actual size of the buffer we got from memory manager
572 MemoryManager.ReturnByteArray(dataout);
573 }
574
575 outStream.Position = 0;
576 return outStream;
577 }

589 public static Stream DecompressPackage(EndianReader raw, long compressionInfoOffset, UnrealPackageFile.CompressionType compressionType = UnrealPackageFile.CompressionType.None,
590 int NumChunks = 0, MEGame game = MEGame.Unknown, GamePlatform platform = GamePlatform.PC, bool canUseLazyDecompression = false)
591 {
592 raw.BaseStream.JumpTo(compressionInfoOffset);
593 if (compressionType == UnrealPackageFile.CompressionType.None)
594 compressionType = (UnrealPackageFile.CompressionType)raw.ReadUInt32();
595
596 if (NumChunks == 0)
597 NumChunks = raw.ReadInt32();
598 var Chunks = new List<Chunk>(NumChunks);
599 var chunkTableStart = raw.Position;
600
601 //DebugOutput.PrintLn("Reading chunk headers...");
602 int maxUncompressedBlockSize = 0;
603 List<byte[]> rentedArrays = null;
604 bool isMemoryStream = false;
605 if (raw.BaseStream is MemoryStream ms && ms.TryGetBuffer(out ArraySegment<byte> seg))
606 {
607 isMemoryStream = true;
608 }
609 else
610 {
611 seg = default;
612 rentedArrays = new List<byte[]>(NumChunks);
613 }
614
615 for (int i = 0; i < NumChunks; i++)
616 {
617 var c = new Chunk
618 {
619 uncompressedOffset = raw.ReadInt32(),
620 uncompressedSize = raw.ReadInt32(),
621 compressedOffset = raw.ReadInt32(),
622 compressedSize = raw.ReadInt32(),
623 };
624
625 var nextChunkPos = raw.Position;
626 if (isMemoryStream)
627 {
628 c.Compressed = seg.AsMemory(c.compressedOffset, c.compressedSize);
629 }
630 else
631 {
632 byte[] arr = MemoryManager.GetByteArray(c.compressedSize);
633 raw.Seek(c.compressedOffset, SeekOrigin.Begin);
634 raw.Read(arr, 0, c.compressedSize);
635 c.Compressed = arr.AsMemory();
636 rentedArrays.Add(arr);
637 }
638
639 c.header = new ChunkHeader
640 {
641 magic = EndianReader.ToInt32(c.Compressed.Span, 0, raw.Endian),
642 // must force block size for ME1 xbox cause in place of block size it seems to list package tag again which breaks loads of things
643 blocksize = (platform == GamePlatform.Xenon && game == MEGame.ME1) ? 0x20000 : EndianReader.ToInt32(c.Compressed.Span, 4, raw.Endian),
644 compressedsize = EndianReader.ToInt32(c.Compressed.Span, 8, raw.Endian),
645 uncompressedsize = EndianReader.ToInt32(c.Compressed.Span, 12, raw.Endian)
646 };
647
648 // Parse block table
649 if (c.header.magic != -1641380927)
650 throw new FormatException("Chunk magic number incorrect");
651 //DebugOutput.PrintLn("Chunkheader read: Magic = " + h.magic + ", Blocksize = " + h.blocksize + ", Compressed Size = " + h.compressedsize + ", Uncompressed size = " + h.uncompressedsize);
652 int pos = 16;
653 int blockCount = c.header.uncompressedsize / c.header.blocksize;
654 if (c.header.uncompressedsize % c.header.blocksize != 0) blockCount++;
655
656 c.blocks = new List<Block>(blockCount);
657 //DebugOutput.PrintLn("\t\t" + count + " Read Blockheaders...");
658 int blockUncompressedOffset = c.uncompressedOffset;
659 for (int j = 0; j < blockCount; j++)
660 {
661 var b = new Block
662 {
663 compressedsize = EndianReader.ToInt32(c.Compressed.Span, pos, raw.Endian),

1003 public PackageDecompressionStream(IEnumerable<Chunk> chunks, int maxBlockSize, UnrealPackageFile.CompressionType compressionType)
1004 {
1005 CompressionType = compressionType;
1006 Chunks = chunks.OrderBy(c => c.uncompressedOffset).ToList();

1009 _length = fullUncompressedSize + firstChunkOffset;
1010 Segment = MemoryManager.GetByteArray(maxBlockSize);
1011 for (chunkIdx = 0; chunkIdx < Chunks.Count; chunkIdx++)

1021 public override int Read(Span<byte> buffer)
1022 {
1023 if (_position == _length)
1024 {
1025 //at end
1026 return 0;
1027 }
1028 int segEndPos = segStartPos + SegmentLength;
1029 if (_position < segStartPos)
1030 {
1031 //go to previous block (hopefully never happens)
1032 chunkIdx = 0;
1033 blockIdx = -1;
1034 return DecompressNewBlockAndRead(buffer);
1035 }
1036 if (_position >= segEndPos)
1037 {
1038 //go to next block
1039 return DecompressNewBlockAndRead(buffer);
1040 }
1041 int count = buffer.Length;
1042 if (_position + count > segEndPos)
1043 {
1044 //multi-segment read
1045 int bytesRemainingInBlock = SegmentLength - SegmentPosition;
1046 Segment.AsSpan(SegmentPosition, bytesRemainingInBlock).CopyTo(buffer);
1047 Seek(bytesRemainingInBlock, SeekOrigin.Current);
1048 return bytesRemainingInBlock + Read(buffer[bytesRemainingInBlock..]);
1049 }
1050 Segment.AsSpan(SegmentPosition, count).CopyTo(buffer);
1051 Seek(count, SeekOrigin.Current);
1052 return count;
1053 }

1060 private int DecompressNewBlockAndRead(Span<byte> buffer)
1061 {
1062 for (; chunkIdx < Chunks.Count; chunkIdx++)
1063 {
1064 Chunk chunk = Chunks[chunkIdx];
1065 if (chunk.uncompressedOffset <= _position && chunk.uncompressedOffset + chunk.uncompressedSize > _position)
1066 {
1067 for (blockIdx++; blockIdx < chunk.blocks.Count; blockIdx++)
1068 {
1069 Block block = chunk.blocks[blockIdx];
1070 if (block.uncompressedOffset <= _position && block.uncompressedOffset + block.uncompressedsize > _position)
1071 {
1072 DecompressBlock();
1073 return Read(buffer);
1074 }
1075 }
1076 }
1077 blockIdx = -1;
1078 }
1079
1080 return 0;

1083 private void DecompressBlock()
1084 {
1085 Chunk chunk = Chunks[chunkIdx];
1086 Block b = chunk.blocks[blockIdx];
1087 segStartPos = b.uncompressedOffset;
1088 int blockstart = 16 + 8 * chunk.blocks.Count;
1089 for (int i = 0; i < blockIdx; i++)
1090 {
1091 blockstart += chunk.blocks[i].compressedsize;
1092 }
1093 SegmentLength = b.uncompressedsize;
1094 ReadOnlySpan<byte> datain = chunk.Compressed.Span.Slice(blockstart, b.compressedsize);
1095 Span<byte> dataOut = Segment.AsSpan(0, b.uncompressedsize);
1096 switch (CompressionType)
1097 {
1098 case UnrealPackageFile.CompressionType.LZO:
1099 if (LZO2.Decompress(datain, dataOut) != b.uncompressedsize)
1100 throw new Exception("LZO decompression failed!");
1101 break;
1102 case UnrealPackageFile.CompressionType.Zlib:
1103 if (Zlib.Decompress(datain, dataOut) != b.uncompressedsize)
1104 throw new Exception("Zlib decompression failed!");
1105 break;
1106 case UnrealPackageFile.CompressionType.LZMA:
1107 if (LZMA.Decompress(datain, dataOut) != 0)
1108 throw new Exception("LZMA decompression failed!");
1109 break;
1110 case UnrealPackageFile.CompressionType.LZX:
1111 if (LZX.Decompress(datain, (uint)datain.Length, Segment, (uint)b.uncompressedsize) != 0)
1112 throw new Exception("LZX decompression failed!");
1113 break;
1114 case UnrealPackageFile.CompressionType.OodleLeviathan:
1115 // Error decompressing exception is thrown in decompress method itself
1116 OodleHelper.Decompress(datain, dataOut);
1117 break;
1118 default:
1119 throw new Exception("Unknown compression type for this package.");
1120 }
