90 public void SetHeaderValuesFromByteArray(byte[] value)
91 {
92 if (value is null)
93 {
94 throw new ArgumentNullException(nameof(value), "Cannot set Header to a null value");
95 }
96 if (value.Length != HeaderLength)
97 {
98 throw new ArgumentException(nameof(value), $"Import header must be exactly {HeaderLength} bytes");
99 }
100 var existingHeader = GenerateHeader();
101 if (existingHeader.AsSpan().SequenceEqual(value))
102 {
103 return; //if the data is the same don't write it and trigger the side effects
104 }
105
106 _header = MemoryMarshal.Read<ImportHeader>(value);
107 if (!FileRef.Endian.IsNative)
108 {
109 _header.ReverseEndianness();
110 }
111 //new header may have changed link or name
112 FileRef.InvalidateLookupTable();
113
114 FileRef.IsModified = true; // mark package as modified if the existing header is changing.
115 HeaderChanged = true;
116 }

128 public byte[] GenerateHeader()
129 {
130 var buff = new byte[HeaderLength];
131 if (FileRef.Endian.IsNative)
132 {
133 MemoryMarshal.Write(buff, ref _header);
134 }
135 else

139 MemoryMarshal.Write(buff, ref _header);
