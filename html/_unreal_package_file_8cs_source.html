351 public void RebuildLookupTable()
352 {
353 EntryLookupTable.Clear();
354
355 //pre-order traversal of entry tree
356 //this is superior to just looping through the export and import arrays and calculating the InstancedFullPath anew for each one
357 //as InstancedFullPath has to recurse up from leaf to root, performing multiple string concats per node.
358 var tree = new EntryTree((IMEPackage)this);
359 var stack = new Stack<(TreeNode<IEntry, int>, string, int)>(8); //max tree depth will rarely be more than 8
360 foreach (TreeNode<IEntry, int> root in tree.Roots)
361 {
362 stack.Clear();
363 string objFullPath = root.Data.ObjectName.Instanced;
364 EntryLookupTable[objFullPath] = root.Data;
365 if (root.Children.Count is 0)
366 {
367 continue;
368 }
369 stack.Push((root, objFullPath, 0));
370 while (true)
371 {
372 if (stack.Count is 0)
373 {
374 break;
375 }
376 int i;
377 TreeNode<IEntry, int> node;
378 (node, objFullPath, i) = stack.Pop();
379 if (i + 1 < node.Children.Count)
380 {
381 stack.Push((node, objFullPath, i + 1));
382 }
383
384 node = tree[node.Children[i]];
385 objFullPath = node.Data.ObjectName.AddToPath(objFullPath);
386 EntryLookupTable[objFullPath] = node.Data;
387 if (node.Children.Count > 0)
388 {
389 stack.Push((node, objFullPath, 0));
390 }
391 }
392 }
393
394 this.tree = tree;
395 lookupTableNeedsToBeRegenerated = false;
396 }

452 public void RemoveTrailingTrash()
453 {
454 ExportEntry trashPackage = FindExport(TrashPackageName);
455 if (trashPackage == null)
456 {
457 return;
458 }
459 int trashPackageUIndex = trashPackage.UIndex;
460 //make sure the first trashed export is the trashpackage
461 foreach (ExportEntry exp in exports)
462 {
463 if (exp == trashPackage)
464 {
465 //trashpackage is the first trashed export, so we're good
466 break;
467 }
468 if (exp.idxLink == trashPackageUIndex)
469 {
470 //turn this into trashpackage, turn old trashpackage into regular Trash, and point all trash entries to the new trashpackage
471 exp.ObjectName = TrashPackageName;
472 exp.idxLink = 0;
473 exp.PackageGUID = TrashPackageGuid;
474
475 trashPackage.ObjectName = "Trash";
476 trashPackage.idxLink = exp.UIndex;
477 trashPackage.PackageGUID = Guid.Empty;
478
479 foreach (IEntry entry in trashPackage.GetChildren())
480 {
481 entry.idxLink = exp.UIndex;
482 }
483
484 trashPackage = exp;
485 trashPackageUIndex = trashPackage.UIndex;
486 break;
487 }
488 }
489
490
491 //remove imports
492 for (int i = ImportCount - 1; i >= 0; i--)
493 {
494 ImportEntry lastImport = imports[i];
495 if (lastImport.idxLink != trashPackageUIndex)
496 {
497 //non-trash import, so stop removing
498 break;
499 }
500
501 lastImport.PropertyChanged -= importChanged;
502 imports.RemoveAt(i);
503 updateTools(PackageChange.ImportRemove, lastImport.UIndex);
504 IsModified = true;
505 }
506 if (ImportCount != imports.Count)
507 {
508 ImportCount = imports.Count;
509 PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(nameof(ImportCount)));
510 }
511
512 //remove exports
513 for (int i = ExportCount - 1; i >= 0; i--)
514 {
515 ExportEntry lastExport = exports[i];
516 if (lastExport.idxLink != trashPackageUIndex)
517 {
518 //non-trash export, so stop removing
519 break;
520 }
521
522 lastExport.PropertyChanged -= importChanged;
523 exports.RemoveAt(i);
524 updateTools(PackageChange.ExportRemove, lastExport.UIndex);
525 IsModified = true;
526 }
527 if (ExportCount != exports.Count)
528 {
529 ExportCount = exports.Count;
530 PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(nameof(ExportCount)));
531 }
532 //if there are no more trashed imports or exports, and if the TrashPackage is the last export, remove it
533 if (exports.LastOrDefault() is ExportEntry finalExport && finalExport == trashPackage)
534 {
535 if (trashPackage.GetChildren().IsEmpty())
536 {
537 trashPackage.PropertyChanged -= importChanged;
538 exports.Remove(trashPackage);
539 updateTools(PackageChange.ExportRemove, trashPackage.UIndex);
540 IsModified = true;
541 }
542 }
543
544 if (ExportCount != exports.Count)
545 {
546 ExportCount = exports.Count;
547 PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(nameof(ExportCount)));
548 }
549 InvalidateLookupTable();
550 }

687 private readonly object _updatelock = new();
688 readonly HashSet<PackageUpdate> pendingUpdates = new();
689 const int queuingDelay = 50;
690 private Timer updateTimer;
691
692 private void UpdateToolsCallback(object _)
693 {
694 lock (_updatelock)
695 {
696 updateTimer.Dispose();
697 updateTimer = null;
698 }
699 new TaskFactory(LegendaryExplorerCoreLib.SYNCHRONIZATION_CONTEXT).StartNew(() =>
700 {
701 List<PackageUpdate> updates;
702 lock (_updatelock)
703 {
704 updates = pendingUpdates.ToList();
705 pendingUpdates.Clear();
706 }
707 var removedImports = updates.Where(u => u.Change == PackageChange.ImportRemove).Select(u => u.Index).ToList();
708 var removedExports = updates.Where(u => u.Change == PackageChange.ExportRemove).Select(u => u.Index).ToList();
709 var pendingUpdatesList = new List<PackageUpdate>();
710 //remove add/change updates for entries that have been removed
711 foreach (PackageUpdate upd in updates)
712 {
713 switch (upd.Change)
714 {
715 case PackageChange.ExportAdd:
716 case PackageChange.ExportData:
717 case PackageChange.ExportHeader:
718 {
719 if (!removedExports.Contains(upd.Index))
720 {
721 pendingUpdatesList.Add(upd);
722 }
723 break;
724 }
725 case PackageChange.ImportAdd:
726 case PackageChange.ImportHeader:
727 {
728 if (!removedImports.Contains(upd.Index))
729 {
730 pendingUpdatesList.Add(upd);
731 }
732 break;
733 }
734 default:
735 pendingUpdatesList.Add(upd);
736 break;
737 }
738 }
739 //WeakUsers needs to come before Users so that FileLib will invalidate BEFORE ScriptEditor refreshes.
740 //This is hacky, and some sort of User priority system should be implemented in the future
741 foreach (var item in WeakUsers.Concat(Users))
742 {
743 item.handleUpdate(pendingUpdatesList);
744 }
745 });
746 }

748 protected void updateTools(PackageChange change, int index)
749 {
750 if (Users.Count == 0 && WeakUsers.Count == 0)
751 {
752 return;
753 }
754 var update = new PackageUpdate(change, index);
755 lock (_updatelock)
756 {
757 pendingUpdates.Add(update);
758 updateTimer ??= new Timer(UpdateToolsCallback);
759 updateTimer.Change(queuingDelay, Timeout.Infinite);
760 }
