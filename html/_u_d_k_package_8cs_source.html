127 private UDKPackage(Stream fs, string filePath) : base(filePath != null ? Path.GetFullPath(filePath) : null)
128 {
129 #region Header
130
131 uint magic = fs.ReadUInt32();
132 if (magic != packageTagLittleEndian)
133 {
134 throw new FormatException("Not an Unreal package!");
135 }
136 ushort unrealVersion = fs.ReadUInt16();
137 ushort licenseeVersion = fs.ReadUInt16();
138 FullHeaderSize = fs.ReadInt32();
139 int foldernameStrLen = fs.ReadInt32();
140 //always "None", so don't bother saving result
141 if (foldernameStrLen > 0)
142 folderName = fs.ReadStringLatin1Null(foldernameStrLen);
143 else
144 folderName = fs.ReadStringUnicodeNull(foldernameStrLen * -2);
145
146 Flags = (EPackageFlags)fs.ReadUInt32();
147
148 //if (Flags.Has(EPackageFlags.Compressed))
149 //{
150 // throw new FormatException("Cannot read compressed UDK packages!");
151 //}
152
153 NameCount = fs.ReadInt32();
154 NameOffset = fs.ReadInt32();
155 ExportCount = fs.ReadInt32();
156 ExportOffset = fs.ReadInt32();
157 ImportCount = fs.ReadInt32();
158 ImportOffset = fs.ReadInt32();
159 DependencyTableOffset = fs.ReadInt32();
160 importExportGuidsOffset = fs.ReadInt32();
161 importGuidsCount = fs.ReadInt32();
162 exportGuidsCount = fs.ReadInt32();
163 thumbnailTableOffset = fs.ReadInt32();
164 PackageGuid = fs.ReadGuid();
165
166 uint generationsTableCount = fs.ReadUInt32();
167 if (generationsTableCount > 0)
168 {
169 generationsTableCount--;
170 Gen0ExportCount = fs.ReadInt32();
171 Gen0NameCount = fs.ReadInt32();
172 Gen0NetworkedObjectCount = fs.ReadInt32();
173 }
174 //don't care about other gens, so skip them
175 fs.Skip(generationsTableCount * 12);
176 engineVersion = fs.ReadInt32();
177 cookedContentVersion = fs.ReadInt32();
178
179 //skip compression type chunks. Decompressor will handle that
180 long compressionInfoOffset = fs.Position;
181 fs.SkipInt32();
182 int numChunks = fs.ReadInt32();
183 fs.Skip(numChunks * 16);
184
185 packageSource = fs.ReadUInt32();
186 //additional packages to cook, and texture allocation, but we don't care about those, so we won't read them in.
187
188 #endregion
189 Stream inStream = fs;
190 if (IsCompressed && numChunks > 0)
191 {
192 inStream = CompressionHelper.DecompressPackage(new EndianReader(fs), compressionInfoOffset);
193 }
194
195 var reader = new EndianReader(inStream); //these will always be little endian so we don't actually use this except for passing
196 //through to methods that can use endianness
197
198 inStream.JumpTo(NameOffset);
199 names = new List<string>(NameCount);
200 for (int i = 0; i < NameCount; i++)
201 {
202 var name = inStream.ReadUnrealString();
203 names.Add(name);
204 nameLookupTable[name] = i;
205 inStream.Skip(8);
206 }
207
208 inStream.JumpTo(ImportOffset);
209 imports = new List<ImportEntry>(ImportCount);
210 for (int i = 0; i < ImportCount; i++)
211 {
212 var imp = new ImportEntry(this, reader) { Index = i };
213 imp.PropertyChanged += importChanged;
214 imports.Add(imp);
215 }
216
217 //read exportTable (ExportEntry constructor reads export data)
218 inStream.JumpTo(ExportOffset);
219 exports = new List<ExportEntry>(ExportCount);
220 for (int i = 0; i < ExportCount; i++)
221 {
222 var e = new ExportEntry(this, reader) { Index = i };
223 e.PropertyChanged += exportChanged;
224 exports.Add(e);
225 }
226
227 EntryLookupTable = new CaseInsensitiveDictionary<IEntry>(ExportCount + ImportCount);
228 RebuildLookupTable();
229 }

231 private static void saveByReconstructing(UDKPackage udkPackage, string path, bool isSaveAs, object diskIOSyncLock = null)
232 {
233 var datastream = udkPackage.SaveToStream(false);
234 // Lock writing with the sync object (if not null) to prevent disk concurrency issues
235 // (the good old 'This file is in use by another process' message)
236 if (diskIOSyncLock == null)
237 {
238 datastream.WriteToFile(path ?? udkPackage.FilePath);
239 }
240 else
241 {
242 lock (diskIOSyncLock)
243 {
244 datastream.WriteToFile(path ?? udkPackage.FilePath);
245 }
246 }
247
248
249 if (!isSaveAs)
250 {
251 udkPackage.AfterSave();
252 }
253 }

262 public MemoryStream SaveToStream(bool compress, bool includeAdditionalPackagesToCook = true, bool includeDependencyTable = true)
263 {
264 if (exports.Any(exp => exp.ClassName == "Level"))
265 {
266 Flags |= EPackageFlags.Map;
267 }
268
269 //UDK does not like it when exports have the forced export flag, which is common in ME files
270 foreach (ExportEntry export in exports)
271 {
272 //if (export.ClassName != "Package")
273 {
274 export.ExportFlags &= ~EExportFlags.ForcedExport;
275 }
276 }
277
278 var ms = MemoryManager.GetMemoryStream();
279
280 //just for positioning. We write over this later when the header values have been updated
281 WriteHeader(ms);
282
283 //name table
284 NameOffset = (int)ms.Position;
285 NameCount = Gen0NameCount = names.Count;
286 foreach (string name in names)
287 {
288 ms.WriteUnrealStringLatin1(name);
289 ms.WriteInt32(0);
290 ms.WriteInt32(458768);
291 }
292
293 //import table
294 ImportOffset = (int)ms.Position;
295 ImportCount = imports.Count;
296 foreach (ImportEntry e in imports)
297 {
298 e.SerializeHeader(ms);
299 }
300
301 //export table
302 ExportOffset = (int)ms.Position;
303 ExportCount = Gen0NetworkedObjectCount = Gen0ExportCount = exports.Count;
304 foreach (ExportEntry e in exports)
305 {
306 e.HeaderOffset = (int)ms.Position;
307 e.SerializeHeader(ms);
308 }
309
310 //dependency table
311 DependencyTableOffset = (int)ms.Position;
312 ms.WriteZeros(4 * ExportCount);
313
314 importExportGuidsOffset = (int)ms.Position;
315 importGuidsCount = exportGuidsCount = 0;
316
317 //generate thumbnails
318 ThumbnailTable.Clear();
319 foreach (ExportEntry export in exports)
320 {
321 if (export.IsTexture())
322 {
323 var tex = new Texture2D(export);
324 var mip = tex.GetTopMip();
325 ThumbnailTable.Add(new Thumbnail
326 {
327 ClassName = export.ClassName,
328 PathName = export.InstancedFullPath,
329 Width = mip.width,
330 Height = mip.height,
331 Data = tex.GetPNG(tex.GetTopMip())
332 });
333 }
334 }
335
336 //write thumbnails
337 foreach (Thumbnail thumbnail in ThumbnailTable)
338 {
339 thumbnail.Offset = (int)ms.Position;
340 ms.WriteInt32(thumbnail.Width);
341 ms.WriteInt32(thumbnail.Height);
342 ms.WriteInt32(thumbnail.Data.Length);
343 ms.WriteFromBuffer(thumbnail.Data);
344 //File.WriteAllBytes(Path.Combine(Path.GetDirectoryName(FilePath), $"{thumbnail.PathName}({thumbnail.ClassName}).png"), thumbnail.Data);
345 }
346
347 thumbnailTableOffset = (int)ms.Position;
348 ms.WriteInt32(ThumbnailTable.Count);
349 foreach (Thumbnail thumbnail in ThumbnailTable)
350 {
351 ms.WriteUnrealStringLatin1(thumbnail.ClassName);
352 ms.WriteUnrealStringLatin1(thumbnail.PathName);
353 ms.WriteInt32(thumbnail.Offset);
354 }
355
356 FullHeaderSize = (int)ms.Position;
357
358 //export data
359 foreach (ExportEntry e in exports)
360 {
361 UpdateUDKOffsets(e, (int)ms.Position);
362 e.DataOffset = (int)ms.Position;
363
364 ms.WriteFromBuffer(e.Data);
365 //update size and offset in already-written header
366 long pos = ms.Position;
367 ms.JumpTo(e.HeaderOffset + 32);
368 ms.WriteInt32(e.DataSize); //DataSize might have been changed by UpdateOffsets
369 ms.WriteInt32(e.DataOffset);
370 ms.JumpTo(pos);
371 }
372
373 //re-write header with updated values
374 ms.JumpTo(0);
375 WriteHeader(ms);
376
377 ms.Position = 0;
378 return ms;
379 }

383 private void WriteHeader(Stream ms)
384 {
385 ms.WriteUInt32(packageTagLittleEndian);
386 //version
387 ms.WriteUInt16(UDKUnrealVersion);
388 ms.WriteUInt16(UDKLicenseeVersion);
389
390 ms.WriteInt32(FullHeaderSize);
391 ms.WriteUnrealStringLatin1(folderName);
392
393 ms.WriteUInt32((uint)Flags);
394 ms.WriteInt32(NameCount);
395 ms.WriteInt32(NameOffset);
396 ms.WriteInt32(ExportCount);
397 ms.WriteInt32(ExportOffset);
398 ms.WriteInt32(ImportCount);
399 ms.WriteInt32(ImportOffset);
400 ms.WriteInt32(DependencyTableOffset);
401 ms.WriteInt32(importExportGuidsOffset);
402 ms.WriteInt32(importGuidsCount);
403 ms.WriteInt32(exportGuidsCount);
404 ms.WriteInt32(thumbnailTableOffset);
405 ms.WriteGuid(PackageGuid);
406
407 //Write 1 generation
408 ms.WriteInt32(1);
409 ms.WriteInt32(Gen0ExportCount);
410 ms.WriteInt32(Gen0NameCount);
411 ms.WriteInt32(Gen0NetworkedObjectCount);
412
413 ms.WriteInt32(engineVersion);
414 ms.WriteInt32(cookedContentVersion);
415
416
417 ms.WriteInt32(0);//CompressiontType.None
418 ms.WriteInt32(0);//numChunks
419
420 ms.WriteUInt32(packageSource);
421
422 ms.WriteInt32(0);//empty additionalPackagesToCook array
423 ms.WriteInt32(0);//empty TextureAllocations
424 }

426 private static void UpdateUDKOffsets(ExportEntry export, int newDataOffset)
427 {
428 if (export.IsDefaultObject)
429 {
430 return; //this is not actually instance of that class
431 }
432 //update offsets for pcc-stored mips in Textures
433 if (export.IsTexture())
434 {
435 int baseOffset = newDataOffset + export.propsEnd();
436 MemoryStream binData = new MemoryStream(export.GetBinaryData());
437 binData.Skip(8);
438 int thumbnailSize = binData.ReadInt32();
439 binData.WriteInt32(baseOffset + (int)binData.Position + 4);
440 binData.Skip(thumbnailSize);
441 for (int i = binData.ReadInt32(); i > 0 && binData.Position < binData.Length; i--)
442 {
443 var storageFlags = (StorageFlags)binData.ReadInt32();
444 if (!storageFlags.Has(StorageFlags.externalFile)) //pcc-stored
445 {
446 int uncompressedSize = binData.ReadInt32();
447 int compressedSize = binData.ReadInt32();
448 binData.WriteInt32(baseOffset + (int)binData.Position + 4);//update offset
449 binData.Seek((storageFlags == StorageFlags.noFlags ? uncompressedSize : compressedSize) + 8, SeekOrigin.Current); //skip texture and width + height values
450 }
451 else
452 {
453 binData.Seek(20, SeekOrigin.Current);//skip whole rest of mip definition
454 }
455 }
456
457 binData.Skip(40);
458 binData.WriteInt32(baseOffset + (int)binData.Position + 4);
459 export.WriteBinary(binData.ToArray());
460 }
461 }
