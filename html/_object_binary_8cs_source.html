235 private static bool IsEither(string className, MEGame game, string baseClass1, string baseClass2)
236 {
237 if (className == baseClass1 || className == baseClass2) return true;
238 var classes = game switch
239 {
240 MEGame.ME1 => ME1UnrealObjectInfo.Classes,
241 MEGame.ME2 => ME2UnrealObjectInfo.Classes,
242 MEGame.ME3 => ME3UnrealObjectInfo.Classes,
243 MEGame.UDK => ME3UnrealObjectInfo.Classes,
244 MEGame.LE1 => LE1UnrealObjectInfo.Classes,
245 MEGame.LE2 => LE2UnrealObjectInfo.Classes,
246 MEGame.LE3 => LE3UnrealObjectInfo.Classes,
247 _ => throw new ArgumentOutOfRangeException(nameof(game), game, null)
248 };
249 while (true)
250 {
251 if (className == baseClass1 || className == baseClass2)
252 {
253 return true;
254 }
255 if (classes.TryGetValue(className, out ClassInfo info))
256 {
257 className = info.baseClass;
258 }
259 else
260 {
261 break;
262 }
263 }
264 return false;
265 }
