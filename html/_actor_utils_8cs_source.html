33 double yaw = rotation.Yaw.UnrealRotationUnitsToRadians();
34 double roll = rotation.Roll.UnrealRotationUnitsToRadians();

36 float sp = (float)Math.Sin(pitch);
37 float sy = (float)Math.Sin(yaw);
38 float sr = (float)Math.Sin(roll);
39 float cp = (float)Math.Cos(pitch);
40 float cy = (float)Math.Cos(yaw);
41 float cr = (float)Math.Cos(roll);
42
43 (float x, float y, float z) = (location.X, location.Y, location.Z);
44 (float sX, float sY, float sZ) = (scale.X, scale.Y, scale.Z);
45 (float pX, float pY, float pZ) = (pivot.X, pivot.Y, pivot.Z);

47 m12: cp * sX * sy,
48 m13: sX * sp,
49 m14: 0f,

51 m22: sY * (cr * cy + sp * sr * sy),
