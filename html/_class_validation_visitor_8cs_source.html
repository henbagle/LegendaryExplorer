56 public bool VisitNode(Class node)
57 {
58 switch (Pass)
59 {
60 case ValidationPass.TypesAndFunctionNamesAndStateNames:
61 {
62 // TODO: allow duplicate names as long as its in different packages!
63 if (node.Name != "Object")//validating Object is a special case, as it is the base class for all classes
64 {
65 //ADD CLASSNAME TO SYMBOLS BEFORE VALIDATION pass!
66 //if (!Symbols.TryAddType(node))
67 //{
68 // return Error($"A class named '{node.Name}' already exists!", node.StartPos, node.EndPos);
69 //}
70 node.Parent.Outer = node;
71 if (!Symbols.TryResolveType(ref node.Parent))
72 return Error($"No parent class named '{node.Parent.Name}' found!", node.Parent.StartPos, node.Parent.EndPos);
73 if (node.Parent.Type != ASTNodeType.Class)
74 return Error($"Parent named '{node.Parent.Name}' is not a class!", node.Parent.StartPos, node.Parent.EndPos);
75
76 if (node._outerClass != null)
77 {
78 node._outerClass.Outer = node;
79 if (!Symbols.TryResolveType(ref node._outerClass))
80 return Error($"No outer class named '{node._outerClass.Name}' found!", node._outerClass.StartPos, node._outerClass.EndPos);
81 if (node.OuterClass.Type != ASTNodeType.Class)
82 return Error($"Outer named '{node.OuterClass.Name}' is not a class!", node.OuterClass.StartPos, node.OuterClass.EndPos);
83 if (node.Parent.Name == "Actor" && !node.OuterClass.Name.Equals("Object", StringComparison.OrdinalIgnoreCase))
84 return Error("Classes extending 'Actor' can not be inner classes!", node.OuterClass.StartPos, node.OuterClass.EndPos);
85 }
86
87 for (int i = 0; i < node.Interfaces.Count; i++)
88 {
89 VariableType nodeInterface = node.Interfaces[i];
90 if (!Symbols.TryResolveType(ref nodeInterface, true))
91 {
92 return Error($"No outer class named '{nodeInterface.Name}' found!", nodeInterface.StartPos, nodeInterface.EndPos);
93 }
94 if (!node.IsNative && ((Class)nodeInterface).IsNative)
95 {
96 return Error($"Only a native class can implement a native interface!", nodeInterface.StartPos, nodeInterface.EndPos);
97 }
98 node.Interfaces[i] = nodeInterface;
99 }
100
101 //specifier validation
102 if (string.Equals(node.ConfigName, "inherit", StringComparison.OrdinalIgnoreCase) && !((Class)node.Parent).Flags.Has(EClassFlags.Config))
103 {
104 return Error($"Cannot inherit config filename from parent class ({node.Parent.Name}) which is not marked as config!", node.StartPos);
105 }
106 //TODO:propagate/check inheritable class flags from parent and implemented interfaces
107 if (node.IsNative && !((Class)node.Parent).IsNative)
108 {
109 return Error($"A native class cannot inherit from a non-native class!", node.StartPos);
110 }
111 Symbols.GoDirectlyToStack(((Class)node.Parent).GetInheritanceString());
112 Symbols.PushScope(node.Name);
113 }
114
115
116
117 //register all the types this class declares
118 foreach (VariableType type in node.TypeDeclarations)
119 {
120 type.Outer = node;
121 Success &= type.AcceptVisitor(this);
122 }
123
124 //register all the function names (do this here so that delegates will resolve correctly)
125 foreach (Function func in node.Functions)
126 {
127 func.Outer = node;
128 Success &= func.AcceptVisitor(this);
129 }
130
131 //register all state names (do this here so that states can extend states that are declared later in the class)
132 foreach (State state in node.States)
133 {
134 state.Outer = node;
135 Success &= state.AcceptVisitor(this);
136 }
137
138 Symbols.RevertToObjectStack();//pops scope until we're in the 'object' scope
139
140 return Success;
141 }
142 case ValidationPass.ClassAndStructMembersAndFunctionParams:
143 {
144 if (node.Name != "Object")
145 {
146 if (((Class)node.Parent).SameAsOrSubClassOf(node)) // TODO: not needed due to no forward declarations?
147 {
148 return Error($"Extending from '{node.Parent.Name}' causes circular extension!", node.StartPos);
149 }
150 if (!((Class)node.OuterClass).SameAsOrSubClassOf(((Class)node.Parent).OuterClass.Name))
151 {
152 return Error("Outer class must be a sub-class of the parents outer class!", node.StartPos);
153 }
154 if (node.SameAsOrSubClassOf("Interface"))
155 {
156 node.Flags |= EClassFlags.Interface;
157 node.PropertyType = EPropertyType.Interface;
158 }
159 Symbols.GoDirectlyToStack(((Class)node.Parent).GetInheritanceString());
160 string outerScope = null;
161 if (node.OuterClass != null && !string.Equals(node.OuterClass.Name, "Object", StringComparison.OrdinalIgnoreCase))
162 {
163 outerScope = ((Class)node.OuterClass).GetInheritanceString();
164 }
165
166 Symbols.PushScope(node.Name); //, outerScope);
167 }
168
169 //second pass over structs to resolve their members
170 foreach (Struct type in node.TypeDeclarations.OfType<Struct>())
171 {
172 Success &= type.AcceptVisitor(this);
173 }
174
175 //resolve instance variables
176 foreach (VariableDeclaration decl in node.VariableDeclarations)
177 {
178 decl.Outer = node;
179 Success &= decl.AcceptVisitor(this);
180
181 if (node.Name != "Object" && Symbols.TryGetSymbolInScopeStack<ASTNode>(decl.Name, out _, node.Parent.GetScope()))
182 {
183 Log.LogWarning($"A symbol named '{decl.Name}' exists in a parent class. Are you sure you want to shadow it?", decl.StartPos, decl.EndPos);
184 }
185 }
186
187 if (node.Name != "Object")
188 {
189 Symbols.TryGetType("Object", out Class objectClass);
190 Symbols.AddSymbol("Class", new VariableDeclaration(new ClassType(node), EPropertyFlags.Const | EPropertyFlags.Native | EPropertyFlags.EditConst, "Class")
191 {
192 Outer = objectClass
193 });
194 Symbols.AddSymbol("Outer", new VariableDeclaration(node.OuterClass, EPropertyFlags.Const | EPropertyFlags.Native | EPropertyFlags.EditConst, "Outer")
195 {
196 Outer = objectClass
197 });
198 }
199
200 //second pass over functions to resolve parameters
201 foreach (Function func in node.Functions)
202 {
203 Success &= func.AcceptVisitor(this);
204 }
205
206 //second pass over states to resolve
207 foreach (State state in node.States)
208 {
209 Success &= state.AcceptVisitor(this);
210 }
211
212 Symbols.RevertToObjectStack();//pops scope until we're in the 'object' scope
213
214 node.Declaration = node;
215 return Success;
216 }
217 case ValidationPass.BodyPass:
218 {
219 //from UDN: "Implementing multiple interface classes which have a common base is not supported and will result in incorrect vtable offsets"
220 if (node.Interfaces.Count > 1)
221 {
222 var interfaceParents = new HashSet<string>();
223 foreach (VariableType interfaceClass in node.Interfaces)
224 {
225 var parentInterface = (interfaceClass as Class)?.Parent as Class;
226 while (parentInterface is not null && !parentInterface.Name.CaseInsensitiveEquals("Interface"))
227 {
228 if (interfaceParents.Contains(parentInterface.Name))
229 {
230 return Error("Cannot implement two interfaces that have a common base interface lower than the Interface class", node.StartPos);
231 }
232 parentInterface = parentInterface.Parent as Class;
233 }
234 }
235 }
236
237 //third pass over structs to check for circular inheritance chains
238 foreach (Struct type in node.TypeDeclarations.OfType<Struct>())
239 {
240 Success &= type.AcceptVisitor(this);
241 }
242
243 //third pass over functions to check overriding rules
244 foreach (Function func in node.Functions)
245 {
246 Success &= func.AcceptVisitor(this);
247 }
248
249 //third pass over states to check function overrides
250 foreach (State state in node.States)
251 {
252 Success &= state.AcceptVisitor(this);
253 }
254
255 //second pass to resolve EPropertyFlags.NeedCtorLink for Struct Properties
256 foreach (VariableDeclaration decl in node.VariableDeclarations)
257 {
258 Success &= decl.AcceptVisitor(this);
259 if (decl.Flags.Has(EPropertyFlags.Component))
260 {
261 node.Flags |= EClassFlags.HasComponents;
262 }
263 if (decl.Flags.Has(EPropertyFlags.CrossLevel))
264 {
265 node.Flags |= EClassFlags.HasCrossLevelRefs;
266 }
267 if (decl.Flags.Has(EPropertyFlags.Config))
268 {
269 node.Flags |= EClassFlags.Config;
270 }
271 if (decl.Flags.Has(EPropertyFlags.Localized))
272 {
273 node.Flags |= EClassFlags.Localized;
274 }
275 if (decl.IsOrHasInstancedObjectProperty())
276 {
277 node.Flags |= EClassFlags.HasInstancedProps;
278 }
279 }
280
281 return Success;
282 }
283 default:
284 return Success;
285 }

293 public bool VisitVarDecl(VariableDeclaration node, bool needsAdd = true)
294 {
295 if (Pass is ValidationPass.ClassAndStructMembersAndFunctionParams)
296 {
297 if (needsAdd)
298 {
299 if (node.VarType is not PrimitiveType)
300 {
301 node.VarType.Outer = node;
302 if (!Symbols.TryResolveType(ref node.VarType))
303 {
304 return Error($"No type named '{node.VarType.Name}' exists!", node.VarType.StartPos, node.VarType.EndPos);
305 }
306 }
307
308 if (Symbols.SymbolExistsInCurrentScope(node.Name))
309 {
310 return Error($"A {(node is FunctionParameter ? "parameter" : "member")} named '{node.Name}' already exists in this {node.Outer.Type}!", node.StartPos, node.EndPos);
311 }
312 Symbols.AddSymbol(node.Name, node);
313 }
314
315 VariableType nodeVarType = (node.VarType as StaticArrayType)?.ElementType ?? node.VarType;
316 if (nodeVarType is DelegateType ||
317 !node.Flags.Has(EPropertyFlags.Native) && nodeVarType is DynamicArrayType or { PropertyType: EPropertyType.String })
318 {
319 node.Flags |= EPropertyFlags.NeedCtorLink;
320 }
321 }
322 else if (Pass is ValidationPass.BodyPass)
323 {
324 //should component flag be set when this is a function parameter?
325 switch ((node.VarType as StaticArrayType)?.ElementType ?? node.VarType)
326 {
327 case DynamicArrayType {ElementType: VariableType elType} dynArrType:
328 if (elType is Class {IsComponent: true})
329 {
330 dynArrType.ElementPropertyFlags |= EPropertyFlags.Component;
331 node.Flags |= EPropertyFlags.Component;
332 }
333 else if (elType is DelegateType ||
334 !node.Flags.Has(EPropertyFlags.Native) && (elType.PropertyType is EPropertyType.String ||
335 elType is Struct elStruct && StructNeedsCtorLink(elStruct, new Stack<Struct> { elStruct })))
336 {
337 dynArrType.ElementPropertyFlags |= EPropertyFlags.NeedCtorLink;
338 }
339 break;
340 case Class {IsComponent: true}:
341 node.Flags |= EPropertyFlags.Component;
342 break;
343 case Struct strct:
344 if (!node.Flags.Has(EPropertyFlags.Native) && StructNeedsCtorLink(strct, new Stack<Struct> { strct }))
345 {
346 node.Flags |= EPropertyFlags.NeedCtorLink;
347 }
348 break;
349 }
350
351 bool StructNeedsCtorLink(Struct s1, Stack<Struct> stack)
352 {
353 foreach (VariableDeclaration strctVariableDeclaration in s1.VariableDeclarations)
354 {
355 if (strctVariableDeclaration.Flags.Has(EPropertyFlags.NeedCtorLink))
356 {
357 return true;
358 }
359 else if (strctVariableDeclaration.VarType is Struct s2)
360 {
361 if (stack.Contains(s2))
362 {
363 stack.Push(s2);
364 Error($"Detected circular reference in these structs: {string.Join(" -> ", stack.Reverse().Select(s3 => s3.Name))}");
365 stack.Pop();
366 return true;
367 }
368 stack.Push(s2);
369 if (StructNeedsCtorLink(s2, stack))
370 {
371 return true;
372 }
373 stack.Pop();
374 }
375 }
376 if (s1.Parent is Struct parentStruct)
377 {
378 stack.Push(parentStruct);
379 if (StructNeedsCtorLink(parentStruct, stack))
380 {
381 return true;
382 }
383 stack.Pop();
384 }
385 return false;
386 }
387 }
388
389 return Success;
390 }

417 public bool VisitNode(Struct node)
418 {
419 if (Pass == ValidationPass.TypesAndFunctionNamesAndStateNames)
420 {
421 if (!Symbols.TryAddType(node))
422 {
423 //Structs do not have to be globally unique, but they do have to be unique within a scope
424 if (node.Outer is ObjectType nodeOuter && nodeOuter.TypeDeclarations.Any(decl => decl != node && decl.Name.CaseInsensitiveEquals(node.Name)))
425 {
426 return Error($"A type named '{node.Name}' already exists in this {nodeOuter.GetType().Name.ToLower()}!", node.StartPos, node.EndPos);
427 }
428 }
429
430 Symbols.PushScope(node.Name);
431
432 //register types of inner structs
433 foreach (VariableType typeDeclaration in node.TypeDeclarations)
434 {
435 typeDeclaration.Outer = node;
436 Success &= typeDeclaration.AcceptVisitor(this);
437 }
438
439 Symbols.PopScope();
440 }
441 else if (Pass == ValidationPass.ClassAndStructMembersAndFunctionParams)
442 {
443 string parentScope = null;
444 if (node.Parent != null)
445 {
446 node.Parent.Outer = node;
447 if (!Symbols.TryResolveType(ref node.Parent))
448 {
449 return Error($"No parent struct named '{node.Parent.Name}' found!", node.Parent.StartPos, node.Parent.EndPos);
450 }
451
452 if (node.Parent.Type != ASTNodeType.Struct)
453 return Error($"Parent named '{node.Parent.Name}' is not a struct!", node.Parent.StartPos, node.Parent.EndPos);
454
455 parentScope = $"{NodeUtils.GetContainingClass(node.Parent).GetInheritanceString()}.{node.Parent.Name}";
456 }
457
458 Symbols.PushScope(node.Name, parentScope);
459
460 //second pass for inner struct members
461 foreach (VariableType typeDeclaration in node.TypeDeclarations)
462 {
463 Success &= typeDeclaration.AcceptVisitor(this);
464 }
465
466 foreach (VariableDeclaration decl in node.VariableDeclarations)
467 {
468 decl.Outer = node;
469 Success = Success && decl.AcceptVisitor(this);
470
471 var parentStruct = node.Parent as Struct;
472 while (parentStruct is not null)
473 {
474 if (parentStruct.VariableDeclarations.Any(parentVarDecl => parentVarDecl.Name.CaseInsensitiveEquals(decl.Name)))
475 {
476 Log.LogWarning($"A member name '{decl.Name}' exists in a parent struct. Are you sure you want to shadow it?", decl.StartPos, decl.EndPos);
477 }
478 parentStruct = parentStruct.Parent as Struct;
479 }
480 }
481
482 Symbols.PopScope();
483
484 node.Declaration = node;
485 }
486 else if (Pass == ValidationPass.BodyPass)
487 {
488 if (node.Parent is Struct parentStruct && parentStruct.SameOrSubStruct(node.Name))
489 {
490 return Error($"Extending from '{parentStruct.Name}' causes circular extension!", parentStruct.StartPos, parentStruct.EndPos);
491 }
492
493 //second pass to resolve EPropertyFlags.NeedCtorLink for Struct Properties
494 foreach (VariableDeclaration decl in node.VariableDeclarations)
495 {
496 Success &= decl.AcceptVisitor(this);
497 }
498 if (HasComponents(node))
499 {
500 node.Flags |= ScriptStructFlags.HasComponents;
501 }
502
503 static bool HasComponents(Struct strct)
504 {
505 bool hasComponents = false;
506 foreach (VariableDeclaration decl in strct.VariableDeclarations)
507 {
508 if (decl.Flags.Has(EPropertyFlags.Component))
509 {
510 hasComponents = true;
511 }
512 var varType = decl.VarType is StaticArrayType staticArrayType ? staticArrayType.ElementType : decl.VarType;
513 if (varType is DynamicArrayType dynArrType)
514 {
515 varType = dynArrType.ElementType;
516 }
517 if (varType is Struct innerStruct && (innerStruct.Flags.Has(ScriptStructFlags.HasComponents) || HasComponents(innerStruct)))
518 {
519 decl.Flags |= EPropertyFlags.Component;
520 hasComponents = true;
521 }
522 }
523 return hasComponents;
524 }
525 }
526 return Success;

588 public bool VisitNode(Function node)
589 {
590 if (Pass == ValidationPass.TypesAndFunctionNamesAndStateNames)
591 {
592 if (Symbols.SymbolExistsInCurrentScope(node.Name))
593 return Error($"The name '{node.Name}' is already in use in this class!", node.StartPos, node.EndPos);
594
595 Symbols.AddSymbol(node.Name, node);
596 return Success;
597 }
598
599 if (Pass == ValidationPass.ClassAndStructMembersAndFunctionParams)
600 {
601 Symbols.PushScope(node.Name);
602
603 if (node.ReturnValueDeclaration != null)
604 {
605 node.ReturnValueDeclaration.Outer = node;
606 Success &= node.ReturnValueDeclaration.AcceptVisitor(this);
607 }
608
609 foreach (FunctionParameter param in node.Parameters)
610 {
611 param.Outer = node;
612 Success &= param.AcceptVisitor(this);
613 }
614
615 //foreach (VariableDeclaration local in node.Locals)
616 //{
617 // local.Outer = node;
618 // Success &= local.AcceptVisitor(this);
619 //}
620 Symbols.PopScope();
621
622 if (Success == false)
623 return Error("Error in function parameters.", node.StartPos, node.EndPos);
624
625
626 if (node.FriendlyName is not null //true in ME1, ME2, LE1, and LE2
627 && node.IsOperator)
628 {
629 if (node.Flags.Has(EFunctionFlags.PreOperator))
630 {
631 if (node.Parameters.Count != 1)
632 {
633 return Error($"{node.Name} is declared as a prefix operator, so it must have exactly one parameter!", node.StartPos, node.EndPos);
634 }
635 Symbols.AddOperator(new PreOpDeclaration(node.FriendlyName, node.ReturnType, node.NativeIndex, node.Parameters[0]) { Implementer = node });
636 }
637 else
638 {
639 switch (node.Parameters.Count)
640 {
641 case 1:
642 Symbols.AddOperator(new PostOpDeclaration(node.FriendlyName, node.ReturnType, node.NativeIndex, node.Parameters[0]) { Implementer = node });
643 break;
644 case 2:
645 Symbols.AddOperator(new InOpDeclaration(node.FriendlyName, node.OperatorPrecedence, node.NativeIndex, node.ReturnType, node.Parameters[0], node.Parameters[1])
646 {
647 Implementer = node
648 });
649 Symbols.InFixOperatorSymbols.Add(node.FriendlyName);
650 break;
651 default:
652 return Error($"{node.Name} is declared as an operator, so it must have either 1 or 2 parameters!", node.StartPos, node.EndPos);
653 }
654 }
655 }
656
657 return Success;
658 }
659
660 //for validating proper override behavior
661 if (Pass == ValidationPass.BodyPass)
662 {
663 Class containingClass = NodeUtils.GetContainingClass(node);
664 Function superFunc = null;
665 if (node.Outer is State state)
666 {
667 state = state.Parent;
668 while (state is not null)
669 {
670 string stateScope = $"{((Class)state.Outer).GetInheritanceString()}.{state.Name}";
671 if (Symbols.TryGetSymbolFromSpecificScope(node.Name, out superFunc, stateScope))
672 {
673 break;
674 }
675
676 state = state.Parent;
677 }
678 }
679 if (superFunc is null)
680 {
681 Class parentScopeClass = node.Outer is State ? containingClass : containingClass.Parent as Class;
682 if (parentScopeClass is not null)
683 {
684 Symbols.TryGetSymbolInScopeStack(node.Name, out superFunc, parentScopeClass.GetInheritanceString());
685 }
686 }
687
688 if (superFunc is not null)
689 {
690 if (superFunc.Flags.Has(EFunctionFlags.Private))
691 {
692 superFunc = null;
693 }
694 else
695 {
696 // If there is a function with this name that we should override, validate the new functions declaration
697 if (superFunc.Flags.Has(EFunctionFlags.Final))
698 return Error($"{node.Name} overrides a function in a parent class, but the parent function is marked as final!", node.StartPos, node.EndPos);
699 if (!NodeUtils.TypeEqual(node.ReturnType, superFunc.ReturnType))
700 return Error($"{node.Name} overrides a function in a parent class, but the functions do not have the same return types!", node.StartPos, node.EndPos);
701
702 if (node.Parameters.Count != superFunc.Parameters.Count)
703 {
704 if (node.Outer is State)
705 {
706 //Contrary to what the unrealscript docs say, states can apparently have functions with the same name as a class function, but with different number of params.
707 superFunc = null;
708 }
709 else
710 {
711 return Error($"{node.Name} overrides a function in a parent class, but the functions do not have the same number of parameters!", node.StartPos, node.EndPos);
712 }
713 }
714 else
715 {
716 for (int n = 0; n < node.Parameters.Count; n++)
717 {
718 if (node.Parameters[n].Type != superFunc.Parameters[n].Type)
719 return Error($"{node.Name} overrides a function in a parent class, but the functions do not have the same parameter types!", node.StartPos, node.EndPos);
720 }
721 }
722
723 node.SuperFunction = superFunc;
724 }
725 }
726
727 if (superFunc is null && node.Outer is State && node.Flags.Has(EFunctionFlags.Net))
728 {
729 return Error("If a state function has the Net flag, it must override a class function", node.StartPos, node.EndPos);
730 }
731
732
733 if (node.ReturnValueDeclaration != null)
734 {
735 Success &= node.ReturnValueDeclaration.AcceptVisitor(this);
736 }
737
738 foreach (FunctionParameter param in node.Parameters)
739 {
740 param.Outer = node;
741 Success &= param.AcceptVisitor(this);
742 }
743
744 if (node.ReturnValueDeclaration is not null)
745 {
746 //if the return type is > 64 bytes, it can't be allocated on the stack.
747 node.RetValNeedsDestruction = node.ReturnValueDeclaration.Flags.Has(EPropertyFlags.NeedCtorLink) || node.ReturnType.Size(Symbols.Game) > 64;
748 }
749 }
750 return Success;

754 {
755 if (Pass == ValidationPass.TypesAndFunctionNamesAndStateNames)
756 {
757 if (Symbols.SymbolExistsInCurrentScope(node.Name))
758 return Error($"The name '{node.Name}' is already in use in this class!", node.StartPos, node.EndPos);
759 Symbols.AddSymbol(node.Name, node);
760 return Success;
761 }
762
763 if (Pass == ValidationPass.ClassAndStructMembersAndFunctionParams)
764 {
765 bool overrides = Symbols.TryGetSymbolInScopeStack(node.Name, out ASTNode overrideState, NodeUtils.GetParentClassScope(node))
766 && overrideState.Type == ASTNodeType.State;
767
768 if (node.Parent is null)
769 {
770 if (overrides)
771 {
772 node.Parent = overrideState as State;
773 }
774 }
775 else
776 {
777 if (overrides)
778 Error("A state is not allowed to both override a parent class's state and extend another state at the same time!", node.StartPos, node.EndPos);
779
780 if (!Symbols.TryGetSymbol(node.Parent.Name, out ASTNode parent))
781 {
782 Error($"No parent state named '{node.Parent.Name}' found in the current class!", node.Parent.StartPos, node.Parent.EndPos);
783 node.Parent = null;
784 }
785
786 if (parent != null)
787 {
788 if (parent.Type != ASTNodeType.State)
789 Error($"Parent named '{node.Parent.Name}' is not a state!", node.Parent.StartPos, node.Parent.EndPos);
790 else
791 node.Parent = parent as State;
792 }
793 }
794
795 string parentScope = node.Parent is not null ? $"{NodeUtils.GetContainingClass(node.Parent)?.GetInheritanceString()}.{node.Parent.Name}" : null;
796 Symbols.PushScope(node.Name, parentScope);
797
798 foreach (Function func in node.Functions)
799 {
800 func.Outer = node;
801 Symbols.AddSymbol(func.Name, func);
802 Success = Success && func.AcceptVisitor(this);
803 }
