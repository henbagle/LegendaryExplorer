169 {
170 if (Game >= MEGame.ME3 && ClassName == "DominantDirectionalLightComponent" || ClassName == "DominantSpotLightComponent")
171 {
172 int minLen = (IsDefaultObject ? 8 : 12);

174 {
175 throw new ArgumentException($"Expected pre-property binary to be at least {minLen} bytes, not {bytes.Length}!", nameof(bytes));
176 }
177 int oldLen = GetPropertyStart();
178 int count = EndianReader.ToInt32(bytes, 0, _fileRef.Endian);

186 ms.WriteFromBuffer(bytes);
187 ms.Write(_data, oldLen, _data.Length - oldLen);

193 {
194 throw new ArgumentException($"Expected pre-property binary to be {GetPropertyStart()} bytes, not {bytes.Length}!", nameof(bytes));

278 public byte[] GenerateHeader(IMEPackage pcc, bool clearComponentMap = false)
279 {
280 using var bin = new EndianWriter(MemoryManager.GetMemoryStream(HeaderLength)) { Endian = pcc.Endian };
281 bin.WriteInt32(idxClass);
282 bin.WriteInt32(idxSuperClass);
283 bin.WriteInt32(idxLink);
284 bin.WriteInt32(idxObjectName);
285 bin.WriteInt32(indexValue);
286 bin.WriteInt32(idxArchetype);
287 bin.WriteUInt64((ulong)_objectFlags);
288 bin.WriteInt32(DataSize);
289 bin.WriteInt32(DataOffset);
290 if (pcc.Game <= MEGame.ME2 && pcc.Platform != MEPackage.GamePlatform.PS3)
291 {
292 if (clearComponentMap)
293 {
294 bin.WriteInt32(0);
295 }
296 else
297 {
298 byte[] componentMap = _componentMap ?? Array.Empty<byte>();
299 bin.WriteInt32(componentMap.Length / 12);
300 bin.Write(componentMap);
301 }
302 }
303 bin.WriteUInt32((uint)ExportFlags);
304 int[] genobjCounts = GenerationNetObjectCount;
305 bin.WriteInt32(genobjCounts.Length);
306 foreach (int count in genobjCounts)
307 {
308 bin.WriteInt32(count);
309 }
310 bin.WriteGuid(PackageGUID);
311 if (pcc.Game != MEGame.ME1 || pcc.Platform != MEPackage.GamePlatform.Xenon)
312 {
313 bin.WriteUInt32((uint)PackageFlags);
314 }
315 return bin.ToArray();
316 }

318 private void DeserializeHeader(EndianReader stream)
319 {
320 _idxClass = stream.ReadInt32();
321 _idxSuperClass = stream.ReadInt32();
322 _idxLink = stream.ReadInt32();
323 _idxObjectName = stream.ReadInt32();
324 _indexValue = stream.ReadInt32();
325 _idxArchetype = stream.ReadInt32();
326 _objectFlags = (EObjectFlags)stream.ReadUInt64();
327 DataSize = stream.ReadInt32();
328 DataOffset = stream.ReadInt32();
329 if (HasComponentMap)
330 {
331 _componentMap = stream.ReadBytes(stream.ReadInt32() * 12);
332 }
333 _exportFlags = (EExportFlags)stream.ReadUInt32();
334 int count = stream.ReadInt32();
335 _generationNetObjectCounts = new int[count];
336 for (int i = 0; i < count; i++)
337 {
338 _generationNetObjectCounts[i] = stream.ReadInt32();
339 }
340 _packageGuid = stream.ReadGuid();
341 if (Game != MEGame.ME1 || _fileRef.Platform != MEPackage.GamePlatform.Xenon)
342 {
343 _packageFlags = (EPackageFlags)stream.ReadUInt32();
344 }

496 public OrderedMultiValueDictionary<NameReference, int> ComponentMap
497 {
498 get
499 {
500 var componentMap = new OrderedMultiValueDictionary<NameReference, int>();
501 if (!HasComponentMap) return componentMap;
502 for (int i = 0; i < _componentMap.Length; i += 12)
503 {
504 string name = _fileRef.GetNameEntry(EndianReader.ToInt32(_componentMap, i, _fileRef.Endian));
505 componentMap.Add(new NameReference(name, EndianReader.ToInt32(_componentMap, i + 4, _fileRef.Endian)),
506 EndianReader.ToInt32(_componentMap, i + 8, _fileRef.Endian));
507 }
508 return componentMap;
509 }
510 set
511 {
512 if (!HasComponentMap) return;
513 if (value is null)
514 {
515 if (_componentMap.Length == 0)
516 {
517 return;
518 }
519 _componentMap = Array.Empty<byte>();
520 HeaderChanged = true;
521 return;
522 }
523 var bin = new MemoryStream(value.Count * 12);
524 foreach ((NameReference name, int _uIndex) in value)
525 {
526 bin.WriteInt32(_fileRef.FindNameOrAdd(name.Name));
527 bin.WriteInt32(name.Number);
528 bin.WriteInt32(_uIndex); // 0-based index
529 }
530 var newMap = bin.ToArray();
531 if (!newMap.AsSpan().SequenceEqual(_componentMap))
532 {
533 _componentMap = newMap;
534 HeaderChanged = true;
535 }

797 public int GetPropertyStart()
798 {
799 if (HasStack)
800 {
801 return stackLength;
802 }
803
804 int start = 0;
805
806
807 if (!IsDefaultObject && this.IsA("Component") || (Game == MEGame.UDK && ClassName.EndsWith("Component")))
808 {
809 if (Game >= MEGame.ME3 && ClassName == "DominantDirectionalLightComponent" || ClassName == "DominantSpotLightComponent")
810 {
811 //DominantLightShadowMap, which goes before everything for some reason
812 int count = EndianReader.ToInt32(_data, 0, _fileRef.Endian);
813 start += count * 2 + 4;
814 }
815 start += 4; //TemplateOwnerClass
816 if (ParentFullPath.Contains("Default__"))
817 {
818 start += 8; //TemplateName
819 }
820 }
821
822 start += 4; //NetIndex
823
824 return start;
