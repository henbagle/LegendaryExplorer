291 protected virtual bool SetEntry(TKey key, TValue value)
292 {
293 var keyExists = KeyedEntryCollection.Contains(key);
294
295 // if identical key/value pair already exists, nothing to do
296 if (keyExists && value.Equals((TValue) KeyedEntryCollection[key].Value))
297 {
298 return false;
299 }
300
301 // otherwise, remove the existing entry
302 if (keyExists)
303 {
304 KeyedEntryCollection.Remove(key);
305 }
306
307 // add the new entry
308 KeyedEntryCollection.Add(new DictionaryEntry(key, value));
309
310 return true;
311 }

334 private bool DoRemoveEntry(TKey key)
335 {
336 DictionaryEntry entry;
337 var index = GetIndexAndEntryForKey(key, out entry);

352 private void DoSetEntry(TKey key, TValue value)
353 {
354 DictionaryEntry entry;
355 var index = GetIndexAndEntryForKey(key, out entry);
356
357 if (SetEntry(key, value))
358 {
359 _version++;
360
361 // if prior entry existed for this key, fire the removed notifications
362 if (index > -1)
363 {
364 FireEntryRemovedNotifications(entry, index);
365
366 // force the property change notifications to fire for the modified entry
367 _countCache--;
368 }
369
370 // then fire the added notifications
371 index = GetIndexAndEntryForKey(key, out entry);
372 FireEntryAddedNotifications(entry, index);
373 }
374 }
