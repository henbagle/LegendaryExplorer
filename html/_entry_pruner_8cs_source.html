17 public static void TrashEntriesAndDescendants(IEnumerable<IEntry> itemsToTrash)
18 {
19 if (!itemsToTrash.Any()) return;
20 var entriesToTrash = new List<IEntry>();
21 entriesToTrash.AddRange(itemsToTrash);
22 foreach (var entry in itemsToTrash)
23 {
24 entriesToTrash.AddRange(entry.GetAllDescendants());
25 }
26
27 // Trash in order of bottom first. This ensures that the package lookup tree stays correct. If we modified top first
28 // it would break the lookup tree as the InstancedFullPath would no longer be accurate
29 TrashEntries(entriesToTrash[0].FileRef, entriesToTrash.OrderByDescending(x => x.InstancedFullPath.Count(y => y == '.')));
30 }
31 public static void TrashEntryAndDescendants(IEntry entry)
32 {
33 var entriesToTrash = new List<IEntry> { entry };
34 entriesToTrash.AddRange(entry.GetAllDescendants());
35 TrashEntries(entry.FileRef, entriesToTrash.OrderByDescending(x => x.InstancedFullPath.Count(y => y == '.')));
36 }
37 public static void TrashEntries(IMEPackage pcc, IEnumerable<IEntry> itemsToTrash)
38 {
39 ExportEntry trashTopLevel = pcc.FindExport(UnrealPackageFile.TrashPackageName);
40 IEntry packageClass = pcc.getEntryOrAddImport("Core.Package");
41
42 foreach (IEntry entry in itemsToTrash)
43 {
44 if (entry == trashTopLevel || entry.ObjectName == "Trash") //don't trash what's already been trashed
45 {
46 continue;
47 }
48 trashTopLevel = TrashEntry(entry, trashTopLevel, packageClass);
49 }
50 pcc.RemoveTrailingTrash();
51 }

60 private static ExportEntry TrashEntry(IEntry entry, ExportEntry trashContainer, IEntry packageClass)
61 {
62 IMEPackage pcc = entry.FileRef;
63 if (entry is ImportEntry imp)
64 {
65 if (trashContainer == null)
66 {
67 trashContainer = new ExportEntry(pcc, 0, UnrealPackageFile.TrashPackageName);
68 pcc.AddExport(trashContainer);
69 trashContainer = TrashEntry(trashContainer, null, packageClass);
70 trashContainer.indexValue = 0;
71 }
72 imp.ClassName = "Package";
73 imp.PackageFile = "Core";
74 imp.idxLink = trashContainer.UIndex;
75 imp.ObjectName = "Trash";
76 imp.indexValue = 0;
77 }
78 else if (entry is ExportEntry exp)
79 {
80 using MemoryStream trashData = MemoryManager.GetMemoryStream();
81 trashData.WriteInt32(-1);
82 trashData.WriteInt32(pcc.FindNameOrAdd("None"));
83 trashData.WriteInt32(0);
84 exp.Data = trashData.ToArray();
85 exp.Archetype = null;
86 exp.SuperClass = null;
87 exp.indexValue = 0;
88 exp.Class = packageClass;
89 exp.ObjectFlags &= ~UnrealFlags.EObjectFlags.HasStack;
90 exp.ObjectFlags &= ~UnrealFlags.EObjectFlags.ArchetypeObject;
91 exp.ObjectFlags &= ~UnrealFlags.EObjectFlags.ClassDefaultObject;
92 exp.ComponentMap = null;
93 if (trashContainer == null)
94 {
95 exp.ObjectName = UnrealPackageFile.TrashPackageName;
96 exp.idxLink = 0;
97 if (exp.idxLink == exp.UIndex)
98 {
99 Debugger.Break(); // RECURSIVE LOOP DETECTION!!
100 }
101 exp.PackageGUID = UnrealPackageFile.TrashPackageGuid;
102 trashContainer = exp;
103 }
104 else
105 {
106 exp.idxLink = trashContainer.UIndex;
107 if (exp.idxLink == exp.UIndex)
108 {
109 //This should not occur
110 Debugger.Break();
111 }
112 exp.ObjectName = "Trash";
113 exp.PackageGUID = Guid.Empty;
114 }
115 //(pcc as UnrealPackageFile).EntryLookupTable[exp.InstancedFullPath] = exp;
116 }
117 return trashContainer;

120 public static void TrashIncompatibleEntries(MEPackage pcc, MEGame oldGame, MEGame newGame)
121 {
122 var entries = pcc.Tree;
123 var oldClasses = GlobalUnrealObjectInfo.GetClasses(oldGame);
124 var newClasses = GlobalUnrealObjectInfo.GetClasses(newGame);
125 var classesToRemove = oldClasses.Keys.Except(newClasses.Keys).ToHashSet();
126 foreach (IEntry entry in entries)
127 {
128 if (classesToRemove.Contains(entry.ClassName) || (entry.ClassName == "Class" && classesToRemove.Contains(entry.ObjectName.Name))
129 || entry is ExportEntry exp && (exp.Archetype?.IsTrash() ?? false))
130 {
131 TrashEntries(pcc, entries.FlattenTreeOf(entry.UIndex));
132 }
133 }

136 public static PropertyCollection RemoveIncompatibleProperties(IMEPackage sourcePcc, PropertyCollection props, string typeName, MEGame newGame, ref bool removedProperties)
137 {
138 var infoProps = GlobalUnrealObjectInfo.GetAllProperties(newGame, typeName);
139 var newProps = new PropertyCollection();
140 foreach (Property prop in props)
141 {
142 if (infoProps.ContainsKey(prop.Name))
143 {
144 switch (prop)
145 {
146 case ArrayProperty<EnumProperty> aep:
147 if (GlobalUnrealObjectInfo.GetEnumValues(newGame, aep.Reference) is List<NameReference> enumValues)
148 {
149 foreach (EnumProperty enumProperty in aep)
150 {
151 if (!enumValues.Contains(enumProperty.Value))
152 {
153 enumProperty.Value = enumValues.First(); //hope that the first value is a reasonable default
154 }
155 }
156 newProps.Add(aep);
157 }
158 else
159 {
160 removedProperties = true;
161 }
162 break;
163 case ArrayProperty<ObjectProperty> asp:
164 for (int i = asp.Count - 1; i >= 0; i--)
165 {
166 if (asp[i].Value == 0 || sourcePcc.GetEntry(asp[i].Value) is IEntry entry && !entry.FullPath.StartsWith(UnrealPackageFile.TrashPackageName))
167 {
168 continue;
169 }
170 //delete if it references a trashed entry or if value is invalid
171 asp.RemoveAt(i);
172 }
173 newProps.Add(asp);
174 break;
175 case ArrayProperty<StructProperty> asp:
176 if (GlobalUnrealObjectInfo.GetStructs(newGame).ContainsKey(asp.Reference))
177 {
178 if (HasIncompatibleImmutabilities(asp.Reference, out bool newImmutability)) break;
179 foreach (StructProperty structProperty in asp)
180 {
181 structProperty.Properties = RemoveIncompatibleProperties(sourcePcc, structProperty.Properties, structProperty.StructType, newGame, ref removedProperties);
182 structProperty.IsImmutable = newImmutability;
183 }
184 newProps.Add(asp);
185 }
186 else
187 {
188 removedProperties = true;
189 }
190 break;
191 case DelegateProperty delegateProperty:
192 //script related, so just delete it.
193 removedProperties = true;
194 break;
195 case EnumProperty enumProperty:
196 if (GlobalUnrealObjectInfo.GetEnumValues(newGame, enumProperty.EnumType) is List<NameReference> values)
197 {
198 if (!values.Contains(enumProperty.Value))
199 {
200 enumProperty.Value = values.First(); //hope that the first value is a reasonable default
201 }
202 newProps.Add(enumProperty);
203 }
204 else
205 {
206 removedProperties = true;
207 }
208 break;
209 case ObjectProperty objectProperty:
210 if (objectProperty.Value == 0 || sourcePcc.GetEntry(objectProperty.Value) is IEntry ent && !ent.FullPath.StartsWith(UnrealPackageFile.TrashPackageName))
211 {
212 newProps.Add(objectProperty);
213 }
214 else
215 {
216 removedProperties = true;
217 }
218 break;
219 case StructProperty structProperty:
220 string structType = structProperty.StructType;
221 if (GlobalUnrealObjectInfo.GetStructs(newGame).ContainsKey(structType))
222 {
223 if (HasIncompatibleImmutabilities(structType, out bool newImmutability)) break;
224 structProperty.Properties = RemoveIncompatibleProperties(sourcePcc, structProperty.Properties, structType, newGame, ref removedProperties);
225 structProperty.IsImmutable = newImmutability;
226 newProps.Add(structProperty);
227 }
228 else
229 {
230 removedProperties = true;
231 }
232 break;
233 default:
234 newProps.Add(prop);
235 break;
236 }
237 }
238 else
239 {
240 removedProperties = true;
241 }
242 }
243
244 return newProps;

246 bool HasIncompatibleImmutabilities(string structType, out bool newImmutability)
247 {
248 bool sourceIsImmutable = GlobalUnrealObjectInfo.IsImmutable(structType, sourcePcc.Game);
249 newImmutability = GlobalUnrealObjectInfo.IsImmutable(structType, newGame);
250
251 if (sourceIsImmutable && newImmutability && !GlobalUnrealObjectInfo.GetClassOrStructInfo(sourcePcc.Game, structType).properties
252 .SequenceEqual(GlobalUnrealObjectInfo.GetClassOrStructInfo(newGame, structType).properties))
253 {
254 //both immutable, but have different properties
255 return true;
256 }
257
258 if (!sourceIsImmutable && newImmutability)
259 {
260 //can't easily guarantee it will have have all neccesary properties
261 return true;
262 }
263
264 return false;
265 }
266 }
