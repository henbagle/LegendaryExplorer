51 public override List<(UIndex, string)> GetUIndexes(MEGame game)
52 {
53 List<(UIndex, string)> uIndices = base.GetUIndexes(game);
54 uIndices.Add((Children, "ChildListStart"));
55
56 if (Export.ClassName is "Function" or "State")
57 {
58 if (Export.Game == MEGame.ME3 || Export.Game.IsLEGame())
59 {
60 try
61 {
62 (List<Token> tokens, _) = Bytecode.ParseBytecode(ScriptBytes, Export);
63 foreach (var t in tokens)
64 {
65 {
66 var refs = t.inPackageReferences.Where(x => x.type == Token.INPACKAGEREFTYPE_ENTRY);
67 uIndices.AddRange(refs.Select(x => (new UIndex(x.value), $"Reference inside of function at 0x{x.position:X}")));
68 }
69 }
70 }
71 catch (Exception e)
72 {
73 Debug.WriteLine($"Error decompiling function {Export.InstancedFullPath}: {e.Message}");
74 }
75 }
76 else
77 {
78 try
79 {
80 var func = Export.ClassName == "State" ? UE3FunctionReader.ReadState(Export) : UE3FunctionReader.ReadFunction(Export);
81 func.Decompile(new TextBuilder(), false, false); //parse bytecode without signature (it does not contain entry refs)
82 var entryRefs = func.EntryReferences;
83 uIndices.AddRange(entryRefs.Select(x =>
84 (new UIndex(x.Value.UIndex), $"Reference inside of function at 0x{x.Key:X}")));
85 }
86 catch (Exception e)

88 Debug.WriteLine($"Error decompiling function {Export.InstancedFullPath}: {e.Message}");

144 public void UpdateChildrenChain(bool relinkChildrenStructs = false)
145 {
146 if (this is UFunction fn)
147 {
148 //UpdateChildrenChain not yet working for function exports, use function compilation instead
149 return;
150 }
151 var children = Export.GetChildren<ExportEntry>().ToList();
152 for (int i = children.Count - 1; i >= 0; --i)
153 {
154 var c = children[i];
155 if (ObjectBinary.From(c) is UField uf)
156 {
157 uf.Next = i == 0 ? 0 : children[i - 1];
158 if (relinkChildrenStructs && uf is UStruct st)
159 {
160 st.UpdateChildrenChain(true);
161 }
162 c.WriteBinary(uf);
163 }
164 else
165 {
166 Debug.WriteLine($"Can't link non UField {c.InstancedFullPath}");
167 }
168 }
169 Children = children.Any() ? children[^1].UIndex : 0;
170 }
