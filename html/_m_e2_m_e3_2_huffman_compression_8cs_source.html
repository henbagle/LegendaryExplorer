138 private void LoadXmlInputData(string fileName, bool debugVersion)
139 {
140 var xmlReader = new XmlTextReader(fileName);
141
142 /* read and store TLK Tool version, which was used to create the XML file */
143 while (xmlReader.Read())
144 {
145 if (xmlReader.NodeType == XmlNodeType.Element && xmlReader.Name.Equals("tlkFile", StringComparison.OrdinalIgnoreCase))
146 {
147 if (xmlReader.GetAttribute("TLKToolVersion") is string toolVersion)
148 {
149 _inputFileVersion = new Version(toolVersion.Trim('v'));
150 }
151 break;
152 }
153 }
154 if (_inputFileVersion == null || _inputFileVersion >= new Version("2.0.12"))
155 {
156 int position = 0;
157 while (xmlReader.Read())
158 {
159 if (xmlReader.NodeType == XmlNodeType.Element && xmlReader.Name.Equals("String", StringComparison.OrdinalIgnoreCase))
160 {
161 int id = 0;
162 if (!int.TryParse(xmlReader.GetAttribute("id"), out id))
163 {
164 throw new XmlException("id not an integer.", null, xmlReader.LineNumber, xmlReader.LinePosition);
165 }
166 string data = xmlReader.ReadElementContentAsString();
167
168 data = data.Replace("\r\n", "\n");
169 /* every string should be NULL-terminated */
170 if (id >= 0)
171 data += '\0';
172 /* only add debug info if we are in debug mode and StringID is positive AND it's localizable */
173 if (id >= 0 && debugVersion && (id & 0x8000000) != 0x8000000)
174 _inputData.Add(new TLKStringRef(id, position, "(#" + id + ") " + data, position));
175 else
176 _inputData.Add(new TLKStringRef(id, position, data, position));
177 position++;
178 }
179 }
180 }
181 else //legacy support
182 {
183 while (xmlReader.Read())
184 {
185 if (xmlReader.NodeType == XmlNodeType.Element && xmlReader.Name.Equals("string", StringComparison.OrdinalIgnoreCase))
186 {
187 int id = 0, position = 0;
188 string data = "";
189 while (xmlReader.Name != "string" || xmlReader.NodeType != XmlNodeType.EndElement)
190 {
191 if (!xmlReader.Read() || xmlReader.NodeType != XmlNodeType.Element)
192 continue;
193 if (xmlReader.Name == "id")
194 id = xmlReader.ReadElementContentAsInt();
195 else if (xmlReader.Name == "position")
196 position = xmlReader.ReadElementContentAsInt();
197 else if (xmlReader.Name == "data")
198 data = xmlReader.ReadString();
199 }
200 data = data.Replace("\r\n", "\n");
201 /* every string should be NULL-terminated */
202 if (id >= 0)
203 data += '\0';
204 /* only add debug info if we are in debug mode and StringID is positive AND it's localizable */
205 if (id >= 0 && debugVersion && (id & 0x8000000) != 0x8000000)
206 _inputData.Add(new TLKStringRef(id, 0, "(#" + id + ") " + data, position));
207 else
208 _inputData.Add(new TLKStringRef(id, 0, data, position));
209 }
210 }
211 }
212 xmlReader.Close();

276 private void TraverseHuffmanTree(HuffmanNode node, List<bool> code)
277 {
278 /* check if both sons are null */
279 if (node.Left == node.Right)
280 {
281 BitArray ba = new BitArray(code.ToArray());
282 _huffmanCodes.Add(node.Data, ba);
283 }
284 else
285 {
286 /* adds 0 to the code - process left son*/
287 code.Add(false);
288 TraverseHuffmanTree(node.Left, code);
289 code.RemoveAt(code.Count() - 1);
290
291 /* adds 1 to the code - process right son*/
292 code.Add(true);
293 TraverseHuffmanTree(node.Right, code);
294 code.RemoveAt(code.Count() - 1);
295 }

302 private List<int> ConvertHuffmanTreeToBuffer()
303 {
304 Queue<HuffmanNode> q = new Queue<HuffmanNode>();
305 Dictionary<int, HuffmanNode> indices = new Dictionary<int, HuffmanNode>();
306
307 int index = 0;
308 q.Enqueue(_huffmanTree[0]);
309
310 while (q.Count > 0)
311 {
312 HuffmanNode node = q.Dequeue();
313 /* if it's a leaf - set it's ID to reflect char data the node contains */
314 if (node.Left == node.Right)
315 {
316 /* store the char data */
317 node.ID = -1 - node.Data;
318
319 /* that's how it's going to be decoded when parsing TLK file:
320 * char c = BitConverter.ToChar(BitConverter.GetBytes(0xffff - node.ID), 0); */
321 }
322 else
323 {
324 node.ID = index++;
325 indices.Add(node.ID, node);
326 }
327 if (node.Right != null)
328 q.Enqueue(node.Right);
329 if (node.Left != null)
330 q.Enqueue(node.Left);
331 }
332
333 List<int> output = new List<int>();
334
335 foreach (HuffmanNode node in indices.Values)
336 {
337 output.Add(node.Left.ID);
338 output.Add(node.Right.ID);
339 }
340
341 return output;
342 }

351 private static byte[] BitArrayListToByteArray(List<BitArray> bitsList, int bitsCount)
352 {
353 const int BITSPERBYTE = 8;
354
355 int bytesize = bitsCount / BITSPERBYTE;
356 if (bitsCount % BITSPERBYTE > 0)
357 bytesize++;
358
359 byte[] bytes = new byte[bytesize];
360 int bytepos = 0;
361 int bitsRead = 0;
362 byte value = 0;
363 byte significance = 1;
364
365 foreach (BitArray bits in bitsList)
366 {
367 int bitpos = 0;
368
369 while (bitpos < bits.Length)
370 {
371 if (bits[bitpos])
372 {
373 value += significance;
374 }
375 ++bitpos;
376 ++bitsRead;
377 if (bitsRead % BITSPERBYTE == 0)
378 {
379 bytes[bytepos] = value;
380 ++bytepos;
381 value = 0;
382 significance = 1;
383 bitsRead = 0;
384 }
385 else
386 {
387 significance <<= 1;
388 }
389 }
390 }
391 if (bitsRead % BITSPERBYTE != 0)
392 bytes[bytepos] = value;
393 return bytes;
394 }

407 public static MemoryStream SaveToTlkStream(List<TLKStringRef> stringRefs)
408 {
409 MemoryStream memStream = MemoryManager.GetMemoryStream();
410 var hc = new HuffmanCompression();
411 if (stringRefs != null)
412 {
413 hc._inputData = stringRefs.OrderBy(x => x.CalculatedID).ToList();
414 hc.PrepareHuffmanCoding();
415 }
416 /* converts Huffmann Tree to binary form */
417 List<int> treeBuffer = hc.ConvertHuffmanTreeToBuffer();
418
419 /* preparing data and entries for writing to file
420 * entries list consists of pairs <String ID, Offset> */
421 var binaryData = new List<BitArray>();
422 var maleStrings = new Dictionary<int, int>();
423 var femaleStrings = new Dictionary<int, int>();
424 int offset = 0;
425
426 foreach (var entry in hc._inputData)
427 {
428 if (entry.StringID < 0)
429 {
430 if (!maleStrings.ContainsKey(entry.StringID))
431 maleStrings.Add(entry.StringID, Convert.ToInt32(entry.ASCIIData));
432 else
433 femaleStrings.Add(entry.StringID, Convert.ToInt32(entry.ASCIIData));
434 continue;
435 }
436
437 if (!maleStrings.ContainsKey(entry.StringID))
438 maleStrings.Add(entry.StringID, offset);
439 else
440 femaleStrings.Add(entry.StringID, offset);
441
442 /* for every character in a string, put it's binary code into data array */
443 foreach (char c in entry.ASCIIData)
444 {
445 binaryData.Add(hc._huffmanCodes[c]);
446 offset += hc._huffmanCodes[c].Count;
447 }
448 }
449
450 /* preparing TLK Header */
451 const int magic = 7040084; //Tlk\0
452 const int ver = 3;
453 const int min_ver = 2;
454 int entry1Count = maleStrings.Count;
455 int entry2Count = femaleStrings.Count;
456 int treeNodeCount = treeBuffer.Count() / 2;
457 int dataLength = offset / 8;
458 if (offset % 8 > 0)
459 ++dataLength;
460
461
462 /* writing TLK Header */
463 memStream.WriteInt32(magic);
464 memStream.WriteInt32(ver);
465 memStream.WriteInt32(min_ver);
466 memStream.WriteInt32(entry1Count);
467 memStream.WriteInt32(entry2Count);
468 memStream.WriteInt32(treeNodeCount);
469 memStream.WriteInt32(dataLength);
470
471 /* writing entries */
472 foreach (var entry in maleStrings)
473 {
474 memStream.WriteInt32(entry.Key);
475 memStream.WriteInt32(entry.Value);
476 }
477 foreach (var entry in femaleStrings)
478 {
479 memStream.WriteInt32(entry.Key);
480 memStream.WriteInt32(entry.Value);
481 }
482
483 /* writing HuffmanTree */
484 foreach (int element in treeBuffer)
485 {
486 memStream.WriteInt32(element);
487 }
488
489 /* writing data */
490 byte[] data = BitArrayListToByteArray(binaryData, offset);
491 memStream.WriteFromBuffer(data);
492 return memStream;
493 }
