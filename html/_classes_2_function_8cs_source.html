246 public void ParseFunction()
247 {
248 var parsedData = Bytecode.ParseBytecode(script, export);
249 ScriptBlocks = parsedData.Item1;
250 SingularTokenList = parsedData.Item2;
251
252 // Calculate memory offsets
253 List<int> objRefPositions = ScriptBlocks.SelectMany(tok => tok.inPackageReferences)
254 .Where(tup => tup.type == Token.INPACKAGEREFTYPE_ENTRY)
255 .Select(tup => tup.position).ToList();
256 int calculatedLength = DiskSize + 4 * objRefPositions.Count;
257
258 DiskToMemPosMap = new int[DiskSize];
259 int iDisk = 0;
260 int iMem = 0;
261 foreach (int objRefPosition in objRefPositions)
262 {
263 while (iDisk < objRefPosition + 4)
264 {
265 DiskToMemPosMap[iDisk] = iMem;
266 iDisk++;
267 iMem++;
268 }
269 iMem += 4;
270 }
271 while (iDisk < DiskSize)
272 {
273 DiskToMemPosMap[iDisk] = iMem;
274 iDisk++;
275 iMem++;
276 }
277
278 foreach (Token t in ScriptBlocks)
279 {
280 var diskPos = t.pos - 32;
281 if (diskPos >= 0 && diskPos < DiskToMemPosMap.Length)
282 {
283 t.memPos = DiskToMemPosMap[diskPos];
284 }
285 }
