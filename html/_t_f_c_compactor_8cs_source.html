128 public TFCInfo GetOutTFC(int requiredSize)
129 {
130 if (!tfcSizes.Any())
131 {
132 return InitTFC();
133 }
134
135 foreach (var tfc in tfcSizes)
136 {
137 if (tfc.currentSize + requiredSize < int.MaxValue)
138 {
139 return new TFCInfo(tfc.tfcInfo);
140 }
141 }
142
143 return InitTFC();
144 }

150 private TFCInfo InitTFC()
151 {
152 int? tfcIdx = infoPackage.UseIndexing ? 0 : null; // 0 cause it gets ++'d before first use
153 while (true)
154 {
155 if (infoPackage.UseIndexing)
156 {
157 tfcIdx++;
158 }
159
160 //var tfcName = $"{infoPackage.TFCType}_{infoPackage.DLCName}{tfcIdx}";
161 var tfcName = $"{infoPackage.TFCType}{tfcIdx}_{infoPackage.DLCName}";
162 var testTFCPath = Path.Combine(infoPackage.StagingPath, $"{tfcName}.tfc");
163 if (File.Exists(testTFCPath))
164 continue; // go to next available name
165
166 // Init TFC
167 using var fs = File.Open(testTFCPath, FileMode.Create, FileAccess.Write);
168 Guid g = Guid.NewGuid();
169 fs.WriteGuid(g);
170 var nTfc = new TFCInfo() { TFCGuid = g, TFCName = tfcName };
171
172 tfcSizes.Add(nTfc, 16);
173 return nTfc;
174 }

177 public static void CompactTFC(TFCCompactorInfoPackage infoPackage, Action<string> errorCallback, Action<string, int, int> progressDelegate = null, TextureMap textureMap = null, CancellationToken cts = default)
178 {
179 TFCCompactor compactor = new TFCCompactor(infoPackage);
180
181 var rootNodes = new List<TextureMapMemoryEntry>();
182 textureMap ??= TextureMapGenerator.GenerateMapForFolder(infoPackage.BaseCompactionPath, x => new TextureMapMemoryEntry(x), x => rootNodes.Add(x), progressDelegate, cts);
183 Debug.WriteLine($@"Texture map count: {textureMap.CalculatedMap.Count}");
184
185 // CRC Check
186 if (textureMap.CalculatedMap.Any(x => x.HasUnmatchedCRCs))
187 {
188 // ERROR: UNMATCHED CRCS
189 // Textures are not properly named and unique
190 errorCallback?.Invoke("There are unmatched CRCs in this directory. Textures with same full instanced paths must have the same textures. Use Texture Studio to identify and fix these issues.");
191 return;
192 }
193
194 // Copy external data to new TFC(s)--------------------
195 foreach (var f in Directory.GetFiles(infoPackage.StagingPath, "*.tfc", SearchOption.TopDirectoryOnly))
196 {
197 File.Delete(f); // Delete existing TFCs in staging dir
198 }
199 var gameFiles = MELoadedFiles.GetFilesLoadedInGame(textureMap.Game, includeTFCs: true, gameRootOverride: infoPackage.GamePath);
200
201 // add TFCs from the directory we're compacting so we can pull data out
202 foreach (var basePathTFC in Directory.GetFiles(infoPackage.BaseCompactionPath, "*.tfc",
203 SearchOption.AllDirectories))
204 {
205 gameFiles[Path.GetFileName(basePathTFC)] = basePathTFC;
206 }
207
208 // Packages are not updated in this step so if something goes wrong it doesn't blow up the files
209 foreach (var v in textureMap.CalculatedMap)
210 {
211 var entries = v.GetAllTextureEntries().ToList();
212 foreach (var entry in entries)
213 {
214 if (compactor.infoPackage.TFCsToCompact.Contains(entry.Instances[0].TFCName))
215 {
216 if (compactor.CRCMap.TryGetValue(entry.GetCRC(), out var portedInfo))
217 {
218 // Texture doesn't need ported into TFC
219 }
220 else
221 {
222 // texture has not been copied to TFC
223 var texInfo = entry.Instances[0];
224 var diskSize = entry.GetExternalDiskSize();
225 var destTFCInfo = compactor.GetOutTFC(diskSize); // Get TFC to write to
226 var destTFCPath = Path.Combine(infoPackage.StagingPath, $"{destTFCInfo.TFCName}.tfc");
227 using var outStream = File.Open(destTFCPath, FileMode.Append, FileAccess.Write);
228 using var inStream = File.OpenRead(gameFiles[$"{texInfo.TFCName}.tfc"]);
229 for (int i = 0; i < texInfo.CompressedMipInfos.Count; i++)
230 {
231 var mipInfo = texInfo.CompressedMipInfos[i];
232 destTFCInfo.MipOffsetMap[i] = (int)outStream.Position;
233 inStream.Seek(mipInfo.Offset, SeekOrigin.Begin);
234
235 if (NeedsCompressionConverted(mipInfo.StorageType, textureMap.Game))
236 {
237 Debug.WriteLine($@"Converting external storage type from {mipInfo.StorageType} to {GetTargetExternalStorageType(textureMap.Game)}");
238 var decompressed = new byte[mipInfo.UncompressedSize];
239 TextureCompression.DecompressTexture(decompressed, inStream, mipInfo.StorageType, mipInfo.UncompressedSize, mipInfo.CompressedSize);
240
241 // Recompress texture
242 var compressed = TextureCompression.CompressTexture(decompressed, GetTargetExternalStorageType(textureMap.Game));
243 outStream.Write(compressed);
244 destTFCInfo.MipCompressedSizeMap[i] = compressed.Length;
245 }
246 else
247 {
248 inStream.CopyToEx(outStream, mipInfo.CompressedSize);
249 destTFCInfo.MipCompressedSizeMap[i] = mipInfo.CompressedSize;
250 }
251 }
252
253 // Add to the map
254 compactor.CRCMap[texInfo.CRC] = destTFCInfo;
255 }
256 }
257 }
258 }
259
260 // Build list of package updates
261 Dictionary<string, List<TextureMapPackageEntry>> packageTextureMap = new(); // Maps a package name to all instances that need to be updated, so we can open a single package at a time and update it
262 foreach (var v in textureMap.CalculatedMap)
263 {
264 // Get all texture leaves
265 var entries = v.GetAllTextureEntries().ToList();
266 foreach (var entry in entries)
267 {
268 foreach (var instance in entry.Instances)
269 {
270 // Build list of TextureMapPackageEntry
271 List<TextureMapPackageEntry> packageTexturesToUpdate;
272 if (!packageTextureMap.TryGetValue(instance.PackageName, out packageTexturesToUpdate))
273 {
274 packageTexturesToUpdate = new List<TextureMapPackageEntry>();
275 packageTextureMap[instance.PackageName] = packageTexturesToUpdate;
276 }
277
278 packageTexturesToUpdate.Add(instance);
279 }
280 }
281 }
282
283 // Update packages
284 var packagePool = Directory.GetFileSystemEntries(infoPackage.BaseCompactionPath, "*.*", SearchOption.AllDirectories).ToList();
285 foreach (var packageNamePair in packageTextureMap)
286 {
287 // Mods may have multiple same-named packages, if for example, they use alternates in M3
288 // So we can't use a mapping of filenames. Hopefully files don't diverge far enough that this won't matter...
289 foreach (var namedPackage in packagePool.Where(x => Path.GetFileName(x).Equals(packageNamePair.Key, StringComparison.InvariantCultureIgnoreCase)))
290 {
291 progressDelegate?.Invoke($"Updating package {packageNamePair.Key}", -1, -1);
292 using var package = MEPackageHandler.OpenMEPackage(namedPackage, forceLoadFromDisk: true);
293 foreach (var tu in packageNamePair.Value)
294 {
295 if (string.IsNullOrWhiteSpace(tu.TFCName))
296 {
297 continue; // PCC Stored
298 }
299 var exportToUpdate = package.FindExport(tu.ExportPath);
300 if (exportToUpdate == null)
301 {
302 Debug.WriteLine($@"Could not find export {tu.ExportPath} in package {namedPackage}, skipping update");
303 }
304 var t2d = ObjectBinary.From<UTexture2D>(exportToUpdate);
305 t2d.Mips.RemoveAll(x => x.StorageType == StorageTypes.empty); // Remove empty mips
306
307 // Update offset
308 if (compactor.CRCMap.TryGetValue(tu.CRC, out var updateInfo))
309 {
310 for (int i = 0; i < updateInfo.MipOffsetMap.Count; i++)
311 {
312 t2d.Mips[i].DataOffset = updateInfo.MipOffsetMap[i];
313
314 // For conversions
315 t2d.Mips[i].StorageType = GetTargetExternalStorageType(textureMap.Game); //tfc compactor changes external types so this should be OK?
316 t2d.Mips[i].CompressedSize = updateInfo.MipCompressedSizeMap[i];
317 }
318
319 exportToUpdate.WriteBinary(t2d);
320
321 // Update TFC properties
322 var properties = exportToUpdate.GetProperties();
323 properties.AddOrReplaceProp(new FGuid(updateInfo.TFCGuid).ToStructProperty("TFCFileGuid"));
324 properties.AddOrReplaceProp(new NameProperty(updateInfo.TFCName, "TextureFileCacheName"));
325 exportToUpdate.WriteProperties(properties);
326 }
327 else if (infoPackage.TFCsToCompact.Contains(tu.TFCName))
328 {
329 Debug.WriteLine($"CRC not found in map 0x{tu.CRC:X8}, {tu.ExportPath} in {tu.TFCName}");
330 }
331 }
332
333 if (package.IsModified)
334 {
335 package.Save();
336 }
337 }
338 }
339
340 var dlcFolderDir = Directory.GetFileSystemEntries(infoPackage.BaseCompactionPath, infoPackage.DLCName, SearchOption.AllDirectories).FirstOrDefault();
341 if (dlcFolderDir == null && Path.GetFileName(infoPackage.BaseCompactionPath) == infoPackage.DLCName)
342 dlcFolderDir = infoPackage.BaseCompactionPath;
343 if (dlcFolderDir != null)
344 {
345 // Delete all existing TFCs
346 var tfcsToDelete = Directory.GetFileSystemEntries(dlcFolderDir, "*.tfc", SearchOption.AllDirectories);
347 foreach (var tfc in tfcsToDelete)
348 {
349 File.Delete(tfc);
350 }
351
352 var destPath = Path.Combine(dlcFolderDir, infoPackage.Game.CookedDirName());
353 foreach (var tfc in compactor.GetAllTFCs())
354 {
355 var tfcSource = Path.Combine(infoPackage.StagingPath, $"{tfc.TFCName}.tfc");
356 File.Move(tfcSource, Path.Combine(destPath, Path.GetFileName(tfcSource)));
357 }
358
359 if (compactor.infoPackage.UseIndexing)
360 {
361 // Stub TFC
362 File.WriteAllBytes(Path.Combine(destPath, $"Textures_{compactor.infoPackage.DLCName}.tfc"), Guid.NewGuid().ToByteArray());
363 }
364 }
365 }

367 private static StorageTypes GetTargetExternalStorageType(MEGame textureMapGame)
368 {
369 if (textureMapGame is MEGame.ME1 or MEGame.ME2) return StorageTypes.extLZO;
370 if (textureMapGame is MEGame.ME3) return StorageTypes.extZlib;
371 if (textureMapGame.IsLEGame()) return StorageTypes.extOodle;
372 throw new ArgumentOutOfRangeException($"Invalid value {textureMapGame} for {nameof(textureMapGame)} as target type for texture");
373 }
374
375 private static bool NeedsCompressionConverted(StorageTypes mipInfoStorageType, MEGame textureMapGame)
376 {
377 if (textureMapGame is MEGame.ME1 or MEGame.ME2 && mipInfoStorageType != StorageTypes.pccLZO && mipInfoStorageType != StorageTypes.extLZO) return true;
378 if (textureMapGame is MEGame.ME3 && mipInfoStorageType != StorageTypes.pccZlib && mipInfoStorageType != StorageTypes.extZlib) return true;
379 if (textureMapGame.IsLEGame() && mipInfoStorageType != StorageTypes.pccOodle && mipInfoStorageType != StorageTypes.extOodle) return true;
380 return false;
381 }
