165 public static string Truncate(this string str, int length, bool ellipsis = false)
166 {
167 if (length < 0)
168 {
169 throw new ArgumentOutOfRangeException(nameof(length), "Length must be >= 0");
170 }
171
172 if (str == null)
173 {
174 return null;
175 }
176
177 if (ellipsis && length > 4 && length < str.Length)
178 {
179 return $"{str.Substring(0, length - 3)}...";
180 }
181
182 int maxLength = Math.Min(str.Length, length);
183 return str.Substring(0, maxLength);
184 }

191 public static string TruncCenter(this string str, int length)
192 {
193 if (str == null)
194 {
195 return null;
196 }
197
198 if (length < 5)
199 {
200 length = 5;
201 }
202
203 if (str.Length <= length)
204 {
205 return str;
206 }
207
208 if (length % 2 == 0)
209 {
210 int sideLen = (length - 2) / 2;
211 return $"{str.Substring(0, sideLen)}..{str.Substring(str.Length - sideLen)}";
212 }
213 else
214 {
215 int sideLen = (length - 3) / 2;
216 return $"{str.Substring(0, sideLen)}...{str.Substring(str.Length - sideLen)}";
217 }

239 public static List<String> WrapLines(this string text, int maxLength)
240 {
241
242 // Return empty list of strings if the text was empty
243 if (text.Length == 0) return new List<string>();
244
245 var words = text.Split(' ');
246 var lines = new List<string>();
247 var currentLine = "";
248
249 foreach (var currentWord in words)
250 {
251
252 if ((currentLine.Length > maxLength) || ((currentLine.Length + currentWord.Length) > maxLength))
253 {
254 lines.Add(currentLine);
255 currentLine = "";
256 }
257
258 if (currentLine.Length > 0)
259 currentLine += " " + currentWord;
260 else
261 currentLine += currentWord;
262
263 }
264
265 if (currentLine.Length > 0)
266 lines.Add(currentLine);
267
268
269 return lines;
270 }
