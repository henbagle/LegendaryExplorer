50 public static void SkipSequenceElement(ExportEntry elementToSkip, string outboundLinkName = null, int outboundLinkIdx = -1)
51 {
52 if (outboundLinkIdx == -1 && outboundLinkName == null)
53 throw new Exception(@"SkipSequenceElement() must have an outboundLinkName or an outboundLinkIdx!");
54
55 if (outboundLinkIdx == -1)
56 {
57 var outboundLinkNames = KismetHelper.GetOutboundLinkNames(elementToSkip);
58 outboundLinkIdx = outboundLinkNames.IndexOf(outboundLinkName);
59 }
60
61
62 // List of outbound link elements on the specified item we want to skip. These will be placed into the inbound item
63 Debug.WriteLine($@"Attempting to skip {elementToSkip.UIndex} in {elementToSkip.FileRef.FilePath}");
64 var outboundLinkLists = SeqTools.GetOutboundLinksOfNode(elementToSkip);
65 var inboundToSkippedNode = SeqTools.FindOutboundConnectionsToNode(elementToSkip, SeqTools.GetAllSequenceElements(elementToSkip).OfType<ExportEntry>());
66
67 var newTargetNodes = outboundLinkLists[outboundLinkIdx];
68
69 foreach (var preNode in inboundToSkippedNode)
70 {
71 // For every node that links to the one we want to skip...
72 var preNodeLinks = GetOutboundLinksOfNode(preNode);
73
74 foreach (var ol in preNodeLinks)
75 {
76 var numRemoved = ol.RemoveAll(x => x.LinkedOp == elementToSkip);
77
78 if (numRemoved > 0)
79 {
80 // At least one was removed. Repoint it
81 ol.AddRange(newTargetNodes);
82 }
83 }
84
85 WriteOutboundLinksToNode(preNode, preNodeLinks);
86 }
87 }

94 public static List<List<OutboundLink>> GetOutboundLinksOfNode(ExportEntry node)
95 {
96 var outputLinksMapping = new List<List<OutboundLink>>();
97 var outlinksProp = node.GetProperty<ArrayProperty<StructProperty>>("OutputLinks");
98 if (outlinksProp != null)
99 {
100 int i = 0;
101 foreach (var ol in outlinksProp)
102 {
103 List<OutboundLink> oLinks = new List<OutboundLink>();
104 outputLinksMapping.Add(oLinks);
105
106 var links = ol.GetProp<ArrayProperty<StructProperty>>("Links");
107 if (links != null)
108 {
109 foreach (var l in links)
110 {
111 oLinks.Add(OutboundLink.FromStruct(l, node.FileRef));
112 }
113 }
114
115 i++;
116 }
117 }
118
119 return outputLinksMapping;

127 public static void WriteOutboundLinksToNode(ExportEntry node, List<List<OutboundLink>> linkSet)

129 var outlinksProp = node.GetProperty<ArrayProperty<StructProperty>>("OutputLinks");

131 if (linkSet.Count != outlinksProp.Count)
132 {
133 Debug.WriteLine("Sets are out of sync for WriteLinksToNode()!");

136

142 }
143

185 public static List<ExportEntry> FindOutboundConnectionsToNode(ExportEntry node, IEnumerable<ExportEntry> sequenceElements)
186 {
187 List<ExportEntry> referencingNodes = new List<ExportEntry>();
188
189 foreach (var seqObj in sequenceElements)
190 {
191 if (seqObj == node) continue; // Skip node pointing to itself
192 var linkSet = GetOutboundLinksOfNode(seqObj);
193 if (linkSet.Any(x => x.Any(y => y.LinkedOp != null && y.LinkedOp.UIndex == node.UIndex)))
194 {
195 referencingNodes.Add(seqObj);
196 }
197 }
198
199 return referencingNodes.Distinct().ToList();
200 }

231 public static List<IEntry> GetAllSequenceElements(ExportEntry export)
232 {
233 if (export.ClassName == "Sequence")
234 {
235 var seqObjs = export.GetProperty<ArrayProperty<ObjectProperty>>("SequenceObjects");
236 if (seqObjs != null)
237 {
238 return seqObjs.Where(x => export.FileRef.IsEntry(x.Value)).Select(x => x.ResolveToEntry(export.FileRef)).ToList();
239 }
240 }
241 else
242 {
243 // Pull from parent sequence.
244 var pSeqObj = export.GetProperty<ObjectProperty>("ParentSequence");
245 if (pSeqObj != null && pSeqObj.ResolveToEntry(export.FileRef) is ExportEntry pSeq && pSeq.ClassName == "Sequence")
246 {
247 return GetAllSequenceElements(pSeq);
248 }
249 }
250
251 return null;
