210 public override List<(UIndex, string)> GetUIndexes(MEGame game)
211 {
212 var uIndexes = new List<(UIndex, string)>();
213
214 uIndexes.Add((Self, nameof(Self)));
215 uIndexes.AddRange(Actors.Select((u, i) => (u, $"{nameof(Actors)}[{i}]")));
216 uIndexes.Add((Model, nameof(Model)));
217 uIndexes.AddRange(ModelComponents.Select((u, i) => (u, $"{nameof(ModelComponents)}[{i}]")));
218 uIndexes.AddRange(GameSequences.Select((u, i) => (u, $"{nameof(GameSequences)}[{i}]")));
219 uIndexes.AddRange(TextureToInstancesMap.Select((kvp, i) => (kvp.Key, $"{nameof(TextureToInstancesMap)}[{i}]")));
220 uIndexes.AddRange(CachedPhysSMDataMap.Select((kvp, i) => (kvp.Key, $"{nameof(CachedPhysSMDataMap)}[{i}]")));
221 uIndexes.AddRange(CachedPhysPerTriSMDataMap.Select((kvp, i) => (kvp.Key, $"{nameof(CachedPhysPerTriSMDataMap)}[{i}]")));
222 uIndexes.AddRange(ForceStreamTextures.Select((kvp, i) => (kvp.Key, $"{nameof(ForceStreamTextures)}[{i}]")));
223 uIndexes.Add((NavListStart, nameof(NavListStart)));
224 uIndexes.Add((NavListEnd, nameof(NavListEnd)));
225 uIndexes.Add((CoverListStart, nameof(CoverListStart)));
226 uIndexes.Add((CoverListEnd, nameof(CoverListEnd)));
227 if (game >= MEGame.ME3)
228 {
229 uIndexes.Add((PylonListStart, nameof(PylonListStart)));
230 uIndexes.Add((PylonListEnd, nameof(PylonListEnd)));
231 }
232 if (game.IsGame3())
233 {
234 uIndexes.AddRange(CoverLinks.Select((u, i) => (u, $"{nameof(CoverLinks)}[{i}]")));
235 uIndexes.AddRange(NavPoints.Select((u, i) => (u, $"{nameof(NavPoints)}[{i}]")));
236 }
237 uIndexes.AddRange(CrossLevelActors.Select((u, i) => (u, $"{nameof(CrossLevelActors)}[{i}]")));
238 if (game.IsGame1())
239 {
240 uIndexes.Add((ArtPlaceable1, nameof(ArtPlaceable1)));
241 uIndexes.Add((ArtPlaceable2, nameof(ArtPlaceable2)));
242 }
243
244 return uIndexes;
