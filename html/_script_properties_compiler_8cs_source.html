94 private PropertyCollection ConvertStatementsToPropertyCollection(List<Statement> statements, ExportEntry export, Dictionary<NameReference, ExportEntry> subObjectDict)
95 {
96 var existingSubObjects = export.GetChildren<ExportEntry>().ToList();
97 var props = new PropertyCollection();
98 var subObjectsToFinish = new List<(Subobject, ExportEntry, int)>();
99 foreach (Statement statement in statements)
100 {
101 switch (statement)
102 {
103 case Subobject subObj:
104 var subObjName = NameReference.FromInstancedString(subObj.NameDeclaration.Name);
105 existingSubObjects.TryRemove(exp => exp.ObjectName == subObjName, out ExportEntry existingSubObject);
106 int netIndex = existingSubObject?.NetIndex ?? 0;
107 CreateSubObject(subObj, export, ref existingSubObject);
108 subObjectDict[subObjName] = existingSubObject;
109 subObjectsToFinish.Add(subObj, existingSubObject, netIndex);
110 break;
111 case AssignStatement assignStatement:
112 Property prop = ConvertToProperty(assignStatement, subObjectDict);
113 props.AddOrReplaceProp(prop);
114 break;
115 default:
116 throw new Exception($"Unexpected statement type: {statement.GetType().Name}");
117 }
118 }
119 //Default__ objects and struct defaults serialize transients, but subobjects dont
120 ShouldStripTransients = true;
121 foreach ((Subobject subobject, ExportEntry subExport, int netIndex) in subObjectsToFinish)
122 {
123 WriteSubObjectData(subobject, subExport, netIndex, subObjectDict);
124 }
125
126 if (existingSubObjects.Any())
127 {
128 EntryPruner.TrashEntriesAndDescendants(existingSubObjects);
129 }
130 return props;
131 }

133 private void CreateSubObject(Subobject subObject, ExportEntry parent, ref ExportEntry subExport)
134 {
135 var objName = NameReference.FromInstancedString(subObject.NameDeclaration.Name);
136 IEntry classEntry = EntryImporter.EnsureClassIsInFile(Pcc, subObject.Class.Name);
137 if (subExport is null)
138 {
139 if (Pcc.TryGetTrash(out subExport))
140 {
141 subExport.Parent = parent;
142 }
143 else
144 {
145 subExport = new ExportEntry(Pcc, parent, objName);
146 Pcc.AddExport(subExport);
147 }
148 }
149 subExport.SuperClass = null;
150 subExport.Class = classEntry;
151 subExport.ObjectName = objName;
152 subExport.Class = classEntry;
153
154 if (subObject.IsTemplate && Default__Archetype is not null)
155 {
156 string defaultArchetypePath = Default__Archetype.InstancedFullPath;
157 string defaultObjectPath = Default__Export.InstancedFullPath;
158 string subObjPath = subExport.InstancedFullPath;
159 string subPath = subObjPath[(defaultObjectPath.Length + 1)..];
160 ExportEntry subObjArchetype = Default__Archetype.FileRef.FindExport($"{defaultArchetypePath}.{subPath}");
161 if (subObjArchetype is null)
162 {
163 //sometimes the subobjects have a flat structure under the Default__
164 subPath = objName.Instanced;
165 subObjArchetype = Default__Archetype.FileRef.FindExport($"{defaultArchetypePath}.{subPath}");
166 }
167
168 if (subObjArchetype is not null && subObjArchetype.ClassName.CaseInsensitiveEquals(subExport.ClassName))
169 {
170 subExport.Archetype = ReferenceEquals(Default__Archetype.FileRef, Pcc)
171 ? subObjArchetype
172 : Pcc.getEntryOrAddImport($"{Default__Export.Archetype.InstancedFullPath}.{subPath}", classEntry.ObjectName.Instanced, classEntry.ParentName);
173 return;
174 }
175 //sometimes the archetype is a subobject of the Default__ for a parent suboject's class.
176 if (parent != Default__Export)
177 {
178 var archetypeRoot = parent;
179 while (archetypeRoot.Archetype is not null)
180 {
181 archetypeRoot = (ExportEntry)archetypeRoot.Parent;
182 }
183 subPath = subObjPath[(archetypeRoot.InstancedFullPath.Length + 1)..];
184 IEntry classDefaultObject = GetClassDefaultObject(archetypeRoot.Class);
185 var baseDefault = classDefaultObject switch
186 {
187 ExportEntry exp => exp,
188 ImportEntry imp => EntryImporter.ResolveImport(imp, packageCache, null, "INT"),
189 _ => null
190 };
191 if (baseDefault is not null)
192 {
193 subObjArchetype = baseDefault.FileRef.FindExport($"{baseDefault.InstancedFullPath}.{subPath}");
194 if (subObjArchetype is null)
195 {
196 subPath = objName.Instanced;
197 subObjArchetype = baseDefault.FileRef.FindExport($"{baseDefault.InstancedFullPath}.{subPath}");
198 }
199 if (subObjArchetype is not null && subObjArchetype.ClassName.CaseInsensitiveEquals(subExport.ClassName))
200 {
201 subExport.Archetype = ReferenceEquals(baseDefault.FileRef, Pcc)
202 ? subObjArchetype
203 : Pcc.getEntryOrAddImport($"{classDefaultObject.InstancedFullPath}.{subPath}", classEntry.ObjectName.Instanced, classEntry.ParentName);
204 return;
205 }
206 }
207 }
208
209 }
210 subExport.Archetype = null;
211 }

213 private Property ConvertToProperty(AssignStatement assignStatement, Dictionary<NameReference, ExportEntry> subObjectDict = null)
214 {
215 var nameRef = (SymbolReference) assignStatement.Target;
216 int staticArrayIndex = 0;
217 if (nameRef is ArraySymbolRef staticArrayRef)
218 {
219 staticArrayIndex = ((IntegerLiteral) staticArrayRef.Index).Value;

231 private Property MakeProperty(NameReference propName, VariableType type, Expression literal, Dictionary<NameReference, ExportEntry> subObjectDict = null)
232 {
233 Property prop;
234 if (type is StaticArrayType sat)
235 {
236 type = sat.ElementType;
237 }
238 switch (type)
239 {
240 case ClassType classType:
241 prop = new ObjectProperty(literal is NoneLiteral ? null : CompilerUtils.ResolveClass((Class)((ClassType)((ObjectLiteral)literal).Class).ClassLimiter, Pcc), propName);
242 break;
243 case Class cls:
244 IEntry entry = literal switch
245 {
246 NoneLiteral => null,
247 SymbolReference {Node: Subobject subobject} => subObjectDict?[NameReference.FromInstancedString(subobject.NameDeclaration.Name)],
248 _ => Pcc.FindEntry(((ObjectLiteral)literal).Name.Value)
249 };
250
251 prop = new ObjectProperty(entry, propName)
252 {
253 InternalPropType = cls.IsInterface ? PropertyType.InterfaceProperty : PropertyType.ObjectProperty
254 };
255 break;
256 case DelegateType:
257 int objUIndex = 0;
258 NameReference funcName;
259 if (literal is NoneLiteral)
260 {
261 funcName = "None";
262 }
263 else
264 {
265 if (literal is CompositeSymbolRef csf)
266 {
267 objUIndex = GetClassDefaultObject(CompilerUtils.ResolveClass((Class)((ClassType)((ObjectLiteral)csf.OuterSymbol).Class).ClassLimiter, Pcc)).UIndex;
268 literal = csf.InnerSymbol;
269 }
270
271 funcName = NameReference.FromInstancedString(((SymbolReference) literal).Name);
272 }
273 prop = new DelegateProperty(objUIndex, funcName, propName);
274 break;
275 case DynamicArrayType dynamicArrayType:
276 VariableType elementType = dynamicArrayType.ElementType;
277 var properties = ((DynamicArrayLiteral)literal).Values.Select(lit => MakeProperty(null, elementType, lit, subObjectDict));
278 switch (elementType)
279 {
280 case ClassType:
281 case Class:
282 prop = new ArrayProperty<ObjectProperty>(properties.Cast<ObjectProperty>().ToList(), propName);
283 break;
284 case DelegateType:
285 prop = new ArrayProperty<DelegateProperty>(properties.Cast<DelegateProperty>().ToList(), propName);
286 break;
287 case Enumeration:
288 prop = new ArrayProperty<EnumProperty>(properties.Cast<EnumProperty>().ToList(), propName);
289 break;
290 case Struct:
291 prop = new ArrayProperty<StructProperty>(properties.Cast<StructProperty>().ToList(), propName);
292 break;
293 default:
294 switch (elementType.PropertyType)
295 {
296 case EPropertyType.Byte:
297 prop = new ArrayProperty<ByteProperty>(properties.Cast<ByteProperty>().ToList(), propName);
298 break;
299 case EPropertyType.Int:
300 prop = new ArrayProperty<IntProperty>(properties.Cast<IntProperty>().ToList(), propName);
301 break;
302 case EPropertyType.Bool:
303 prop = new ArrayProperty<BoolProperty>(properties.Cast<BoolProperty>().ToList(), propName);
304 break;
305 case EPropertyType.Float:
306 prop = new ArrayProperty<FloatProperty>(properties.Cast<FloatProperty>().ToList(), propName);
307 break;
308 case EPropertyType.Name:
309 prop = new ArrayProperty<NameProperty>(properties.Cast<NameProperty>().ToList(), propName);
310 break;
311 case EPropertyType.String:
312 prop = new ArrayProperty<StrProperty>(properties.Cast<StrProperty>().ToList(), propName);
313 break;
314 case EPropertyType.StringRef:
315 prop = new ArrayProperty<StringRefProperty>(properties.Cast<StringRefProperty>().ToList(), propName);
316 break;
317 default:
318 throw new ArgumentOutOfRangeException(nameof(elementType.PropertyType), elementType.PropertyType, $"{elementType.PropertyType} is not a valid array element type!");
319 }
320 break;
321 }
322 break;
323 case Enumeration enumeration:
324 NameReference value;
325 if (literal is NoneLiteral)
326 {
327 value = "None";
328 }
329 else
330 {
331 value = NameReference.FromInstancedString(((EnumValue) ((SymbolReference) literal).Node).Name);
332 }
333 prop = new EnumProperty(value, NameReference.FromInstancedString(enumeration.Name), Pcc.Game, propName);
334 break;
335 case Struct @struct:
336 //todo: Spec says that unspecified properties on a struct value should be inherited from base class's default for that property
337 var structProps = (IsStructDefaults || @struct.IsAtomic) ? @struct.GetDefaultPropertyCollection(Pcc, ShouldStripTransients, packageCache) : new PropertyCollection();
338 foreach (Statement statement in ((StructLiteral)literal).Statements)
339 {
340 structProps.AddOrReplaceProp(ConvertToProperty((AssignStatement)statement, subObjectDict));
341 }
342 prop = new StructProperty(@struct.Name, structProps, propName, @struct.IsImmutable);
343 break;
344 default:
345 switch (type.PropertyType)
346 {
347 case EPropertyType.Byte:
348 prop = new ByteProperty((byte) ((IntegerLiteral) literal).Value, propName);
349 break;
350 case EPropertyType.Int:
351 prop = new IntProperty(((IntegerLiteral) literal).Value, propName);
352 break;
353 case EPropertyType.Bool:
354 prop = new BoolProperty(((BooleanLiteral) literal).Value, propName);
355 break;
356 case EPropertyType.Float:
357 prop = new FloatProperty(((FloatLiteral) literal).Value, propName);
358 break;
359 case EPropertyType.Name:
360 prop = new NameProperty(NameReference.FromInstancedString(((NameLiteral) literal).Value), propName);
361 break;
362 case EPropertyType.String:
363 prop = new StrProperty(((StringLiteral) literal).Value, propName);
364 break;
365 case EPropertyType.StringRef:
366 prop = new StringRefProperty(((StringRefLiteral) literal).Value, propName);
367 break;
368 default:
369 throw new ArgumentOutOfRangeException();
370 }
371
372 break;
373 }
374
375 return prop;
376 }

378 private void WriteSubObjectData(Subobject subObject, ExportEntry subExport, int netIndex, Dictionary<NameReference, ExportEntry> parentSubObjectDict)
379 {
380 PropertyCollection props = ConvertStatementsToPropertyCollection(subObject.Statements, subExport, new(parentSubObjectDict));
381 var binary = ObjectBinary.Create(subExport.ClassName, subExport.Game, props);
382
383 //this code should probably be somewhere else, perhaps integrated into ObjectBinary.Create somehow?
384 if (binary is BioDynamicAnimSet dynAnimSet && props.GetProp<ArrayProperty<ObjectProperty>>("Sequences") is {} sequences)
385 {
386 var setName = props.GetProp<NameProperty>("m_nmOrigSetName");
387 foreach (ObjectProperty objProp in sequences)
388 {
389 switch (objProp.ResolveToEntry(Pcc))
390 {
391 case ExportEntry exportEntry when exportEntry.GetProperty<NameProperty>("SequenceName") is {} seqNameProperty:
392 dynAnimSet.SequenceNamesToUnkMap.Add(seqNameProperty.Value, 1);
393 break;
394 case IEntry entry:
395 if (setName is null)
396 {
397 throw new Exception($"{nameof(BioDynamicAnimSet)} must have m_nmOrigSetName property defined!");
398 }
399 dynAnimSet.SequenceNamesToUnkMap.Add(entry.ObjectName.Instanced[(setName.Value.Instanced.Length + 1)..], 1);
400 break;
401 }
402 }
403 }
404
405
406 if (subExport.ClassName is "DominantDirectionalLightComponent" or "DominantSpotLightComponent")
407 {
408 Span<byte> preProps = stackalloc byte[20];
409 const int templateOwnerClass = 0; //todo: When is this not 0?
410
411 EndianBitConverter.WriteAsBytes(0, preProps, Pcc.Endian);
412 EndianBitConverter.WriteAsBytes(templateOwnerClass, preProps[4..], Pcc.Endian);
413 EndianBitConverter.WriteAsBytes(Pcc.FindNameOrAdd(subExport.ObjectName.Name), preProps[8..], Pcc.Endian);
414 EndianBitConverter.WriteAsBytes(subExport.ObjectName.Number, preProps[12..], Pcc.Endian);
415 EndianBitConverter.WriteAsBytes(netIndex, preProps[16..], Pcc.Endian);
416 subExport.WritePrePropsAndPropertiesAndBinary(preProps.ToArray(), props, binary);
417 }
418 else if (subObject.Class.IsComponent)
419 {
420 Span<byte> preProps = stackalloc byte[16];
421 const int templateOwnerClass = 0; //todo: When is this not 0?
422
423 EndianBitConverter.WriteAsBytes(templateOwnerClass, preProps, Pcc.Endian);
424 EndianBitConverter.WriteAsBytes(Pcc.FindNameOrAdd(subExport.ObjectName.Name), preProps[4..], Pcc.Endian);
425 EndianBitConverter.WriteAsBytes(subExport.ObjectName.Number, preProps[8..], Pcc.Endian);
426 EndianBitConverter.WriteAsBytes(netIndex, preProps[12..], Pcc.Endian);
427 subExport.WritePrePropsAndPropertiesAndBinary(preProps.ToArray(), props, binary);
428 }
429 else
430 {
431 subExport.WritePropertiesAndBinary(props, binary);
432 }
433 }
