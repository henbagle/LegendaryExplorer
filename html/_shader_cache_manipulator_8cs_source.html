17 {
18 var staticParamSetsInFile = new HashSet<StaticParameterSet>();
19 //figure out which MaterialShaderMaps to keep
20 foreach (ExportEntry export in pcc.Exports)
21 {
22 if (export.ClassName == "Material")
23 {
24 staticParamSetsInFile.Add((StaticParameterSet)ObjectBinary.From<Material>(export).SM3MaterialResource.ID);

30 }

34

80 public static ShaderCache GetLocalShaders(HashSet<StaticParameterSet> staticParamSets, ExportEntry seekFreeShaderCacheExport)
81 {
82 var localCache = seekFreeShaderCacheExport.GetBinaryData<ShaderCache>();
83
84 var tempCache = new ShaderCache
85 {
86 Shaders = new OrderedMultiValueDictionary<Guid, Shader>(),
87 MaterialShaderMaps = new OrderedMultiValueDictionary<StaticParameterSet, MaterialShaderMap>(),
88 ShaderTypeCRCMap = localCache.ShaderTypeCRCMap,
89 VertexFactoryTypeCRCMap = localCache.VertexFactoryTypeCRCMap
90 };
91
92 //get corresponding MaterialShaderMap for each StaticParameterSet
93 foreach ((StaticParameterSet key, MaterialShaderMap msm) in localCache.MaterialShaderMaps)
94 {
95 if (staticParamSets.Contains(key) && !tempCache.MaterialShaderMaps.ContainsKey(key))
96 {
97 tempCache.MaterialShaderMaps.Add(key, msm);
98 }
99 }
100
101 //get the guids for every shader referenced by the MaterialShaderMaps
102 var shaderGuids = new HashSet<Guid>();
103 foreach (MaterialShaderMap materialShaderMap in tempCache.MaterialShaderMaps.Values())
104 {
105 foreach (ShaderReference shaderRef in materialShaderMap.Shaders.Values())
106 {
107 shaderGuids.Add(shaderRef.Id);
108 }
109
110 foreach (MeshShaderMap meshShaderMap in materialShaderMap.MeshShaderMaps)
111 {
112 foreach (ShaderReference shaderRef in meshShaderMap.Shaders.Values())
113 {
114 shaderGuids.Add(shaderRef.Id);
115 }
116 }
117 }
118 foreach ((Guid key, Shader shader) in localCache.Shaders)
119 {
120 if (shaderGuids.Contains(key) && !tempCache.Shaders.ContainsKey(key))
121 {
122 tempCache.Shaders.Add(key, shader);
123 }
124 }
125
126 return tempCache;
127 }

129 public static void AddShadersToFile(IMEPackage destFile, ShaderCache shadersToAdd)
130 {
131 const string seekfreeshadercache = "SeekFreeShaderCache";
132 var destCacheExport = destFile.FindExport(seekfreeshadercache);
133
134 if (destCacheExport is null)
135 {
136 destFile.AddExport(new ExportEntry(destFile, 0, seekfreeshadercache, BitConverter.GetBytes(-1), binary: shadersToAdd)
137 {
138 Class = destFile.getEntryOrAddImport("Engine.ShaderCache"),
139 ObjectFlags = EObjectFlags.LoadForClient | EObjectFlags.LoadForEdit | EObjectFlags.LoadForServer | EObjectFlags.Standalone
140 });
141 return;
142 }
143
144 var destCache = destCacheExport.GetBinaryData<ShaderCache>();
145
146 foreach ((StaticParameterSet key, MaterialShaderMap materialShaderMap) in shadersToAdd.MaterialShaderMaps)
147 {
148 if (!destCache.MaterialShaderMaps.ContainsKey(key))
149 {
150 destCache.MaterialShaderMaps.Add(key, materialShaderMap);
151 }
152 }
153
154 foreach ((Guid key, Shader shader) in shadersToAdd.Shaders)
155 {
156 if (!destCache.Shaders.ContainsKey(key))
157 {
158 destCache.Shaders.Add(key, shader);
159 }
160 }
161
162 destCacheExport.WriteBinary(destCache);
163 }
164 }
