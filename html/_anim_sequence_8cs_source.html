113 public void DecompressAnimationData()
114 {
115 var ms = new MemoryStream(CompressedAnimationData);
116 RawAnimationData = new List<AnimTrack>();
117
118 for (int i = 0; i < Bones.Count; i++)
119 {
120 int posOff = TrackOffsets[i * 4];
121 int numPosKeys = TrackOffsets[i * 4 + 1];
122 int rotOff = TrackOffsets[i * 4 + 2];
123 int numRotKeys = TrackOffsets[i * 4 + 3];
124
125 var track = new AnimTrack
126 {
127 Positions = new List<Vector3>(numPosKeys),
128 Rotations = new List<Quaternion>(numRotKeys)
129 };
130
131 if (numPosKeys > 0)
132 {
133 ms.JumpTo(posOff);
134
135 AnimationCompressionFormat compressionFormat = posCompression;
136
137 if (numPosKeys == 1)
138 {
139 compressionFormat = AnimationCompressionFormat.ACF_None;
140 }
141
142 for (int j = 0; j < numPosKeys; j++)
143 {
144 switch (compressionFormat)
145 {
146 case AnimationCompressionFormat.ACF_None:
147 case AnimationCompressionFormat.ACF_Float96NoW:
148 track.Positions.Add(new Vector3(ms.ReadFloat(), ms.ReadFloat(), ms.ReadFloat()));
149 break;
150 case AnimationCompressionFormat.ACF_IntervalFixed32NoW:
151 case AnimationCompressionFormat.ACF_Fixed48NoW:
152 case AnimationCompressionFormat.ACF_Fixed32NoW:
153 case AnimationCompressionFormat.ACF_Float32NoW:
154 case AnimationCompressionFormat.ACF_BioFixed48:
155 throw new NotImplementedException($"Translation keys in format {compressionFormat} cannot be read yet!");
156 }
157 }
158
159 if (keyEncoding == AnimationKeyFormat.AKF_VariableKeyLerp && numPosKeys > 1)
160 {
161 ms.JumpTo(ms.Position.Align(4));
162
163 var keyTimes = new List<int>(numPosKeys);
164 for (int j = 0; j < numPosKeys; j++)
165 {
166 keyTimes.Add(NumFrames > 0xFF ? ms.ReadUInt16() : ms.ReadByte());
167 }
168 //RawAnimationData should have either 1 key, or the same number of keys as frames.
169 //Lerp any missing keys
170 List<Vector3> tempPositions = track.Positions;
171 track.Positions = new List<Vector3>(NumFrames)
172 {
173 tempPositions[0]
174 };
175 for (int frameIdx = 1, keyIdx = 1; frameIdx < NumFrames; keyIdx++, frameIdx++)
176 {
177 if (keyIdx >= keyTimes.Count)
178 {
179 track.Positions.Add(track.Positions[frameIdx - 1]);
180 }
181 else if (keyTimes[keyIdx] == frameIdx)
182 {
183 track.Positions.Add(tempPositions[keyIdx]);
184 }
185 else
186 {
187 int nextFrame = keyTimes[keyIdx];
188 int prevFrame = frameIdx - 1;
189 for (int j = frameIdx; j < nextFrame; j++)
190 {
191 float amount = (float)(j - prevFrame) / (nextFrame - prevFrame);
192 track.Positions.Add(Vector3.Lerp(track.Positions[prevFrame], tempPositions[keyIdx], amount));
193 }
194 track.Positions.Add(tempPositions[keyIdx]);
195 frameIdx = nextFrame;
196 }
197 }
198 }
199 }
200
201 if (numRotKeys > 0)
202 {
203 ms.JumpTo(rotOff);
204
205 AnimationCompressionFormat compressionFormat = rotCompression;
206
207 if (numRotKeys == 1)
208 {
209 compressionFormat = AnimationCompressionFormat.ACF_Float96NoW;
210 }
211 else if (compressedDataSource != MEGame.UDK)
212 {
213 ms.Skip(12 * 2); //skip mins and ranges
214 }
215
216 for (int j = 0; j < numRotKeys; j++)
217 {
218 switch (compressionFormat)
219 {
220 case AnimationCompressionFormat.ACF_None:
221 track.Rotations.Add(new Quaternion(ms.ReadFloat(), ms.ReadFloat(), ms.ReadFloat(), ms.ReadFloat()));
222 break;
223 case AnimationCompressionFormat.ACF_Float96NoW:
224 {
225 float x = ms.ReadFloat();
226 float y = ms.ReadFloat();
227 float z = ms.ReadFloat();
228 track.Rotations.Add(new Quaternion(x, y, z, getW(x, y, z)));
229 break;
230 }
231 case AnimationCompressionFormat.ACF_BioFixed48:
232 {
233 const float shift = 0.70710678118f;
234 const float scale = 1.41421356237f;
235 const float precisionMult = 32767.0f;
236 ushort a = ms.ReadUInt16();
237 ushort b = ms.ReadUInt16();
238 ushort c = ms.ReadUInt16();
239 float x = (a & 0x7FFF) / precisionMult * scale - shift;
240 float y = (b & 0x7FFF) / precisionMult * scale - shift;
241 float z = (c & 0x7FFF) / precisionMult * scale - shift;
242 float w = getW(x, y, z);
243 int wPos = ((a >> 14) & 2) | ((b >> 15) & 1);
244 track.Rotations.Add(wPos switch
245 {
246 0 => new Quaternion(w, x, y, z),
247 1 => new Quaternion(x, w, y, z),
248 2 => new Quaternion(x, y, w, z),
249 _ => new Quaternion(x, y, z, w)
250 });
251
252 break;
253 }
254 case AnimationCompressionFormat.ACF_Fixed48NoW:
255 {
256
257 const float scale = 32767.0f;
258 const ushort shift = 32767;
259 float x = (ms.ReadUInt16() - shift) / scale;
260 float y = (ms.ReadUInt16() - shift) / scale;
261 float z = (ms.ReadUInt16() - shift) / scale;
262 track.Rotations.Add(new Quaternion(x, y, z, getW(x, y, z)));
263 break;
264 }
265 case AnimationCompressionFormat.ACF_IntervalFixed32NoW:
266 case AnimationCompressionFormat.ACF_Fixed32NoW:
267 case AnimationCompressionFormat.ACF_Float32NoW:
268 throw new NotImplementedException($"Rotation keys in format {compressionFormat} cannot be read yet!");
269 }
270 }
271
272 if (keyEncoding == AnimationKeyFormat.AKF_VariableKeyLerp && numRotKeys > 1)
273 {
274 ms.JumpTo(ms.Position.Align(4));
275
276 var keyTimes = new List<int>(numRotKeys);
277 for (int j = 0; j < numRotKeys; j++)
278 {
279 keyTimes.Add(NumFrames > 0xFF ? ms.ReadUInt16() : ms.ReadByte());
280 }
281 //RawAnimationData should have either 1 key, or the same number of keys as frames.
282 //Slerp any missing keys
283 List<Quaternion> tempRotations = track.Rotations;
284 track.Rotations = new List<Quaternion>(NumFrames)
285 {
286 tempRotations[0]
287 };
288 for (int frameIdx = 1, keyIdx = 1; frameIdx < NumFrames; keyIdx++, frameIdx++)
289 {
290 if (keyIdx >= keyTimes.Count)
291 {
292 track.Rotations.Add(track.Rotations[frameIdx - 1]);
293 }
294 else if (keyTimes[keyIdx] == frameIdx)
295 {
296 track.Rotations.Add(tempRotations[keyIdx]);
297 }
298 else
299 {
300 int nextFrame = keyTimes[keyIdx];
301 int prevFrame = frameIdx - 1;
302 for (int j = frameIdx; j < nextFrame; j++)
303 {
304 float amount = (float)(j - prevFrame) / (nextFrame - prevFrame);
305 track.Rotations.Add(Quaternion.Slerp(track.Rotations[prevFrame], tempRotations[keyIdx], amount));
306 }
307 track.Rotations.Add(tempRotations[keyIdx]);
308 frameIdx = nextFrame;
309 }
310 }
311 }
312 }
313
314 RawAnimationData.Add(track);
315 }
316 static float getW(float x, float y, float z)
317 {
318 float wSquared = 1.0f - (x * x + y * y + z * z);
319 return (float)(wSquared > 0f ? Math.Sqrt(wSquared) : 0f);
320 }

323 void CompressAnimationData(MEGame game, AnimationCompressionFormat newRotationCompression)
324 {
325 if (RawAnimationData is null)
326 {
327 DecompressAnimationData();
328 }
329
330 keyEncoding = AnimationKeyFormat.AKF_ConstantKeyLerp;
331 posCompression = AnimationCompressionFormat.ACF_None;
332 rotCompression = newRotationCompression;
333 TrackOffsets = new int[Bones.Count * 4];
334 using var ms = MemoryManager.GetMemoryStream();
335
336 for (int i = 0; i < Bones.Count; i++)
337 {
338 AnimTrack track = RawAnimationData[i];
339
340 TrackOffsets[i * 4] = (int)ms.Position;
341 int numPosKeys = track.Positions.Count;
342 TrackOffsets[i * 4 + 1] = numPosKeys;
343
344 if (numPosKeys > 0)
345 {
346 AnimationCompressionFormat compressionFormat = posCompression;
347
348 if (numPosKeys == 1)
349 {
350 compressionFormat = AnimationCompressionFormat.ACF_None;
351 }
352
353 for (int j = 0; j < numPosKeys; j++)
354 {
355 switch (compressionFormat)
356 {
357 case AnimationCompressionFormat.ACF_None:
358 case AnimationCompressionFormat.ACF_Float96NoW:
359 ms.WriteFloat(track.Positions[j].X);
360 ms.WriteFloat(track.Positions[j].Y);
361 ms.WriteFloat(track.Positions[j].Z);
362 break;
363 default:
364 throw new NotImplementedException($"Translation keys in format {compressionFormat} cannot be written yet!");
365 }
366 }
367 PadTo4();
368 }
369
370 TrackOffsets[i * 4 + 2] = (int)ms.Position;
371 int numRotKeys = track.Rotations.Count;
372 TrackOffsets[i * 4 + 3] = numRotKeys;
373
374 if (numRotKeys > 0)
375 {
376 AnimationCompressionFormat compressionFormat = rotCompression;
377
378 if (numRotKeys == 1)
379 {
380 compressionFormat = AnimationCompressionFormat.ACF_Float96NoW;
381 }
382 else if (game != MEGame.UDK)
383 {
384 float xMin, yMin, zMin, xMax, yMax, zMax;
385 xMin = yMin = zMin = float.MaxValue;
386 xMax = yMax = zMax = float.MinValue;
387
388 foreach (Quaternion quat in track.Rotations)
389 {
390 xMin = Math.Min(xMin, quat.X);
391 yMin = Math.Min(yMin, quat.Y);
392 zMin = Math.Min(zMin, quat.Z);
393 xMax = Math.Max(xMax, quat.X);
394 yMax = Math.Max(yMax, quat.Y);
395 zMax = Math.Max(zMax, quat.Z);
396 }
397
398 ms.WriteFloat(xMin);
399 ms.WriteFloat(yMin);
400 ms.WriteFloat(zMin);
401 ms.WriteFloat(xMax - xMin);
402 ms.WriteFloat(yMax - yMin);
403 ms.WriteFloat(zMax - zMin);
404 }
405
406 for (int j = 0; j < numRotKeys; j++)
407 {
408 Quaternion rot = track.Rotations[j];
409 switch (compressionFormat)
410 {
411 case AnimationCompressionFormat.ACF_None:
412 ms.WriteFloat(rot.X);
413 ms.WriteFloat(rot.Y);
414 ms.WriteFloat(rot.Z);
415 ms.WriteFloat(rot.W);
416 break;
417 case AnimationCompressionFormat.ACF_Float96NoW:
418 ms.WriteFloat(rot.X);
419 ms.WriteFloat(rot.Y);
420 ms.WriteFloat(rot.Z);
421 break;
422 case AnimationCompressionFormat.ACF_BioFixed48:
423 {
424 const float shift = 0.70710678118f;
425 const float scale = 1.41421356237f;
426 const float precisionMult = 32767.0f;
427 //smallest three compression: https://gafferongames.com/post/snapshot_compression/
428 //omit the largest component, and store its index
429 int wPos = 0;
430 float max = 0f;
431 float[] rotArr = {rot.X, rot.Y, rot.Z, rot.W};
432 for (int k = 0; k < 4; k++)
433 {
434 if (Math.Abs(rotArr[k]) > max)
435 {
436 max = Math.Abs(rotArr[k]);
437 wPos = k;
438 }
439 }
440 //remap the smallest three components to the lower 15 bits of a ushort
441 ushort a, b, c;
442
443 static ushort compress(float f) => (ushort)((f + shift) / scale * precisionMult).Clamp(0, 0x7FFF);
444
445 switch (wPos)
446 {
447 case 0:
448 a = compress(rot.Y);
449 b = compress(rot.Z);
450 c = compress(rot.W);
451 break;
452 case 1:
453 a = compress(rot.X);
454 b = compress(rot.Z);
455 c = compress(rot.W);
456 break;
457 case 2:
458 a = compress(rot.X);
459 b = compress(rot.Y);
460 c = compress(rot.W);
461 break;
462 default:
463 a = compress(rot.X);
464 b = compress(rot.Y);
465 c = compress(rot.Z);
466 break;
467 }
468
469 //stuff the 2 bit index of the omitted component into the high bits of a and b
470 a |= (ushort)((wPos & 2) << 14);
471 b |= (ushort)((wPos & 1) << 15);
472
473 ms.WriteUInt16(a);
474 ms.WriteUInt16(b);
475 ms.WriteUInt16(c);
476 }
477 break;
478 case AnimationCompressionFormat.ACF_Fixed48NoW:
479 {
480
481 const float scale = 32767.0f;
482 const ushort shift = 32767;
483 ms.WriteUInt16((ushort)(rot.X * scale + shift).Clamp(0, ushort.MaxValue));
484 ms.WriteUInt16((ushort)(rot.Y * scale + shift).Clamp(0, ushort.MaxValue));
485 ms.WriteUInt16((ushort)(rot.Z * scale + shift).Clamp(0, ushort.MaxValue));
486 break;
487 }
488 case AnimationCompressionFormat.ACF_IntervalFixed32NoW:
489 case AnimationCompressionFormat.ACF_Fixed32NoW:
490 case AnimationCompressionFormat.ACF_Float32NoW:
491 default:
492 throw new NotImplementedException($"Rotation keys in format {compressionFormat} cannot be written yet!");
493 }
494 }
495 PadTo4();
496 }
