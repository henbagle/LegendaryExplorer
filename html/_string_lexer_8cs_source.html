80 private ScriptToken GetNextTokenImproved()
81 {
82 if (Data.AtEnd())
83 {
84 return new ScriptToken(TokenType.EOF, null, StreamPosition, StreamPosition);
85 }
86
87 ScriptToken result = null;
88
89 char peek = Data.CurrentItem;
90 if (peek == '/' && Data.LookAhead(1) == '/')
91 {
92 result = SingleLineCommentMatcher.MatchComment(Data, ref StreamPosition);
93 }
94 else if (peek == '"')
95 {
96 result = StringLiteralMatcher.MatchString(Data, ref StreamPosition, Log);
97 }
98 else if (peek == '\'')
99 {
100 result = NameLiteralMatcher.MatchName(Data, ref StreamPosition, Log);
101 }
102 else if (peek == '$' && (Data.LookAhead(1).IsDigit() ||
103 Data.LookAhead(1) == '-' && Data.LookAhead(2).IsDigit()))
104 {
105 result = StringRefLiteralMatcher.MatchStringRef(Data, ref StreamPosition);
106 }
107 else if (GlobalLists.IsDelimiterChar(peek))
108 {
109 //looping over every single one is a terrible way of doing this.
110 foreach (SymbolMatcher matcher in GlobalLists.DelimitersAndOperators)
111 {
112 Data.PushSnapshot();
113
114 result = matcher.Match(Data, ref StreamPosition, Log);
115 if (result == null)
116 {
117 Data.PopSnapshot();
118 }
119 else
120 {
121 Data.DiscardSnapshot();
122 break;
123 }
124 }
125 }
126 else if (char.IsWhiteSpace(peek))
127 {
128 result = WhiteSpaceMatcher.MatchWhiteSpace(Data, ref StreamPosition);
129 }
130 else if (peek.IsDigit())
131 {
132 result = NumberMatcher.MatchNumber(Data, ref StreamPosition);
133 }
134 else
135 {
136 result = WordMatcher.MatchWord(Data, ref StreamPosition);
137 }
138
139
140
141 if (result == null)
142 {
143 Log.LogError($"Could not lex '{Data.CurrentItem}'",
144 StreamPosition, StreamPosition.GetModifiedPosition(0, 1, 1));
145 Data.Advance();
146 return new ScriptToken(TokenType.INVALID, Data.CurrentItem.ToString(), StreamPosition, StreamPosition.GetModifiedPosition(0, 1, 1)) { SyntaxType = EF.ERROR };
147 }
148 return result;
149 }

151 private List<ScriptToken> LexData()
152 {
153 var tokens = new List<ScriptToken>();
154 StreamPosition = new SourcePosition(1, 0, 0);
155 ScriptToken token = GetNextTokenImproved();
156 while (token.Type != TokenType.EOF)
157 {
158 if (token.Type is not TokenType.WhiteSpace and not TokenType.SingleLineComment and not TokenType.MultiLineComment)
159 {
160 tokens.Add(token);
161 }
162
163 token = GetNextTokenImproved();
164 }
165 return tokens;
166 }
