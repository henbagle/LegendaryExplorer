27 {

30 LECLog.Information(@"Loading property db for ME2");
31 try
32 {
33 var infoText = jsonTextOverride ?? ObjectInfoLoader.LoadEmbeddedJSONText(MEGame.ME2);
34 if (infoText != null)
35 {
36 var blob = JsonConvert.DeserializeAnonymousType(infoText, new { SequenceObjects, Classes, Structs, Enums });
37 SequenceObjects = blob.SequenceObjects;
38 Classes = blob.Classes;
39 Structs = blob.Structs;
40 Enums = blob.Enums;
41
42 AddCustomAndNativeClasses(Classes, SequenceObjects);
43
44 foreach ((string className, ClassInfo classInfo) in Classes)
45 {
46 classInfo.ClassName = className;
47 }
48 foreach ((string className, ClassInfo classInfo) in Structs)
49 {
50 classInfo.ClassName = className;
51 }
52 IsLoaded = true;
53 }
54 }
55 catch (Exception ex)
56 {
57 LECLog.Information($@"Property database load failed for ME2: {ex.Message}");
58 return;
59 }

95 {

97 if (temp.ContainsKey(className))

103 if (Enums.ContainsKey(propInfo.Reference))

108 //look in structs
109 else
110 {

116 if (vals != null)

120 }
121 }
122 }
123 //look in base class
124 if (temp.ContainsKey(info.baseClass))
125 {
126 List<NameReference> vals = getEnumfromProp(info.baseClass, propName, inStruct);
127 if (vals != null)
128 {
129 return vals;
130 }
131 }
132 }
133 return null;

229 public static PropertyInfo getPropertyInfo(string className, NameReference propName, bool inStruct = false, ClassInfo nonVanillaClassInfo = null, bool reSearch = true, ExportEntry containingExport = null)
230 {
231 if (className.StartsWith("Default__", StringComparison.OrdinalIgnoreCase))
232 {
233 className = className.Substring(9);
234 }
235 Dictionary<string, ClassInfo> temp = inStruct ? Structs : Classes;
236 bool infoExists = temp.TryGetValue(className, out ClassInfo info);
237 if (!infoExists && nonVanillaClassInfo != null)
238 {
239 info = nonVanillaClassInfo;
240 infoExists = true;
241 }
242 if (infoExists) //|| (temp = !inStruct ? Structs : Classes).ContainsKey(className))
243 {
244 //look in class properties
245 if (info.properties.TryGetValue(propName, out var propInfo))
246 {
247 return propInfo;
248 }
249 //look in structs
250
251 foreach (PropertyInfo p in info.properties.Values())
252 {
253 if ((p.Type is PropertyType.StructProperty or PropertyType.ArrayProperty) && reSearch)
254 {
255 PropertyInfo val = getPropertyInfo(p.Reference, propName, true, nonVanillaClassInfo, reSearch: false);
256 if (val != null)

258 return val;

260 }
261 }
262 //look in base class

272 {
273 //Baseclass may be modified as well...
274 if (containingExport?.SuperClass is ExportEntry parentExport)
275 {
276 //Class parent is in this file. Generate class parent info and attempt refetch
277 return getPropertyInfo(parentExport.SuperClassName, propName, inStruct, generateClassInfo(parentExport), reSearch: true, parentExport);

279 }
280 }

282 //if (reSearch)

301 public static Property getDefaultProperty(NameReference propName, PropertyInfo propInfo, PackageCache packageCache, bool stripTransients = true, bool isImmutable = false)
302 {
303 switch (propInfo.Type)
304 {
305 case PropertyType.IntProperty:
306 return new IntProperty(0, propName);
307 case PropertyType.FloatProperty:
308 return new FloatProperty(0f, propName);
309 case PropertyType.DelegateProperty:
310 return new DelegateProperty(0, "None");
311 case PropertyType.ObjectProperty:
312 return new ObjectProperty(0, propName);
313 case PropertyType.NameProperty:
314 return new NameProperty("None", propName);
315 case PropertyType.BoolProperty:
316 return new BoolProperty(false, propName);
317 case PropertyType.ByteProperty when propInfo.IsEnumProp():
318 return new EnumProperty(propInfo.Reference, MEGame.ME2, propName);
319 case PropertyType.ByteProperty:
320 return new ByteProperty(0, propName);
321 case PropertyType.StrProperty:
322 return new StrProperty("", propName);
323 case PropertyType.StringRefProperty:
324 return new StringRefProperty(propName);
325 case PropertyType.BioMask4Property:
326 return new BioMask4Property(0, propName);
327 case PropertyType.ArrayProperty:
328 switch (getArrayType(propInfo))
329 {
330 case ArrayType.Object:
331 return new ArrayProperty<ObjectProperty>(propName);
332 case ArrayType.Name:
333 return new ArrayProperty<NameProperty>(propName);
334 case ArrayType.Enum:
335 return new ArrayProperty<EnumProperty>(propName);
336 case ArrayType.Struct:
337 return new ArrayProperty<StructProperty>(propName);
338 case ArrayType.Bool:
339 return new ArrayProperty<BoolProperty>(propName);
340 case ArrayType.String:
341 return new ArrayProperty<StrProperty>(propName);
342 case ArrayType.Float:
343 return new ArrayProperty<FloatProperty>(propName);
344 case ArrayType.Int:
345 return new ArrayProperty<IntProperty>(propName);
346 case ArrayType.Byte:
347 return new ImmutableByteArrayProperty(propName);
348 default:
349 return null;
350 }
351 case PropertyType.StructProperty:
352 isImmutable = isImmutable || GlobalUnrealObjectInfo.IsImmutable(propInfo.Reference, MEGame.ME2);
353 return new StructProperty(propInfo.Reference, GlobalUnrealObjectInfo.getDefaultStructValue(MEGame.ME2, propInfo.Reference, stripTransients, packageCache), propName, isImmutable);
354 case PropertyType.None:
355 case PropertyType.Unknown:
356 default:
357 return null;
358 }

548 public static ClassInfo generateClassInfo(ExportEntry export, bool isStruct = false)
549 {
550 IMEPackage pcc = export.FileRef;
551 ClassInfo info = new()
552 {
553 baseClass = export.SuperClassName,
554 exportIndex = export.UIndex,
555 ClassName = export.ObjectName
556 };
557 if (!isStruct)
558 {
559 var classBinary = ObjectBinary.From<UClass>(export);
560 info.isAbstract = classBinary.ClassFlags.Has(UnrealFlags.EClassFlags.Abstract);
561 }
562
563 if (pcc.FilePath != null)
564 {
565 if (pcc.FilePath.Contains("BioGame"))
566 {
567 info.pccPath = new string(pcc.FilePath.Skip(pcc.FilePath.LastIndexOf("BioGame") + 8).ToArray());
568 }
569 else
570 {
571 info.pccPath = pcc.FilePath; //used for dynamic resolution of files outside the game directory.
572 }
573 }
574
575 int nextExport = EndianReader.ToInt32(export.DataReadOnly, isStruct ? 0x18 : 0x10, export.FileRef.Endian);
576 while (nextExport > 0)
577 {
578 var entry = pcc.GetUExport(nextExport);
579 if (entry.ClassName != "ScriptStruct" && entry.ClassName != "Enum"
580 && entry.ClassName != "Function" && entry.ClassName != "Const" && entry.ClassName != "State")
581 {
582 if (!info.properties.ContainsKey(entry.ObjectName))
583 {
584 PropertyInfo p = getProperty(entry);
585 if (p != null)
586 {
587 info.properties.Add(entry.ObjectName, p);
588 }
589 }
590 }
591 nextExport = EndianReader.ToInt32(entry.DataReadOnly, 0x10, export.FileRef.Endian);
592 }
593 return info;
594 }

596 private static void generateEnumValues(ExportEntry export, Dictionary<string, List<NameReference>> NewEnums = null)
597 {
598 var enumTable = NewEnums ?? Enums;
599 string enumName = export.ObjectName.Instanced;
600 if (!enumTable.ContainsKey(enumName))
601 {
602 var values = new List<NameReference>();
603 var buff = export.DataReadOnly;
604 int count = EndianReader.ToInt32(buff, 20, export.FileRef.Endian) - 1;
605 for (int i = 0; i < count; i++)
606 {
607 int enumValIndex = 24 + i * 8;
608 values.Add(new NameReference(export.FileRef.Names[EndianReader.ToInt32(buff, enumValIndex, export.FileRef.Endian)], EndianReader.ToInt32(buff, enumValIndex + 4, export.FileRef.Endian)));
609 }
610
611 enumTable.Add(enumName, values);
612 }

615 private static PropertyInfo getProperty(ExportEntry entry)
616 {
617 IMEPackage pcc = entry.FileRef;
618
619 string reference = null;
620 PropertyType type;
621 switch (entry.ClassName)
622 {
623 case "IntProperty":
624 type = PropertyType.IntProperty;
625 break;
626 case "StringRefProperty":
627 type = PropertyType.StringRefProperty;
628 break;
629 case "FloatProperty":
630 type = PropertyType.FloatProperty;
631 break;
632 case "BoolProperty":
633 type = PropertyType.BoolProperty;
634 break;
635 case "StrProperty":
636 type = PropertyType.StrProperty;
637 break;
638 case "NameProperty":
639 type = PropertyType.NameProperty;
640 break;
641 case "DelegateProperty":
642 type = PropertyType.DelegateProperty;
643 break;
644 case "ClassProperty":
645 case "ObjectProperty":
646 case "ComponentProperty":
647 type = PropertyType.ObjectProperty;
648 reference = pcc.getObjectName(EndianReader.ToInt32(entry.DataReadOnly, entry.DataSize - 4, entry.FileRef.Endian));
649 break;
650 case "StructProperty":
651 type = PropertyType.StructProperty;
652 reference = pcc.getObjectName(EndianReader.ToInt32(entry.DataReadOnly, entry.DataSize - 4, entry.FileRef.Endian));
653 break;
654 case "BioMask4Property":
655 case "ByteProperty":
656 type = PropertyType.ByteProperty;
657 reference = pcc.getObjectName(EndianReader.ToInt32(entry.DataReadOnly, entry.DataSize - 4, entry.FileRef.Endian));
658 break;
659 case "ArrayProperty":
660 type = PropertyType.ArrayProperty;
661 PropertyInfo arrayTypeProp = getProperty(pcc.GetUExport(EndianReader.ToInt32(entry.DataReadOnly, 44, entry.FileRef.Endian)));
662 if (arrayTypeProp != null)
663 {
664 switch (arrayTypeProp.Type)
665 {
666 case PropertyType.ObjectProperty:
667 case PropertyType.StructProperty:
668 case PropertyType.ArrayProperty:
669 reference = arrayTypeProp.Reference;
670 break;
671 case PropertyType.ByteProperty:
672 //if (arrayTypeProp.reference == "")
673 if (arrayTypeProp.Reference == "Class")
674 reference = arrayTypeProp.Type.ToString();
675 else
676 reference = arrayTypeProp.Reference;
677 break;
678 case PropertyType.IntProperty:
679 case PropertyType.FloatProperty:
680 case PropertyType.NameProperty:
681 case PropertyType.BoolProperty:
682 case PropertyType.StrProperty:
683 case PropertyType.StringRefProperty:
684 case PropertyType.DelegateProperty:
685 reference = arrayTypeProp.Type.ToString();
686 break;
687 case PropertyType.None:
688 case PropertyType.Unknown:
689 default:
690 Debugger.Break();
691 return null;
692 }
693 }
694 else
695 {
696 return null;
697 }
698 break;
699 case "InterfaceProperty":
700 default:
701 return null;
702 }
703
704 bool transient = ((UnrealFlags.EPropertyFlags)EndianReader.ToUInt64(entry.DataReadOnly, 24, entry.FileRef.Endian)).Has(UnrealFlags.EPropertyFlags.Transient);
705 int arrayLength = EndianReader.ToInt32(entry.DataReadOnly, 20, entry.FileRef.Endian);
706 return new PropertyInfo(type, reference, transient, arrayLength);
707 }
