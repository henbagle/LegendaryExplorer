226 public ExportEntry ParseSingleNodeInterpData(DialogueNodeExtended node, Dictionary<int, ExportEntry> convStarts = null)
227 {
228 if (Sequence == null || node == null || Sequence.UIndex < 1)
229 return null;
230
231 if (convStarts == null && Sequence is ExportEntry sequence)
232 {
233 var seqobjs = sequence.GetProperty<ArrayProperty<ObjectProperty>>("SequenceObjects");
234 convStarts = new Dictionary<int, ExportEntry>();
235 foreach (var prop in seqobjs)
236 {
237 var seqobj = Export.FileRef.GetUExport(prop.Value);
238 if (seqobj.ClassName == "BioSeqEvt_ConvNode")
239 {
240 int key = seqobj.GetProperty<IntProperty>("m_nNodeID"); //ME3
241 if (!convStarts.ContainsKey(key))
242 {
243 convStarts.Add(key, seqobj);
244 }
245 }
246 }
247 }
248
249 //Match to export id => SeqAct_Interp => Interpdata
250 node.ExportID = node.NodeProp.GetProp<IntProperty>("nExportID");
251 if (node.ExportID != 0)
252 {
253 var convstart = convStarts.FirstOrDefault(s => s.Key == node.ExportID).Value;
254 if (convstart != null)
255 {
256 // Find the interp data
257 List<ExportEntry> searchingExports = new List<ExportEntry>();
258 searchingExports.Add(convstart);
259 var seqActInterp = RecursiveFindSeqActInterp(searchingExports, new List<ExportEntry>(), 10);
260
261 var varLinksProp = seqActInterp.GetProperty<ArrayProperty<StructProperty>>("VariableLinks");
262 if (varLinksProp != null)
263 {
264 foreach (var prop in varLinksProp)
265 {
266 var desc = prop.GetProp<StrProperty>("LinkDesc").Value; //ME3/ME2/ME1
267 if (desc == "Data") //ME3/ME1
268 {
269 var linkedVars = prop.GetProp<ArrayProperty<ObjectProperty>>("LinkedVariables");
270 if (linkedVars != null && linkedVars.Count > 0)
271 {
272 var datalink = linkedVars[0].Value;
273 return Export.FileRef.GetUExport(datalink);
274
275 }
276 break;
277 }
278 }
279 }
280 }
281 }
282 return null;
283 }

285 private ExportEntry RecursiveFindSeqActInterp(List<ExportEntry> nodesToSearch, List<ExportEntry> nodesSearched, int searchDepthRemaining)
286 {
287 if (searchDepthRemaining <= 0)
288 return null; // NOT FOUND, NO FURTHER SEARCH
289
290 List<ExportEntry> nextNodesToSearch = new();
291 foreach (var searchingExport in nodesToSearch)
292 {
293 if (nodesSearched.Contains(searchingExport))
294 continue; // Do not enumerate existing items we've found, if there's some sort of loop
295 if (searchingExport.ClassName == "SeqAct_Interp")
296 {
297 return searchingExport;
298 }
299 else
300 {
301 nodesSearched.Add(searchingExport);
302 }
303
304 var outLinks = KismetHelper.GetOutboundLinksOfNode(searchingExport);
305 foreach(var outbound in outLinks)
306 {
307 nextNodesToSearch.AddRange(outbound.Where(x => x.LinkedOp is ExportEntry).Select(x=>x.LinkedOp as ExportEntry));
308 }
309 }
310
311 return RecursiveFindSeqActInterp(nextNodesToSearch, nodesSearched, --searchDepthRemaining);
312 }

403 public DialogueNodeExtended ParseSingleLine(StructProperty Node, int count, bool isReply, Func<int, IMEPackage, string> tlkLookupFunc = null)
404 {
405 int linestrref = 0;
406 int spkridx = -2;
407 int cond = -1;
408 string line = "Unknown Reference";
409 int stevent = -1;
410 bool bcond = false;
411 EReplyTypes eReply = EReplyTypes.REPLY_STANDARD;
412 try
413 {
414 linestrref = Node.GetProp<StringRefProperty>("srText")?.Value ?? 0;
415 line = tlkLookupFunc?.Invoke(linestrref, Export.FileRef);
416 cond = Node.GetProp<IntProperty>("nConditionalFunc")?.Value ?? -1;
417 stevent = Node.GetProp<IntProperty>("nStateTransition")?.Value ?? -1;
418 bcond = Node.GetProp<BoolProperty>("bFireConditional");
419 if (isReply)
420 {
421 Enum.TryParse(Node.GetProp<EnumProperty>("ReplyType").Value.Name, out eReply);
422 }
423 else
424 {
425 spkridx = Node.GetProp<IntProperty>("nSpeakerIndex");
426 }
427
428 return new DialogueNodeExtended(Node, isReply, count, spkridx, linestrref, line, bcond, cond, stevent, eReply);
429 }
430 catch (Exception e)
431 {
432#if DEBUG
433 throw new Exception($"List Parse failed: N{count} Reply?:{isReply}, {linestrref}, {line}, {cond}, {stevent}, {bcond.ToString()}, {eReply.ToString()}", e); //Note some convos don't have replies.
434#endif
435 return new DialogueNodeExtended(Node, isReply, count, spkridx, linestrref, line, bcond, cond, stevent, eReply);
436 }

555 public void ParseWwiseBank()
556 {
557 WwiseBank = null;
558 if (!Export.FileRef.Game.IsGame1())
559 {
560 try
561 {
562 ArrayProperty<ObjectProperty> wwevents;
563 IEntry ffxo = GetFaceFX(-1, true); //find owner animset
564
565 if (ffxo == null) //if no facefx then maybe soundobject conversation
566 {
567 wwevents = Export.GetProperty<ArrayProperty<ObjectProperty>>("m_aMaleSoundObjects");
568
569 }
570 else
571 {
572 ExportEntry ffxoExport = (ExportEntry)ffxo;
573
574 wwevents = ffxoExport.GetProperty<ArrayProperty<ObjectProperty>>("ReferencedSoundCues"); //pull an owner wwiseevent array
575 if (wwevents == null || wwevents.Count == 0 || wwevents[0].Value == 0)
576 {
577 IEntry ffxp = GetFaceFX(-2, true); //find player as alternative
578 if (!Export.FileRef.IsUExport(ffxp.UIndex))
579 return;
580 ExportEntry ffxpExport = (ExportEntry)ffxp;
581 wwevents = ffxpExport.GetProperty<ArrayProperty<ObjectProperty>>("ReferencedSoundCues");
582 }
583 if (wwevents == null || wwevents.Count == 0 || wwevents[0].Value == 0)
584 {
585 IEntry ffxS = GetFaceFX(0, true); //find speaker 1 as alternative
586 if (ffxS == null || !Export.FileRef.IsUExport(ffxS.UIndex))
587 return;
588 ExportEntry ffxSExport = (ExportEntry)ffxS;
589 wwevents = ffxSExport.GetProperty<ArrayProperty<ObjectProperty>>("ReferencedSoundCues");
590 }
591 }
592
593 if (wwevents == null || wwevents.Count == 0 || wwevents[0].Value == 0)
594 {
595 WwiseBank = null;
596 return;
597 }
598
599 if (Export.FileRef.Game.IsGame3())
600 {
601 StructProperty r = Export.FileRef.GetUExport(wwevents[0].Value).GetProperty<StructProperty>("Relationships"); //lookup bank
602 var bank = r.GetProp<ObjectProperty>("Bank");
603 WwiseBank = Export.FileRef.GetUExport(bank.Value);
604 }
605 else if (Export.FileRef.Game == MEGame.ME2) //Game is ME2. Wwisebank ref in Binary.
606 {
607 var wwiseEvent = Export.FileRef.GetUExport(wwevents[0].Value).GetBinaryData<WwiseEvent>();
608 foreach (var link in wwiseEvent.Links)
609 {
610 if (link.WwiseBanks.FirstOrDefault() is UIndex bankIdx)
611 {
612 WwiseBank = Export.FileRef.GetUExport(bankIdx);
613 break;
614 }
615 }
616 }
617 else if (Export.FileRef.Game is MEGame.LE2)
618 {
619 var r = Export.FileRef.GetUExport(wwevents[0].Value).GetProperty<ArrayProperty<StructProperty>>("References"); //lookup PlatformRelationships
620 var wr = r[0].GetProp<StructProperty>("Relationships");
621 var bank = wr.GetProp<ObjectProperty>("Bank");
622 WwiseBank = Export.FileRef.GetUExport(bank.Value);
623 }
624 }
625 catch (Exception e)
626 {
627#if DEBUG
628 throw new Exception($"WwiseBank Parse Failed. {ConvName}", e);
629#endif
630 }
631 }

717 {
718 AutoGenerateSpeakerArrays();
719 var newstartlist = new ArrayProperty<IntProperty>("m_StartingList");
720 foreach ((var _, int value) in StartingList)
721 {
722 newstartlist.Add(value);
723 }
724
725 var newentryList = new ArrayProperty<StructProperty>("m_EntryList");
726 foreach (var entry in EntryList.OrderBy(entry => entry.NodeCount))
727 {
728 newentryList.Add(entry.NodeProp);
729 }
730 var newreplyList = new ArrayProperty<StructProperty>("m_ReplyList");
731 foreach (var reply in ReplyList.OrderBy(reply => reply.NodeCount))
732 {
733 newreplyList.Add(reply.NodeProp);
734 }
735
736
737 if (Export.Game.IsGame3())
738 {
739 var newSpeakerList = new ArrayProperty<NameProperty>( "m_aSpeakerList");
740 foreach (var speaker in Speakers.OrderBy(x => x.SpeakerID))
741 {
742 if (speaker.SpeakerID < 0)
743 continue; // They don't belong here
744 newSpeakerList.Add(new NameProperty(speaker.SpeakerNameRef));
745 }
746 BioConvo.AddOrReplaceProp(newSpeakerList);
747 }
748 else
749 {
750 var newSpeakerList = new ArrayProperty<StructProperty>("m_SpeakerList");
751 foreach (var speaker in Speakers.OrderBy(x => x.SpeakerID))
752 {
753 if (speaker.SpeakerID < 0)
754 continue; // They don't belong here
755 PropertyCollection ssProps = new PropertyCollection();
756 ssProps.Add(new NameProperty(speaker.SpeakerNameRef, "sSpeakerTag"));
757 var speakerStruct = new StructProperty("BioDialogSpeaker", ssProps);
758 newSpeakerList.Add(speakerStruct);
759 }
760 BioConvo.AddOrReplaceProp(newSpeakerList);
761 }
762
763
764 if (newstartlist.Count > 0)
765 {
766 BioConvo.AddOrReplaceProp(newstartlist);
767 }
768
769 if (newentryList.Count > 0)
770 {
771 BioConvo.AddOrReplaceProp(newentryList);
772 }
773
774 if (newreplyList.Count >= 0)
775 {
776 BioConvo.AddOrReplaceProp(newreplyList);
777 }
778
779 if (commitToExport)
780 Export.WriteProperties(BioConvo);
781 }

783 public bool AutoGenerateSpeakerArrays()
784 {
785 bool hasLoopingPaths = false;
786
787 // Set blank speakers/listeners
788 var blankaSpkr = new ArrayProperty<IntProperty>("aSpeakerList");
789 foreach (var dnode in EntryList)
790 {
791 dnode.NodeProp.Properties.AddOrReplaceProp(blankaSpkr);
792 }
793
794 // Traverse conversation graph
795 foreach (int entryIndex in StartingList.Values)
796 {
797 var aSpkrs = new SortedSet<int>();
798 var startNode = EntryList[entryIndex];
799 var visitedNodes = new HashSet<DialogueNodeExtended>();
800 var newNodes = new Queue<DialogueNodeExtended>();
