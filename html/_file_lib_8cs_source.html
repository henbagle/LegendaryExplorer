120 private bool InternalInitialize(PackageCache packageCache, string gameRootPath = null)
121 {
122 try
123 {
124 LECLog.Information($@"Game Root Path for FileLib Init: {gameRootPath}. Has package cache: {packageCache != null}");
125 InitializationLog = new MessageLog();
126 _cacheEnabled = false;
127 _baseSymbols = null;
128 var gameFiles = MELoadedFiles.GetFilesLoadedInGame(Pcc.Game, gameRootOverride: gameRootPath);
129 string[] baseFileNames = BaseFileNames(Pcc.Game);
130 bool isBaseFile = false;
131 if (baseFileNames.IndexOf(Path.GetFileName(Pcc.FilePath)) is var fileNameIdx and >= 0)
132 {
133 isBaseFile = true;
134 baseFileNames = baseFileNames.Slice(0, fileNameIdx);
135 }
136 foreach (string fileName in baseFileNames)
137 {
138 if (gameFiles.TryGetValue(fileName, out string path) && File.Exists(path))
139 {
140 using var pcc = MEPackageHandler.OpenMEPackage(path);
141 if (!ResolveAllClassesInPackage(pcc, ref _baseSymbols, InitializationLog, packageCache))
142 {
143 return false;
144 }
145 }
146 else
147 {
148 InitializationLog.LogError($"Could not find required base file: {fileName}");
149 }
150 }
151 if (!isBaseFile)
152 {
153 var associatedFiles = EntryImporter.GetPossibleAssociatedFiles(Pcc, includeNonBioPRelated: false);
154 if (Pcc.Game is MEGame.ME3)
155 {
156 if (Pcc.FindEntry("SFXGameMPContent") is IEntry { ClassName: "Package" } && !associatedFiles.Contains("BIOP_MP_COMMON.pcc"))
157 {
158 associatedFiles.Add("BIOP_MP_COMMON.pcc");
159 }
160 if (Pcc.FindEntry("SFXGameContentDLC_CON_MP2") is IEntry { ClassName: "Package" })
161 {
162 associatedFiles.Add("Startup_DLC_CON_MP2_INT.pcc");
163 }
164 if (Pcc.FindEntry("SFXGameContentDLC_CON_MP3") is IEntry { ClassName: "Package" })
165 {
166 associatedFiles.Add("Startup_DLC_CON_MP3_INT.pcc");
167 }
168 if (Pcc.FindEntry("SFXGameContentDLC_CON_MP4") is IEntry { ClassName: "Package" })
169 {
170 associatedFiles.Add("Startup_DLC_CON_MP4_INT.pcc");
171 }
172 if (Pcc.FindEntry("SFXGameContentDLC_CON_MP5") is IEntry { ClassName: "Package" })
173 {
174 associatedFiles.Add("Startup_DLC_CON_MP5_INT.pcc");
175 }
176 }
177 foreach (var fileName in Enumerable.Reverse(associatedFiles))
178 {
179 if (gameFiles.TryGetValue(fileName, out string path) && File.Exists(path))
180 {
181 using var pcc = MEPackageHandler.OpenMEPackage(path);
182 if (!ResolveAllClassesInPackage(pcc, ref _baseSymbols, InitializationLog, packageCache))
183 {
184 return false;
185 }
186 }
187 }
188 }
189 _symbols = _baseSymbols?.Clone();
190 _cacheEnabled = true;
191 return ResolveAllClassesInPackage(Pcc, ref _symbols, InitializationLog, packageCache);
192 }
193 catch when (!LegendaryExplorerCoreLib.IsDebug)
194 {
195 return false;
196 }
197 }

322 private bool ResolveAllClassesInPackage(IMEPackage pcc, ref SymbolTable symbols, MessageLog log, PackageCache packageCache = null, Class classOverride = null)
323 {
324 string fileName = Path.GetFileNameWithoutExtension(pcc.FilePath);
325#if DEBUGSCRIPT
326 string dumpFolderPath = Path.Combine(MEDirectories.GetDefaultGamePath(pcc.Game), "ScriptDump", fileName);
327 Directory.CreateDirectory(dumpFolderPath);
328#endif
329 LECLog.Debug($"{fileName}: Beginning Parse.");
330 var classes = new List<(Class ast, string scriptText)>();
331 foreach (ExportEntry export in pcc.Exports.Where(exp => exp.IsClass))
332 {
333 Class cls;
334 if (classOverride != null && export.ObjectNameString.CaseInsensitiveEquals(classOverride.Name))
335 {
336 cls = classOverride;
337 }
338 else
339 {
340 cls = ScriptObjectToASTConverter.ConvertClass(GetCachedObjectBinary<UClass>(export, packageCache), false, this, packageCache);
341 }
342 log.CurrentClass = cls;
343 if (!cls.IsFullyDefined)
344 {
345 continue;
346 }
347 string scriptText = "";
348 try
349 {
350#if DEBUGSCRIPT
351 var codeBuilder = new CodeBuilderVisitor();
352 cls.AcceptVisitor(codeBuilder);
353 scriptText = codeBuilder.GetOutput();
354 File.WriteAllText(Path.Combine(dumpFolderPath, $"{cls.Name}.uc"), scriptText);
355 var parser = new ClassOutlineParser(new TokenStream<string>(new StringLexer(scriptText, log)), log);
356 cls = parser.TryParseClass();
357 if (cls == null || log.Content.Any())
358 {
359 DisplayError(scriptText, log.ToString());
360 return false;
361 }
362#endif
363 if (pcc.Game <= MEGame.ME2 && export.ObjectName == "Package")
364 {
365 symbols.TryGetType("Class", out Class classClass);
366 classClass.OuterClass = cls;
367 }
368 if (export.ObjectName == "Object")
369 {
370 symbols = SymbolTable.CreateIntrinsicTable(cls, pcc.Game);
371 }
372 else if (!symbols.AddType(cls))
373 {
374 continue; //class already defined
375 }
376
377 classes.Add(cls, scriptText);
378 }
379 catch (Exception e)// when (!LegendaryExplorerCoreLib.IsDebug)
380 {
381 log.LogError(e.FlattenException());
382 DisplayError(scriptText, log.ToString());
383 return false;
384 }
385 }
386 LECLog.Debug($"{fileName}: Finished parse.");
387 foreach (var validationPass in Enums.GetValues<ValidationPass>())
388 {
389 foreach ((Class cls, string scriptText) in classes)
390 {
391 log.CurrentClass = cls;
392 try
393 {
394 var validator = new ClassValidationVisitor(log, symbols, validationPass);
395 cls.AcceptVisitor(validator);
396 if (log.HasErrors)
397 {
398 DisplayError(scriptText, log.ToString());
399 return false;
400 }
401 }
402 catch (Exception e)// when(!ME3ExplorerCoreLib.IsDebug)
403 {
404 log.LogError(e.FlattenException());
405 DisplayError(scriptText, log.ToString());
406 return false;
407 }
408 }
409 LECLog.Debug($"{fileName}: Finished validation pass {validationPass}.");
410 }
411
412 switch (fileName)
413 {
414 case "Core" when pcc.Game.IsGame3():
415 symbols.InitializeME3LE3Operators();
416 break;
417 case "Engine":
418 symbols.ValidateIntrinsics();
419 break;
420 }
421
422#if DEBUGSCRIPT
423 //parse function bodies for testing purposes
424 foreach ((Class ast, string scriptText) in classes)
425 {
426 symbols.RevertToObjectStack();
427 if (!ast.Name.CaseInsensitiveEquals("Object"))
428 {
429 symbols.GoDirectlyToStack(((Class)ast.Parent).GetInheritanceString());
430 symbols.PushScope(ast.Name);
431 }
432
433 foreach (Function function in ast.Functions.Where(func => !func.IsNative && func.IsDefined))
434 {
435 CodeBodyParser.ParseFunction(function, pcc.Game, scriptText, symbols, log);
436 if (log.Content.Any())
437 {
438 DisplayError(scriptText, log.ToString());
439 }
440 }
441 }
442#endif
443
444 symbols.RevertToObjectStack();
445
446 return true;
447 }
