58 public static int Decompress(ReadOnlySpan<byte> src, Span<byte> dst)
59 {
60 uint srcLen = (uint)src.Length;
61 uint dstLen = (uint)dst.Length;
62 unsafe
63 {
64 fixed (byte* inPtr = &MemoryMarshal.GetReference(src))
65 fixed (byte* outPtr = &MemoryMarshal.GetReference(dst))
66 {
67 return SevenZipDecompress(inPtr, srcLen, outPtr, ref dstLen);
68 }
69 }

72 public static byte[] Decompress(byte[] src, uint dstLen)
73 {
74 uint len = dstLen;

77 int status = SevenZipDecompress(src, (uint)src.Length, dst, ref len);
78 if (status != 0)

84 public static byte[] Compress(byte[] src, int compressionLevel = 9)
85 {
86 uint dstLen = (uint)(src.Length * 2 + 8);
87 byte[] tmpbuf = new byte[dstLen];
88
89 int status = SevenZipCompress(compressionLevel, src, (uint)src.Length, tmpbuf, ref dstLen);
90 if (status != 0)
91 return new byte[0];
92
93 byte[] dst = new byte[dstLen];
94 Array.Copy(tmpbuf, dst, (int)dstLen);
95
96 return dst;
97 }

112 public static byte[] CompressToLZMAFile(byte[] src)
113 {
114 var compressedBytes = LZMA.Compress(src);
115 byte[] fixedBytes = new byte[compressedBytes.Length + 8]; //needs 8 byte header written into it (only mem version needs this)
116 Buffer.BlockCopy(compressedBytes, 0, fixedBytes, 0, 5);
117 fixedBytes.OverwriteRange(5, BitConverter.GetBytes(src.Length));
118 Buffer.BlockCopy(compressedBytes, 5, fixedBytes, 13, compressedBytes.Length - 5);
119 return fixedBytes;
120 }

122 public static byte[] DecompressLZMAFile(byte[] lzmaFile)
123 {
124 int len = (int)BitConverter.ToInt32(lzmaFile, 5); //this is technically a 64-bit value, but since MEM code can't handle 64 bit sizes we are just going to use 32bit. We aren't going to have a 2GB+ single LZMA file
125
126 if (len >= 0)
127 {
128 byte[] strippedData = new byte[lzmaFile.Length - 8];
129 //Non-Streamed (made from disk)
130 Buffer.BlockCopy(lzmaFile, 0, strippedData, 0, 5);
131 Buffer.BlockCopy(lzmaFile, 13, strippedData, 5, lzmaFile.Length - 13);
132 return Decompress(strippedData, (uint)len);
133 }
134 else if (len == -1)
135 {
136 throw new InvalidOperationException("Cannot decompress streamed LZMA with this implementation!");
137 }
138 else
139 {
140 Debug.WriteLine(@"Cannot decompress LZMA array: Length is not positive or -1 (" + len + @")! This is not an LZMA array");
141 return null; //Not LZMA!
142 }
143 }

145 public static void DecompressLZMAStream(MemoryStream compressedStream, MemoryStream decompressedStream)
146 {
147 compressedStream.Seek(5, SeekOrigin.Begin);
148 int len = compressedStream.ReadInt32();
149 compressedStream.Seek(0, SeekOrigin.Begin);
150
151 if (len >= 0)
152 {
153 byte[] strippedData = new byte[compressedStream.Length - 8];
154 compressedStream.Read(strippedData, 0, 5);
155 compressedStream.Seek(8, SeekOrigin.Current); //Skip 8 bytes for length.
156 compressedStream.Read(strippedData, 5, (int)compressedStream.Length - 13);
157 var decompressed = Decompress(strippedData, (uint)len);
158 decompressedStream.WriteFromBuffer(decompressed);
159 }
160 else if (len == -1)
161 {
162 throw new Exception("Cannot decompress streamed LZMA with this implementation!");
163 }
164 else
165 {
166 Debug.WriteLine(@"LZMA Stream to decompress has wrong length: " + len);
167 }
168 }
