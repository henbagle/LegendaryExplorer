49 public static uint Decompress(ReadOnlySpan<byte> src, Span<byte> dst)
50 {
51 uint srcLen = (uint)src.Length;
52 uint dstLen = (uint)dst.Length;
53
54 int status;
55 unsafe
56 {
57 fixed (byte* inPtr = &MemoryMarshal.GetReference(src))
58 fixed (byte* outPtr = &MemoryMarshal.GetReference(dst))
59 {
60 status = ZlibDecompress(inPtr, srcLen, outPtr, ref dstLen);
61 }
62 }
63 if (status != 0)
64 return 0;
65
66 return dstLen;
67 }

69 public static byte[] Compress(ReadOnlySpan<byte> src, int compressionLevel = -1)
70 {
71 byte[] tmpbuf = new byte[GetCompressionBound(src.Length)];
72 uint dstLen = (uint)tmpbuf.Length;
73
74 unsafe
75 {
76 fixed (byte* ptr = &MemoryMarshal.GetReference(src))
77 {
78 int status = ZlibCompress(compressionLevel, ptr, (uint)src.Length, tmpbuf, ref dstLen);
79 if (status != 0)
80 return new byte[0];
81 }
82 }
83
84 byte[] dst = new byte[dstLen];
85 Array.Copy(tmpbuf, dst, (int)dstLen);
86
87 return dst;
88 }

90 public static int Compress(ReadOnlySpan<byte> inputBuffer, Span<byte> outputBuffer)
91 {
92 uint compressedCount = (uint)outputBuffer.Length;
93 unsafe
94 {
95 fixed (byte* inPtr = &MemoryMarshal.GetReference(inputBuffer))
96 fixed (byte* outPtr = &MemoryMarshal.GetReference(outputBuffer))
97 {
98 int status = ZlibCompress(-1, Unsafe.AsRef<byte>(inPtr), (uint)inputBuffer.Length, Unsafe.AsRef<byte>(outPtr), ref compressedCount);
99 if (status != 0)
100 {
101 return 0;
102 }
103 }
104 }
105 return (int)compressedCount;
106 }

108 public static int GetCompressionBound(int length)
109 {
110 return length + ((length + 7) >> 3) + ((length + 63) >> 6) + 11;
111 }
