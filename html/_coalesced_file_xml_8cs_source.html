277 public void Deserialize(Stream input)
278 {
279 var magic = input.ReadUInt32();
280
281 if (magic != 0x666D726D && magic.Swap() != 0x666D726D)
282 {
283 throw new FormatException();
284 }
285
286 var endian = magic == 0x666D726D ? ByteOrder.LittleEndian : ByteOrder.BigEndian;
287 var version = input.ReadUInt32(endian);
288
289 if (version != 1)

293
294 Version = version;

296 input.ReadInt32(endian);

306 using (var data = input.ReadToMemoryStream(stringTableSize))
307 {
308 var localStringTableSize = data.ReadUInt32(endian);
309
310 if (localStringTableSize != stringTableSize)
311 {
312 throw new FormatException();
313 }
314
315 var count = data.ReadUInt32(endian);
316
317 var offsets = new List<KeyValuePair<uint, uint>>();
318
319 for (uint i = 0; i < count; i++)
320 {
321 var hash = data.ReadUInt32(endian);
322 var offset = data.ReadUInt32(endian);
323
324 offsets.Add(new KeyValuePair<uint, uint>(hash, offset));
325 }
326
327 foreach (var kv in offsets)
328 {
329 var hash = kv.Key;
330 var offset = kv.Value;
331
332 data.Seek(8 + offset, SeekOrigin.Begin);
333 var length = data.ReadUInt16(endian);

335
