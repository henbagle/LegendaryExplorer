104 public override List<(UIndex, string)> GetUIndexes(MEGame game)
105 {
106 var uIndexes = new List<(UIndex, string)>();
107
108 uIndexes.Add((Self, nameof(Self)));
109 for (int i = 0; i < Surfs.Length; i++)
110 {
111 BspSurf surf = Surfs[i];
112 uIndexes.Add((surf.Material, $"Surfs[{i}].Material"));
113 uIndexes.Add((surf.Actor, $"Surfs[{i}].Actor"));
114 }
115
116 uIndexes.AddRange(Zones.Select((zone, i) => (zone.ZoneActor, $"Zones[{i}].ZoneActor")));
117 uIndexes.Add((Polys, nameof(Polys)));
118
119 return uIndexes;
120 }
