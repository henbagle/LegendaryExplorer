143 public static bool TryParse(string input, bool ignoreCase, out TEnum? result)
144 {
145 result = null;
146
147 if (!Enum.IsDefined(typeof (TEnum), input))
148 {
149 return false;
150 }
151
152 try
153 {
154 result = (TEnum) Enum.Parse(typeof (TEnum), input, ignoreCase);
155 }
156 catch (Exception)
157 {
158 return false;
159 }
160
161 return true;
162 }

171 public static bool TryParse(string input, bool ignoreCase, out TEnum result)
172 {
173 TEnum? temp;
174
175 if (!TryParse(input, ignoreCase, out temp))
176 {
177 // input not found in the Enum, fill the out parameter with the first item from the enum
178 var values = GetValues().ToArray();
179
180 result = (TEnum) values.GetValue(values.GetLowerBound(0));
181
182 return false;
183 }
184
185 result = temp ?? default(TEnum);
186
187 return true;
188 }
