73 public bool VisitNode(Class node)
74 {
75 Write(CLASS, EF.Keyword);
76 Space();
77 Append(node.Name, EF.TypeName);
78
79 if (node.Parent != null && !node.Parent.Name.Equals("Object", StringComparison.OrdinalIgnoreCase))
80 {
81 Space();
82 Append(EXTENDS, EF.Keyword);
83 Space();
84 Append(node.Parent.Name, EF.TypeName);
85 }
86 if (node.OuterClass != null && !node.OuterClass.Name.Equals("Object", StringComparison.OrdinalIgnoreCase))
87 {
88 Space();
89 Append(WITHIN, EF.Keyword);
90 Space();
91 Append(node.OuterClass.Name, EF.TypeName);
92 }
93
94 NestingLevel++;
95
96 if (node.Interfaces.Any())
97 {
98 Write("implements", EF.Keyword);
99 Append("(");
100 Join(node.Interfaces.Select(i => i.Name).ToList(), ", ", EF.TypeName);
101 Append(")");
102 }
103
104
105 EClassFlags flags = node.Flags;
106 if (flags.Has(EClassFlags.Native))
107 {
108 Write("native", EF.Specifier);
109 }
110 if (flags.Has(EClassFlags.NativeOnly))
111 {
112 Write("nativeonly", EF.Specifier);
113 }
114 if (flags.Has(EClassFlags.NoExport))
115 {
116 Write("noexport", EF.Specifier);
117 }
118 if (flags.Has(EClassFlags.EditInlineNew))
119 {
120 Write("editinlinenew", EF.Specifier);
121 }
122 if (flags.Has(EClassFlags.Placeable))
123 {
124 Write("placeable", EF.Specifier);
125 }
126 if (flags.Has(EClassFlags.HideDropDown))
127 {
128 Write("hidedropdown", EF.Specifier);
129 }
130 if (flags.Has(EClassFlags.NativeReplication))
131 {
132 Write("nativereplication", EF.Specifier);
133 }
134 if (flags.Has(EClassFlags.PerObjectConfig))
135 {
136 Write("perobjectconfig", EF.Specifier);
137 }
138 if (flags.Has(EClassFlags.Abstract))
139 {
140 Write("abstract", EF.Specifier);
141 }
142 if (flags.Has(EClassFlags.Deprecated))
143 {
144 Write("deprecated", EF.Specifier);
145 }
146 if (flags.Has(EClassFlags.Transient))
147 {
148 Write("transient", EF.Specifier);
149 }
150 if (flags.Has(EClassFlags.Config))
151 {
152 Write("config", EF.Specifier);
153 Append($"({node.ConfigName})");
154 }
155 if (flags.Has(EClassFlags.SafeReplace))
156 {
157 Write("safereplace", EF.Specifier);
158 }
159 if (flags.Has(EClassFlags.Hidden))
160 {
161 Write("hidden", EF.Specifier);
162 }
163 if (flags.Has(EClassFlags.CollapseCategories))
164 {
165 Write("collapsecategories", EF.Specifier);
166 }
167
168 NestingLevel--;
169 Append(";");
170
171 // print the rest of the class, according to the standard "anatomy of an unrealscript" article.
172 if (node.TypeDeclarations.Count > 0)
173 {
174 Write();
175 Write("// Types", EF.Comment);
176 foreach (VariableType type in node.TypeDeclarations)
177 type.AcceptVisitor(this);
178 }
179
180 if (node.VariableDeclarations.Count > 0)
181 {
182 Write();
183 Write("// Variables", EF.Comment);
184 foreach (VariableDeclaration decl in node.VariableDeclarations)
185 decl.AcceptVisitor(this);
186 }
187
188 if (node.Functions.Count > 0)
189 {
190 Write();
191 Write("// Functions", EF.Comment);
192 foreach (Function func in node.Functions)
193 func.AcceptVisitor(this);
194 }
195
196 if (node.States.Count > 0)
197 {
198 Write();
199 Write("// States", EF.Comment);
200 foreach (State state in node.States)
201 state.AcceptVisitor(this);
202 }
203
204 if (node.ReplicationBlock?.Statements.Count > 0)
205 {
206 Write();
207 if (node.Flags.Has(EClassFlags.NativeReplication))
208 {
209 Write("//Replication conditions for this class are native. This block has no effect", EF.Comment);
210 }
211 Write(REPLICATION, EF.Keyword);
212 Write("{");
213 NestingLevel++;
214 node.ReplicationBlock.AcceptVisitor(this);
215 NestingLevel--;
216 Write("}");
217 }
218
219 Write();
220 Write("//class default properties can be edited in the Properties tab for the class's Default__ object.", EF.Comment);
221 node.DefaultProperties?.AcceptVisitor(this);
222
223 return true;
224 }

329 {

331 Write(STRUCT, EF.Keyword);
332 Space();
333 var specs = new List<string>();
334 ScriptStructFlags flags = node.Flags;
335 if (flags.Has(ScriptStructFlags.Native))
336 {
337 specs.Add("native");
338 }
339 if (flags.Has(ScriptStructFlags.Export))
340 {
341 specs.Add("export");
342 }
343 if (flags.Has(ScriptStructFlags.Transient))
344 {
345 specs.Add("transient");
346 }
347 if (flags.Has(ScriptStructFlags.Immutable))
348 {
349 specs.Add("immutable");
350 }
351 else if (flags.Has(ScriptStructFlags.Atomic))
352 {
353 specs.Add("atomic");
354 }

356 {
357 specs.Add("immutablewhencooked");
358 }
359 if (flags.Has(ScriptStructFlags.StrictConfig))
360 {
361 specs.Add("strictconfig");
362 }

367
368 foreach (string spec in specs)
369 {
370 Append(spec, EF.Specifier);
371 Space();
372 }
373
374 Append(node.Name, EF.TypeName);
375 Space();
376 if (node.Parent != null)
377 {
378 Append(EXTENDS, EF.Keyword);
379 Space();
380 Append(node.Parent.Name, EF.TypeName);
381 Space();
382 }
383
384 Write("{");
385 NestingLevel++;
386
387 foreach (VariableType typeDeclaration in node.TypeDeclarations)
388 {
389 typeDeclaration.AcceptVisitor(this);
390 }
391
392 foreach (VariableDeclaration member in node.VariableDeclarations)
393 member.AcceptVisitor(this);
394
395 if (node.DefaultProperties.Statements.Any())
396 {
397 Write();
398 node.DefaultProperties.AcceptVisitor(this);
399 }
400
401 NestingLevel--;
402 Write("};");
403
404 return true;

448 {
449 // [specifiers] function [returntype] functionname ( [parameter declarations] ) body_or_semicolon
450 Write();
451
452 var specs = new List<string>();
453 EFunctionFlags flags = node.Flags;
454
455 if (flags.Has(EFunctionFlags.Private))
456 {
457 specs.Add("private");
458 }
459 if (flags.Has(EFunctionFlags.Protected))
460 {
461 specs.Add("protected");
462 }
463 if (flags.Has(EFunctionFlags.Public))
464 {
465 specs.Add("public");
466 }
467 if (flags.Has(EFunctionFlags.Static))
468 {
469 specs.Add("static");
470 }
471 if (flags.Has(EFunctionFlags.Final))
472 {
473 specs.Add("final");
474 }
475 if (flags.Has(EFunctionFlags.Delegate))
476 {
477 specs.Add("delegate");
478 }
479 if (flags.Has(EFunctionFlags.Event))
480 {
481 specs.Add("event");
482 }
483 if (flags.Has(EFunctionFlags.PreOperator))
484 {
485 specs.Add("preoperator");
486 }
487 else if (flags.Has(EFunctionFlags.Operator))
488 {
489 specs.Add("operator");
490 }
491 if (flags.Has(EFunctionFlags.Iterator))
492 {
493 specs.Add("iterator");
494 }
495 if (flags.Has(EFunctionFlags.Singular))
496 {
497 specs.Add("singular");
498 }
499 if (flags.Has(EFunctionFlags.Latent))
500 {
501 specs.Add("latent");
502 }
503 if (flags.Has(EFunctionFlags.Exec))
504 {
505 specs.Add("exec");
506 }
507 if (flags.Has(EFunctionFlags.NetReliable))
508 {
509 specs.Add("reliable");
510 }
511 else if (flags.Has(EFunctionFlags.Net))
512 {
513 specs.Add("unreliable");
514 }
515 if (flags.Has(EFunctionFlags.NetServer))
516 {
517 specs.Add("server");
518 }
519 if (flags.Has(EFunctionFlags.NetClient))
520 {
521 specs.Add("client");
522 }
523 else if (flags.Has(EFunctionFlags.Simulated))
524 {
525 specs.Add("simulated");
526 }
527
528 foreach (string spec in specs)
529 {
530 Append(spec, EF.Specifier);
531 Space();
532 }
533 if (flags.Has(EFunctionFlags.Native))
534 {
535 Append("native", EF.Specifier);
536 if (node.NativeIndex > 0)
537 {
538 Append("(");
539 Append(node.NativeIndex.ToString(), EF.Number);
540 Append(")");
541 }
542 Space();
543 }
544
545 Append(FUNCTION, EF.Keyword);
546 Space();
547 if (node.ReturnType != null)
548 {
549 if (node.CoerceReturn)
550 {
551 Append("coerce", EF.Specifier);
552 Space();
553 }
554 AppendTypeName(node.ReturnType);
555 Space();
556 }
557 Append(node.Name, EF.Function);
558 Append("(");
559 if (node.Parameters.Any())
560 {
561 node.Parameters[0].AcceptVisitor(this);
562 for (int i = 1; i < node.Parameters.Count; i++)
563 {
564 Append(",");
565 Space();
566 node.Parameters[i].AcceptVisitor(this);
567 }
568 }
569
570 Append(")");
571
572 if (flags.Has(EFunctionFlags.Defined) && node.Body.Statements != null)
573 {
574 var tmp = LabelNest;
575 LabelNest = NestingLevel;
576
577 Write("{");
578 NestingLevel++;
579 if (node.Locals.Any())
580 {
581 foreach (VariableDeclaration v in node.Locals)
582 v.AcceptVisitor(this);
583 Write();
584 }
585 node.Body.AcceptVisitor(this);
586 NestingLevel--;
587 Write("}");
588
589 LabelNest = tmp;
590 }
591 else
592 {
593 Append(";");
594 Write();
595 }
596
597 return true;
598 }

625 {
626 // [specifiers] state statename [extends parentstruct] { \n contents \n };
627 Write();
628
629 var specs = new List<string>();
630 EStateFlags flags = node.Flags;
631
632 if (flags.Has(EStateFlags.Simulated))
633 {
634 specs.Add("simulated");
635 }
636 if (flags.Has(EStateFlags.Auto))
637 {
638 specs.Add("auto");
639 }
640
641 foreach (string spec in specs)
642 {
643 Append(spec, EF.Specifier);
644 Space();
645 }
646
647 Append(STATE, EF.Keyword);
648 if (flags.Has(EStateFlags.Editable))
649 {
650 Append("()");
651 }
652 Space();
653 Append(node.Name, EF.State);
654 Space();
655 if (node.Parent != null)
656 {
657 Append(EXTENDS, EF.Keyword);
658 Space();
659 Append(node.Parent.Name, EF.State);
660 Space();
661 }
662
663 var nestTmp = LabelNest;
664 LabelNest = NestingLevel;
665
666 Write("{");
667 NestingLevel++;
668
669 if (node.IgnoreMask != (EProbeFunctions)ulong.MaxValue)
670 {
671 Write(IGNORES, EF.Keyword);
672 Space();
673 Join((~node.IgnoreMask).MaskToList().Select(flag => flag.ToString()).ToList(), ", ", EF.Function);
674 Write(";");
675 }
676
677 Write("// State Functions", EF.Comment);
678 foreach (Function func in node.Functions)
679 func.AcceptVisitor(this);
680
681 Write();
682 Write("// State code", EF.Comment);
683 if (node.Body.Statements.Count != 0)
684 {
685 node.Body.AcceptVisitor(this);
686 }
687
688 NestingLevel--;
689 Write("};");
690
691 LabelNest = nestTmp;
692
693 return true;

1512 private static string FormatFloat(float single)
1513 {
1514 //G9 ensures a fully accurate version of the float (no rounding) is written.
1515 //more details here: https://docs.microsoft.com/en-us/dotnet/standard/base-types/standard-numeric-format-strings#the-round-trip-r-format-specifier
1516 string floatString = single.ToString("G9", NumberFormatInfo.InvariantInfo).Replace("E+", "e");
1517
1518 if (floatString.Contains("E-"))
1519 {
1520 //unrealscript does not support negative exponents in literals, so we have to format it manually
1521 //for example, 1.401298E-45 would be formatted as 0.00000000000000000000000000000000000000000000140129846
1522 //This code assumes there is exactly 1 digit before the decimal point, which will always be the case when formatted as scientific notation with the G specifier
1523 string minus = null;
1524 if (floatString[0] == '-')
1525 {
1526 minus = "-";
1527 floatString = floatString.Substring(1, floatString.Length - 1);
1528 }
1529 int ePos = floatString.IndexOf("E-");
1530 int exponent = int.Parse(floatString.Substring(ePos + 2));
1531 string digits = floatString.Substring(0, ePos).Replace(".", "");
1532 floatString = $"{minus}0.{new string('0', exponent - 1)}{digits}";
1533 }
1534 else if (!floatString.Contains(".") && !floatString.Contains("e"))
1535 {
1536 //need a decimal place in the float so that it does not get parsed as an int
1537 floatString += $".0";
1538 }
1539
1540 return floatString;
1541 }
