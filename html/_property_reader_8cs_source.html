123 public static NameReference FromInstancedString(string s)
124 {
125 int num = 0;
126 int _Idx = s.LastIndexOf('_');
127 if (_Idx > 0)
128 {
129 string numComponent = s.Substring(_Idx + 1);
130 if (numComponent.Length > 0
131 && !(numComponent.Length > 1 && numComponent[0] == '0') //if there's more than one character and a leading zero, it's just part of the string
132 && int.TryParse(numComponent, NumberStyles.None, null, out num))
133 {
134 s = s.Substring(0, _Idx);
135 num += 1;
136 }
137 }
138 return new NameReference(s, num);
