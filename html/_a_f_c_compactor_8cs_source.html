39 public static AFCInventory GetInventory(string inputPath, MEGame game, Action<string> currentScanningFileCallback = null, Action<string> debugOut = null)
40 {
41 var inventory = new AFCInventory();
42 inventory.LocalFolderAFCFiles.ReplaceAll(Directory.GetFiles(inputPath, "*.afc", SearchOption.AllDirectories));
43
44
45 debugOut?.Invoke($"Beginning AFC references scan. Input path: {inputPath}, game {game}");
46 inventory.BasegameAFCFiles.ReplaceAll(MELoadedFiles.GetCookedFiles(game, GameFilesystem.MEDirectories.GetBioGamePath(game), includeAFCs: true).Where(x => Path.GetExtension(x) == ".afc"));
47 foreach (var oafc in inventory.BasegameAFCFiles)
48 {
49 debugOut?.Invoke($@" >> Found Basegame AFC {oafc}");
50 }
51 inventory.OfficialDLCAFCFiles = MELoadedFiles.GetOfficialDLCFolders(game).SelectMany(x => Directory.GetFiles(x, "*.afc", SearchOption.AllDirectories)).ToList();
52 foreach (var oafc in inventory.OfficialDLCAFCFiles)
53 {
54 debugOut?.Invoke($@" >> Found AFC in DLC directory {oafc}");
55 }
56
57 // ME3: Inspect SFARs for AFCs
58 if (game == MEGame.ME3 && Directory.Exists(ME3Directory.DLCPath))
59 {
60 debugOut?.Invoke($@"DLC directory: {ME3Directory.DLCPath}");
61 foreach (var officialDLC in ME3Directory.OfficialDLC)
62 {
63 var sfarPath = Path.Combine(ME3Directory.DLCPath, officialDLC, "CookedPCConsole", "Default.sfar");
64 if (File.Exists(sfarPath))
65 {
66 currentScanningFileCallback?.Invoke(ME3Directory.OfficialDLCNames[officialDLC]);
67 debugOut?.Invoke($@"{ME3Directory.OfficialDLCNames[officialDLC]} is installed ({officialDLC})");
68
69 var dlc = new DLCPackage(sfarPath);
70 inventory.SFARAFCsMap[officialDLC] = dlc.Files.Where(x => x.FileName.EndsWith(".afc")).Select(x => x.FileName).ToList();
71 foreach (var oafc in inventory.SFARAFCsMap[officialDLC])
72 {
73 debugOut?.Invoke($@" >> Found SFAR AFC {oafc}");
74 }
75
76 inventory.OfficialDLCAFCFiles.AddRange(inventory.SFARAFCsMap[officialDLC]);
77 }
78 }
79 }
80 return inventory;
81 }

126 public static (List<ReferencedAudio> missingAFCReferences, List<ReferencedAudio> availableAFCReferences) GetReferencedAudio(MEGame game, string inputPath,
127 Action<long,long> notifyProgress = null,
128 Action<string> currentScanningFileCallback = null,
129 Action<string> debugOut = null)
130 {
131 var sizesJsonStr = new StreamReader(LegendaryExplorerCoreUtilities.LoadFileFromCompressedResource("Infos.zip", $"{game}-vanillaaudiosizes.json")).ReadToEnd();
132 var vanillaSizesMap = JsonConvert.DeserializeObject<CaseInsensitiveDictionary<int>>(sizesJsonStr);
133 var pccFiles = Directory.GetFiles(inputPath, "*.pcc", SearchOption.AllDirectories);
134
135 AFCInventory afcInventory = AFCInventory.GetInventory(inputPath, game, debugOut);
136 var referencedAFCAudio = new List<ReferencedAudio>();
137 var missingAFCReferences = new List<ReferencedAudio>();
138 int i = 1;
139
140 foreach (string pccPath in pccFiles)
141 {
142 notifyProgress?.Invoke(i - 1, pccFiles.Count());
143 debugOut?.Invoke($@"SCANNING {pccPath}");
144 currentScanningFileCallback?.Invoke(pccPath);
145 //NotifyStatusUpdate?.Invoke($"Finding all referenced audio ({i}/{pccFiles.Length})");
146 using (var pack = MEPackageHandler.OpenMEPackage(pccPath))
147 {
148 List<ExportEntry> wwiseStreamExports = pack.Exports.Where(x => x.ClassName == "WwiseStream").ToList();
149 foreach (ExportEntry exp in wwiseStreamExports)
150 {
151 debugOut?.Invoke($@" >> WwiseStream {exp.UIndex} {exp.ObjectName}");
152
153 var afcNameProp = exp.GetProperty<NameProperty>("Filename");
154 if (afcNameProp != null)
155 {
156 debugOut?.Invoke($@" >>>> AFC filename: {afcNameProp.Value.Name}");
157
158 bool isBasegame = false;
159 bool isOfficialDLC = false;
160 var afcFile = afcInventory.LocalFolderAFCFiles.FirstOrDefault(x => Path.GetFileNameWithoutExtension(x).Equals(afcNameProp.Value, StringComparison.InvariantCultureIgnoreCase));
161
162 bool logged = false;
163 if (afcFile != null)
164 {
165 debugOut?.Invoke($@" >>>> AFC found locally: {afcFile}");
166 logged = true;
167 }
168 if (afcFile == null)
169 {
170 // Try to find basegame version
171 afcFile = afcInventory.BasegameAFCFiles.FirstOrDefault(x => Path.GetFileNameWithoutExtension(x).Equals(afcNameProp.Value, StringComparison.InvariantCultureIgnoreCase));
172 isBasegame = afcFile != null;
173 }
174 if (afcFile != null && !logged)
175 {
176 debugOut?.Invoke($@" >>>> AFC found in basegame: {afcFile}");
177 logged = true;
178 }
179
180
181 if (afcFile == null)
182 {
183 // Try to find official DLC version
184 afcFile = afcInventory.OfficialDLCAFCFiles.FirstOrDefault(x => Path.GetFileNameWithoutExtension(x).Equals(afcNameProp.Value, StringComparison.InvariantCultureIgnoreCase));
185 isOfficialDLC = afcFile != null;
186 }
187 if (afcFile != null && !logged)
188 {
189 debugOut?.Invoke($@" >>>> AFC found in official DLC: {afcFile}");
190 logged = true;
191 }
192
193 string afcName = afcNameProp.ToString().ToLower();
194 int audioSize = EndianReader.ToInt32(exp.DataReadOnly, exp.DataSize - 8, exp.FileRef.Endian);
195 int audioOffset = EndianReader.ToInt32(exp.DataReadOnly, exp.DataSize - 4, exp.FileRef.Endian);
196 bool isModified = false;
197 if (afcFile != null)
198 {
199 var source = !isBasegame && !isOfficialDLC ? "Modified" : null;
200 if (isBasegame || isOfficialDLC)
201 {
202 // Check if offset indicates this is official bioware afc territory
203 if (vanillaSizesMap.TryGetValue(afcName, out var vanillaSize))
204 {
205 if (audioOffset < vanillaSize)
206 {
207 if (isOfficialDLC)
208 {
209 source = "Official DLC";
210 }
211 else if (isBasegame)
212 {
213 debugOut?.Invoke($" >>>> Dropping fully basegame audio reference: {exp.ObjectName}");
214 continue; //Fully basegame audio is never returned as it will always be available
215 }
216 }
217 else
218 {
219 // Out of range?
220 isModified = true;
221 source = "Modified official DLC";
222 }
223 }
224 else
225 {
226 debugOut?.Invoke($"!!! Vanilla sizes map doesn't include file being checked: {afcName}");
227 }
228 }
229 else
230 {
231 isModified = true;
232 source = "Modified unofficial";
233 }
234
235 referencedAFCAudio.Add(new ReferencedAudio()
236 {
237 afcName = afcName,
238 audioSize = audioSize,
239 audioOffset = audioOffset,
240 uiOriginatingExportName = exp.ObjectName,
241 uiAFCSourceType = source,
242 isModified = isModified
243 });
244 }
245 else
246 {
247 debugOut?.Invoke($@" !!!! AFC NOT FOUND: {afcNameProp.Value}");
248
249 missingAFCReferences.Add(new ReferencedAudio()
250 {
251 afcName = afcNameProp.Value,
252 audioSize = audioSize,
253 audioOffset = audioOffset,
254 uiOriginatingExportName = exp.ObjectName,
255 uiAFCSourceType = "AFC unavailable"
256 });
257 }
258 }
259 }
260 }
261 i++;
262 }
263 referencedAFCAudio = referencedAFCAudio.Distinct().ToList();
264 return (missingAFCReferences, referencedAFCAudio);
265 }

267 public static bool CompactAFC(MEGame game, string inputPath, string newAFCBaseName, List<ReferencedAudio> referencesToCompact,
268 Func<List<(ReferencedAudio, string)>, bool> notifyBrokenAudio,
269 Action<long, long> notifyProgress,
270 Action<string> notifyStatusUpdate = null,
271 Action<string> notifyFinalAfcPath = null,
272 Action<string> debugOut = null)
273 {
274 notifyStatusUpdate?.Invoke("Preparing to compact AFC");
275
276 var afcInventory = AFCInventory.GetInventory(inputPath, game, notifyStatusUpdate, null); //Null lets me know there's something here to add for debug
277 // Order by AFC name so we can just open a single stream to pull from rather than 800 times
278 referencesToCompact = referencesToCompact.OrderBy(x => x.afcName).ToList();
279
280 #region EXTRACT AND BUILD NEW AFC FILE
281 string currentOpenAfc = null;
282 Stream currentOpenAfcStream = null;
283
284 notifyStatusUpdate?.Invoke("Creating reference map to new AFC");
285
286 // Mapping of old reference => new reference
287 var referenceMap = new Dictionary<AFCCompactor.ReferencedAudio, AFCCompactor.ReferencedAudio>();
288
289 using MemoryStream memoryNewAfc = MemoryManager.GetMemoryStream();
290 var brokenAudio = new List<(ReferencedAudio brokenRef, string brokenReason)>();
291 int i = 0;
292 foreach (var referencedAudio in referencesToCompact)
293 {
294 notifyProgress?.Invoke(i, referencesToCompact.Count);
295 i++;
296 if (referencedAudio.afcName != currentOpenAfc)
297 {
298 currentOpenAfcStream?.Dispose();
299 currentOpenAfcStream = fetchAfcStream(referencedAudio.afcName, afcInventory, debugOut);
300 currentOpenAfc = referencedAudio.afcName;
301 }
302
303 if (currentOpenAfcStream == null)
304 {
305 Debug.WriteLine($"AFC could not be found: {referencedAudio.afcName}");
306 brokenAudio.Add((referencedAudio, $"AFC could not be found: {referencedAudio.afcName}"));
307 continue;
308 }
309
310 var referencePos = memoryNewAfc.Position;
311 try
312 {
313 if (currentOpenAfcStream.Length <= referencedAudio.audioOffset)
314 {
315 brokenAudio.Add((referencedAudio, $"Audio pointer is outside of AFC {referencedAudio.afcName} @ 0x{referencedAudio.audioOffset:X8}"));
316 continue;
317 }
318
319 if (currentOpenAfcStream.Length < referencedAudio.audioOffset + referencedAudio.audioSize)
320 {
321 brokenAudio.Add((referencedAudio, $"Audio size causes reference to extend beyond end of AFC {referencedAudio.afcName} @ 0x{referencedAudio.audioOffset:X8} for length 0x{referencedAudio.audioSize:X6}. The AFC is only 0x{currentOpenAfcStream.Length:X8} in size"));
322 continue;
323 }
324
325 // Read header
326 currentOpenAfcStream.Position = referencedAudio.audioOffset;
327 var header = currentOpenAfcStream.ReadStringLatin1(4);
328 if (header != "RIFF")
329 {
330 brokenAudio.Add((referencedAudio, $"Audio pointer doesn't point to data that doesn't start with the RIFF tag. This is an invalid pointer as all audio will start with RIFF. AFC {referencedAudio.afcName} @ 0x{referencedAudio.audioOffset:X8}"));
331 continue;
332 }
333
334 currentOpenAfcStream.Position = referencedAudio.audioOffset;
335 currentOpenAfcStream.CopyToEx(memoryNewAfc, (int)referencedAudio.audioSize);
336
337 referenceMap[referencedAudio] = new ReferencedAudio()
338 {
339 afcName = newAFCBaseName,
340 audioOffset = referencePos,
341 audioSize = referencedAudio.audioSize,
342 };
343 }
344 catch (Exception e)
345 {
346 brokenAudio.Add((referencedAudio, $"AFC could not be found: {referencedAudio.afcName} (Error: {e.Message})"));
347 }
348
349 var test = referenceMap[referencedAudio];
350 }
351 currentOpenAfcStream?.Dispose();
352 Debug.WriteLine($"New AFC size: 0x{memoryNewAfc.Length:X8} ({FileSize.FormatSize(memoryNewAfc.Length)})");
353
354 if (brokenAudio.Any())
355 {
356 var shouldContinue = notifyBrokenAudio?.Invoke(brokenAudio);
357 if (!shouldContinue.HasValue || !shouldContinue.Value)
358 {
359 return false;
360 }
361 }
362
363 // Write temp to make sure we don't update references and then find out we can't actually write to disk
364 var finalAfcPath = inputPath;
365 if (!Path.GetFileName(finalAfcPath).StartsWith("CookedPC", StringComparison.InvariantCultureIgnoreCase))
366 {
367 finalAfcPath = Path.Combine(finalAfcPath, MEDirectories.CookedName(game));
368 }
369 finalAfcPath = Path.Combine(finalAfcPath, $"{newAFCBaseName}.afc");
370 var tempAfcPath = Path.Combine(inputPath, $"TEMP_{newAFCBaseName}.afc");
371 memoryNewAfc.WriteToFile(tempAfcPath);
372 #endregion
373
374 #region UPDATE AUDIO REFERENCES
375 notifyStatusUpdate?.Invoke("Updating audio references to point to new AFC");
376 var pccFiles = Directory.GetFiles(inputPath, "*.pcc", SearchOption.AllDirectories);
377
378 // Update audio references
379 i = 0;
380 foreach (string pccPath in pccFiles)
381 {
382 notifyProgress?.Invoke(i, pccFiles.Count());
383 i++;
384 notifyStatusUpdate?.Invoke($"Updating {Path.GetFileName(pccPath)}");
385 using var pack = MEPackageHandler.OpenMEPackage(pccPath);
386 bool shouldSave = false;
387 List<ExportEntry> wwiseStreamExports = pack.Exports.Where(x => x.ClassName == "WwiseStream").ToList();
388 foreach (ExportEntry exp in wwiseStreamExports)
389 {
390 // Check if this needs updated by finding it in the reference map
391 var wwiseStream = ObjectBinary.From<WwiseStream>(exp);
392 if (wwiseStream.IsPCCStored) continue; //Nothing to update here
393
394 var key = new ReferencedAudio()
395 { afcName = wwiseStream.Filename, audioSize = wwiseStream.DataSize, audioOffset = wwiseStream.DataOffset };
396
397 if (referenceMap.TryGetValue(key, out var newInfo))
398 {
399 //Write new filename
400 exp.WriteProperty(new NameProperty(newInfo.afcName, "FileName"));
401 byte[] newData = exp.Data;
402 // Write new offset
403 Buffer.BlockCopy(BitConverter.GetBytes((int)newInfo.audioOffset), 0, newData,
404 newData.Length - 4, 4); //update AFC audio offset
405 exp.Data = newData;
406
407 //don't mark for saving if the data didn't actually change (e.g. trying to compact a compacted AFC).
408 shouldSave |= exp.DataChanged;
409 }
410 }
411 if (shouldSave)
412 {
413 pack.Save();
414 }
415 }
416 #endregion
417
418 // write final afc
419 if (File.Exists(finalAfcPath))
420 File.Delete(finalAfcPath);
421 File.Move(tempAfcPath, finalAfcPath);
422 notifyFinalAfcPath?.Invoke(finalAfcPath);
423 return true;
424 }

426 private static Stream fetchAfcStream(string referencedAudioAfcName, AFCInventory inventory, Action<string> debugOut = null)
427 {
428 var fname = referencedAudioAfcName.ToLower() + ".afc";
429
430 var localAFCFile = inventory.LocalFolderAFCFiles.FirstOrDefault(x => Path.GetFileName(x).ToLower() == fname);
431
432 if (localAFCFile != null)
433 {
434 return File.OpenRead(localAFCFile);
435 }
436
437
438 var basegameAFCFile = inventory.BasegameAFCFiles.FirstOrDefault(x => Path.GetFileName(x).ToLower() == fname);
439 if (basegameAFCFile != null)
440 {
441 return File.OpenRead(basegameAFCFile);
442 }
443
444 if (inventory.SFARAFCsMap.Any())
445 {
446 var relevantDLC = inventory.SFARAFCsMap.FirstOrDefault(x => x.Value.Exists(y => Path.GetFileName(y).Equals(fname, StringComparison.InvariantCultureIgnoreCase)));
447 if (relevantDLC.Key != null)
448 {
449 DLCPackage dlc = new DLCPackage(Path.Combine(ME3Directory.DLCPath, relevantDLC.Key, "CookedPCConsole", "Default.sfar"));
450 return dlc.DecompressEntry(dlc.Files.First(x => Path.GetFileName(x.FileName).ToLower() == fname));
451 }
452 }
453
454 var officialDlcFile = inventory.OfficialDLCAFCFiles.FirstOrDefault(x => Path.GetFileName(x).ToLower() == fname);
455 if (officialDlcFile != null)
456 {
457 return File.OpenRead(officialDlcFile);
458 }
459
460 debugOut?.Invoke($"!!! Could not find AFC {referencedAudioAfcName} for compaction!");
461 // Could not find file! This shouldn't happen, technically...
462 return null;
463 }
464 }
