71 public static ArrayType getArrayType(string className, NameReference propName, ExportEntry export = null)
72 {
73 if (!IsLoaded) loadfromJSON();
74 PropertyInfo p = getPropertyInfo(className, propName, false, containingExport: export);
75 if (p == null)
76 {
77 p = getPropertyInfo(className, propName, true, containingExport: export);
78 }
79 if (p == null && export != null)
80 {
81 if (!export.IsClass && export.Class is ExportEntry classExport)
82 {
83 export = classExport;
84 }
85 if (export.IsClass)
86 {
87 ClassInfo currentInfo = generateClassInfo(export);
88 currentInfo.baseClass = export.SuperClassName;
89 p = getPropertyInfo(className, propName, false, currentInfo, containingExport: export);
90 if (p == null)
91 {
92 p = getPropertyInfo(className, propName, true, currentInfo, containingExport: export);
93 }
94 }
95 }
96 return getArrayType(p);

157 {
158 if (!IsLoaded) loadfromJSON();

161 className = className.Substring(9);

163 Dictionary<string, ClassInfo> temp = inStruct ? Structs : Classes;
164 ClassInfo info;
165 bool infoExists = temp.TryGetValue(className, out info);
166 if (!infoExists && nonVanillaClassInfo != null)

168 info = nonVanillaClassInfo;
169 infoExists = true;

171 if (infoExists) //|| (temp = !inStruct ? Structs : Classes).ContainsKey(className))
172 {
173 //look in class properties

178 //look in structs

184 {

186 if (val != null)

207 //Class parent is in this file. Generate class parent info and attempt refetch

221 public static bool inheritsFrom(IEntry entry, string baseClass)
222 {
223 if (!IsLoaded) loadfromJSON();
224 string className = entry.ClassName;
225 while (Classes.ContainsKey(className))
226 {
227 if (className == baseClass)
228 {
229 return true;
230 }
231 className = Classes[className].baseClass;
232 }
233 return false;
234 }

297 private static PropertyInfo getProperty(UDKPackage pcc, ExportEntry entry)
298 {
299 if (!IsLoaded) loadfromJSON();
300
301 string reference = null;
302 PropertyType type;
303 switch (entry.ClassName)
304 {
305 case "IntProperty":
306 type = PropertyType.IntProperty;
307 break;
308 case "StringRefProperty":
309 type = PropertyType.StringRefProperty;
310 break;
311 case "FloatProperty":
312 type = PropertyType.FloatProperty;
313 break;
314 case "BoolProperty":
315 type = PropertyType.BoolProperty;
316 break;
317 case "StrProperty":
318 type = PropertyType.StrProperty;
319 break;
320 case "NameProperty":
321 type = PropertyType.NameProperty;
322 break;
323 case "DelegateProperty":
324 type = PropertyType.DelegateProperty;
325 break;
326 case "ObjectProperty":
327 case "ClassProperty":
328 case "ComponentProperty":
329 type = PropertyType.ObjectProperty;
330 reference = pcc.getObjectName(EndianReader.ToInt32(entry.DataReadOnly, entry.DataSize - 4, entry.FileRef.Endian));
331 break;
332 case "StructProperty":
333 type = PropertyType.StructProperty;
334 reference = pcc.getObjectName(EndianReader.ToInt32(entry.DataReadOnly, entry.DataSize - 4, entry.FileRef.Endian));
335 break;
336 case "BioMask4Property":
337 case "ByteProperty":
338 type = PropertyType.ByteProperty;
339 reference = pcc.getObjectName(EndianReader.ToInt32(entry.DataReadOnly, entry.DataSize - 4, entry.FileRef.Endian));
340 break;
341 case "ArrayProperty":
342 type = PropertyType.ArrayProperty;
343 PropertyInfo arrayTypeProp = getProperty(pcc, pcc.Exports[EndianReader.ToInt32(entry.DataReadOnly, 44, entry.FileRef.Endian) - 1]);
344 if (arrayTypeProp != null)
345 {
346 switch (arrayTypeProp.Type)
347 {
348 case PropertyType.ObjectProperty:
349 case PropertyType.StructProperty:
350 case PropertyType.ArrayProperty:
351 reference = arrayTypeProp.Reference;
352 break;
353 case PropertyType.ByteProperty:
354 if (arrayTypeProp.Reference == "Class")
355 reference = arrayTypeProp.Type.ToString();
356 else
357 reference = arrayTypeProp.Reference;
358 break;
359 case PropertyType.IntProperty:
360 case PropertyType.FloatProperty:
361 case PropertyType.NameProperty:
362 case PropertyType.BoolProperty:
363 case PropertyType.StrProperty:
364 case PropertyType.StringRefProperty:
365 case PropertyType.DelegateProperty:
366 reference = arrayTypeProp.Type.ToString();
367 break;
368 case PropertyType.None:
369 case PropertyType.Unknown:
370 default:
371 System.Diagnostics.Debugger.Break();
372 return null;
373 }
374 }
375 else
376 {
377 return null;
378 }
379 break;
380 case "InterfaceProperty":
381 default:
382 return null;
383 }
384
385 bool transient = (EndianReader.ToUInt64(entry.DataReadOnly, 24, entry.FileRef.Endian) & 0x0000000000002000) != 0;
386 return new PropertyInfo(type, reference, transient);
387 }
