178 public static void Convert(CoalescedType SourceType, string SourcePath, string DestinationPath)
179 {
180 if (!Path.IsPathRooted(SourcePath))
181 {
182 SourcePath = Path.GetFullPath(SourcePath);
183 }
184
185 if (!Path.IsPathRooted(DestinationPath))
186 {
187 DestinationPath = Path.GetFullPath(DestinationPath);
188 }
189
190 if (!File.Exists(SourcePath))
191 {
192 throw new FileNotFoundException("Source file not found.");
193 }
194
195 switch (SourceType)
196 {
197
198 case CoalescedType.Binary:
199 var ConvertingLECoalesced = !CoalescedConverter.IsGame3Coalesced(SourcePath);
200 if (!Directory.Exists(Path.GetDirectoryName(DestinationPath) ?? DestinationPath))
201 {
202 Directory.CreateDirectory(DestinationPath);
203 }
204
205 if (ConvertingLECoalesced)
206 {
207 LECoalescedConverter.Unpack(SourcePath, DestinationPath);
208 }
209 else
210 {
211 CoalescedConverter.ConvertToXML(SourcePath, DestinationPath);
212 }
213 break;
214 case CoalescedType.Xml:
215 CoalescedConverter.ConvertToBin(SourcePath, DestinationPath);
216 break;
217 case CoalescedType.ExtractedBin:
218 var containingFolder = Path.GetDirectoryName(SourcePath);
219 LECoalescedConverter.Pack(containingFolder, DestinationPath);
220 break;
221 default:
222 throw new ArgumentOutOfRangeException();
223 }

487 public static void ConvertToBin(string source, string destination, XDocument preloadedDoc = null)
488 {
489 var inputPath = Path.IsPathRooted(source) ? source : Path.Combine(GetExePath(), source);
490 var outputPath = !string.IsNullOrEmpty(destination) ? destination : Path.ChangeExtension(inputPath, ".bin");
491
492 if (!Path.IsPathRooted(outputPath))
493 {
494 outputPath = Path.Combine(GetExePath(), outputPath);
495 }
496
497 if (!File.Exists(inputPath))
498 {
499 return;
500 }
501
502 if (preloadedDoc == null) preloadedDoc = XDocument.Load(source);
503 var file = XmlCoalesceFile.LoadXmlDocument(inputPath, preloadedDoc);
504
505 var coal = new CoalescedFileXml
506 {
507 ByteOrder = ByteOrder.LittleEndian,
508 Version = 1
509 };
510
511 foreach (var asset in file.Assets)
512 {
513 var entry =
514 new FileEntry(asset.Source)
515 {
516 Sections = new Dictionary<string, Dictionary<string, List<PropertyValue>>>()
517 };
518
519 foreach (var section in asset.Sections)
520 {
521 var eSection = new Dictionary<string, List<PropertyValue>>();
522
523 foreach (var property in section.Value)
524 {
525 var eProperty = new List<PropertyValue>();
526
527 foreach (var value in property.Value)
528 {
529 if (!file.Settings.CompileTypes.Contains(value.ValueType))
530 {
531 continue;
532 }
533
534 var valueValue = value.Value;
535
536 if (!string.IsNullOrEmpty(valueValue))
537 {
538 valueValue = SpecialCharacters.Aggregate(valueValue, (current, c) => current.Replace(c.Value, c.Key));
539 }
540
541 eProperty.Add(new PropertyValue(value.ValueType, valueValue));
542 }
543
544 eSection.Add(property.Key, eProperty);
545 }
546
547 entry.Sections.Add(section.Key, eSection);
548 }
549
550 coal.Files.Add(entry);
551 }
552
553 using (var output = File.Create(outputPath))
554 {
555 if (file.Settings != null)
556 {
557 coal.OverrideCompileValueTypes = file.Settings.OverrideCompileValueTypes;
558 coal.CompileTypes = file.Settings.CompileTypes;
559 }
560
561 coal.Serialize(output);
562 }
