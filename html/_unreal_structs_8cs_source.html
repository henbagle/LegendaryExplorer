128 public static Rotator FromDirectionVector(Vector3 dirVec)
129 {
130 float x = dirVec.X;
131 float y = dirVec.Y;
132 float z = dirVec.Z;
133 var pitch = Math.Atan2(z, Math.Sqrt(Math.Pow(x, 2) + Math.Pow(y, 2)));
134 var yaw = Math.Atan2(y, x);
135 return new Rotator(((float)pitch).RadiansToUnrealRotationUnits(), ((float)yaw).RadiansToUnrealRotationUnits(), 0);
136 }
137
138 public Vector3 GetDirectionalVector()
139 {
140 var cp = Math.Cos(Pitch.UnrealRotationUnitsToRadians());
141 var cy = Math.Cos(Yaw.UnrealRotationUnitsToRadians());
142 var sp = Math.Sin(Pitch.UnrealRotationUnitsToRadians());
143 var sy = Math.Sin(Yaw.UnrealRotationUnitsToRadians());
144 return new Vector3((float)(cp * cy), (float)(cp * sy), (float)sp);
145 }
146
147 public bool IsZero => Pitch == 0 && Yaw == 0 && Roll == 0;
148 public override string ToString()
149 {
150 return $"Pitch:{Pitch} Yaw:{Yaw} Roll:{Roll}";
