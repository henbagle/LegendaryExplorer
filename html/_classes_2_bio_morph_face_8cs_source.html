45 private SkeletalMesh LoadMeshFromEntry(IEntry mOBaseHead)
46 {
47 if (mOBaseHead is null) return null;

65 public SkeletalMesh Apply()
66 {
67 // apply vertices morph first
68 // in skeletalMesh, we load only LOD0, so we only apply for lod0
69 for (int lod = 0; lod < 1; lod++)
70 {
71 for (int v=0; v < BaseHead.LODModels[lod].VertexBufferGPUSkin.VertexData.Length; v++)
72 {
73 var vertex = BaseHead.LODModels[lod].VertexBufferGPUSkin.VertexData[v];
74 vertex.Position.X = MorphFace.LODs[lod][v].X;
75 vertex.Position.Y = MorphFace.LODs[lod][v].Y;
76 vertex.Position.Z = MorphFace.LODs[lod][v].Z;
77 BaseHead.LODModels[lod].VertexBufferGPUSkin.VertexData[v] = vertex;
78 }
79 }
80
81 // return mesh
82 return BaseHead;
83 }
