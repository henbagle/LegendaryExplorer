81 public static void RegisterUnknownBytecode(byte b, UnBytecodeOwner function, byte[] subsequentBytes)
82 {
83 if (subsequentBytes.Length < 4) return; // most likely bogus bytes after 'return'
84 if (!_unknownBytecodes.TryGetValue(b, out List<UnknownBytecodeOccurrence> occurrences))

92 public static void DumpUnknownBytecodes(StreamWriter writer)
93 {
94 foreach (byte b in _unknownBytecodes.Keys)
95 {
96 writer.WriteLine(b.ToString("X2"));
97 var occurrences = _unknownBytecodes[b];
98 foreach (var occurrence in occurrences)
99 {
100 var function = occurrence.Function;
101 writer.Write(" ");
102 WriteFunctionName(writer, function);
103
104 var bytes = occurrence.SubsequentBytes;
105 int skip = 0;
106 if (b == 0x0E)
107 {
108 skip = DumpNextBytecodes(writer, function, bytes, 4, 1);
109 }
110 writer.WriteLine($" {BytecodeReader.DumpBytes(bytes, skip, 32)}");
111 }
112 }
