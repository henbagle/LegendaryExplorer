215 public static byte[] Compress(ReadOnlySpan<byte> buffer, int size, OodleFormat format, OodleCompressionLevel level)
216 {
217 int compressedBufferSize = GetCompressionBound(size);
218 byte[] compressedBuffer = MemoryManager.GetByteArray(compressedBufferSize); // we will not use all of this. someday we will want to improve this i think
219
220 int compressedCount;
221 unsafe
222 {
223 fixed (byte* ptr = &MemoryMarshal.GetReference(buffer))
224 {
225 compressedCount = OodleLZ_Compress(format, Unsafe.AsRef<byte>(ptr), buffer.Length, compressedBuffer, level, 0, 0, 0, 0, 0, 0);
226 }
227 }
228
229
230 byte[] outputBuffer = new byte[compressedCount];
231 Buffer.BlockCopy(compressedBuffer, 0, outputBuffer, 0, compressedCount);
232
233 MemoryManager.ReturnByteArray(compressedBuffer);
234 return outputBuffer;
235 }
