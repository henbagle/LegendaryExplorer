484 private void BinScrubCheck(ObjectBinary objBin1, ObjectBinary objBin2, ExportEntry exp1, ExportEntry exp2)
485 {
486 bool isDebug = false;
487 //bool isDebug = exp1.UIndex == 1682;
488 //if (!isDebug) return;
489 EndianReader er1 = new EndianReader(new MemoryStream());
490 objBin1.WriteTo(er1.Writer, exp1.FileRef);
491 var originalWrittenBin1 = er1.ToArray();
492
493 EndianReader er2 = new EndianReader(new MemoryStream());
494 objBin2.WriteTo(er2.Writer, exp2.FileRef);
495 var originalWrittenBin2 = er2.ToArray();
496
497 // Package-Specific Variables
498 var names1 = objBin1.GetNames(exp1.FileRef.Game);
499 var names2 = objBin2.GetNames(exp2.FileRef.Game);
500 var uindices1 = objBin1.GetUIndexes(exp1.FileRef.Game);
501 var uindices2 = objBin2.GetUIndexes(exp2.FileRef.Game);
502
503 // Now we iterate through written binary looking for values
504 for (int pos = 0; pos < originalWrittenBin1.Length; pos++)
505 {
506 if (isDebug && pos == 0x171)
507 Debug.Write("");
508
509 if (pos <= originalWrittenBin1.Length - 8 && pos <= originalWrittenBin2.Length - 8)
510 {
511 // Check name
512 var nameIdx = BitConverter.ToInt32(originalWrittenBin1, pos);
513 if (nameIdx != 0 && exp1.FileRef.IsName(nameIdx))
514 {
515
516 var name1 = exp1.FileRef.GetNameEntry(nameIdx);
517 //if (isDebug)
518 // Debugger.Break();
519 //var name2 = exp2.FileRef.GetNameEntry(BitConverter.ToInt32(originalWrittenBin2, pos));
520
521 var index1 = BitConverter.ToInt32(originalWrittenBin1, pos + 4);
522 //var index2 = BitConverter.ToInt32(originalWrittenBin1, pos + 4);
523
524 var generatedName = new NameReference(name1, index1);
525 if (isDebug)
526 Debug.Write("");
527 if (names1.Any(x => x.Item1 == generatedName))
528 {
529 // SCRUB
530 if (isDebug)
531 Debug.WriteLine($"Scrubbing N {generatedName.Instanced} at 0x{pos:X4}");
532 originalWrittenBin1.OverwriteRange(pos, new byte[8]);
533 originalWrittenBin2.OverwriteRange(pos, new byte[8]);
534 pos += 7; // will +1 on loop
535 continue;
536 }
537 }
538 }
539
540 if (pos <= originalWrittenBin1.Length - 4 && pos <= originalWrittenBin2.Length - 4)
541 {
542 // Check UIndex
543 var uindex1 = BitConverter.ToInt32(originalWrittenBin1, pos);
544 //var name2 = exp2.FileRef.GetNameEntry(BitConverter.ToInt32(originalWrittenBin2, pos));
545
546 if (uindex1 != 0 && uindices1.Any(x => x.Item1.value == uindex1))
547 {
548 // SCRUB
549 //if (isDebug)
550 // Debug.WriteLine($"Scrubbing U {uindex1} at 0x{pos:X4}");
551 originalWrittenBin1.OverwriteRange(pos, new byte[4]);
552 originalWrittenBin2.OverwriteRange(pos, new byte[4]);
553 pos += 3; // will +1 on loop
554 }
555 }
556 }
557
558 long binStopPosition = -1;
559 if (objBin1 is UScriptStruct uss1 && objBin2 is UScriptStruct uss2)
560 {
561 MatchPropertyCollections(uss1.Defaults, uss2.Defaults, exp1, exp2);

564
565 // Find difference
566 int binStart = exp1.propsEnd();
567 for (int i = 0; i < originalWrittenBin1.Length; i++)
568 {
569 if (binStopPosition != -1 && i >= binStopPosition)

571 if (originalWrittenBin1[i] != originalWrittenBin2[i])
572 {
573 Debug.WriteLine($"BINARY DIFF ON {exp1.UIndex} {exp1.InstancedFullPath} at 0x{(binStart + i):X5} ({i:X5}): Left: {originalWrittenBin1[i]:X2} Right {originalWrittenBin2[i]:X2}");
574 if (!isDebug)
575 break;
576 }

587 private void MatchProperty(Property dProp, Property pProp, ExportEntry exp1, ExportEntry exp2)
588 {
589 if (dProp.GetType() != pProp.GetType())
590 {
591 Debug.WriteLine("NON-MATCHING PROPERTY TYPES");
592 return;
593 }
594
595 if (dProp.Name != pProp.Name)
596 {
597 Debug.WriteLine(@"NON-MATCHING PROPERTY NAMES!");
598 }
599
600 if (dProp is ObjectProperty dOp && pProp is ObjectProperty pOp)
601 {
602 if (dOp.Value == 0 && pOp.Value == dOp.Value)
603 {
604 // Zero, same
605 }
606 else
607 {
608 // Ensure they resolve to same variable
609 var d = dOp.ResolveToEntry(exp1.FileRef);
610 var p = pOp.ResolveToEntry(exp2.FileRef);
611
612 if (d.GetType() != p.GetType())
613 {
614 Debug.WriteLine($@"ObjectProp referenced object types differ on {exp1.UIndex} {exp1.InstancedFullPath}! Left: {d.GetType()} Right: {p.GetType()}");
615 }
616
617 if (d.InstancedFullPath != p.InstancedFullPath)
618 {
619 Debug.WriteLine($@"Referenced ObjectProperty value differs on {exp1.UIndex} {exp1.InstancedFullPath}! Left: {d.InstancedFullPath} Right: {p.InstancedFullPath}");
620 }
621 }
622 }
623 else if (dProp is ArrayPropertyBase dArrayP && pProp is ArrayPropertyBase pArrayP)
624 {
625 if (dArrayP.Count != pArrayP.Count)
626 {
627 Debug.WriteLine($"Different sized arrays on {exp1.UIndex} {exp1.InstancedFullPath}!");
628 }
629
630 for (int i = 0; i < dArrayP.Properties.Count && i < pArrayP.Properties.Count; i++)
631 {
632 MatchProperty(dArrayP.Properties[i], pArrayP.Properties[i], exp1, exp2);
633 }
634 }
635 else if (dProp is StructProperty dStructP && pProp is StructProperty pStructP)
636 {
637 if (dStructP.Properties.Count != pStructP.Properties.Count)
638 {
639 Debug.WriteLine($"Different sized Structs on {exp1.UIndex} {exp1.InstancedFullPath}!");
640 }
641
642 for (int i = 0; i < dStructP.Properties.Count && i < pStructP.Properties.Count; i++)
643 {
644 MatchProperty(dStructP.Properties[i], pStructP.Properties[i], exp1, exp2);
645 }
646 }
647 else if (dProp is DelegateProperty dDelP && pProp is DelegateProperty pDelP)
648 {
649 // Ensure they resolve to same variable
650 var d = exp1.FileRef.GetEntry(dDelP.Value.Object);
651 var p = exp2.FileRef.GetEntry(pDelP.Value.Object);
652
653 if (d != null && p != null)
654 {
655 if (d.GetType() != p.GetType())
656 {
657 Debug.WriteLine(@"DelegateProperty ObjectProp types differ!");
658 }
659
660 if (d.InstancedFullPath != p.InstancedFullPath)
661 {
662 Debug.WriteLine(@"Referenced DelegateProperty value differs!");
663 }
664 }
665 else if (d != p)
666 {
667 Debug.WriteLine("XOR'd script delegates!");
668 }
669 }
670 else if (dProp is BoolProperty dBool && pProp is BoolProperty pBool)
671 {
672 if (dBool.Value != pBool.Value)
673 {
674 Debug.WriteLine($"Bool difference on {exp1.UIndex} {exp1.InstancedFullPath}: Left: {dBool.Value} Right {pBool.Value}");
675 }
676 }
677 else if (dProp is StrProperty dStr && pProp is StrProperty pStr)
678 {
679 if (dStr.Value != pStr.Value)
680 {
681 Debug.WriteLine($"Str difference on {exp1.UIndex} {exp1.InstancedFullPath}: Left: {dStr.Value} Right {pStr.Value}");
682 }
683 }
684 else if (dProp is FloatProperty dFloat && pProp is FloatProperty pFloat)
685 {
686 if (dFloat.Value != pFloat.Value)
687 {
688 Debug.WriteLine($"Float difference on {exp1.UIndex} {exp1.InstancedFullPath}: Left: {dFloat.Value} Right {pFloat.Value}");
689 }
690 }
691 else if (dProp is IntProperty dInt && pProp is IntProperty pInt)
692 {
693 // DEBUG: Ignore GUIDs as they seem to differ across files
694 if (dProp.Name == "A" || dProp.Name == "B" || dProp.Name == "C" || dProp.Name == "D")
695 return;
696 if (dInt.Value != pInt.Value)
697 {
698 Debug.WriteLine($"Int difference in {exp1.UIndex} {exp1.InstancedFullPath}: Left: {dInt.Value} Right {pInt.Value}");
699 }
700 }
701 else if (dProp is NameProperty dName && pProp is NameProperty pName)
702 {
703 if (dName.Value.Name != pName.Value.Name || dName.Value.Number != pName.Value.Number)
704 {
705 Debug.WriteLine($"Name difference on {exp1.UIndex} {exp1.InstancedFullPath}: Left: {dName.Value.Instanced} Right {pName.Value.Instanced}");
706 }
707 }
708 else if (dProp is EnumProperty dEnum && pProp is EnumProperty pEnum)
709 {
710 if (dEnum.EnumType != pEnum.EnumType)
711 {
