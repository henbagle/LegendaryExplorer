182 public virtual List<(UIndex, string)> GetUIndexes(MEGame game)
183 {
184 List<(UIndex uIndex, string)> uIndexes = TextureDependencyLengthMap.Keys().Select((uIndex, i) => (uIndex, $"TextureDependencyLengthMap[{i}]")).ToList();
185 if (game >= MEGame.ME3)
186 {
187 uIndexes.AddRange(UniformExpressionTextures.Select((uIndex, i) => (uIndex, $"UniformExpressionTextures[{i}]")));
188 }
189 else
190 {
191 uIndexes.AddRange(UniformPixelVectorExpressions.OfType<MaterialUniformExpressionFlipbookParameter>()
192 .Select((flipParam, i) => (flipParam.TextureIndex, $"UniformPixelVectorExpressions[{i}]")));
193
194 // Used in ME2 Carnage fireballPlasma Material
195 uIndexes.AddRange(UniformPixelVectorExpressions.OfType<MaterialUniformExpressionTexture>()
196 .Select((flipParam, i) => (flipParam.TextureIndex, $"UniformPixelVectorExpressions[{i}]")));
197
198 uIndexes.AddRange(UniformPixelScalarExpressions.OfType<MaterialUniformExpressionFlipbookParameter>()
199 .Select((flipParam, i) => (flipParam.TextureIndex, $"UniformPixelScalarExpressions[{i}]")));
200 uIndexes.AddRange(Uniform2DTextureExpressions.Select((texParam, i) => (texParam.TextureIndex, $"Uniform2DTextureExpressions[{i}]")));
201 uIndexes.AddRange(UniformCubeTextureExpressions.Select((texParam, i) => (texParam.TextureIndex, $"UniformCubeTextureExpressions[{i}]")));
202 if (game == MEGame.ME1)
203 {
204 int j = 0;
205 foreach (ME1MaterialUniformExpressionsElement expressionsElement in Me1MaterialUniformExpressionsList)
206 {
207 uIndexes.AddRange(expressionsElement.UniformPixelVectorExpressions.OfType<MaterialUniformExpressionFlipbookParameter>()
208 .Select((flipParam, i) => (flipParam.TextureIndex, $"MaterialUniformExpressions[{j}].UniformPixelVectorExpressions[{i}]")));
209 uIndexes.AddRange(expressionsElement.UniformPixelScalarExpressions.OfType<MaterialUniformExpressionFlipbookParameter>()
210 .Select((flipParam, i) => (flipParam.TextureIndex, $"MaterialUniformExpressions[{j}].UniformPixelScalarExpressions[{i}]")));
211 uIndexes.AddRange(expressionsElement.Uniform2DTextureExpressions.Select((texParam, i) => (texParam.TextureIndex, $"MaterialUniformExpressions[{j}].Uniform2DTextureExpressions[{i}]")));
212 uIndexes.AddRange(expressionsElement.UniformCubeTextureExpressions.Select((texParam, i) => (texParam.TextureIndex, $"MaterialUniformExpressions[{j}].UniformCubeTextureExpressions[{i}]")));
213 ++j;
214 }
215 }
216 }
217 return uIndexes;

220 public List<(NameReference, string)> GetNames(MEGame game)
221 {
222 var names = new List<(NameReference, string)>();
223
224 if (game <= MEGame.ME2)
225 {
226 var uniformExpressionArrays = new List<(string, MaterialUniformExpression[])>
227 {
228 (nameof(UniformPixelVectorExpressions), UniformPixelVectorExpressions),
229 (nameof(UniformPixelScalarExpressions), UniformPixelScalarExpressions),
230 (nameof(Uniform2DTextureExpressions), Uniform2DTextureExpressions),
231 (nameof(UniformCubeTextureExpressions), UniformCubeTextureExpressions),
232 };
233 if (game == MEGame.ME1)
234 {
235 int j = 0;
236 foreach (ME1MaterialUniformExpressionsElement expressionsElement in Me1MaterialUniformExpressionsList)
237 {
238 uniformExpressionArrays.Add(($"MaterialUniformExpressions[{j}].UniformPixelVectorExpressions", expressionsElement.UniformPixelVectorExpressions));
239 uniformExpressionArrays.Add(($"MaterialUniformExpressions[{j}].UniformPixelScalarExpressions", expressionsElement.UniformPixelScalarExpressions));
240 uniformExpressionArrays.Add(($"MaterialUniformExpressions[{j}].Uniform2DTextureExpressions", expressionsElement.Uniform2DTextureExpressions));
241 uniformExpressionArrays.Add(($"MaterialUniformExpressions[{j}].UniformCubeTextureExpressions", expressionsElement.UniformCubeTextureExpressions));
242 ++j;
243 }
244 }
245
246 foreach ((string prefix, MaterialUniformExpression[] expressions) in uniformExpressionArrays)
247 {
248 for (int i = 0; i < expressions.Length; i++)
249 {
250 MaterialUniformExpression expression = expressions[i];
251 names.Add((expression.ExpressionType, $"{prefix}[{i}].ExpressionType"));
252 switch (expression)
253 {
254 case MaterialUniformExpressionTextureParameter texParamExpression:
255 names.Add((texParamExpression.ParameterName, $"{prefix}[{i}].ParameterName"));
256 break;
257 case MaterialUniformExpressionScalarParameter scalarParameterExpression:
258 names.Add((scalarParameterExpression.ParameterName, $"{prefix}[{i}].ParameterName"));
259 break;
260 case MaterialUniformExpressionVectorParameter vecParameterExpression:
261 names.Add((vecParameterExpression.ParameterName, $"{prefix}[{i}].ParameterName"));
262 break;
263 }
264
265 names.AddRange(expression.GetNames(game));
266 }
267 }
268 }
269
270 return names;
271 }

380 public bool Equals(StaticParameterSet other)
381 {
382 if (other is null || other.BaseMaterialId != BaseMaterialId || other.StaticSwitchParameters.Length != StaticSwitchParameters.Length ||
383 other.StaticComponentMaskParameters.Length != StaticComponentMaskParameters.Length || other.NormalParameters.Length != NormalParameters.Length)
384 {
385 return false;
386 }
387 //bOverride is intentionally left out of the following comparisons
388 for (int i = 0; i < StaticSwitchParameters.Length; i++)
389 {
390 var a = StaticSwitchParameters[i];
391 var b = other.StaticSwitchParameters[i];
392 if (a.ParameterName != b.ParameterName || a.ExpressionGUID != b.ExpressionGUID || a.Value != b.Value)
393 {
394 return false;
395 }
396 }
397 for (int i = 0; i < StaticComponentMaskParameters.Length; i++)
398 {
399 var a = StaticComponentMaskParameters[i];
400 var b = other.StaticComponentMaskParameters[i];
401 if (a.ParameterName != b.ParameterName || a.ExpressionGUID != b.ExpressionGUID || a.R != b.R || a.G != b.G || a.B != b.B || a.A != b.A)
402 {
403 return false;
404 }
405 }
406 for (int i = 0; i < NormalParameters.Length; i++)
407 {
408 var a = NormalParameters[i];
409 var b = other.NormalParameters[i];
410 if (a.ParameterName != b.ParameterName || a.ExpressionGUID != b.ExpressionGUID || a.CompressionSettings != b.CompressionSettings)
411 {
412 return false;
413 }
414 }
415
416 return true;
417 }
418
419 public override bool Equals(object obj)
420 {
421 if (obj is null) return false;
422 if (ReferenceEquals(this, obj)) return true;

503 public static MaterialUniformExpression Create(SerializingContainer2 sc)
504 {
505 NameReference expressionType = sc.ms.ReadNameReference(sc.Pcc);
506 sc.ms.Skip(-8);//ExpressionType will be read again during serialization, so back the stream up.
507 switch (expressionType.Name)
508 {
509 case "FMaterialUniformExpressionAbs":
510 case "FMaterialUniformExpressionCeil":
511 case "FMaterialUniformExpressionFloor":
512 case "FMaterialUniformExpressionFrac":
513 case "FMaterialUniformExpressionPeriodic":
514 case "FMaterialUniformExpressionSquareRoot":
515 return new MaterialUniformExpressionUnaryOp();
516 case "FMaterialUniformExpressionAppendVector":
517 return new MaterialUniformExpressionAppendVector();
518 case "FMaterialUniformExpressionClamp":
519 return new MaterialUniformExpressionClamp();
520 case "FMaterialUniformExpressionConstant":
521 return new MaterialUniformExpressionConstant();
522 case "FMaterialUniformExpressionFmod":
523 case "FMaterialUniformExpressionMax":
524 case "FMaterialUniformExpressionMin":
525 return new MaterialUniformExpressionBinaryOp();
526 case "FMaterialUniformExpressionFoldedMath":
527 return new MaterialUniformExpressionFoldedMath();
528 case "FMaterialUniformExpressionTime":
529 case "FMaterialUniformExpressionRealTime":
530 case "FMaterialUniformExpressionFractionOfEffectEnabled":
531 return new MaterialUniformExpression();
532 case "FMaterialUniformExpressionScalarParameter":
533 return new MaterialUniformExpressionScalarParameter();
534 case "FMaterialUniformExpressionSine":
535 return new MaterialUniformExpressionSine();
536 case "FMaterialUniformExpressionTexture":
537 case "FMaterialUniformExpressionFlipBookTextureParameter":
538 return new MaterialUniformExpressionTexture();
539 case "FMaterialUniformExpressionTextureParameter":
540 return new MaterialUniformExpressionTextureParameter();
541 case "FMaterialUniformExpressionVectorParameter":
542 return new MaterialUniformExpressionVectorParameter();
543 case "FMaterialUniformExpressionFlipbookParameter":
544 return new MaterialUniformExpressionFlipbookParameter();
545 default:
546 throw new ArgumentException(expressionType.Instanced);
547 }
548 }
