203 public static void CheckReferences(ReferenceCheckPackage item, string basePath, GetLocalizedStringDelegate localizationDelegate, Action<string> statusUpdateDelegate, Action<string> logMessageDelegate = null, List<string> referencedFiles = null, CancellationTokenSource cts = null)
204 {
205 referencedFiles ??= Directory.GetFiles(basePath, "*.*", SearchOption.AllDirectories).Where(x => x.RepresentsPackageFilePath()).ToList();
206 int numChecked = 0;
207 Parallel.ForEach(referencedFiles,
208 new ParallelOptions()
209 {
210 MaxDegreeOfParallelism = Math.Min(3, Environment.ProcessorCount)
211 },
212 f =>
213 {
214 if (cts != null && cts.IsCancellationRequested)
215 return;
216 //if (!f.Contains("BioA_Nor")) return;
217 var lnumChecked = Interlocked.Increment(ref numChecked);
218 statusUpdateDelegate?.Invoke(localizationDelegate(ME3XL.string_checkingNameAndObjectReferences) + $@" [{lnumChecked - 1}/{referencedFiles.Count}]");
219
220 var relativePath = f.Substring(basePath.Length + 1);
221 logMessageDelegate?.Invoke($@"Checking package and name references in {relativePath}");
222 var package = MEPackageHandler.OpenMEPackage(f, forceLoadFromDisk: true);
223 CheckReferences(item, package, localizationDelegate, relativePath, cts);
224 });

227 public static void CheckReferences(ReferenceCheckPackage item, IMEPackage package, GetLocalizedStringDelegate localizationDelegate, string relativePath = null, CancellationTokenSource cts = null)
228 {
229 string fName = Path.GetFileName(package.FilePath);
230 foreach (ExportEntry exp in package.Exports)
231 {
232 if (cts != null && cts.IsCancellationRequested)
233 return;
234
235 // Has to be done before accessing the name because it will cause infinite crash loop
236 //Debug.WriteLine($"Checking {exp.UIndex} {exp.InstancedFullPath} in {exp.FileRef.FilePath}");
237 if (exp.idxLink == exp.UIndex)
238 {
239 item.AddBlockingError(localizationDelegate(ME3XL.string_interp_fatalExportCircularReference, relativePath ?? fName, exp.UIndex));
240 continue;
241 }
242
243 var prefix = localizationDelegate(ME3XL.string_interp_warningGenericExportPrefix, relativePath ?? fName, exp.UIndex, exp.ObjectName.Name, exp.ClassName);
244 try
245 {
246 checkName(item, localizationDelegate, () => exp.ObjectName, "Object Name", $"export {exp.UIndex}", relativePath, fName, exp);
247
248 if (exp.idxArchetype != 0 && !package.IsEntry(exp.idxArchetype))
249 {
250 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningArchetypeOutsideTables, prefix, exp.idxArchetype), exp);
251 }
252
253 if (exp.idxSuperClass != 0 && !package.IsEntry(exp.idxSuperClass))
254 {
255 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningSuperclassOutsideTables, prefix, exp.idxSuperClass), exp);
256 }
257
258 if (exp.idxClass != 0 && !package.IsEntry(exp.idxClass))
259 {
260 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningClassOutsideTables, prefix, exp.idxClass), exp);
261 }
262
263 if (exp.idxLink != 0 && !package.IsEntry(exp.idxLink))
264 {
265 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningLinkOutsideTables, prefix, exp.idxLink), exp);
266 }
267
268
269
270 if (exp.HasComponentMap)
271 {
272 foreach (var c in exp.ComponentMap)
273 {
274 if (c.Value != 0 && !package.IsEntry(c.Value))
275 {
276 // Can components point to 0? I don't think so
277 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningComponentMapItemOutsideTables, prefix, c.Value), exp);
278 }
279 }
280 }
281
282 //find stack references
283 if (exp.HasStack && exp.DataReadOnly is var data)
284 {
285 var stack1 = EndianReader.ToInt32(data, 0, exp.FileRef.Endian);
286 var stack2 = EndianReader.ToInt32(data, 4, exp.FileRef.Endian);
287 if (stack1 != 0 && !package.IsEntry(stack1))
288 {
289 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningExportStackElementOutsideTables, prefix, 0, stack1), exp);
290 }
291
292 if (stack2 != 0 && !package.IsEntry(stack2))
293 {
294 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningExportStackElementOutsideTables, prefix, 1, stack2), exp);
295 }
296 }
297 else if (exp.TemplateOwnerClassIdx is var toci && toci >= 0)
298 {
299 var TemplateOwnerClassIdx = EndianReader.ToInt32(exp.DataReadOnly, toci, exp.FileRef.Endian);
300 if (TemplateOwnerClassIdx != 0 && !package.IsEntry(TemplateOwnerClassIdx))
301 {
302 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningTemplateOwnerClassOutsideTables, prefix, toci.ToString(@"X6"), TemplateOwnerClassIdx), exp);
303 }
304 }
305
306 var props = exp.GetProperties();
307 foreach (var p in props)
308 {
309 recursiveCheckProperty(item, localizationDelegate, relativePath, exp.ClassName, exp, p);
310 }
311 }
312 catch (Exception e)
313 {
314 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningExceptionParsingProperties, prefix, e.Message), exp);
315 continue;
316 }
317
318 //find binary references
319 try
320 {
321 if (!exp.IsDefaultObject && ObjectBinary.From(exp) is ObjectBinary objBin)
322 {
323 List<(UIndex, string)> indices = objBin.GetUIndexes(exp.FileRef.Game);
324 foreach ((UIndex uIndex, string propName) in indices)
325 {
326 if (uIndex.value != 0 && !exp.FileRef.IsEntry(uIndex.value))
327 {
328 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningBinaryReferenceOutsideTables, prefix, uIndex.value), exp);
329 }
330 else if (exp.FileRef.GetEntry(uIndex.value)?.ObjectName.ToString() == @"Trash")
331 {
332 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningBinaryReferenceTrashed, prefix, uIndex.value), exp);
333 }
334 else if (exp.FileRef.GetEntry(uIndex.value)?.ObjectName.ToString() == @"ME3ExplorerTrashPackage")
335 {
336 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningBinaryReferenceTrashed, prefix, uIndex.value), exp);
337 }
338 }
339
340 var nameIndicies = objBin.GetNames(exp.FileRef.Game);
341 foreach (var ni in nameIndicies)
342 {
343 if (ni.Item1 == "")
344 {
345 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningBinaryNameReferenceOutsideNameTable, prefix), exp);
346 }
347 }
348 }
349 }
350 catch (Exception e) /* when (!App.IsDebug)*/
351 {
352 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningUnableToParseBinary, prefix, e.Message), exp);
353 }
354 }
355
356 foreach (ImportEntry imp in package.Imports)
357 {
358 if (imp.idxLink != 0 && !package.TryGetEntry(imp.idxLink, out _))
359 {
360 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningImportLinkOutideOfTables, relativePath ?? fName, imp.UIndex, imp.idxLink), imp);
361 }
362 else if (imp.idxLink == imp.UIndex)
363 {
364 item.AddBlockingError(localizationDelegate(ME3XL.string_interp_fatalImportCircularReference, relativePath ?? fName, imp.UIndex), imp);
365 }
366
367 // Values check
368 checkName(item, localizationDelegate, () => imp.PackageFile, "Package file", $"import {imp.UIndex}", relativePath, fName, imp);
369 checkName(item, localizationDelegate, () => imp.ClassName, "Class name", $"import {imp.UIndex}", relativePath, fName, imp);
370 }
371 }

388 private static void recursiveCheckProperty(ReferenceCheckPackage item, GetLocalizedStringDelegate localizationDelegate, string relativePath, string containingClassOrStructName, IEntry entry, Property property)
389 {
390 var prefix = localizationDelegate(ME3XL.string_interp_warningPropertyTypingWrongPrefix, relativePath, entry.UIndex, entry.ObjectName.Name, entry.ClassName, property.StartOffset.ToString(@"X6"));
391 if (property is UnknownProperty up)
392 {
393 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningFoundBrokenPropertyData, prefix), entry);
394
395 }
396 else if (property is ObjectProperty op)
397 {
398 bool validRef = true;
399 if (op.Value > 0 && op.Value > entry.FileRef.ExportCount)
400 {
401 //bad
402 if (op.Name.Name != null)
403 {
404 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningReferenceNotInExportTable, prefix, op.Name.Name, op.Value), entry);
405 validRef = false;
406 }
407 else
408 {
409 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_nested_warningReferenceNoInExportTable, prefix, op.Value), entry);
410 validRef = false;
411 }
412 }
413 else if (op.Value < 0 && Math.Abs(op.Value) > entry.FileRef.ImportCount)
414 {
415 //bad
416 if (op.Name.Name != null)
417 {
418 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningReferenceNotInImportTable, prefix, op.Name.Name, op.Value), entry);
419 validRef = false;
420
421 }
422 else
423 {
424 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_nested_warningReferenceNoInImportTable, prefix, op.Value), entry);
425 validRef = false;
426
427 }
428 }
429 else if (entry.FileRef.GetEntry(op.Value)?.ObjectName.ToString() == @"Trash" || entry.FileRef.GetEntry(op.Value)?.ObjectName.ToString() == UnrealPackageFile.TrashPackageName)
430 {
431 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_nested_warningTrashedExportReference, prefix, op.Value), entry);
432 validRef = false;
433 }
434
435 // Check object is of correct typing?
436 if (validRef && op.Value != 0)
437 {
438 var referencedEntry = op.ResolveToEntry(entry.FileRef);
439 if (referencedEntry.FullPath.Equals(@"SFXGame.BioDeprecated", StringComparison.InvariantCulture)) return; //This will appear as wrong even though it's technically not
440 if (entry.FileRef.Game == MEGame.ME2)
441 {
442 if (op.Name == "m_oAreaMap" && referencedEntry.ClassName == @"BioSWF") return; //This will appear as wrong even though it's technically not (deprecated leftover)
443 if (op.Name == "AIController" && referencedEntry.ObjectName.Name.StartsWith("BioAI_")) return; //These are all deprecated. A few things use them still but the inheritance is wrong
444 if (op.Name == "TrackingSound" && entry.ClassName == "SFXSeqAct_SecurityCam" && referencedEntry.ClassName == "WwiseEvent") return; // Appears to be incorrect in vanilla. Don't report it as an issue
445 }
446
447 var propInfo = GlobalUnrealObjectInfo.GetPropertyInfo(entry.Game, op.Name, containingClassOrStructName, containingExport: entry as ExportEntry);
448 var customClassInfos = new Dictionary<string, ClassInfo>();
449
450 if (referencedEntry.ClassName == @"Class" && op.Value > 0)
451 {
452
453 // Make sure we have info about this class.
454 var lookupEnt = referencedEntry as ExportEntry;
455 while (lookupEnt != null && lookupEnt.IsClass && !GlobalUnrealObjectInfo.GetClasses(entry.FileRef.Game).ContainsKey(lookupEnt.ObjectName))
456 {
457 // Needs dynamically generated
458 var cc = GlobalUnrealObjectInfo.generateClassInfo(lookupEnt);
459 customClassInfos[lookupEnt.ObjectName] = cc;
460 lookupEnt = lookupEnt.Parent as ExportEntry;
461 }
462
463 // If we did not pull it previously, we should try again with our custom info.
464 if (propInfo == null && customClassInfos.Any())
465 {
466 propInfo = GlobalUnrealObjectInfo.GetPropertyInfo(entry.Game, op.Name,
467 containingClassOrStructName, customClassInfos[referencedEntry.ObjectName],
468 containingExport: entry as ExportEntry);
469 }
470 }
471
472 if (propInfo != null && propInfo.Reference != null)
473 {
474 // We can't resolve if an object inherits from a class object that's only defined in native.
475 // This is only possible if the refernce is an import and it's importing native class object.
476 // Like Engine.CodecBinkMovie
477 if (!referencedEntry.IsAKnownNativeClass())
478 {
479 if (referencedEntry.ClassName == @"Class")
480 {
481 // Inherits
482 if (!GlobalUnrealObjectInfo.IsA(referencedEntry.ObjectName.Name, propInfo.Reference, referencedEntry.Game, customClassInfos, (entry as ExportEntry)?.SuperClassName))
483 {
484 if (op.Name.Name != null)
485 {
486 item.AddSignificantIssue(localizationDelegate(
487 ME3XL.string_interp_warningWrongPropertyTypingWrongMessage, prefix,
488 op.Name.Name, op.Value, op.ResolveToEntry(entry.FileRef).InstancedFullPath,
489 propInfo.Reference, referencedEntry.ClassName), entry);
490 }
491 else
492 {
493 item.AddSignificantIssue(localizationDelegate(
494 ME3XL
495 .string_interp_nested_warningWrongClassPropertyTypingWrongMessage,
496 prefix, op.Value, op.ResolveToEntry(entry.FileRef).InstancedFullPath,
497 propInfo.Reference, referencedEntry.ClassName), entry);
498 }
499 }
500 }
501 else if (!referencedEntry.IsA(propInfo.Reference, customClassInfos))
502 {
503 // Is instance of
504 if (op.Name.Name != null)
505 {
506 item.AddSignificantIssue(localizationDelegate(
507 ME3XL.string_interp_warningWrongObjectPropertyTypingWrongMessage,
508 prefix, op.Name.Name, op.Value,
509 op.ResolveToEntry(entry.FileRef).InstancedFullPath, propInfo.Reference,
510 referencedEntry.ClassName), entry);
511 }
512 else
513 {
514 item.AddSignificantIssue(localizationDelegate(
515 ME3XL.string_interp_nested_warningWrongObjectPropertyTypingWrongMessage,
516 prefix, op.Value, op.ResolveToEntry(entry.FileRef).InstancedFullPath,
517 propInfo.Reference, referencedEntry.ClassName), entry);
518 }
519 }
520 }
521 }
522 }
523 }
524 else if (property is ArrayProperty<ObjectProperty> aop)
525 {
526 foreach (var p in aop)
527 {
528 recursiveCheckProperty(item, localizationDelegate, relativePath, aop.Name, entry, p);
529 }
530 }
531 else if (property is StructProperty sp)
532 {
533 foreach (var p in sp.Properties)
534 {
535 recursiveCheckProperty(item, localizationDelegate, relativePath, sp.StructType, entry, p);
536 }
537 }
538 else if (property is ArrayProperty<StructProperty> asp)
539 {
540 foreach (var p in asp)
541 {
542 recursiveCheckProperty(item, localizationDelegate, relativePath, p.StructType, entry, p);
543 }
544 }
545 else if (property is DelegateProperty dp)
546 {
547 if (dp.Value.Object != 0 && !entry.FileRef.IsEntry(dp.Value.Object))
548 {
549 item.AddSignificantIssue(localizationDelegate(ME3XL.string_interp_warningDelegatePropertyIsOutsideOfExportTable, prefix, dp.Name.Name), entry);
550 }
551 }
552 }
553 }
