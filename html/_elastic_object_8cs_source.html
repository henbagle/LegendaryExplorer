172 public override bool TryBinaryOperation(BinaryOperationBinder binder, object arg, out object result)
173 {
174 if (binder.Operation == ExpressionType.LeftShiftAssign && _nodeType == NodeType.Element)
175 {
176 InternalContent = arg;
177 result = this;
178 return true;
179 }
180
181 if (binder.Operation == ExpressionType.LeftShiftAssign && _nodeType == NodeType.Attribute)
182 {
183 InternalValue = arg;
184 result = this;
185 return true;
186 }
187
188 switch (binder.Operation)
189 {
190 case ExpressionType.LeftShift:
191 {
192 if (arg is string)
193 {
194 var exp = new ElasticObject(arg as string, null)
195 {
196 _nodeType = NodeType.Element
197 };
198
199 AddElement(exp);
200 result = exp;
201 return true;
202 }
203
204 if (!(arg is ElasticObject))
205 {
206 return base.TryBinaryOperation(binder, arg, out result);
207 }
208
209 var eobj = arg as ElasticObject;
210
211 if (!Elements.Contains(eobj))
212 {
213 AddElement(eobj);
214 }
215
216 result = eobj;
217
218 return true;
219 }
220 case ExpressionType.LessThan:
221 {
222 var memberName = "";
223
224 if (arg is string)
225 {
226 memberName = arg as string;
227
228 if (HasAttribute(memberName))
229 {
230 throw new InvalidOperationException("An attribute with name" + memberName + " already exists");
231 }
232
233 var att = new ElasticObject(memberName, null);
234 AddAttribute(memberName, att);
235 result = att;
236
237 return true;
238 }
239
240 if (!(arg is ElasticObject))
241 {
242 return base.TryBinaryOperation(binder, arg, out result);
243 }
244
245 var eobj = arg as ElasticObject;

269 {
