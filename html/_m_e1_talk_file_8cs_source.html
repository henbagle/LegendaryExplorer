189 private void LoadTlkData(IMEPackage pcc)
190 {
191 var r = new EndianReader(pcc.GetUExport(UIndex).GetReadOnlyBinaryStream(), Encoding.Unicode)
192 {
193 Endian = pcc.Endian
194 };
195 //hashtable
196 int entryCount = r.ReadInt32();
197 StringRefs = new TLKStringRef[entryCount];
198 for (int i = 0; i < entryCount; i++)
199 {
200 StringRefs[i] = new TLKStringRef(r, true);
201 }
202
203 //Huffman tree
204 int nodeCount = r.ReadInt32();
205 nodes = new List<HuffmanNode>(nodeCount);
206 for (int i = 0; i < nodeCount; i++)
207 {
208 bool leaf = r.ReadBoolean();
209 if (leaf)
210 {
211 nodes.Add(new HuffmanNode(r.ReadChar()));
212 }
213 else
214 {
215 nodes.Add(new HuffmanNode(r.ReadInt16(), r.ReadInt16()));
216 }
217 }
218 //TraverseHuffmanTree(nodes[0], new List<bool>());
219
220 //encoded data
221 int stringCount = r.ReadInt32();
222 byte[] data = new byte[r.BaseStream.Length - r.BaseStream.Position];
223 r.Read(data, 0, data.Length);
224 var bits = new BitArray(data);
225
226 //decompress encoded data with huffman tree
227 int offset = 4;
228 var rawStrings = new List<string>(stringCount);
229 while (offset * 8 < bits.Length)
230 {
231 int size = BitConverter.ToInt32(data, offset);
232 offset += 4;
233 string s = GetString(offset * 8, bits);
234 offset += size + 4;
235 rawStrings.Add(s);
236 }
237
238 //associate StringIDs with strings
239 foreach (TLKStringRef strRef in StringRefs)
240 {
241 if (strRef.Flags == 1)
242 {
243 strRef.Data = rawStrings[strRef.Index];
244 }
245 }
246 }

248 private string GetString(int bitOffset, BitArray bits)
249 {
250 HuffmanNode root = nodes[0];
251 HuffmanNode curNode = root;
252
253 var builder = new StringBuilder();
254 int i;
255 for (i = bitOffset; i < bits.Length; i++)
256 {
257 /* reading bits' sequence and decoding it to Strings while traversing Huffman Tree */
258 int nextNodeID = bits[i] ? curNode.RightNodeID : curNode.LeftNodeID;
259
260 /* it's an internal node - keep looking for a leaf */
261 if (nextNodeID >= 0)
262 curNode = nodes[nextNodeID];
263 else
264 /* it's a leaf! */
265 {
266 char c = curNode.data;
267 if (c != '\0')
268 {
269 /* it's not NULL */
270 builder.Append(c);
271 curNode = root;
272 i--;
273 }
274 else
275 {
276 /* it's a NULL terminating processed string, we're done */
277 //skip ahead approximately 9 bytes to the next string
278 return builder.ToString();
279 }
280 }
281 }
282
283 if (curNode.LeftNodeID == curNode.RightNodeID)
284 {
285 char c = curNode.data;
286 //We hit edge case where final bit is on a byte boundary and there is nothing left to read. This is a leaf node.
287 if (c != '\0')
288 {
289 /* it's not NULL */
290 builder.Append(c);
291 curNode = root;
292 }
293 else
294 {
295 /* it's a NULL terminating processed string, we're done */
296 //skip ahead approximately 9 bytes to the next string
297 return builder.ToString();
298 }
299 }
300
301 Debug.WriteLine("RETURNING NULL STRING (NOT NULL TERMINATED)!");
302 return null;
303 }

305 private void TraverseHuffmanTree(HuffmanNode node, List<bool> code)
306 {
307 /* check if both sons are null */
308 if (node.LeftNodeID == node.RightNodeID)
309 {
310 BitArray ba = new BitArray(code.ToArray());
311 string c = "";
312 foreach (bool b in ba)
313 {
314 c += b ? '1' : '0';
315 }
316 }
317 else
318 {
319 /* adds 0 to the code - process left son*/
320 code.Add(false);
321 TraverseHuffmanTree(nodes[node.LeftNodeID], code);
322 code.RemoveAt(code.Count - 1);
323
324 /* adds 1 to the code - process right son*/
325 code.Add(true);
326 TraverseHuffmanTree(nodes[node.RightNodeID], code);
327 code.RemoveAt(code.Count - 1);
328 }

342 private static void WriteXML(IEnumerable<TLKStringRef> tlkStringRefs, string name, XmlTextWriter writer)
343 {
344 writer.Formatting = Formatting.Indented;
345 writer.Indentation = 4;
346
347 writer.WriteStartDocument();
348 writer.WriteStartElement("tlkFile");
349 writer.WriteAttributeString("Name", name);
350
351 foreach (TLKStringRef tlkStringRef in tlkStringRefs)
352 {
353 writer.WriteStartElement("string");
354 writer.WriteStartElement("id");
355 writer.WriteValue(tlkStringRef.StringID);
356 writer.WriteEndElement(); // </id>
357 writer.WriteStartElement("flags");
358 writer.WriteValue(tlkStringRef.Flags);
359 writer.WriteEndElement(); // </flags>
360 if (tlkStringRef.Flags != 1)
361 writer.WriteElementString("data", "-1");
362 else
363 writer.WriteElementString("data", tlkStringRef.Data);
364 writer.WriteEndElement(); // </string>
365 }
366 writer.WriteEndElement(); // </tlkFile>
367 }
