95 public CodeBody Decompile()
96 {
97 // Skip native funcs
98 if (DataContainer is UFunction Func && Func.FunctionFlags.Has(EFunctionFlags.Native))
99 {
100 var comment = new ExpressionOnlyStatement(new SymbolReference(null, "// Native function"));
101 return new CodeBody(new List<Statement> { comment });
102 }
103
104 if (ContainingClass.Export.ObjectName == "SFXGalaxyMapObject")
105 {
106 //these functions are broken and cannot be decompiled. instead of trying, we construct a simpler, functionally identical version
107 if (DataContainer.Export.ObjectName == "GetEditorLabel")
108 {
109 return new CodeBody(new List<Statement> { new ReturnStatement(new SymbolReference(null, "sLabel")) });
110 }
111 if (DataContainer.Export.ObjectName == "InitializeAppearance")
112 {
113 return new CodeBody(new List<Statement> { new ReturnStatement() });
114 }
115 }
116
117 Position = 0;
118 _totalPadding = 0;
119 CurrentScope = new Stack<int>();
120 var statements = new List<Statement>();
121 var codeBody = new CodeBody(statements);
122 StatementLocations = new Dictionary<ushort, Statement>();
123 StartPositions = new Stack<ushort>();
124 Scopes = new List<List<Statement>>();
125 LabelTable = new List<LabelTableEntry>();
126 ForEachScopes = new Stack<ushort>();
127
128 DecompileDefaultParameterValues(statements);
129
130 Scopes.Add(statements);
131 CurrentScope.Push(Scopes.Count - 1);
132 while (Position < Size && !CurrentIs(OpCodes.EndOfScript))
133 {
134 Statement current;
135 try
136 {
137 current = DecompileStatement();
138 if (current == null && PeekByte == (byte)OpCodes.EndOfScript)
139 break; // Natural end after label table, no error
140 }
141 catch when(!LegendaryExplorerCoreLib.IsDebug)
142 {
143 current = null;
144 }
145
146 if (current == null)
147 {
148 ClearToDecompilationError(statements);
149 return codeBody;
150 }
151
152 statements.Add(current);
153 }
154 CurrentScope.Pop();
155
156 Dictionary<Statement, ushort> LocationStatements = StatementLocations.ToDictionary(kvp => kvp.Value, kvp => kvp.Key);
157
158 if (DataContainer is UClass && ReplicatedProperties is not null)
159 {
160 var newStatements = new List<Statement>();
161 foreach (Statement statement in statements)
162 {
163 if (statement is not ExpressionOnlyStatement exprStatement)
164 {
165 ClearToDecompilationError(statements);
166 return codeBody;
167 }
168 ushort loc = LocationStatements[statement];
169 if (!ReplicatedProperties.TryGetValue(loc, out List<string> propNames))
170 {
171 propNames = new() { "#ERROR_MISSING_PROPERTY#" };
172 }
173 List<SymbolReference> replicatedVariables = propNames.Select(s => new SymbolReference(null, s)).ToList();
174 newStatements.Add(new ReplicationStatement(exprStatement.Value, replicatedVariables));
175 }
176 return new CodeBody(newStatements);
177 }
178
179 try
180 {
181 DecompileLoopsAndIfs(codeBody, LocationStatements);
182 }
183 catch (Exception e)
184 {
185 ClearToDecompilationError(statements);
186 statements.Add(new ExpressionOnlyStatement(new SymbolReference(null, e.FlattenException())));
187 return codeBody;
188 }
189
190 //insert state Labels
191 foreach (LabelTableEntry label in LabelTable)
192 {
193 InsertLabel(new Label(label.NameRef, (ushort)label.Offset));
194 }
195
196 //insert labels for any (non-state) gotos
197 foreach ((ushort labelPos, List<Goto> gotos) in LabelLocationsToFix)
198 {
199 var label = new Label($"label_0x{labelPos:X}", labelPos);
200 bool hasValidGotos = false;
201 foreach (Goto g in gotos)
202 {
203 //some gotos may have been converted into breaks or continues
204 if (StatementLocations[LocationStatements[g]] == g)
205 {
206 hasValidGotos = true;
207 }
208 g.Label = label;
209 g.LabelName = label.Name;
210 }
211 if (!hasValidGotos)
212 {
213 continue;
214 }
215 InsertLabel(label);
216 }
217
218 //a void return at the end of a function is a bytecode implementation detail, get rid of it.
219 //This will also get rid of returnnothings, so loop to make sure we get both
220 while (statements.Count > 0 && statements.Last() is ReturnStatement {Value: null})
221 {
222 statements.RemoveAt(statements.Count - 1);
223 }
224
225 return codeBody;
226 }

239 private void InsertLabel(Label label)
240 {
241 var statementAtLabelLocation = StatementLocations[label.StartOffset];
242 while (statementAtLabelLocation is UnconditionalJump unJump)
243 {
244 statementAtLabelLocation = StatementLocations[unJump.JumpLoc];
245 }
246
247 switch (statementAtLabelLocation.Outer)
248 {
249 case CodeBody cb:
250 {
251 int stmntIdx = cb.Statements.IndexOf(statementAtLabelLocation);
252 label.Outer = cb;
253 cb.Statements.Insert(stmntIdx, label);
254 break;
255 }
256 case ForLoop forLoop:
257 if (statementAtLabelLocation == forLoop.Update)
258 {
259 label.Outer = forLoop.Body;
260 forLoop.Body.Statements.Add(label);
261 }
262 else //statementAtLabelLocation is the initializer
263 {
264 var cb = (CodeBody) forLoop.Outer;
265 int stmntIdx = cb.Statements.IndexOf(forLoop);
266 label.Outer = cb;
267 cb.Statements.Insert(stmntIdx, label);
268 }
269
270 break;
271 case ForEachLoop forEachLoop:
272 if (statementAtLabelLocation is IteratorNext)
273 {
274 label.Outer = forEachLoop.Body;
275 forEachLoop.Body.Statements.Add(label);
276 }
277 else
278 {
279 throw new Exception("Invalid control flow!");
280 }
281
282 break;
283 default:
284 throw new Exception("Invalid control flow!");
285 }

290 private void DecompileLoopsAndIfs(CodeBody outer, Dictionary<Statement, ushort> positions, bool inLoop = false)
291 {
292 List<Statement> statements = outer.Statements;
293 var defaultCaseStatements = new Stack<Statement>();
294 var switchEnds = new Dictionary<int, ushort>();
295 for (int i = statements.Count - 1; i >= 0; i--)
296 {
297 Statement cur = statements[i];
298 cur.Outer = outer;
299 if (!positions.TryGetValue(cur, out ushort curPos)) continue; //default param values, labels
300
301 if (cur is UnconditionalJump loopEnd && loopEnd.JumpLoc < curPos)
302 {
303 //end of while or for loop
304 ushort continueToPos = curPos;
305 Statement statementBeforeEndOfLoop = statements[i - 1];
306 bool isForLoop = IsIncrementDecrement(statementBeforeEndOfLoop, out Statement update);
307 Statement loopStart = StatementLocations[loopEnd.JumpLoc];
308 int conditionIdx = statements.IndexOf(loopStart);
309 int skipStart = conditionIdx;
310 int numToSkip = i - conditionIdx + 1;
311
312 if (statements[conditionIdx] is not ConditionalJump conditionalJump)
313 {
314 //no loop condition, so this is a backwards goto instead
315 ReplaceWithGoto(loopEnd, outer, i);
316 continue;
317 }
318 var loopBody = new CodeBody(statements.GetRange(conditionIdx + 1, i - conditionIdx - 1));
319 if (!isForLoop && statementBeforeEndOfLoop is not Jump)
320 {
321 //check to see if there is an unconditional jump to the statement before the end of the loop. This indicates a continue inside a for loop
322 ushort checkPos = positions[statementBeforeEndOfLoop];
323 if (loopBody.Statements.OfType<UnconditionalJump>().Any(possibleContinue => possibleContinue.JumpLoc == checkPos))
324 {
325 update = statementBeforeEndOfLoop;
326 isForLoop = true;
327 }
328 }
329 if (isForLoop)
330 {
331 //check to see if there is an unconditional jump to the end of the loop.
332 //This indicates that the loop is NOT a for loop, as there is no way to skip the increment statement in a for loop
333 for (int j = i - 2; j > conditionIdx; j--)
334 {
335 if (statements[j] is UnconditionalJump unj && unj.JumpLoc == curPos)
336 {
337 isForLoop = false;
338 update = null;
339 break;
340 }
341 }
342 }
343 AssignStatement forInit = null;
344 ushort loopLoc = loopEnd.JumpLoc;
345 if (isForLoop && conditionIdx > 0 && statements[conditionIdx - 1] is AssignStatement assignStatement)
346 {
347 forInit = assignStatement;
348 skipStart = conditionIdx - 1;
349 numToSkip++;
350 loopLoc = positions[assignStatement];
351 }
352
353 Statement loop;
354 Expression condition = conditionalJump.Condition;
355 if (isForLoop)
356 {
357 continueToPos = positions[statementBeforeEndOfLoop];
358 loopBody.Statements.RemoveAt(loopBody.Statements.Count - 1);
359 loop = new ForLoop(forInit, condition, update, loopBody) { Outer = outer };
360 }
361 else
362 {
363 loop = new WhileLoop(condition, loopBody) { Outer = outer };
364 }
365
366 //recurse into body of loop
367 DecompileLoopsAndIfs(loopBody, positions, true);
368
369 //convert unconditional jumps into continue and break
370 const int sizeOfUnconditionalJump = 3;
371 ConvertJumps(loopBody, continueToPos, loopEnd.JumpLoc, curPos + sizeOfUnconditionalJump);
372
373 statements.RemoveRange(skipStart, numToSkip);
374 statements.Insert(skipStart, loop);
375 StatementLocations[loopLoc] = loop;
376 positions[loop] = loopLoc;
377
378 i = skipStart;
379 }
380 else if (cur is ConditionalJump inj && inj.JumpLoc < curPos)
381 {
382 //end of do until loop
383 Statement loopStart = StatementLocations[inj.JumpLoc];
384 int loopStartIdx = statements.IndexOf(loopStart);
385 int loopLength = i - loopStartIdx;
386 var loop = new DoUntilLoop(inj.Condition, new CodeBody(statements.GetRange(loopStartIdx, loopLength))) { Outer = outer };
387
388 //recurse into body of loop
389 DecompileLoopsAndIfs(loop.Body, positions, inLoop);
390
391 //convert unconditional jumps into continue and break
392 ConvertJumps(loop.Body, curPos, inj.JumpLoc, curPos + inj.SizeOfExpression);
393
394 statements.RemoveRange(loopStartIdx, loopLength + 1);
395 statements.Insert(loopStartIdx, loop);
396 StatementLocations[inj.JumpLoc] = loop;
397 positions[loop] = inj.JumpLoc;
398
399 i = loopStartIdx;
400 }
401 else if (cur is ConditionalJump ifJump)
402 {
403 //Just a plain if (and maybe else)
404 Statement jumpToStatement = StatementLocations[ifJump.JumpLoc];
405 int jumpToStatementIdx = statements.IndexOf(jumpToStatement);
406 if (jumpToStatementIdx == -1)
407 {
408 jumpToStatementIdx = statements.Count;
409 }
410
411 int thenBodyStartIdx = i + 1;
412 int thenBodyLength = jumpToStatementIdx - thenBodyStartIdx;
413
414 int totalLength = thenBodyLength + 1;
415
416 CodeBody elseBody = null;
417 if (jumpToStatementIdx < statements.Count)
418 {
419 if (inLoop && thenBodyLength == 1 && statements[thenBodyStartIdx] is UnconditionalJump possibleContinue && possibleContinue.JumpLoc > positions[statements[^1]])
420 {
421 //this is a continue, not an else. Will be converted elsewhere
422 }
423 else if (statements[jumpToStatementIdx - 1] is UnconditionalJump elseJump)
424 {
425 if (inLoop && elseJump.JumpLoc > positions[statements[^1]])
426 {
427 //this is a break or goto. Will be converted elsewhere
428 }
429 else
430 {
431 Statement elseJumpToStatement = StatementLocations[elseJump.JumpLoc];
432 int elseJumpToStatementIdx = statements.IndexOf(elseJumpToStatement);
433 if (elseJumpToStatementIdx == -1)
434 {
435 elseJumpToStatementIdx = statements.Count;
436 }
437
438 elseBody = new CodeBody(statements.GetRange(jumpToStatementIdx, elseJumpToStatementIdx - jumpToStatementIdx));
439 thenBodyLength--;
440 totalLength += elseBody.Statements.Count;
441 }
442 }
443 }
444 var thenBody = new CodeBody(statements.GetRange(thenBodyStartIdx, thenBodyLength));
445
446 //shouldn't be neccesary? we're working backwards, so all the statements in the bodies will have been processed already
447 //DecompileLoopsAndIfs(thenBody.Statements, positions, inLoop);
448 //FinishSwitch(thenBody.Statements);
449 //if (elseBody != null)
450 //{
451 // DecompileLoopsAndIfs(elseBody.Statements, positions, inLoop);
452 // FinishSwitch(elseBody.Statements);
453 //}
454
455 Expression condition = ifJump.Condition;
456 statements.RemoveRange(i, totalLength);
457 var ifStatement = new IfStatement(condition, thenBody, elseBody) { Outer = outer };
458
459 statements.Insert(i, ifStatement);
460 StatementLocations[curPos] = ifStatement;
461 positions[ifStatement] = curPos;
462 }
463 else if (cur is ForEachLoop fel && !decompiledForEachLoops.Contains(fel))
464 {
465 decompiledForEachLoops.Add(fel);
466 DecompileLoopsAndIfs(fel.Body, positions, false);
467 ConvertJumps(fel.Body, fel.iteratorPopPos - 1, -1, fel.iteratorPopPos);
468 }
469 else if (cur is DefaultCaseStatement)
470 {
471 defaultCaseStatements.Push(cur);
472 int breakIdx = i - 1;
473 while (breakIdx >= 0 && (statements[breakIdx] is not UnconditionalJump uj || uj.JumpLoc <= curPos))
474 {
475 --breakIdx;
476 }
477
478 if (breakIdx >= 0)
479 {
480 var switchEndJump = (UnconditionalJump)statements[breakIdx];
481 switchEnds[defaultCaseStatements.Count] = switchEndJump.JumpLoc;
482 var brk = new BreakStatement { Outer = outer };
483 StatementLocations[positions[switchEndJump]] = statements[breakIdx] = brk;
484 positions[brk] = positions[switchEndJump];
485
486 var defaultBreakIdx = statements.IndexOf(StatementLocations[switchEndJump.JumpLoc]) - 1;
487 if (defaultBreakIdx < 0)
488 {
489 defaultBreakIdx = statements.Count - 1;
490 }
491
492 if (statements[defaultBreakIdx] is UnconditionalJump defaultBreak && defaultBreak.JumpLoc == switchEndJump.JumpLoc)
493 {
494 var defBrk = new BreakStatement { Outer = outer };
495 StatementLocations[positions[defaultBreak]] = statements[defaultBreakIdx] = defBrk;
496 positions[defBrk] = positions[defaultBreak];
497 }
498 }
499 }
500 else if (cur is UnconditionalJump possibleBreak && defaultCaseStatements.Count > 0 && switchEnds.ContainsValue(possibleBreak.JumpLoc))
501 {
502 var brk = new BreakStatement { Outer = outer };
503 StatementLocations[positions[possibleBreak]] = statements[i] = brk;
504 positions[brk] = positions[possibleBreak];
505 }
506 else if (cur is SwitchStatement sw)
507 {
508 int lastStatementIdx;
509
510 //if this switch has breaks in it, get the end of the switch, else the last statement is the default case
511 if (switchEnds.TryGetValue(defaultCaseStatements.Count, out ushort jumpLoc))
512 {
513 switchEnds.Remove(defaultCaseStatements.Count);
514 defaultCaseStatements.Pop();
515 Statement jumpToStatement = StatementLocations[jumpLoc];
516 lastStatementIdx = statements.IndexOf(jumpToStatement) - 1;
517 if (lastStatementIdx < i)
518 {
519 lastStatementIdx = statements.Count - 1;
520 }
521 }
522 else
523 {
524 lastStatementIdx = statements.IndexOf(defaultCaseStatements.Pop());
525 }
526
527 int length = lastStatementIdx - i;
528
529 sw.Body = new CodeBody(statements.GetRange(i + 1, length)) {Outer = sw};
530 statements.RemoveRange(i + 1, length);
531
532 //if this is switching on an enum property, we should convert integer literals in case statements to enum values
533 if (sw.Expression is SymbolReference {Node: Enumeration enm})
534 {
535 int valuesCount = enm.Values.Count;
536 foreach (CaseStatement caseStatement in sw.Body.Statements.OfType<CaseStatement>())
537 {
538 if (caseStatement.Value is IntegerLiteral {Value: >= 0} intLit && intLit.Value < valuesCount)
539 {
540 EnumValue enumValue = enm.Values[intLit.Value];
541 caseStatement.Value = new SymbolReference(enumValue, enumValue.Name);
542 }
543 }
544 }
545 }
546 }

549 ConvertJumps(outer, -1, -1, -1);

566 void ConvertJumps(CodeBody codeBody, int continueToPos, int loopStartPos, int breakPos)
567 {
568 for (int j = 0; j < codeBody.Statements.Count; j++)
569 {
570 Statement bodyStatement = codeBody.Statements[j];
571 switch (bodyStatement)
572 {
573 case UnconditionalJump unJump when unJump.JumpLoc == continueToPos || unJump.JumpLoc == loopStartPos:
574 {
575 var continueStatement = new ContinueStatement { Outer = codeBody };
576 ushort position = positions[bodyStatement];
577 StatementLocations[position] = codeBody.Statements[j] = continueStatement;
578 positions[continueStatement] = position;
579 break;
580 }
581 case UnconditionalJump unJump when unJump.JumpLoc == breakPos:
582 {
583 var breakStatement = new BreakStatement { Outer = codeBody };
584 ushort position = positions[bodyStatement];
585 StatementLocations[position] = codeBody.Statements[j] = breakStatement;
586 positions[breakStatement] = position;
587 break;
588 }
589 case UnconditionalJump unJump and not Goto:
590 {
591 ReplaceWithGoto(unJump, codeBody, j);
592 break;
593 }
594 case IfStatement ifStatement:
595 if (ifStatement.Then?.Statements != null) ConvertJumps(ifStatement.Then, continueToPos, loopStartPos, breakPos);
596 if (ifStatement.Else?.Statements != null) ConvertJumps(ifStatement.Else, continueToPos, loopStartPos, breakPos);
597 break;
598 case SwitchStatement switchStatement:
599 if (switchStatement.Body?.Statements != null) ConvertJumps(switchStatement.Body, continueToPos, loopStartPos, breakPos);
600 break;
601 }
602 }
603 }

615 private void DecompileDefaultParameterValues(List<Statement> statements)
616 {
617 if (DataContainer is UFunction) // Gets all optional params for default value parsing
618 {
619 OptionalParams = new Queue<FunctionParameter>();
620 foreach (FunctionParameter param in Parameters.Where(param => param.IsOptional))
621 {
622 OptionalParams.Enqueue(param);
623 }
624 while (PeekByte is (byte)OpCodes.DefaultParmValue or (byte)OpCodes.Nothing)
625 {
626 StartPositions.Push((ushort)Position);
627 byte token = PopByte();
628 if (token == (byte)OpCodes.DefaultParmValue) // default value assigned
629 {
630
631 ReadInt16(); //MemLength of value
632 var value = DecompileExpression();
633 PopByte(); // Opcodes.EndParmValue
634
635
636 if (OptionalParams.Count != 0)
637 {
638 var parm = OptionalParams.Dequeue();
639 parm.DefaultParameter = value;
640 StartPositions.Pop();
641 }
642 else
643 { // TODO: weird, research how to deal with this
644 var builder = new CodeBuilderVisitor(); // what a wonderful hack, TODO.
645 value.AcceptVisitor(builder);
646 var comment = new SymbolReference(null, "// Orphaned Default Parm: " + builder.GetOutput(), null, null);
647 var statement = new ExpressionOnlyStatement(comment, null, null);
648 StatementLocations.Add(StartPositions.Pop(), statement);
649 statements.Add(statement);

654 OptionalParams.Dequeue();
