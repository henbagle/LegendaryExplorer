42 public AudioInfo GetAudioInfo()
43 {
44 try
45 {
46 AudioInfo ai = new AudioInfo();
47 Stream dataStream;
48 if (IsPCCStored)
49 {
50 dataStream = new MemoryStream(EmbeddedData);
51 }
52 else
53 {
54 var afc = GetPathToAFC();
55 if (string.IsNullOrWhiteSpace(afc) || !File.Exists(afc))
56 {
57 return null; // We have access to where the audio is contained
58 }
59
60 dataStream = ExternalFileHelper.ReadExternalData(afc, DataOffset, DataSize);
61 }
62
63 EndianReader er = new EndianReader(dataStream);
64 var header = er.ReadStringASCII(4);
65 if (header == "RIFX") er.Endian = Endian.Big;
66 if (header == "RIFF") er.Endian = Endian.Little;
67 // Position 4
68
69 er.Seek(0xC, SeekOrigin.Current); // Post 'fmt ', get fmt size
70 var fmtSize = er.ReadInt32();
71 var postFormatPosition = er.Position;
72 ai.CodecID = er.ReadUInt16();
73
74 switch (ai.CodecID)
75 {
76 case 0xFFFF:
77 ai.CodecName = "Vorbis";
78 break;
79 case 0x0166:
80 ai.CodecName = "XMA2";
81 break;
82 default:
83 ai.CodecName = $"Unknown codec ID {ai.CodecID}";
84 break;
85 }
86
87 ai.Channels = er.ReadUInt16();
88 ai.SampleRate = er.ReadUInt32();
89 er.ReadInt32(); //Average bits per second
90 er.ReadUInt16(); //Alignment. VGMStream shows this is 16bit but that doesn't seem right
91 ai.BitsPerSample = er.ReadUInt16(); //Bytes per sample. For vorbis this is always 0!
92 var extraSize = er.ReadUInt16();
93 if (extraSize == 0x30)
94 {
95 // Newer Wwise
96 er.Seek(postFormatPosition + 0x18, SeekOrigin.Begin);
97 ai.SampleCount = er.ReadUInt32();
98 }
99 else
100 {
101 if (ai.CodecID == 0xFFFF)
102 {
103 // Vorbis
104 er.Seek(0x14 + fmtSize, SeekOrigin.Begin);
105 var chunkName = er.ReadStringASCII(4);
106 while (!chunkName.Equals("vorb", StringComparison.InvariantCultureIgnoreCase))
107 {
108 er.Seek(er.ReadInt32(), SeekOrigin.Current);
109 chunkName = er.ReadStringASCII(4);
110 }
111 er.SkipInt32(); //Skip vorb size
112 ai.SampleCount = er.ReadUInt32();
113 }
114 else if (ai.CodecID == 0x0166)
115 {
116 // XMA2 (360)
117
118 // This calculation is wrong.
119 // See https://github.com/losnoco/vgmstream/blob/master/src/meta/wwise.c#L484
120 // and
121 // https://github.com/losnoco/vgmstream/blob/b61908f3af892714dda09c143a52fe0d65228985/src/coding/coding_utils.c#L767
122 // Seems correct, but will need to investigate why it's wrong. Example file is 543 in BioSnd_OmgPrA.xxx Xenon ME2
123 er.Seek(0x14 + 0x18, SeekOrigin.Begin); //Start of fmt + 0x18
124 ai.SampleCount = er.ReadUInt32();
125 }
126 else
127 {
128 // UNKNOWN!!
129 Debug.WriteLine("Unknown codec ID!");
130 }
131 }
132
133 // We don't care about the rest.
134 return ai;
135 }
136 catch (Exception)
137 {
138 return null;
139 }
140 }
