48 protected override void Serialize(SerializingContainer2 sc)
49 {
50 if (sc.Game != MEGame.ME3 && sc.Game != MEGame.ME2 && sc.Game != MEGame.LE3 && sc.Game != MEGame.LE2)
51 {
52 throw new Exception($"WwiseBank is not a valid class for {sc.Game}!");
53 }
54 if (sc.Game.IsGame2())
55 {
56 sc.Serialize(ref Unk1);
57 sc.Serialize(ref Unk2);
58 if (Unk1 == 0 && Unk2 == 0)
59 {
60 return; //not sure what's going on here
61 }
62 }
63 sc.SerializeConstInt(0);//BulkDataFlags
64 var dataSizePos = sc.ms.Position; //come back to write size at the end
65 int dataSize = 0;
66 sc.Serialize(ref dataSize);
67 sc.Serialize(ref dataSize);
68 sc.SerializeFileOffset();
69 if (sc.IsLoading && dataSize == 0 || sc.IsSaving && Version == 0)
70 {
71 return;
72 }
73
74 if (sc.IsLoading)
75 {
76 sc.ms.Skip(4);
77 }
78 else
79 {
80 sc.ms.Writer.WriteUInt32(bkhd);
81 }
82
83 int bkhdLen = 8 + (bkhdUnks?.Length ?? 0) * 4;
84 sc.Serialize(ref bkhdLen);
85 sc.Serialize(ref Version);
86 sc.Serialize(ref ID);
87 if (sc.IsLoading)
88 {
89 bkhdUnks = new uint[(bkhdLen - 8) / 4];
90 }
91
92 for (int i = 0; i < bkhdUnks.Length; i++)
93 {
94 sc.Serialize(ref bkhdUnks[i]);
95 }
96
97 if (Version is 38 || //strangely formatted Wwisebank, unused maybe? We're going to ignore it.
98 sc.Game is MEGame.LE2 && Version is 44) //temporary hack. Todo: WwiseBank parsing should be refactored to parse based on Version, not game
99 {
100 if (sc.IsLoading)
101 {
102 var amountRead = (sc.ms.Position - dataSizePos - 12);
103 var amountRemaining = dataSize - amountRead;
104 ME2STMGFallback = sc.ms.ReadBytes((int)amountRemaining);
105 }
106 else
107 {
108 sc.ms.Writer.WriteFromBuffer(ME2STMGFallback);
109 var endPos = sc.ms.Position;
110 sc.ms.JumpTo(dataSizePos);
111 sc.ms.Writer.WriteInt32((int)(endPos - dataSizePos - 12));
112 sc.ms.Writer.WriteInt32((int)(endPos - dataSizePos - 12));
113 sc.ms.JumpTo(endPos);
114 }
115 }
116 else
117 {
118 if (sc.IsLoading)
119 {
120 ReadChunks(sc);
121 }
122 else
123 {
124 WriteChunks(sc);
125 var endPos = sc.ms.Position;
126 sc.ms.JumpTo(dataSizePos);
127 sc.ms.Writer.WriteInt32((int)(endPos - dataSizePos - 12));
128 sc.ms.Writer.WriteInt32((int)(endPos - dataSizePos - 12));
129 sc.ms.JumpTo(endPos);
130 }
131 }

142 private void ReadChunks(SerializingContainer2 sc)
143 {
144 while (sc.ms.Position < sc.ms.Length)
145 {
146 // It looks like on consoles this is not endian
147 string chunkID = sc.ms.BaseStream.ReadStringLatin1(4);
148
149 int chunkSize = sc.ms.ReadInt32();
150 switch (chunkID)
151 {
152 case "STMG":
153 {
154 if (sc.Game.IsGame2() || sc.Game is MEGame.LE3)
155 {
156 ME2STMGFallback = sc.ms.ReadBytes(chunkSize);
157 break;
158 }
159 InitStateManagement = new WwiseStateManagement
160 {
161 VolumeThreshold = sc.ms.ReadFloat(),
162 MaxVoiceInstances = sc.ms.ReadUInt16()
163 };
164 int stateGroupCount = sc.ms.ReadInt32();
165 InitStateManagement.StateGroups = new OrderedMultiValueDictionary<uint, WwiseStateManagement.StateGroup>();
166 for (int i = 0; i < stateGroupCount; i++)
167 {
168 uint id = sc.ms.ReadUInt32();
169 var stateGroup = new WwiseStateManagement.StateGroup
170 {
171 ID = id,
172 DefaultTransitionTime = sc.ms.ReadUInt32(),
173 CustomTransitionTimes = new List<WwiseStateManagement.CustomTransitionTime>()
174 };
175 int transTimesCount = sc.ms.ReadInt32();
176 for (int j = 0; j < transTimesCount; j++)
177 {
178 stateGroup.CustomTransitionTimes.Add(new WwiseStateManagement.CustomTransitionTime
179 {
180 FromStateID = sc.ms.ReadUInt32(),
181 ToStateID = sc.ms.ReadUInt32(),
182 TransitionTime = sc.ms.ReadUInt32(),
183 });
184 }
185 InitStateManagement.StateGroups.Add(id, stateGroup);
186 }
187
188 int switchGroupCount = sc.ms.ReadInt32();
189 InitStateManagement.SwitchGroups = new OrderedMultiValueDictionary<uint, WwiseStateManagement.SwitchGroup>();
190 for (int i = 0; i < switchGroupCount; i++)
191 {
192 uint id = sc.ms.ReadUInt32();
193 var switchGroup = new WwiseStateManagement.SwitchGroup
194 {
195 ID = id,
196 GameParamID = sc.ms.ReadUInt32(),
197 Points = new List<WwiseStateManagement.SwitchPoint>()
198 };
199 int pointsCount = sc.ms.ReadInt32();
200 for (int j = 0; j < pointsCount; j++)
201 {
202 switchGroup.Points.Add(new WwiseStateManagement.SwitchPoint
203 {
204 GameParamValue = sc.ms.ReadFloat(),
205 SwitchID = sc.ms.ReadUInt32(),
206 CurveShape = sc.ms.ReadUInt32()
207 });
208 }
209 InitStateManagement.SwitchGroups.Add(id, switchGroup);
210 }
211
212 int gameParamsCount = sc.ms.ReadInt32();
213 InitStateManagement.GameParameterDefaultValues = new OrderedMultiValueDictionary<uint, float>();
214 for (int i = 0; i < gameParamsCount; i++)
215 {
216 InitStateManagement.GameParameterDefaultValues.Add(sc.ms.ReadUInt32(), sc.ms.ReadFloat());
217 }
218 break;
219 }
220 case "DIDX":
221 int numFiles = chunkSize / 12;
222 var infoPos = sc.ms.Position;
223 sc.ms.Skip(chunkSize);
224 if (sc.ms.BaseStream.ReadUInt32() != data) //not endian swapped
225 {
226 throw new Exception("DIDX chunk is not followed by DATA chunk in WwiseBank!");
227 }
228
229 var dataBytes = sc.ms.ReadBytes(sc.ms.ReadInt32());
230 var dataEndPos = sc.ms.Position;
231 sc.ms.JumpTo(infoPos);
232
233 for (int i = 0; i < numFiles; i++)
234 {
235 EmbeddedFiles.Add(sc.ms.ReadUInt32(), dataBytes.Slice(sc.ms.ReadInt32(), sc.ms.ReadInt32()));
236 }
237 sc.ms.JumpTo(dataEndPos);
238 break;
239 case "HIRC":
240 int count = sc.ms.ReadInt32();
241 for (int i = 0; i < count; i++)
242 {
243 var ho = HIRCObject.Create(sc);
244 HIRCObjects.Add(ho.ID, ho);
245 }
246 break;
247 case "STID":
248 sc.ms.Skip(4);//unknown uint (always 1)
249 int numBanks = sc.ms.ReadInt32();
250 for (int i = 0; i < numBanks; i++)
251 {
252 uint id = sc.ms.ReadUInt32();
253 byte strLen = sc.ms.ReadByte();
254 ReferencedBanks.Add(id, sc.ms.ReadStringASCII(strLen));
255 }
256 break;
257 case "FXPR":
258 //no idea what's in this chunk
259 FXPR_Chunk = sc.ms.ReadBytes(chunkSize);
260 break;
261 case "ENVS":
262 //no idea what's in this chunk
263 ENVS_Chunk = sc.ms.ReadBytes(chunkSize);
264 break;
265 case "INIT":
266 //no idea what's in this chunk
267 INIT_Chunk = sc.ms.ReadBytes(chunkSize);
268 break;
269 case "PLAT":
270 //no idea what's in this chunk
271 Platform = sc.ms.ReadUnrealString();
272 break;
273 default:
274 throw new Exception($"Unknown Chunk: {sc.ms.ReadEndianASCIIString(4)} at {sc.ms.Position - 4}");
275 }
276 }
277 }

279 private void WriteChunks(SerializingContainer2 sc)
280 {
281 EndianWriter writer = sc.ms.Writer;
282 if (EmbeddedFiles.Count > 0)
283 {
284 writer.WriteUInt32(didx);
285 writer.WriteInt32(EmbeddedFiles.Count * 12);
286 using var dataChunk = MemoryManager.GetMemoryStream();
287 foreach ((uint id, byte[] bytes) in EmbeddedFiles)
288 {
289 dataChunk.WriteZeros((int)(dataChunk.Position.Align(16) - dataChunk.Position)); //files must be 16-byte aligned in the data chunk
290 writer.WriteUInt32(id); //writing to DIDX
291 writer.WriteInt32((int)dataChunk.Position); //Writing to DIDX
292 writer.WriteInt32(bytes.Length); //Writing to DIDX
293 dataChunk.WriteFromBuffer(bytes); //Writing to DATA
294 }
295
296 writer.WriteUInt32(data);
297 writer.WriteInt32((int)dataChunk.Length);
298 writer.WriteFromBuffer(dataChunk.ToArray());
299 }
300
301 if (INIT_Chunk != null)
302 {
303 writer.WriteUInt32(init);
304 writer.WriteInt32(INIT_Chunk.Length);
305 writer.WriteFromBuffer(INIT_Chunk);
306 }
307
308 if (ME2STMGFallback != null)
309 {
310 writer.WriteUInt32(stmg);
311 writer.WriteInt32(ME2STMGFallback.Length);
312 writer.WriteFromBuffer(ME2STMGFallback);
313 }
314
315 if (InitStateManagement != null)
316 {
317 writer.WriteUInt32(stmg);
318 var lenPos = sc.ms.Position;
319 writer.WriteUInt32(0);
320 writer.WriteFloat(InitStateManagement.VolumeThreshold);
321 writer.WriteUInt16(InitStateManagement.MaxVoiceInstances);
322 writer.WriteInt32(InitStateManagement.StateGroups.Count);
323 foreach ((uint _, WwiseStateManagement.StateGroup stateGroup) in InitStateManagement.StateGroups)
324 {
325 writer.WriteUInt32(stateGroup.ID);
326 writer.WriteUInt32(stateGroup.DefaultTransitionTime);
327 writer.WriteInt32(stateGroup.CustomTransitionTimes.Count);
328 foreach (var transTime in stateGroup.CustomTransitionTimes)
329 {
330 writer.WriteUInt32(transTime.FromStateID);
331 writer.WriteUInt32(transTime.ToStateID);
332 writer.WriteUInt32(transTime.TransitionTime);
333 }
334 }
335 writer.WriteInt32(InitStateManagement.SwitchGroups.Count);
336 foreach ((uint _, WwiseStateManagement.SwitchGroup switchGroup) in InitStateManagement.SwitchGroups)
337 {
338 writer.WriteUInt32(switchGroup.ID);
339 writer.WriteUInt32(switchGroup.GameParamID);
340 writer.WriteInt32(switchGroup.Points.Count);
341 foreach (var point in switchGroup.Points)
342 {
343 writer.WriteFloat(point.GameParamValue);
344 writer.WriteUInt32(point.SwitchID);
345 writer.WriteUInt32(point.CurveShape);
346 }
347 }
348 writer.WriteInt32(InitStateManagement.GameParameterDefaultValues.Count);
349 foreach ((uint id, float defaultValue) in InitStateManagement.GameParameterDefaultValues)
350 {
351 writer.WriteUInt32(id);
352 writer.WriteFloat(defaultValue);
353 }
354 var endPos = sc.ms.Position;
355 sc.ms.JumpTo(lenPos);
356 writer.WriteInt32((int)(endPos - lenPos - 4));
357 sc.ms.JumpTo(endPos);
358 }
359
360 if (HIRCObjects.Count > 0)
361 {
362 writer.WriteUInt32(hirc);
363 var lengthPos = sc.ms.Position;
364 writer.WriteUInt32(0);
365 writer.WriteInt32(HIRCObjects.Count);
366 foreach ((uint _, HIRCObject h) in HIRCObjects)
367 {
368 writer.WriteFromBuffer(h.ToBytes(sc.Game));
369 }
370
371 var endPos = sc.ms.Position;
372 sc.ms.JumpTo(lengthPos);
373 writer.WriteInt32((int)(endPos - lengthPos - 4));
374 sc.ms.JumpTo(endPos);
375 }
376
377 if (ReferencedBanks.Count > 0)
378 {
379 writer.WriteUInt32(stid);
380 var lengthPos = sc.ms.Position;
381 writer.WriteUInt32(0);
382 writer.WriteUInt32(1);
383 writer.WriteInt32(ReferencedBanks.Count);
384 foreach ((uint id, string name) in ReferencedBanks)
385 {
386 writer.WriteUInt32(id);
387 writer.WriteByte((byte)name.Length);
388 writer.WriteStringLatin1(name);
389 }
390
391 var endPos = sc.ms.Position;
392 sc.ms.JumpTo(lengthPos);
393 writer.WriteInt32((int)(endPos - lengthPos - 4));
394 sc.ms.JumpTo(endPos);
395 }
396
397 if (FXPR_Chunk != null)
398 {
399 writer.WriteUInt32(fxpr);
400 writer.WriteInt32(FXPR_Chunk.Length);
401 writer.WriteFromBuffer(FXPR_Chunk);
402 }
403
404 if (ENVS_Chunk != null)
405 {
406 writer.WriteUInt32(envs);
407 writer.WriteInt32(ENVS_Chunk.Length);
408 writer.WriteFromBuffer(ENVS_Chunk);
409 }
410
411 if (Platform != null)
412 {
413 writer.WriteUInt32(plat);
414 if (Platform.Length > 0)
415 {
416 writer.WriteInt32(Platform.Length + 5);
417 }
418 else
419 {
420 writer.WriteInt32(4);
421 }
422 writer.WriteUnrealStringLatin1(Platform);
423 }
424 }

430 public HIRCType Type;

432 public virtual int DataLength(MEGame game) => unparsed.Length + 4;

435 public static HIRCObject Create(SerializingContainer2 sc)
436 {
437 HIRCType type = (HIRCType)((sc.Game is MEGame.ME2) ? (byte)sc.ms.ReadInt32() : sc.ms.ReadByte());
438 int len = sc.ms.ReadInt32();
439 uint id = sc.ms.ReadUInt32();
440 return type switch
441 {
442 //TODO: figure out SoundSXFSoundVoice for LE
443 HIRCType.SoundSXFSoundVoice when sc.Game.IsOTGame() => SoundSFXVoice.Create(sc, id, len),
444 HIRCType.Event => Event.Create(sc, id),
445 HIRCType.EventAction => EventAction.Create(sc, id, len),
446 _ => new HIRCObject
447 {
448 Type = type,
449 ID = id,
450 unparsed = sc.ms.ReadBytes(len - 4)
451 }
452 };
453 }
454
455 public virtual byte[] ToBytes(MEGame game)
456 {
457 using MemoryStream ms = WriteHIRCObjectHeader(game);
458 ms.WriteFromBuffer(unparsed);
459 return ms.ToArray();
460 }
461
462 protected MemoryStream WriteHIRCObjectHeader(MEGame game)
463 {
464 var ms = MemoryManager.GetMemoryStream();
465 if (game is MEGame.ME2)
466 {
467 ms.WriteInt32((byte)Type);
468 }
469 else
470 {
471 ms.WriteByte((byte)Type);
472 }
473
474 ms.WriteInt32(DataLength(game));
475 ms.WriteUInt32(ID);
476 return ms;
477 }
478
479 public virtual HIRCObject Clone()
480 {
481 HIRCObject clone = (HIRCObject)MemberwiseClone();
482 clone.unparsed = unparsed?.ArrayClone();
483 return clone;
484 }

487 public class SoundSFXVoice : HIRCObject
488 {
489 public uint Unk1;
490 public SoundState State;
491 public uint AudioID;
492 public uint SourceID;
493 public int UnkType;
494 public int UnkPrefetchLength;
495 public SoundType SoundType; //0=SFX, 1=Voice
496
497 private int ParsedLength => 21 + (State == SoundState.Streamed ? 0 : 8);
498 public override int DataLength(MEGame game) => unparsed.Length + ParsedLength;
499
500 public override byte[] ToBytes(MEGame game)
501 {
502 using MemoryStream ms = WriteHIRCObjectHeader(game);
503 ms.WriteUInt32(Unk1);
504 ms.WriteUInt32((uint)State);
505 ms.WriteUInt32(AudioID);
506 ms.WriteUInt32(SourceID);
507 if (State != SoundState.Streamed)
508 {
509 ms.WriteInt32(UnkType);
510 ms.WriteInt32(UnkPrefetchLength);
511 }
512 ms.WriteByte((byte)SoundType);
513 ms.WriteFromBuffer(unparsed);
514 return ms.ToArray();
515 }
