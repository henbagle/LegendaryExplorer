59 [DebuggerDisplay("MEM MipMap {width}x{height}")]
60 public class MipMap
61 {
62 public byte[] data { get; private set; }
63 public int width { get; private set; }
64 public int height { get; private set; }
65 public int origWidth { get; private set; }
66 public int origHeight { get; private set; }
67
68 public MipMap(int w, int h, PixelFormat format)
69 {
70 width = origWidth = w;
71 height = origHeight = h;
72
73 if (format == PixelFormat.DXT1 ||
74 format == PixelFormat.DXT3 ||
75 format == PixelFormat.DXT5)
76 {
77 if (width < 4)
78 width = 4;
79 if (height < 4)
80 height = 4;
81 }
82
83 data = new byte[getBufferSize(width, height, format)];
84 }
85
86 public MipMap(byte[] src, int w, int h, PixelFormat format)
87 {
88 width = origWidth = w;
89 height = origHeight = h;
90
91 if (format is PixelFormat.DXT1 or PixelFormat.DXT3 or PixelFormat.DXT5 or PixelFormat.BC7)
92 {
93 if (width < 4)
94 width = 4;
95 if (height < 4)
96 height = 4;
97 }
98
99 if (src.Length != getBufferSize(width, height, format))
100 throw new Exception("data size is not valid");
101 data = src;
102 }
103
104 public static int getBufferSize(int w, int h, PixelFormat format)
105 {
106 switch (format)
107 {
108 case PixelFormat.ARGB:
109 return 4 * w * h;
110 case PixelFormat.RGB:
111 return 3 * w * h;
112 case PixelFormat.V8U8:
113 return 2 * w * h;
114 case PixelFormat.DXT3:
115 case PixelFormat.DXT5:
116 case PixelFormat.ATI2:
117 case PixelFormat.BC5:
118 case PixelFormat.G8:
119 case PixelFormat.BC7:
120 return w * h;
121 case PixelFormat.DXT1:
122 return (w * h) / 2;
123 default:
124 throw new Exception("unknown format");
125 }
126 }
127 }

284 public static byte[] convertRawToARGB(byte[] src, ref int w, ref int h, PixelFormat format, bool clearAlpha = false)
285 {
286 byte[] tmpData;
287
288 switch (format)
289 {
290 case PixelFormat.DXT1:
291 case PixelFormat.DXT3:
292 case PixelFormat.DXT5:
293 {
294 if (w < 4 || h < 4)
295 {
296 if (w < 4)
297 w = 4;
298 if (h < 4)
299 h = 4;
300 return new byte[w * h * 4];
301 }
302 tmpData = Image.decompressMipmap(format, src, w, h);
303 break;
304 }
305 case PixelFormat.BC5:
306 case PixelFormat.ATI2:
307 if (w < 4 || h < 4)
308 return new byte[w * h * 4];
309 tmpData = Image.decompressMipmap(format, src, w, h);
310 if (format == PixelFormat.BC5)
311 {
312 // Swap R and G
313 SwapChannelsARGB(tmpData, 1, 2);
314 }
315 break;
316 case PixelFormat.ARGB: tmpData = src; break;
317 case PixelFormat.RGB: tmpData = RGBToARGB(src, w, h); break;
318 case PixelFormat.V8U8: tmpData = V8U8ToARGB(src, w, h); break;
319 case PixelFormat.G8: tmpData = G8ToARGB(src, w, h); break;
320 case PixelFormat.BC7: tmpData = BC7ToARGB(src, w, h); break;
321 default:
322 throw new Exception("invalid texture format " + format);
323 }
324
325 if (clearAlpha)
326 clearAlphaFromARGB(tmpData, w, h);
327
328 return tmpData;
329 }

332 {
333 var decoder = new BcDecoder();
334 var decoded = decoder.DecodeRaw(src, w, h, CompressionFormat.Bc7);
335 byte[] bytes = MemoryMarshal.AsBytes(decoded.AsSpan()).ToArray();
336 // Swap red and blue channels.
337 // idk if there is faster way to do this
338 SwapChannelsARGB(bytes, 0, 2); // decoded data is in ABGR in memory
339
340 return bytes;
341 }

350 private static byte[] SwapChannelsARGB(byte[] bytes, int c1, int c2)
351 {
352 for (int i = 0; i < bytes.Length / 4; i++)
353 {
354 var b1 = bytes[i * 4 + c1];
355 bytes[i * 4 + c1] = bytes[i * 4 + c2];
356 bytes[i * 4 + c2] = b1;
357 }
358
359 return bytes;
360 }

362 private static byte[] ShiftChannels(byte[] bytes, int numRight)
363 {
364 for (int i = 0; i < bytes.Length / 4; i++)
365 {
366 var b1 = bytes[i * 4 + 3];
367 bytes[i * 4 + 1] = bytes[i * 4];
368 bytes[i * 4 + 2] = bytes[i * 4 + 1];
369 bytes[i * 4 + 3] = bytes[i * 4 + 2];
370 bytes[i * 4] = b1;
371 }
372
373 return bytes;
374 }

415 {

418 {

439 private static byte[] V8U8ToARGB(byte[] src, int w, int h)
440 {
441 byte[] tmpData = new byte[w * h * 4];
442 for (int i = 0; i < w * h; i++)
443 {
444 tmpData[4 * i + 0] = 255;
445 tmpData[4 * i + 1] = (byte)(((sbyte)src[2 * i + 1]) + 128);
446 tmpData[4 * i + 2] = (byte)(((sbyte)src[2 * i + 0]) + 128);
447 tmpData[4 * i + 3] = 255;
448 }
449 return tmpData;

489 private static byte[] downscaleARGB(byte[] src, int w, int h)
490 {
491 if (w == 1 && h == 1)
492 throw new Exception("1x1 can not be downscaled");
493
494 byte[] tmpData;
495 if (w == 1 || h == 1)
496 {
497 tmpData = new byte[w * h * 2];
498 for (int srcPos = 0, dstPos = 0; dstPos < w * h * 2; srcPos += 8)
499 {
500 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 0] + src[srcPos + 4 + 0]) >> 1);
501 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 1] + src[srcPos + 4 + 1]) >> 1);
502 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 2] + src[srcPos + 4 + 2]) >> 1);
503 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 3] + src[srcPos + 4 + 3]) >> 1);
504 }
505 }
506 else
507 {
508 tmpData = new byte[w * h];
509 int pitch = w * 4;
510 for (int srcPos = 0, dstPos = 0; dstPos < w * h; srcPos += pitch)
511 {
512 for (int x = 0; x < (w / 2); x++, srcPos += 8)
513 {
514 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 0] + src[srcPos + 4 + 0] + src[srcPos + pitch + 0] + src[srcPos + pitch + 4 + 0]) >> 2);
515 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 1] + src[srcPos + 4 + 1] + src[srcPos + pitch + 1] + src[srcPos + pitch + 4 + 1]) >> 2);
516 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 2] + src[srcPos + 4 + 2] + src[srcPos + pitch + 2] + src[srcPos + pitch + 4 + 2]) >> 2);
517 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 3] + src[srcPos + 4 + 3] + src[srcPos + pitch + 3] + src[srcPos + pitch + 4 + 3]) >> 2);
518 }
519 }
520 }
521
522 return tmpData;
523 }

525 private static byte[] downscaleRGB(byte[] src, int w, int h)
526 {
527 if (w == 1 && h == 1)
528 throw new Exception("1x1 can not be downscaled");
529
530 byte[] tmpData;
531 if (w == 1 || h == 1)
532 {
533 tmpData = new byte[(w * h * 3) / 2];
534 for (int srcPos = 0, dstPos = 0; dstPos < (w * h * 3) / 2; srcPos += 6)
535 {
536 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 0] + src[srcPos + 3 + 0]) >> 1);
537 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 1] + src[srcPos + 3 + 1]) >> 1);
538 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 2] + src[srcPos + 3 + 2]) >> 1);
539 }
540 }
541 else
542 {
543 tmpData = new byte[(w * h * 3) / 4];
544 int pitch = w * 3;
545 for (int srcPos = 0, dstPos = 0; dstPos < (w * h * 3) / 4; srcPos += pitch)
546 {
547 for (int x = 0; x < (w / 2); x++, srcPos += 6)
548 {
549 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 0] + src[srcPos + 3 + 0] + src[srcPos + pitch + 0] + src[srcPos + pitch + 3 + 0]) >> 2);
550 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 1] + src[srcPos + 3 + 1] + src[srcPos + pitch + 1] + src[srcPos + pitch + 3 + 1]) >> 2);
551 tmpData[dstPos++] = (byte)((uint)(src[srcPos + 2] + src[srcPos + 3 + 2] + src[srcPos + pitch + 2] + src[srcPos + pitch + 3 + 2]) >> 2);
552 }
553 }
554 }
555
556 return tmpData;
557 }

643 public void correctMips(PixelFormat dstFormat, bool dxt1HasAlpha = false, byte dxt1Threshold = 128)
644 {
645 byte[] tempData;
646
647 if (pixelFormat != PixelFormat.ARGB)
648 {
649 int w = mipMaps[0].width;
650 int h = mipMaps[0].height;
651 tempData = convertRawToARGB(mipMaps[0].data, ref w, ref h, pixelFormat);
652 }
653 else
654 {
655 tempData = mipMaps[0].data;
656 }
657
658 int width = mipMaps[0].origWidth;
659 int height = mipMaps[0].origHeight;
660
661 if (mipMaps.Count > 1)
662 mipMaps.RemoveRange(1, mipMaps.Count - 1);
663
664 if (dstFormat != pixelFormat || (dstFormat == PixelFormat.DXT1 && !dxt1HasAlpha))
665 {
666 byte[] top = convertToFormat(PixelFormat.ARGB, tempData, width, height, dstFormat, dxt1HasAlpha, dxt1Threshold);
667 mipMaps.RemoveAt(0);
668 mipMaps.Add(new MipMap(top, width, height, dstFormat));
669 pixelFormat = dstFormat;
670 }
671
672 int prevW, prevH;
673 int origW = width;
674 int origH = height;
675 for (; ; )
676 {
677 prevW = width;
678 prevH = height;
679 origW >>= 1;
680 origH >>= 1;
681 if (origW == 0 && origH == 0)
682 break;
683 if (origW == 0)
684 origW = 1;
685 if (origH == 0)
686 origH = 1;
687 width = origW;
688 height = origH;
689
690 if (pixelFormat is PixelFormat.ATI2 or PixelFormat.BC5 && (width < 4 || height < 4))
691 {
692 mipMaps.Add(new MipMap(width, height, pixelFormat));
693 continue;
694 }
695
696 if (pixelFormat is PixelFormat.DXT1 or PixelFormat.DXT3 or PixelFormat.DXT5)
697 {
698 if (width < 4 || height < 4)
699 {
700 if (width < 4)
701 width = 4;
702 if (height < 4)
703 height = 4;
704 mipMaps.Add(new MipMap(origW, origH, pixelFormat));
705 continue;
706 }
707 }
708
709 tempData = downscaleARGB(tempData, prevW, prevH);
710 if (pixelFormat != PixelFormat.ARGB)
711 {
712 byte[] converted = convertToFormat(PixelFormat.ARGB, tempData, origW, origH, pixelFormat, dxt1HasAlpha, dxt1Threshold);
713 mipMaps.Add(new MipMap(converted, origW, origH, pixelFormat));
714 }
715 else
716 {
717 mipMaps.Add(new MipMap(tempData, origW, origH, pixelFormat));
718 }
719 }

722 public static PixelFormat getPixelFormatType(string format)
723 {
724 switch (format)
725 {
726 case "PF_BC1":
727 case "PF_DXT1":
728 return PixelFormat.DXT1;
