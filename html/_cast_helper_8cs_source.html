37 public static ECast GetConversion(VariableType dest, VariableType src)
38 {
39 var destType = dest?.PropertyType ?? EPropertyType.None;
40 var srcType = src?.PropertyType ?? EPropertyType.None;
41
42 if (dest is Class dCls && dCls.SameAsOrSubClassOf("Interface"))
43 {
44 destType = EPropertyType.Interface;
45 }
46 if (src is Class sCls && sCls.SameAsOrSubClassOf("Interface"))
47 {
48 srcType = EPropertyType.Interface;
49 }
50
51 if (src is null && (destType is EPropertyType.Object or EPropertyType.Delegate))
52 {
53 return ECast.AutoConvert;
54 }
55
56 if (src is null && destType is EPropertyType.String)
57 {
58 return ECast.ObjectToString;
59 }
60
61 return Conversions[(int)destType][(int)srcType];
62 }

66 public static int ConversionCost(FunctionParameter dest, VariableType src)
67 {
68 if (dest.VarType == src)
69 {
70 return 0; //exact match
71 }
72
73 if (src?.PropertyType == EPropertyType.Vector && dest.VarType?.PropertyType == EPropertyType.Vector
74 || src?.PropertyType == EPropertyType.Rotator && dest.VarType?.PropertyType == EPropertyType.Rotator)
75 {
76 return 0;
77 }
78 if (dest.VarType is Class c && (src is null || src is ClassType && !c.IsInterface))
79 {
80 return 0;
81 }
82 if (dest.VarType is DelegateType && src is null)
83 {
84 return 0;
85 }
86 if (dest.IsOut)
87 {
88 return int.MaxValue;
89 }
90 if (INTERFACE.CaseInsensitiveEquals(dest.VarType?.Name) && src is Class cls && cls.SameAsOrSubClassOf(INTERFACE))
91 {
92 return 1; //Interface subclass
93 }
94 if (!INTERFACE.CaseInsensitiveEquals(dest.VarType?.Name) && dest.VarType is Class && src is Class)
95 {
96 return 2;
97 }
98 ECast conversion = GetConversion(dest.VarType, src);
99 //if it has 'coerce', any valid conversion is acceptable, otherwise only autoconversions are acceptable
100 if (dest.Flags.Has(UnrealFlags.EPropertyFlags.CoerceParm) ? conversion != ECast.Max : conversion.Has(ECast.AutoConvert))
101 {
102 if (conversion.Has(ECast.Truncate))
103 {
104 return 104; //lossy conversion
105 }
106
107 if (dest.VarType == SymbolTable.FloatType && (src == SymbolTable.IntType || src?.PropertyType == EPropertyType.Byte))
108 {
109 return 103; //int to float conversion
110 }
111
112 return 101; //lossless conversion
113 }
114
115 return int.MaxValue;
116 }
