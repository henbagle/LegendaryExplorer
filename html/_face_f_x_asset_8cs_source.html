25 protected override void Serialize(SerializingContainer2 sc)
26 {
27 if (sc.Game == MEGame.ME2) throw new NotSupportedException("ME2 FaceFXAsset parsing is not supported");
28
29 int int0 = 0;
30
31 var startPos = sc.ms.Position;//come back here to serialize length at the end
32 int length = 0;
33 sc.Serialize(ref length);
34
35 sc.SerializeFaceFXHeader(ref Version);
36
37 sc.Serialize(ref HNodes, SCExt.Serialize);
38 sc.Serialize(ref Names, SCExt.SerializeFaceFXString);
39 sc.Serialize(ref int0);
40 sc.Serialize(ref int0);
41
42 sc.Serialize(ref BoneNodes, SCExt.Serialize);
43 sc.Serialize(ref CombinerNodes, SCExt.Serialize);
44
45 // Do not serialize TableC count - it is the same length as combiner nodes (except not really)
46 // Table C sucks. That's all i'm gonna say about it
47 if (sc.IsLoading)
48 {
49 TableC = new FXATableCElement[CombinerNodes.Count];
50 }
51 for (int i = 0; i < TableC.Length; i++)
52 {
53 if (sc.IsSaving && TableC[i] == null) continue;
54 sc.Serialize(ref TableC[i]);
55 if (TableC[i].StringTuples.Length > 1)
56 {
57 i += TableC[i].StringTuples.Length - 1;
58 }
59 }
60
61 sc.Serialize(ref unk1);
62 sc.Serialize(ref Name);
63
64 sc.Serialize(ref Lines, SCExt.Serialize);
65 sc.Serialize(ref int0);
66
67 sc.Serialize(ref TableD, SCExt.Serialize);
68 sc.Serialize(ref LipSyncPhonemeNames, SCExt.Serialize);
69
70
71 // Serialize length (at the start of the binary)
72 var endingPosition = sc.ms.Position;
73 length = (int)(endingPosition - startPos - 4);
74 sc.ms.JumpTo(startPos);
75 sc.Serialize(ref length);
76
77 // Come back to the end to finish serialization
78 sc.ms.JumpTo(endingPosition);
79 sc.Serialize(ref int0);
80 if (sc.Game is MEGame.LE1 or MEGame.LE2)
81 {
82 sc.Serialize(ref EndingInts, SCExt.Serialize);
83 }
84
85 if (sc.IsLoading)
86 {
87 foreach (var line in Lines)
88 {
89 line.NameAsString = Names[line.NameIndex];
90 }
91 }
92 }

366 public static void Serialize(this SerializingContainer2 sc, ref FaceFXAsset.FXATableCElement el)
367 {
368 if (sc.IsLoading)
369 {
370 el = new FaceFXAsset.FXATableCElement();
371 }
372 sc.Serialize(ref el.Name);
373 sc.Serialize(ref el.unk1);
374 if (sc.IsLoading)
375 {
376 el.StringTuples = new (int, string)[sc.ms.ReadInt32()];
377 }
378 else sc.ms.Writer.WriteInt32(el.StringTuples.Length);
379
380 for (int i = 0; i < el.StringTuples.Length; i++)
381 {
382 (int, string) entry = el.StringTuples[i];
383
384 // Format is String, Int String, Int String...
385 // Store them as int,string tuples and don't serializing the first int.
386 // Every extra string takes one away from the outer loop of TableC entries. I have no idea why.
387 if (i > 0)
388 {
389 sc.Serialize(ref entry.Item1);
390 }
391 sc.SerializeFaceFXString(ref entry.Item2);
392 el.StringTuples[i] = entry;
393 }
