181 static byte CompressAxis(float targetBound, float otherBound, float left, float right)
182 {
183 float range = otherBound - targetBound;
184 if (range == 0.0f)
185 {
186 return 1;
187 }
188 float lFrac = (left - targetBound) / range;
189 float rFrac = (right - targetBound) / range;
190 return (byte)(lFrac >= rFrac
191 ? 128 + Math.Max((int)Math.Floor(127.0f * lFrac) - 1, 0)
192 : 127 - Math.Max((int)Math.Floor(127.0f * rFrac) - 1, 0));
193 }

195 static int FindBestPlane(kDopBuildTriangle[] triangles, int start, int numTris)
196 {
197 int bestPlane = -1;
198 float bestVariance = 0;
199
200 for (int i = 0; i< 3; ++i)
201 {
202 float mean = 0;
203 float variance = 0;
204 for (int j = 0; j < start + numTris; ++j)
205 {
206 mean += Vector3.Dot(triangles[j].Centroid, PlaneNormals[i]);
207 }
208 mean /= numTris;
209 for (int j = 0; j<start + numTris; ++j)
210 {
211 float dotProd = Vector3.Dot(triangles[j].Centroid, PlaneNormals[i]);
212 variance += (dotProd - mean) * (dotProd - mean);
213 }
214 variance /= numTris;
215 if (variance >= bestVariance)
216 {
217 bestPlane = i;
218 bestVariance = variance;
219 }
220 }
221 return bestPlane;
222 }

224 static void PartialSort(int start, int end, int split, kDopBuildTriangle[] triangles, Vector3 bestPlaneNormal)
225 {
226 while (start < end)
227 {

240 {

242 (triangles[end], triangles[pivot]) = (triangles[pivot], triangles[end]);

244

246 {
