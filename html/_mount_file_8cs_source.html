217 public MountFile(string filepath)
218 {
219 using var ms = new MemoryStream(File.ReadAllBytes(filepath));
220 byte b = (byte)ms.ReadByte();
221 switch (b)
222 {
223 case 0:
224 Game = MEGame.ME2;
225 LoadMountFileME2(ms);
226 break;
227 case 0xAC:
228 Game = MEGame.LE2;
229 LoadMountFileME2(ms);
230 break;
231 default:
232 ms.JumpTo(0x4);
233 Game = ms.ReadInt32() == 0x2AC ? MEGame.ME3 : MEGame.LE3;
234 LoadMountFileME3(ms);
235 break;
236 }
237 }

241 private void LoadMountFileME2(MemoryStream ms)
242 {
243 ms.Seek(0x28, SeekOrigin.Begin);
244 MountFlags = new MountFlag(ms.ReadInt32(), true);
245 ms.Seek(0xC, SeekOrigin.Begin);
246 MountPriority = ms.ReadUInt16();
247 ms.Seek(0x2C, SeekOrigin.Begin);
248 ME2Only_DLCHumanName = ms.ReadUnrealString();
249 TLKID = ms.ReadInt32();
250 ME2Only_DLCFolderName = ms.ReadUnrealString();
251 }
252
253 private void LoadMountFileME3(MemoryStream ms)
254 {
255 ms.Seek(0x10, SeekOrigin.Begin);
256 MountPriority = ms.ReadUInt16();
257 ms.Seek(0x18, SeekOrigin.Begin);
258 MountFlags = new MountFlag(ms.ReadInt32(), false);
259 ms.Seek(0x1C, SeekOrigin.Begin);
260 TLKID = ms.ReadInt32();
261 }

267 public void WriteMountFile(string path)
268 {
269 using MemoryStream ms = MemoryManager.GetMemoryStream();
270 switch (Game)
271 {
272 case MEGame.ME2:
273 WriteME2Mount(ms);
274 break;
275 case MEGame.ME3:
276 WriteME3Mount(ms);
277 break;
278 case MEGame.LE2:
279 WriteLE2Mount(ms);
280 break;
281 case MEGame.LE3:
282 WriteLE3Mount(ms);
283 break;
284 default:
285 throw new Exception($"Cannot write a mount file for {Game}!");
286 }
287 ms.WriteToFile(path);
288 }
