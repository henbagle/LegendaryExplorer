103 public static PSK CreateFromSkeletalMesh(SkeletalMesh skelMesh, int lodIdx = 0)
104 {
105 var lod = skelMesh.LODModels[0];
106
107 int numVertices = (int)lod.NumVertices;
108 var psk = new PSK
109 {
110 Points = new List<Vector3>(numVertices),
111 Wedges = new List<PSKWedge>(),
112 Faces = new List<PSKTriangle>(),
113 Materials = new List<PSKMaterial>(),
114 Bones = new List<PSA.PSABone>(),
115 Weights = new List<PSKWeight>()
116 };
117 int numTriangles = 0;
118 var matIndices = new byte[numVertices];
119 foreach (SkelMeshSection section in lod.Sections)
120 {
121 numTriangles += section.NumTriangles;
122 for (int i = 0; i < section.NumTriangles * 3; i++)
123 {
124 uint baseIndex = section.BaseIndex;
125 byte materialIndex = (byte)section.MaterialIndex;
126 matIndices[lod.IndexBuffer[baseIndex + i]] = materialIndex;
127 psk.Faces.Add(new PSKTriangle
128 {
129 //intentionally flipped
130 WedgeIdx1 = (ushort)(baseIndex + i * 3 + 0),
131 WedgeIdx0 = (ushort)(baseIndex + i * 3 + 1),
132 WedgeIdx2 = (ushort)(baseIndex + i * 3 + 2),
133 MatIndex = materialIndex
134 });
135 }
136 }
137
138 foreach (UIndex uIndex in skelMesh.Materials)
139 {
140 psk.Materials.Add(new PSKMaterial
141 {
142 Name = skelMesh.Export.FileRef.GetEntry(uIndex)?.ObjectName.Instanced ?? ""
143 });
144 }
145
146 const float weightUnpackScale = 1f / 255;
147 if (lod.ME1VertexBufferGPUSkin != null)
148 {
149 for (int i = 0; i < lod.ME1VertexBufferGPUSkin.Length; i++)
150 {
151 SoftSkinVertex vertex = lod.ME1VertexBufferGPUSkin[i];
152 psk.Points.Add(new Vector3(vertex.Position.X, vertex.Position.Y * -1, vertex.Position.Z * -1));
153 psk.Wedges.Add(new PSKWedge
154 {
155 MatIndex = matIndices[i],
156 PointIndex = (ushort)i,
157 U = vertex.UV.X,
158 V = vertex.UV.Y
159 });
160 for (int j = 0; j < 4; j++)
161 {
162 if (vertex.InfluenceBones[j] == 0)
163 {
164 break;
165 }
166
167 psk.Weights.Add(new PSKWeight
168 {
169 Bone = vertex.InfluenceBones[j],
170 Weight = vertex.InfluenceWeights[j] * weightUnpackScale,
171 Point = i
172 });
173 }
174 }
175 }
176 else
177 {
178 for (int i = 0; i < lod.VertexBufferGPUSkin.VertexData.Length; i++)
179 {
180 GPUSkinVertex vertex = lod.VertexBufferGPUSkin.VertexData[i];
181 psk.Points.Add(new Vector3(vertex.Position.X, vertex.Position.Y * -1, vertex.Position.Z * -1));
182 psk.Wedges.Add(new PSKWedge
183 {
184 MatIndex = matIndices[i],
185 PointIndex = (ushort)i,
186 U = vertex.UV.X,
187 V = vertex.UV.Y
188 });
189 for (int j = 0; j < 4; j++)
190 {
191 if (vertex.InfluenceBones[j] == 0)
192 {
193 break;
194 }
195
196 psk.Weights.Add(new PSKWeight
197 {
198 Bone = vertex.InfluenceBones[j],
199 Weight = vertex.InfluenceWeights[j] * weightUnpackScale,
200 Point = i
201 });
202 }
203 }
204 }
205 foreach (MeshBone meshBone in skelMesh.RefSkeleton)
206 {
207 psk.Bones.Add(new PSA.PSABone
208 {
209 Name = meshBone.Name.Instanced,
210 Flags = meshBone.Flags,
211 ParentIndex = meshBone.ParentIndex,
212 NumChildren = meshBone.NumChildren,
213 Position = new Vector3(meshBone.Position.X, meshBone.Position.Y * -1, meshBone.Position.Z),
214 Rotation = new Quaternion(meshBone.Orientation.X, meshBone.Orientation.Y * -1, meshBone.Orientation.Z, meshBone.Orientation.W * -1)
215 });
216 }
217
218 return psk;
219 }
