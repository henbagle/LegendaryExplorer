234 public static bool IsInBasegame(string path, MEGame game, string gameRootOverride = null)
235 {
236 if (game is MEGame.UDK or MEGame.LELauncher) return false;
237 if (gameRootOverride is null && GetDefaultGamePath(game) is null)
238 {
239 return false;
240 }
241 if (game == MEGame.LE1 && path.StartsWith(LE1Directory.GetISACTPath(gameRootOverride))) return true;
242 return path.StartsWith(GetCookedPath(game, gameRootOverride));
243 }

247 public static bool IsInOfficialDLC(string path, MEGame game, string gameRootOverride = null)
248 {
249 if (game is MEGame.UDK or MEGame.LELauncher or MEGame.LE1)
250 {
251 return false;
252 }
253 string dlcPath = GetDLCPath(game, gameRootOverride);
254 if (dlcPath is null)
255 {
256 return false;
257 }
258 return OfficialDLC(game).Any(dlcFolder => path.StartsWith(Path.Combine(dlcPath, dlcFolder)));
259 }

278 public static void SaveSettings(List<string> BIOGames)
279 {
280 if (BIOGames.Count != 4)
281 throw new Exception("SaveSettings() requires 4 items in the parameter");
282 try
283 {
284 if (!string.IsNullOrEmpty(BIOGames[0]))
285 {
286 LegendaryExplorerCoreLibSettings.Instance.ME1Directory = BIOGames[0];
287 ME1Directory.DefaultGamePath = BIOGames[0];
288 }
289
290 if (!string.IsNullOrEmpty(BIOGames[1]))
291 {
292 LegendaryExplorerCoreLibSettings.Instance.ME2Directory = BIOGames[1];
293 ME2Directory.DefaultGamePath = BIOGames[1];
294 }
295
296 if (!string.IsNullOrEmpty(BIOGames[2]))
297 {
298 LegendaryExplorerCoreLibSettings.Instance.ME3Directory = BIOGames[2];
299 ME3Directory.DefaultGamePath = BIOGames[2];
300 }
301
302 if (!string.IsNullOrEmpty(BIOGames[3]))
303 {
304 LegendaryExplorerCoreLibSettings.Instance.LEDirectory = BIOGames[3];
305 LE1Directory.ReloadDefaultGamePath();
306 LE2Directory.ReloadDefaultGamePath();
307 LE3Directory.ReloadDefaultGamePath();
308 }
309 }
310 catch (Exception e)
311 {
312 Debug.WriteLine($"Error saving pathing: {e.Message}");
313 }

332 public static List<string> EnumerateGameFiles(MEGame game, List<string> files, Predicate<string> predicate = null)
333 {
334 if (predicate == null)
335 {
336 // KFreon: Set default search predicate.
337 switch (game)
338 {
339 case MEGame.ME1:
340 predicate = s => s.ToLowerInvariant().EndsWith(".upk", true, null) || s.ToLowerInvariant().EndsWith(".u", true, null) || s.ToLowerInvariant().EndsWith(".sfm", true, null);
341 break;
342 case MEGame.ME2:
343 case MEGame.ME3:
344 case MEGame.LE1:
345 case MEGame.LE2:
346 case MEGame.LE3:
347 predicate = s => s.ToLowerInvariant().EndsWith(".pcc", true, null) || s.ToLowerInvariant().EndsWith(".tfc", true, null);
348 break;
349 }
350 }
351
352 return files.Where(t => predicate(t)).ToList();
353 }
