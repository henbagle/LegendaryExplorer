98 public static async Task SaveAsync(this IMEPackage package, string savePath = null, bool? compress = null, bool includeAdditionalPackagesToCook = true, bool includeDependencyTable = true, object diskIOSyncLock = null)
99 {
100//#if !DEBUG && !AZURE

102//#endif

105 foreach (IPackageUser packageUser in package.Users)

107 packageUser.HandleSaveStateChange(true);

109 await Task.Run(() => Save(package, savePath, compress, includeAdditionalPackagesToCook, includeDependencyTable, diskIOSyncLock));

152 private static void MESave(MEPackage pcc, string savePath, bool compress = false, bool includeAdditionalPackagesToCook = true, bool includeDependencyTable = true, object diskIOSyncLock = null)
153 {
154 bool isSaveAs = savePath != null && savePath != pcc.FilePath;
155 int originalLength = -1;
156 if (pcc.Game == MEGame.ME3 && CheckME3Running is not null && NotifyRunningTOCUpdateRequired is not null && !isSaveAs
157 && ME3Directory.GetBioGamePath() != null && pcc.FilePath.StartsWith(ME3Directory.GetBioGamePath()) && CheckME3Running.Invoke())
158 {
159 try
160 {
161 originalLength = (int)new FileInfo(pcc.FilePath).Length;
162 }
163 catch
164 {
165 originalLength = -1;
166 }
167 }
168 try
169 {
170 if (CanReconstruct(pcc, savePath ?? pcc.FilePath))
171 {
172 MESaveDelegate(pcc, savePath ?? pcc.FilePath, isSaveAs, compress, includeAdditionalPackagesToCook, includeDependencyTable, diskIOSyncLock);
173 }
174 else
175 {
176 PackageSaveFailedCallback?.Invoke($"Cannot save ME1 packages with externally referenced textures. Please make an issue on github: {LegendaryExplorerCoreLib.BugReportURL}");
177 }
178 }
179 catch (Exception ex) when (!LegendaryExplorerCoreLib.IsDebug)
180 {
181 PackageSaveFailedCallback?.Invoke($"Error saving {pcc.FilePath}:\n{ex.FlattenException()}");
182 }
183
184 if (originalLength > 0)
185 {
186 string relativePath = Path.GetFullPath(pcc.FilePath).Substring(Path.GetFullPath(ME3Directory.DefaultGamePath).Length);
187 using var bin = MemoryManager.GetMemoryStream();
188 bin.WriteInt32(originalLength);
189 bin.WriteStringLatin1Null(relativePath);
190 File.WriteAllBytes(Path.Combine(ME3Directory.ExecutableFolder, "tocupdate"), bin.ToArray());
191 NotifyRunningTOCUpdateRequired();
192 }
193 }
