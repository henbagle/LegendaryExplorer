263 private static PropertyCollection RemoveDefaultValues(PropertyCollection props, MEGame game)
264 {
265 var result = new PropertyCollection();
266
267 foreach (Property prop in props)
268 {
269 switch (prop)
270 {
271 case ArrayPropertyBase arrayPropertyBase:
272 if (arrayPropertyBase.Count > 0)
273 {
274 result.Add(prop);
275 }
276 break;
277 case BioMask4Property bioMask4Property:
278 if (bioMask4Property.Value != 0)
279 {
280 result.Add(prop);
281 }
282 break;
283 case BoolProperty boolProperty:
284 if (boolProperty.Value)
285 {
286 result.Add(prop);
287 }
288 break;
289 case ByteProperty byteProperty:
290 if (byteProperty.Value != 0)
291 {
292 result.Add(prop);
293 }
294 break;
295 case DelegateProperty delegateProperty:
296 if (delegateProperty.Value != ScriptDelegate.Empty)
297 {
298 result.Add(prop);
299 }
300 break;
301 case EnumProperty enumProperty:
302 if (enumProperty.Value != GlobalUnrealObjectInfo.GetEnumValues(game, enumProperty.EnumType).FirstOrDefault())
303 {
304 result.Add(prop);
305 }
306 break;
307 case FloatProperty floatProperty:
308 if (floatProperty.Value != 0f)
309 {
310 result.Add(prop);
311 }
312 break;
313 case IntProperty intProperty:
314 if (intProperty.Value != 0)
315 {
316 result.Add(prop);
317 }
318 break;
319 case NameProperty nameProperty:
320 if (nameProperty.Value != "None")
321 {
322 result.Add(prop);
323 }
324 break;
325 case ObjectProperty objectProperty:
326 if (objectProperty.Value != 0)
327 {
328 result.Add(prop);
329 }
330 break;
331 case StringRefProperty stringRefProperty:
332 if (stringRefProperty.Value != 0)
333 {
334 result.Add(prop);
335 }
336 break;
337 case StrProperty strProperty:
338 if (strProperty.Value != string.Empty)
339 {
340 result.Add(prop);
341 }
342 break;
343 case StructProperty structProperty:
344 structProperty.Properties = RemoveDefaultValues(structProperty.Properties, game);
345 if (structProperty.Properties.Count > 0)
346 {
347 result.Add(structProperty);
348 }
349 break;
350 case NoneProperty _:
351 break;
352 default:
353 throw new ArgumentOutOfRangeException($"RemoveDefaultValue() invalid property type for {nameof(prop)}: {prop?.PropType}");
354 }
355 }
356
357 return result;

393 private static VariableType GetPropertyType(UProperty obj, FileLib fileLib, PackageCache packageCache = null)
394 {
395 string typeStr = "UNKNOWN";
396 switch (obj)
397 {
398 case UArrayProperty arrayProperty:
399 return new DynamicArrayType(GetPropertyType(fileLib.GetCachedObjectBinary(obj.Export.FileRef.GetUExport(arrayProperty.ElementType), packageCache) as UProperty, fileLib, packageCache));
400 case UBioMask4Property:
401 return SymbolTable.BioMask4Type;
402 case UBoolProperty:
403 return SymbolTable.BoolType;
404 case UByteProperty byteProperty:
405 if (byteProperty.IsEnum)
406 {
407 IEntry enumDef = byteProperty.Enum.GetEntry(obj.Export.FileRef);
408 if (enumDef is ExportEntry enumExp)
409 {
410 return ConvertEnum(fileLib.GetCachedObjectBinary<UEnum>(enumExp, packageCache));
411 }
412 typeStr = enumDef.ObjectName.Instanced;
413 }
414 else
415 {
416 return SymbolTable.ByteType;
417 }
418 break;
419 case UClassProperty classProp:
420 return new ClassType(new VariableType(obj.Export.FileRef.GetEntry(classProp.ClassRef).ObjectName));
421 case UDelegateProperty delegateProperty:
422 {
423 IEntry entry = obj.Export.FileRef.GetEntry(delegateProperty.Function);
424 IEntry functionClass = entry.Parent;
425 for (IEntry delPropClass = delegateProperty.Export; delPropClass != null; delPropClass = delPropClass.Parent)
426 {
427 if (delPropClass.ClassName == "Class")
428 {
429 while (delPropClass != null)
430 {
431 if (delPropClass == functionClass)
432 {
433 return new DelegateType(new Function(entry.ObjectName.Instanced, default, null, null, null));
434 }
435
436 delPropClass = (delPropClass as ExportEntry)?.SuperClass;
437 }
438 break;
439 }
440 }
441 //function is not in scope, fully qualify it
442 string qualifiedFunctionName = entry.ObjectName;
443 while (entry.Parent != null && entry.Parent.ClassName != "Package")
444 {
445 entry = entry.Parent;
446 qualifiedFunctionName = $"{entry.ObjectName.Instanced}.{qualifiedFunctionName}";
447 }
448 return new DelegateType(new Function(qualifiedFunctionName, default, null, null, null));
449 }
450 case UFloatProperty:
451 return SymbolTable.FloatType;
452 case UIntProperty:
453 return SymbolTable.IntType;
454 case UNameProperty:
455 return SymbolTable.NameType;
456 case UStringRefProperty:
457 return SymbolTable.StringRefType;
458 case UStrProperty:
459 return SymbolTable.StringType;
460 case UStructProperty structProperty:
461 typeStr = structProperty.Struct.GetEntry(obj.Export.FileRef)?.ObjectName.Instanced ?? typeStr;
462 break;
463 //if we're just getting the name of the objectref, then Interface and Component are the same as Object
464 //Leave these here in case we do something fancier
465 //case UInterfaceProperty interfaceProperty:
466 // typeStr = interfaceProperty.ObjectRef.GetEntry(obj.Export.FileRef)?.ObjectName.Instanced ?? typeStr; // ?
467 // break;
468 //case UComponentProperty componentProperty:
469 // typeStr = componentProperty.ObjectRef.GetEntry(obj.Export.FileRef)?.ObjectName.Instanced ?? typeStr; // ?
470 // break;
471 case UObjectProperty objectProperty:
472 typeStr = objectProperty.ObjectRef.GetEntry(obj.Export.FileRef)?.ObjectName.Instanced ?? typeStr; // ?
473 break;
474 default:
475 {
476 //if (obj is UObject)
477 typeStr = "Object";
478 break;
479 }
480 }
481
482 return new VariableType(typeStr);

486 {

492 classExport = classExport.Parent as ExportEntry;

530 nextItem = 0;
531 break;
532 }
533 }
534
535 CodeBody body = null;
536 if (decompileBytecode)
537 {
538 body = new ByteCodeDecompiler(obj, containingClass, fileLib, parameters, returnVal?.VarType).Decompile();

541

581 private static List<Statement> ConvertProperties(PropertyCollection properties, ExportEntry defaultsExport, string objectName, bool isStruct, FileLib fileLib)
582 {
583 var statements = new List<Statement>();
584 foreach (var prop in properties)
585 {
586 if (prop is NoneProperty)
587 {
588 continue;
589 }
590
591 Expression name = new SymbolReference(null, prop.Name.Instanced);
592
593 if (fileLib.IsInitialized && fileLib.ReadonlySymbolTable is SymbolTable symbols)
594 {
595 string scope = null;
596 if (isStruct && !defaultsExport.ClassName.CaseInsensitiveEquals("ScriptStruct"))
597 {
598 if (symbols.TryGetType(objectName, out Struct strct))
599 {
600 scope = strct.GetScope();
601 }
602 }
603 else if (symbols.TryGetType(isStruct ? defaultsExport.Parent.ObjectName.Instanced : objectName, out Class containingClass))
604 {
605 if (isStruct)
606 {
607 foreach (Struct strct in containingClass.TypeDeclarations.OfType<Struct>())
608 {
609 if (strct.Name.CaseInsensitiveEquals(objectName))
610 {
611 scope = strct.GetScope();
612 break;
613 }
614 }
615 }
616 else
617 {
618 scope = containingClass.GetScope();
619 }
620 }
621
622 if (scope is not null && symbols.TryGetSymbolInScopeStack(prop.Name, out VariableDeclaration decl, scope) && decl.IsStaticArray)
623 {
624 name = new ArraySymbolRef(name, new IntegerLiteral(prop.StaticArrayIndex), null, null);
625 }
626 }
627 else if (prop.StaticArrayIndex > 0)
628 {
629 name = new ArraySymbolRef(name, new IntegerLiteral(prop.StaticArrayIndex), null, null);
630 }
631 var value = ConvertPropertyValue(prop);
632 statements.Add(new AssignStatement(name, value));
633 }
634
635
636
637 return statements;
638
639 Expression ConvertPropertyValue(Property prop)
640 {
641 IMEPackage pcc = defaultsExport.FileRef;
642 switch (prop)
643 {
644 case BoolProperty boolProperty:
645 return new BooleanLiteral(boolProperty.Value);
646 case ByteProperty byteProperty:
647 return new IntegerLiteral(byteProperty.Value) { NumType = Keywords.BYTE };
648 case BioMask4Property bioMask4Property:
649 return new IntegerLiteral(bioMask4Property.Value) { NumType = Keywords.BIOMASK4 };
650 case DelegateProperty delegateProperty:
651 string funcName = delegateProperty.Value.FunctionName.Instanced;
652 var symRef = new SymbolReference(null, funcName);
653 if (pcc.TryGetEntry(delegateProperty.Value.Object, out IEntry containingObject))
654 {
655 symRef = new CompositeSymbolRef(new ObjectLiteral(new NameLiteral(containingObject.ClassName), new VariableType("Class")), symRef);
656 }
657 return symRef;
658 case EnumProperty enumProperty:
659 if (enumProperty.Value.Instanced.CaseInsensitiveEquals("None"))
660 {
661 return new NoneLiteral();
662 }
663 return new SymbolReference(new EnumValue(enumProperty.Value.Instanced, 0) {Enum = new Enumeration(enumProperty.EnumType.Instanced, new List<EnumValue>(), null, null)}, enumProperty.Value.Instanced);
664 case FloatProperty floatProperty:
665 return new FloatLiteral(floatProperty.Value);
666 case IntProperty intProperty:
667 return new IntegerLiteral(intProperty.Value);
668 case NameProperty nameProperty:
669 return new NameLiteral(nameProperty.Value.Instanced);
670 case ObjectProperty objectProperty:
671 var objRef = objectProperty.Value;
672 if (objRef == 0)
673 return new NoneLiteral();
674 var objEntry = pcc.GetEntry(objRef);
675 if (objEntry is ExportEntry objExp && objExp.InstancedFullPath.StartsWith(defaultsExport.InstancedFullPath, StringComparison.OrdinalIgnoreCase))
676 {
677 //subObject reference
678 return new SymbolReference(null, objExp.ObjectName.Instanced);
679 }
680 else
681 {
682 return new ObjectLiteral(new NameLiteral(objEntry.ClassName.CaseInsensitiveEquals("Class") ? objEntry.ObjectName.Instanced : objEntry.InstancedFullPath), new VariableType(objEntry.ClassName));
683 }
684 case StringRefProperty stringRefProperty:
685 return new StringRefLiteral(stringRefProperty.Value);
686 case StrProperty strProperty:
687 return new StringLiteral(strProperty.Value);
688 case StructProperty structProperty:
689 return new StructLiteral(null, ConvertProperties(structProperty.Properties, defaultsExport, structProperty.StructType, true, fileLib));
690 case ArrayPropertyBase arrayPropertyBase:
691 return new DynamicArrayLiteral(null, arrayPropertyBase.Properties.Select(ConvertPropertyValue).ToList());
692 default:
693 return new SymbolReference(null, "UNSUPPORTED:" + prop.PropType);
