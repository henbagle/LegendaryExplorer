221 public static ExportEntry ImportExport(IMEPackage destPackage, ExportEntry sourceExport, int link, bool importExportDependencies = false,
222 IDictionary<IEntry, IEntry> objectMapping = null, Action<string> errorOccuredCallback = null)
223 {
224 byte[] prePropBinary;
225 if (sourceExport.HasStack)
226 {
227 byte[] dummy = GetStackDummy(destPackage.Game);
228 prePropBinary = new byte[8 + dummy.Length];
229 sourceExport.DataReadOnly[..8].CopyTo(prePropBinary);
230 dummy.CopyTo(prePropBinary, 8);
231 }
232 else
233 {
234 int start = sourceExport.GetPropertyStart();
235 if (start == 16)
236 {
237 var sourceSpan = sourceExport.DataReadOnly[..16];
238 int newNameIdx = destPackage.FindNameOrAdd(sourceExport.FileRef.GetNameEntry(MemoryMarshal.Read<int>(sourceSpan[4..])));
239 prePropBinary = sourceSpan.ToArray();
240 MemoryMarshal.Write(prePropBinary.AsSpan(4), ref newNameIdx);
241 }
242 else
243 {
244 prePropBinary = sourceExport.DataReadOnly[..start].ToArray();
245 }
246 }
247
248 PropertyCollection props = sourceExport.GetProperties();
249
250 //store copy of names list in case something goes wrong
251 if (sourceExport.Game != destPackage.Game)
252 {
253 List<string> names = destPackage.Names.ToList();
254 try
255 {
256 if (sourceExport.Game != destPackage.Game)
257 {
258 bool removedProperties = false;
259 props = EntryPruner.RemoveIncompatibleProperties(sourceExport.FileRef, props, sourceExport.ClassName, destPackage.Game, ref removedProperties);
260 }
261 }
262 catch (Exception exception) when (!LegendaryExplorerCoreLib.IsDebug)
263 {
264 //restore namelist in event of failure.
265 destPackage.restoreNames(names);
266 errorOccuredCallback?.Invoke($"Error occurred while trying to import {sourceExport.ObjectName.Instanced} : {exception.Message}");
267 throw; //should we throw?
268 }
269 }
270
271
272 //takes care of slight header differences between ME1/2 and ME3
273 byte[] newHeader = sourceExport.GenerateHeader(destPackage, false); //The header needs relinked or it will be wrong if it has a component map!
274
278 ObjectBinary binaryData = ExportBinaryConverter.ConvertPostPropBinary(sourceExport, destPackage.Game, props);
279
280 //Set class.
281 IEntry classValue = null;
282 switch (sourceExport.Class)
283 {
284 case ImportEntry sourceClassImport:
285 //The class of the export we are importing is an import. We should attempt to relink this.
286 classValue = GetOrAddCrossImportOrPackage(sourceClassImport.InstancedFullPath, sourceExport.FileRef, destPackage, objectMapping: objectMapping);
287 break;
288 case ExportEntry sourceClassExport:
289 if (IsSafeToImportFrom(sourceExport.FileRef.FilePath, destPackage.Game))
290 {
291 classValue = GetOrAddCrossImportOrPackageFromGlobalFile(sourceClassExport.InstancedFullPath, sourceExport.FileRef, destPackage, objectMapping);
292 break;
293 }
294 classValue = destPackage.FindExport(sourceClassExport.InstancedFullPath);
295 if (classValue is null && importExportDependencies)
296 {
297 IEntry classParent = GetOrAddCrossImportOrPackage(sourceClassExport.ParentFullPath, sourceExport.FileRef, destPackage, true, objectMapping);
298 classValue = ImportExport(destPackage, sourceClassExport, classParent?.UIndex ?? 0, true, objectMapping);
299 }
300 break;
301 }
302
303 //Set superclass
304 IEntry superclass = null;
305 switch (sourceExport.SuperClass)
306 {
307 case ImportEntry sourceSuperClassImport:
308 //The class of the export we are importing is an import. We should attempt to relink this.
309 superclass = GetOrAddCrossImportOrPackage(sourceSuperClassImport.InstancedFullPath, sourceExport.FileRef, destPackage, objectMapping: objectMapping);
310 break;
311 case ExportEntry sourceSuperClassExport:
312 if (IsSafeToImportFrom(sourceExport.FileRef.FilePath, destPackage.Game))
313 {
314 superclass = GetOrAddCrossImportOrPackageFromGlobalFile(sourceSuperClassExport.InstancedFullPath, sourceExport.FileRef, destPackage, objectMapping);
315 break;
316 }
317 superclass = destPackage.FindExport(sourceSuperClassExport.InstancedFullPath);
318 if (superclass is null && importExportDependencies)
319 {
320 IEntry superClassParent = GetOrAddCrossImportOrPackage(sourceSuperClassExport.ParentFullPath, sourceExport.FileRef, destPackage,
321 true, objectMapping);
322 superclass = ImportExport(destPackage, sourceSuperClassExport, superClassParent?.UIndex ?? 0, true, objectMapping);
323 }
324 break;
325 }
326
327 //Check archetype.
328 IEntry archetype = null;
329 switch (sourceExport.Archetype)
330 {
331 case ImportEntry sourceArchetypeImport:
332 archetype = GetOrAddCrossImportOrPackage(sourceArchetypeImport.InstancedFullPath, sourceExport.FileRef, destPackage, objectMapping: objectMapping);
333 break;
334 case ExportEntry sourceArchetypeExport:
335 if (IsSafeToImportFrom(sourceExport.FileRef.FilePath, destPackage.Game))
336 {
337 archetype = GetOrAddCrossImportOrPackageFromGlobalFile(sourceArchetypeExport.InstancedFullPath, sourceExport.FileRef, destPackage, objectMapping);
338 break;
339 }
340 archetype = destPackage.FindExport(sourceArchetypeExport.InstancedFullPath);
341 if (archetype is null && importExportDependencies)
342 {
343 IEntry archetypeParent = GetOrAddCrossImportOrPackage(sourceArchetypeExport.ParentInstancedFullPath, sourceExport.FileRef, destPackage,
344 true, objectMapping);
345 archetype = ImportExport(destPackage, sourceArchetypeExport, archetypeParent?.UIndex ?? 0, true, objectMapping);
346 }
347 break;
348 }
349
350 EndianBitConverter.WriteAsBytes(destPackage.FindNameOrAdd(sourceExport.ObjectName.Name), newHeader.AsSpan(ExportEntry.OFFSET_idxObjectName), destPackage.Endian);
351 EndianBitConverter.WriteAsBytes(sourceExport.ObjectName.Number, newHeader.AsSpan(ExportEntry.OFFSET_indexValue), destPackage.Endian);
352 EndianBitConverter.WriteAsBytes(link, newHeader.AsSpan(ExportEntry.OFFSET_idxLink), destPackage.Endian);
353
354 var newExport = new ExportEntry(destPackage, newHeader, prePropBinary, props, binaryData, sourceExport.IsClass)
355 {
356 Class = classValue,
357 SuperClass = superclass,
358 Archetype = archetype,
359 };
360 destPackage.AddExport(newExport);
361 if (objectMapping != null)
362 {
363 objectMapping[sourceExport] = newExport;
364 }
365
366 return newExport;
367 }

369 public static bool ReplaceExportDataWithAnother(ExportEntry incomingExport, ExportEntry targetExport, Action<string> errorOccuredCallback = null)
370 {
371
372 using var res = new EndianReader(MemoryManager.GetMemoryStream()) { Endian = targetExport.FileRef.Endian };
373 if (incomingExport.HasStack)
374 {
375 res.Writer.Write(incomingExport.DataReadOnly.Slice(0, 8));
376 res.Writer.WriteFromBuffer(GetStackDummy(targetExport.Game));
377 }
378 else
379 {
380 //int start = incomingExport.GetPropertyStart();
381 res.Writer.WriteZeros(incomingExport.GetPropertyStart());
382 //res.Writer.Write(new byte[start], 0, start);
383 }
384
385 //store copy of names list in case something goes wrong
386 List<string> names = targetExport.FileRef.Names.ToList();
387 try
388 {
389 PropertyCollection props = incomingExport.GetProperties();
390 ObjectBinary binary = ExportBinaryConverter.ConvertPostPropBinary(incomingExport, targetExport.Game, props);
391 props.WriteTo(res.Writer, targetExport.FileRef);
392 res.Writer.WriteFromBuffer(binary.ToBytes(targetExport.FileRef));
393 }
394 catch (Exception exception)
395 {
396 //restore namelist in event of failure.
397 targetExport.FileRef.restoreNames(names);
398 errorOccuredCallback?.Invoke($"Error occurred while replacing data in {incomingExport.ObjectName.Instanced} : {exception.Message}");
399 return false;
400 }
401 targetExport.Data = res.ToArray();
402 return true;
403 }

1060 {
1061 // Add objects that we have not referenced yet.

1066 }
1067

1075
1076 //Relink stack

1078 {

1083 AddReferenceLocal(uIndex);
1084 }

1086 else if (relinkingExport.TemplateOwnerClassIdx is var toci && toci >= 0)

1088

1090 AddReferenceLocal(uIndex);
1091 }

1094 if (relinkingExport.SuperClass != null)
1095 AddReferenceLocal(relinkingExport.idxSuperClass);
1096 if (relinkingExport.Archetype != null)
1097 AddReferenceLocal(relinkingExport.idxArchetype);
1098 if (relinkingExport.Class != null)
1099 AddReferenceLocal(relinkingExport.idxClass);
1100

1102 var props = relinkingExport.GetProperties();
1103 foreach (var prop in props)
1104 {
1105 RecursiveGetPropDependencies(prop, AddReferenceLocal);
1106 }
1107

1110 if (bin != null)

1113 foreach (var binUIndex in binUIndexes)
1114 {
1115 AddReferenceLocal(binUIndex.Item1);
1116 }
1117 }
1118
1119 // We have now collected all local references
1120 // We should reach out and see if we need to index others.
1121 foreach (var v in localExportReferences)
1122 {
1123 RecursiveGetDependencies(v, referencedItems, true);
1124 }
