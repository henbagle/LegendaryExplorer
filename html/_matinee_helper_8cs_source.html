20 private static ExportEntry InternalAddGroup(string className, ExportEntry interpData, string groupName)
21 {
22 var properties = new PropertyCollection{new ArrayProperty<ObjectProperty>("InterpTracks")};
23 if (!string.IsNullOrEmpty(groupName))
24 {
25 properties.Add(new NameProperty(groupName, "GroupName"));
26 }
27 properties.Add(CommonStructs.ColorProp(className == "InterpGroup" ? Color.Green : Color.Purple, "GroupColor"));
28 ExportEntry group = CreateNewExport(className, interpData, properties);
29
30 var props = interpData.GetProperties();
31 var groupsProp = props.GetProp<ArrayProperty<ObjectProperty>>("InterpGroups") ?? new ArrayProperty<ObjectProperty>("InterpGroups");
32 groupsProp.Add(new ObjectProperty(group));
33 props.AddOrReplaceProp(groupsProp);
34 interpData.WriteProperties(props);
35
36 return group;
37 }

274 private static void PresetAddTracks(string preset, ExportEntry interpGroup, MEGame game, string param1 = null)
275 {
276 switch (preset)
277 {
278 case "Camera":
279 var move = AddNewTrackToGroup(interpGroup, "InterpTrackMove");
280 AddDefaultPropertiesToTrack(move);
281
282 var fov = AddNewTrackToGroup(interpGroup, "InterpTrackFloatProp");
283 fov.WriteProperty(new InterpCurveFloat().ToStructProperty(fov.Game, "FloatTrack"));
284 fov.WriteProperty(new StrProperty("FOVAngle", "TrackTitle"));
285 fov.WriteProperty(new NameProperty("FOVAngle", "PropertyName"));
286 break;
287
288 case "Director":
289 var dir = AddNewTrackToGroup(interpGroup, "InterpTrackDirector");
290 AddDefaultPropertiesToTrack(dir);
291
292 var dof = AddNewTrackToGroup(interpGroup, "BioEvtSysTrackDOF");
293 AddDefaultPropertiesToTrack(dof);
294 break;
295
296 case "Gesture":
297 var ges = AddNewTrackToGroup(interpGroup, "BioEvtSysTrackGesture");
298 ges.WriteProperty(new ArrayProperty<StructProperty>("m_aGestures"));
299 ges.WriteProperty(new ArrayProperty<StructProperty>("m_aTrackKeys"));
300 ges.WriteProperty(new NameProperty("None", "nmStartingPoseSet"));
301 ges.WriteProperty(new NameProperty("None", "nmStartingPoseAnim"));
302 ges.WriteProperty(new FloatProperty(0, "m_fStartPoseOffset"));
303 ges.WriteProperty(new EnumProperty("None", "EBioTrackAllPoseGroups", game, "ePoseFilter"));
304 ges.WriteProperty(new EnumProperty("None", "EBioGestureAllPoses", game, "eStartingPose"));
305 ges.WriteProperty(new BoolProperty(true, "m_bUseDynamicAnimSets"));
306 ges.WriteProperty(new NameProperty(param1, "m_nmFindActor"));
307 ges.WriteProperty(new StrProperty($"Gesture -- {param1}", "TrackTitle"));
308 break;
309
310 case "Gesture2":
311 var ges2 = AddNewTrackToGroup(interpGroup, "BioEvtSysTrackGesture");
312 var m_aGestures = new ArrayProperty<StructProperty>("m_aGestures");
313 var gesProps = PresetCreateProperties("Gesture2-gesture", game);
314 if (gesProps != null)
315 {
316 m_aGestures.Add(new StructProperty("BioGestureData", gesProps, "BioGestureData"));
317 ges2.WriteProperty(m_aGestures);
318 }
319 ges2.WriteProperty(new NameProperty("None", "nmStartingPoseSet"));
320 ges2.WriteProperty(new NameProperty("None", "nmStartingPoseAnim"));
321 ges2.WriteProperty(new FloatProperty(0, "m_fStartPoseOffset"));
322 ges2.WriteProperty(new BoolProperty(true, "m_bUseDynamicAnimSets"));
323 ges2.WriteProperty(new EnumProperty("None", "EBioTrackAllPoseGroups", game, "ePoseFilter"));
324 ges2.WriteProperty(new NameProperty(param1, "m_nmFindActor"));
325 var m_aTrackKeys = new ArrayProperty<StructProperty>("m_aTrackKeys");
326 var keyProps = PresetCreateProperties("Gesture2-key", game);
327 if (keyProps != null)
328 {
329 m_aTrackKeys.Add(new StructProperty("BioTrackKey", keyProps, "BioTrackKey"));
330 ges2.WriteProperty(m_aTrackKeys);
331 }
332 ges2.WriteProperty(new StrProperty($"Gesture -- {param1}", "TrackTitle"));
333 break;
334
335 case "Actor":
336 var actMove = AddNewTrackToGroup(interpGroup, "InterpTrackMove");
337 AddDefaultPropertiesToTrack(actMove);
338 PresetAddTracks("Gesture", interpGroup, game, param1);
339 break;
340
341 default:
342 return;
343 }
