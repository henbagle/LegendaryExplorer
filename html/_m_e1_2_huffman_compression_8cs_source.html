94 public void serializeTalkfileToExport(ExportEntry export, bool savePackage = false)
95 {
96 /* converts Huffmann Tree to binary form */
97 byte[] treeBuffer = ConvertHuffmanTreeToBuffer();
98
99 var encodedStrings = new List<EncodedString>();
100 int i = 0;
101 foreach (var entry in _inputData)
102 {
103 if (entry.Flags == 0)
104 {
105 if (entry.StringID > 0)
106 entry.Index = -1;
107 else
108 entry.Index = 0;
109 }
110 else
111 {
112 entry.Index = i;
113 i++;
114 var binaryData = new List<BitArray>();
115 int binaryLength = 0;
116
117 /* for every character in a string, put it's binary code into data array */
118 foreach (char c in entry.ASCIIData)
119 {
120 binaryData.Add(_huffmanCodes[c]);
121 binaryLength += _huffmanCodes[c].Count;
122 }
123 byte[] buffer = BitArrayListToByteArray(binaryData, binaryLength);
124 encodedStrings.Add(new EncodedString(entry.ASCIIData.Length, buffer.Length, buffer));
125 }
126 }
127
128
129 /* writing properties */
130 export.WriteProperty(new IntProperty(_inputData.Count, "m_nHashTableSize"));
131
132 using MemoryStream m = MemoryManager.GetMemoryStream();
133 /* writing entries */
134 m.Write(BitConverter.GetBytes(_inputData.Count), 0, 4);
135 foreach (TLKStringRef entry in _inputData)
136 {
137 m.Write(BitConverter.GetBytes(entry.StringID), 0, 4);
138 m.Write(BitConverter.GetBytes(entry.Flags), 0, 4);
139 m.Write(BitConverter.GetBytes(entry.Index), 0, 4);
140 }
141
142 /* writing HuffmanTree */
143 m.Write(treeBuffer, 0, treeBuffer.Length);
144
145 /* writing data */
146 m.Write(BitConverter.GetBytes(encodedStrings.Count), 0, 4);
147 foreach (EncodedString enc in encodedStrings)
148 {
149 m.Write(BitConverter.GetBytes(enc.stringLength), 0, 4);
150 m.Write(BitConverter.GetBytes(enc.encodedLength), 0, 4);
151 m.Write(enc.binaryData, 0, enc.encodedLength);
152 }
153
154 byte[] buff = m.ToArray();
155 export.WriteBinary(buff);
156 if (savePackage)
157 {
158 export.FileRef.Save(export.FileRef.FilePath);
159 }

202 private void PrepareHuffmanCoding()
203 {

207 if (entry.StringID <= 0)

258 private void TraverseHuffmanTree(HuffmanNode node, List<bool> code)
259 {
260 /* check if both sons are null */
261 if (node.Left == node.Right)
262 {
263 BitArray ba = new BitArray(code.ToArray());
264 _huffmanCodes.Add(node.Data, ba);
265 }
266 else
267 {
268 /* adds 0 to the code - process left son*/
269 code.Add(false);
270 TraverseHuffmanTree(node.Left, code);
271 code.RemoveAt(code.Count() - 1);
272
273 /* adds 1 to the code - process right son*/
274 code.Add(true);
275 TraverseHuffmanTree(node.Right, code);
276 code.RemoveAt(code.Count() - 1);
277 }

284 private byte[] ConvertHuffmanTreeToBuffer()
285 {
286 var nodes = new List<HuffmanNode>();
287 var q = new Queue<HuffmanNode>();
288
289 ushort index = 0;
290 q.Enqueue(_huffmanTree[0]);
291
292 while (q.Count > 0)
293 {
294 HuffmanNode node = q.Dequeue();
295 nodes.Add(node);
296 node.ID = index;
297 index++;
298 if (node.Right != null)
299 {
300 q.Enqueue(node.Right);
301 }
302 if (node.Left != null)
303 {
304 q.Enqueue(node.Left);
305 }
306 }
307
308 var output = new List<byte>();
309 output.AddRange(BitConverter.GetBytes((int)index));
310 foreach (HuffmanNode node in nodes)
311 {
312 if (node.leaf)
313 {
314 output.Add(1);
315 output.AddRange(BitConverter.GetBytes(node.Data));
316 }
317 else
318 {
319 output.Add(0);
320 output.AddRange(BitConverter.GetBytes(node.Right.ID));
321 output.AddRange(BitConverter.GetBytes(node.Left.ID));
322 }
323 }
324
325 return output.ToArray();
326 }

335 private static byte[] BitArrayListToByteArray(List<BitArray> bitsList, int bitsCount)
336 {
337 const int BITSPERBYTE = 8;
338
339 int bytesize = bitsCount / BITSPERBYTE;
340 if (bitsCount % BITSPERBYTE > 0)
341 bytesize++;
342
343 byte[] bytes = new byte[bytesize];
344 int bytepos = 0;
345 int bitsRead = 0;
346 byte value = 0;
347 byte significance = 1;
348
349 foreach (BitArray bits in bitsList)
350 {
351 int bitpos = 0;
352
353 while (bitpos < bits.Length)
354 {
355 if (bits[bitpos])
356 {
357 value += significance;
358 }
359 ++bitpos;
360 ++bitsRead;
361 if (bitsRead % BITSPERBYTE == 0)
362 {
363 bytes[bytepos] = value;
364 ++bytepos;
365 value = 0;
366 significance = 1;
367 bitsRead = 0;
368 }
369 else
370 {
371 significance <<= 1;
372 }
373 }
374 }
375 if (bitsRead % BITSPERBYTE != 0)
376 bytes[bytepos] = value;
377 return bytes;
378 }
