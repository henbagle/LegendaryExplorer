11 public static bool TryGetTrash<T>(this IMEPackage pcc, out T entry) where T : class, IEntry
12 {
13 IEntry trashPackage = pcc.FindEntry(UnrealPackageFile.TrashPackageName);
14 if (trashPackage is null)
15 {
16 entry = null;
17 return false;
18 }
19 var trashChildren = trashPackage.GetChildren();
20 bool hasChildren = false;
21 foreach (IEntry trashChild in trashChildren)
22 {
23 if (trashChild is T tChild)
24 {
25 entry = tChild;
26 return true;
27 }
28 hasChildren = true;
29 }
30 if (!hasChildren && trashPackage is T tPackage)
31 {
32 entry = tPackage;
33 return true;
34 }
35 entry = null;
36 return false;
37 }
