93 uIndices.AddRange(Interfaces.SelectMany((kvp, i) => new[] { (kvp.Key, $"InterfacesMap[{i}]"), (kvp.Value, $"InterfacesMap[{i}].PropertyPointer") }));
94
95 uIndices.Add((Defaults, "Defaults"));
96 if (game is MEGame.UDK or MEGame.ME3 or MEGame.LE3)

98 uIndices.AddRange(VirtualFunctionTable.Select((u, i) => (u, $"FullFunctionsList[{i}]")));
99 }
