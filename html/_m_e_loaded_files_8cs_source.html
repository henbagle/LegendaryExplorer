96 }
97 if (fileName != null) loadedFiles[fileName] = filePath;
98 }
99 }
100 }
101
102 if (gameRootOverride == null)
103 {
104 // Cache results
105 if (game == MEGame.ME1) cachedME1LoadedFiles = loadedFiles;
106 else if (game == MEGame.ME2) cachedME2LoadedFiles = loadedFiles;
107 else if (game == MEGame.ME3) cachedME3LoadedFiles = loadedFiles;

109 else if (game == MEGame.LE2) cachedLE2LoadedFiles = loadedFiles;
110 else if (game == MEGame.LE3) cachedLE3LoadedFiles = loadedFiles;
111 }
112
113 return loadedFiles;

132 if (!forceReload)

154 }

157 if (game == MEGame.ME2) cachedME2TargetFiles = loadedFiles;

160 if (game == MEGame.LE2) cachedLE2TargetFiles = loadedFiles;

206 public static IEnumerable<string> GetCookedFiles(MEGame game, string directory, bool includeTFCs = false, bool includeAFCs = false, string[] additionalExtensions = null)
207 {
208 if (game == MEGame.ME1)
209 {
210 var patterns = ME1FilePatterns.ToList();
211 if (additionalExtensions != null)
212 patterns.AddRange(additionalExtensions);
213 return patterns.SelectMany(pattern => Directory.EnumerateFiles(Path.Combine(directory, game.CookedDirName()), pattern, SearchOption.AllDirectories));
214
215 }
216
217 List<string> extensions = new List<string>();
218 if (includeTFCs) extensions.Add("*.tfc");
219 if (includeAFCs) extensions.Add("*.afc");
220 if (additionalExtensions != null)
221 extensions.AddRange(additionalExtensions.Select(x => $"*{x}"));
222 extensions.Add("*.pcc"); //This is last, as any of the lookup methods will see if any of these files types exist in-order. By putting pcc's last, the lookups will be searched first when using
223 //includeTFC or includeAFC.
224 return extensions.SelectMany(pattern => Directory.EnumerateFiles(Path.Combine(directory, game.CookedDirName()), pattern, SearchOption.AllDirectories));
225 }

246 public static bool IsEnabledDLC(string dir, MEGame game)
247 {
248 string dlcName = Path.GetFileName(dir);
249 if (game == MEGame.ME1)
250 {
251 return ME1Directory.OfficialDLC.Contains(dlcName) || File.Exists(Path.Combine(dir, "AutoLoad.ini"));
252 }
253 else if (game == MEGame.LE1) return dlcName.StartsWith("DLC_MOD") && File.Exists(Path.Combine(dir, "AutoLoad.ini"));
254 return dlcName.StartsWith("DLC_") && File.Exists(GetMountDLCFromDLCDir(dir, game));
255 }
