112 {
113 get

116 {
117 if (string.IsNullOrEmpty(LegendaryExplorerCoreLibSettings.Instance?.LEDirectory))
118 {
119 return null;
120 }
121 _gamePath = Path.Combine(LegendaryExplorerCoreLibSettings.Instance.LEDirectory, "Game", "ME1");
122 }
123
124 return Path.GetFullPath(_gamePath); //normalize
125 }
126 set
127 {
128 if (value != null)
129 {
130 if (value.Contains("BioGame", StringComparison.OrdinalIgnoreCase))
131 value = value.Substring(0, value.LastIndexOf("BioGame", StringComparison.OrdinalIgnoreCase));
132 }
133 _gamePath = value;
134 }
