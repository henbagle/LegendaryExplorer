153 public static int IndexOf<T>(this IReadOnlyList<T> readOnlyList, T value)
154 {
155 Contract.Requires(readOnlyList != null, "readOnlyList must not be null");
156 var count = readOnlyList.Count;
157 var equalityComparer = EqualityComparer<T>.Default;
158 for (var i = 0; i < count; i++)
159 {
160 var current = readOnlyList[i];
161 if (equalityComparer.Equals(current, value))
162 {
163 return i;
164 }
165 }
166 return -1;
167 }

198 public static ILookup<TKey, T> ToMultiKeyLookup<T, TKey>(this IEnumerable<T> seq, Func<T, IEnumerable<TKey>> func)
199 {
200 Contract.Requires(seq != null, "seq must not be null");
201 Contract.Requires(func != null, "func must not be null");
202 Contract.Ensures(Contract.Result<ILookup<TKey, T>>() != null, "returned lookup object is not null");
203 Debug.Assert(seq != null);
204 var dicoLookup = new DicoLookup<TKey, T>();
205 foreach (var elem in seq)
206 {
207 var keys = func(elem);
208 if (keys == null)
209 {
210 continue;
211 }
212 foreach (var key in keys)
213 {
214 IEnumerable<T> seqTmp;
215 if (!dicoLookup.TryGetValue(key, out seqTmp))
216 {
217 // A single sequence is added first, and if several elements share the same key we transform the single sequence into a list.
218 // This is an optimisation because a single sequence is cheaper than a List<T>,
219 // and we hope that most keys are not not shared by several elements.
220 dicoLookup.Add(key, elem.ToEnumerable());
221 continue;
222 }
223 var list = seqTmp as List<T>;
224 if (list != null)
225 {
226 list.Add(elem);
227 continue;
228 }
229 list = new List<T> { seqTmp.First(), elem };
230 dicoLookup[key] = list;
231 }
232 }
233 return dicoLookup;
234 }
