79 public static List<Texture2DMipInfo> GetTexture2DMipInfos(ExportEntry exportEntry, string cacheName)
80 {
81 UTexture2D texBin = exportEntry.GetBinaryData<UTexture2D>();
82 var mips = new List<Texture2DMipInfo>();
83 for (int i = 0; i < texBin.Mips.Count; i++)
84 {
85 UTexture2D.Texture2DMipMap binMip = texBin.Mips[i];
86 Texture2DMipInfo mip = new Texture2DMipInfo
87 {
88 Export = exportEntry,
89 index = texBin.Mips.Count - i,
90 storageType = binMip.StorageType,
91 uncompressedSize = binMip.UncompressedSize,
92 compressedSize = binMip.CompressedSize,
93 externalOffset = binMip.DataOffset,
94 Mip = binMip.Mip,
95 TextureCacheName = cacheName, //If this is ME1, this will simply be ignored in the setter
96 width = binMip.SizeX,
97 height = binMip.SizeY
98 };
99
100 if (mip.width == 4 && mips.Exists(m => m.width == mip.width))
101 mip.width = mips.Last().width / 2;
102 if (mip.height == 4 && mips.Exists(m => m.height == mip.height))
103 mip.height = mips.Last().height / 2;
104 if (mip.width == 0)
105 mip.width = 1;
106 if (mip.height == 0)
107 mip.height = 1;
108 mips.Add(mip);
109 }
110
111 return mips;
112 }

146 public void SerializeNewData(Stream ms)
147 {
148 if (!Export.FileRef.Game.IsGame3()) // Is this rig
149 {
150 for (int i = 0; i < 12; i++)
151 ms.WriteByte(0); //12 0s
152 for (int i = 0; i < 4; i++)
153 ms.WriteByte(0); //position in the package. will be updated later
154 }
155
156 ms.WriteInt32(Mips.Count);
157 foreach (var mip in Mips)
158 {
159 ms.WriteUInt32((uint)mip.storageType);
160 ms.WriteInt32(mip.uncompressedSize);
161 ms.WriteInt32(mip.compressedSize);
162 ms.WriteInt32(mip.externalOffset);
163 if (mip.storageType is StorageTypes.pccUnc or StorageTypes.pccLZO or StorageTypes.pccZlib or StorageTypes.pccOodle)
164 {
165 ms.Write(mip.Mip, 0, mip.Mip.Length);
166 }
167 ms.WriteInt32(mip.width);
168 ms.WriteInt32(mip.height);
169 }
170
171 if (Export.Game != MEGame.UDK)
172 {
173 ms.WriteInt32(0);
174 }
175 if (Export.Game != MEGame.ME1)
176 {
177 ms.WriteGuid(TextureGuid);
178 }
179 if (Export.Game == MEGame.UDK)
180 {
181 ms.WriteZeros(4 * 8);
182 }
183 if (Export.Game == MEGame.ME3 || Export.Game.IsLEGame())
184 {
185 ms.WriteInt32(0);
186 if (Export.ClassName == "LightMapTexture2D")
187 {
188 ms.WriteInt32(0);
189 }
190 }

200 public static byte[] GetTextureData(Texture2DMipInfo mipToLoad, MEGame game, string gamePathToUse = null, bool decompress = true, List<string> additionalTFCs = null)
201 {
202 bool dataLoaded = false;
203 var imagebytes = new byte[decompress ? mipToLoad.uncompressedSize : mipToLoad.compressedSize];
204 //Debug.WriteLine("getting texture data for " + mipToLoad.Export.FullPath);
205 if (mipToLoad.storageType == StorageTypes.pccUnc)
206 {
207 Buffer.BlockCopy(mipToLoad.Mip, 0, imagebytes, 0, mipToLoad.uncompressedSize);
208 }
209 else if (mipToLoad.storageType is StorageTypes.pccLZO or StorageTypes.pccZlib or StorageTypes.pccOodle)
210 {
211 if (decompress)
212 {
213 try
214 {
215 TextureCompression.DecompressTexture(imagebytes,
216 new MemoryStream(mipToLoad.Mip),
217 mipToLoad.storageType, mipToLoad.uncompressedSize, mipToLoad.compressedSize);
218 }
219 catch (Exception e)
220 {
221 throw new Exception(GetLocalizedTextureExceptionInternalMessage(e.Message, mipToLoad.storageType.ToString()));
222 }
223 }
224 else
225 {
226 Buffer.BlockCopy(mipToLoad.Mip, 0, imagebytes, 0, mipToLoad.compressedSize);
227 }
228 }
229 else if (((int)mipToLoad.storageType & (int)StorageFlags.externalFile) != 0)
230 {
231 // external
232 string filename = null;
233 List<string> loadedFiles = MELoadedFiles.GetAllGameFiles(gamePathToUse, game, false, true);
234 if (mipToLoad.Export.Game == MEGame.ME1)
235 {
236 var fullPath = loadedFiles.FirstOrDefault(x => Path.GetFileName(x).Equals(mipToLoad.TextureCacheName, StringComparison.InvariantCultureIgnoreCase));
237
238 if (fullPath != null)
239 {
240 filename = fullPath;
241 }
242 else
243 {
244 fullPath = AdditionalME1MasterTexturePackages.FirstOrDefault(x => Path.GetFileName(x).Equals(mipToLoad.TextureCacheName, StringComparison.InvariantCultureIgnoreCase));
245 filename = fullPath ?? throw new FileNotFoundException(GetLocalizedCouldNotFindME1TexturePackageMessage(mipToLoad.TextureCacheName));
246 }
247 }
248 else
249 {
250 string archive = mipToLoad.TextureCacheName + @".tfc";
251
252 var localDirectoryTFCPath = Path.Combine(Path.GetDirectoryName(mipToLoad.Export.FileRef.FilePath), archive);
253 if (File.Exists(localDirectoryTFCPath))
254 {
255 filename = localDirectoryTFCPath;
256 }
257 else if (additionalTFCs != null && additionalTFCs.Any(x => Path.GetFileName(x).Equals(archive, StringComparison.InvariantCultureIgnoreCase)))
258 {
259 filename = additionalTFCs.First(x => Path.GetFileName(x).Equals(archive, StringComparison.InvariantCultureIgnoreCase));
260 }
261 else
262 {
263 //var tfcs = loadedFiles.Where(x => x.EndsWith(@".tfc")).ToList();
264
265 var fullPath = loadedFiles.FirstOrDefault(x => Path.GetFileName(x).Equals(archive, StringComparison.InvariantCultureIgnoreCase));
266 if (fullPath != null)
267 {
268 filename = fullPath;
269 }
270 else if (game == MEGame.ME3 && mipToLoad.TextureCacheName.StartsWith(@"Textures_DLC"))
271 {
272 // Check SFAR
273 var dlcName = mipToLoad.TextureCacheName.Substring(9);
274 if (MEDirectories.OfficialDLC(MEGame.ME3).Contains(dlcName) && ME3Directory.DLCPath != null)
275 {
276 var sfarPath = Path.Combine(ME3Directory.DLCPath, dlcName, game.CookedDirName(), "Default.sfar");
277 if (File.Exists(sfarPath))
278 {
279 var dpackage = new DLCPackage(sfarPath);
280 var entryId = dpackage.FindFileEntry(archive);
281 if (entryId >= 0)
282 {
283 // TFC is in this SFAR
284 imagebytes = dpackage.ReadFromEntry(entryId, mipToLoad.externalOffset, mipToLoad.uncompressedSize);
285 dataLoaded = true;
286 }
287 else
288 {
289 // File not in archive
290 throw new FileNotFoundException(GetLocalizedCouldNotFindME2ME3TextureCacheMessage(archive));
291 }
292 }
293 else
294 {
295 // SFAR not in folder
296 throw new FileNotFoundException(GetLocalizedCouldNotFindME2ME3TextureCacheMessage(archive));
297 }
298 }
299 else
300 {
301 // Not an official DLC
302 throw new FileNotFoundException(GetLocalizedCouldNotFindME2ME3TextureCacheMessage(archive));
303 }
304 }
305 else
306 {
307 throw new FileNotFoundException(GetLocalizedCouldNotFindME2ME3TextureCacheMessage(archive));
308 }
309 }
310 }
311
312 //exceptions above will prevent filename from being null here
313
314 if (!dataLoaded) // The data hasn't been extracted yet (sfar fetch)
315 {
316 try
317 {
318 try
319 {
320 using var fs = new FileStream(filename, FileMode.Open, FileAccess.Read);
321 fs.Seek(mipToLoad.externalOffset, SeekOrigin.Begin);
322 if (mipToLoad.storageType is StorageTypes.extLZO or StorageTypes.extZlib or StorageTypes.extOodle)
323 {
324 if (decompress)
325 {
326 try
327 {
328 using MemoryStream tmpStream = fs.ReadToMemoryStream(mipToLoad.compressedSize);
329 TextureCompression.DecompressTexture(imagebytes, tmpStream, mipToLoad.storageType, mipToLoad.uncompressedSize, mipToLoad.compressedSize);
330 }
331 catch (Exception e)
332 {
333 throw new Exception(GetLocalizedTextureExceptionExternalMessage(e.Message, filename, mipToLoad.storageType.ToString(), mipToLoad.externalOffset.ToString()));
334 }
335 }
336 else
337 {
338 fs.Read(imagebytes, 0, mipToLoad.compressedSize);
339 }
340 }
341 else
342 {
343 fs.Read(imagebytes, 0, mipToLoad.uncompressedSize);
344 }
345 }
346 catch (Exception e)
347 {
348 throw new Exception(GetLocalizedTextureExceptionExternalMessage(e.Message, filename, mipToLoad.storageType.ToString(), mipToLoad.externalOffset.ToString()));
349 }
350 }
351 catch (Exception e)
352 {
353 throw new Exception(GetLocalizedTextureExceptionExternalMessage(e.Message, filename, mipToLoad.storageType.ToString(), mipToLoad.externalOffset.ToString()));
354 }
355 }
356 }
357 return imagebytes;
358 }

382 public string Replace(Image image, PropertyCollection props, string fileSourcePath = null, string forcedTFCName = null, string forcedTFCPath = null, bool isPackageStored = false)
383 {
384 string errors = "";
385 var textureCache = forcedTFCName ?? GetTopMip().TextureCacheName;
386 if (isPackageStored) textureCache = null;
387 string fmt = TextureFormat;
388 PixelFormat pixelFormat = Image.getPixelFormatType(fmt);
389 RemoveEmptyMipsFromMipList();
390
391 PixelFormat newPixelFormat = pixelFormat;
392
393 // Generate mips if necessary
394 if (!image.checkDDSHaveAllMipmaps() || Mips.Count > 1 && image.mipMaps.Count <= 1 || image.pixelFormat != newPixelFormat)
395 {
396 bool dxt1HasAlpha = false;
397 byte dxt1Threshold = 128;
398 if (pixelFormat == PixelFormat.DXT1 && props.GetProp<EnumProperty>("CompressionSettings") is { Value: { Name: "TC_OneBitAlpha" } })
399 {
400 dxt1HasAlpha = true;
401 if (image.pixelFormat is PixelFormat.ARGB or PixelFormat.DXT3 or PixelFormat.DXT5)
402 {
403 errors += "Warning: Texture was converted from full alpha to binary alpha." + Environment.NewLine;
404 }
405 }
406
407 //Generate lower mips
408 image.correctMips(newPixelFormat, dxt1HasAlpha, dxt1Threshold);
409 }
410
411 if (Mips.Count == 1)
412 {
413 var topMip = image.mipMaps[0];
414 image.mipMaps.Clear();
415 image.mipMaps.Add(topMip);
416 }
417 else
418 {
419 // remove lower mipmaps from source image which not exist in game data
420 //Not sure what this does since we just generated most of these mips
421 for (int t = 0; t < image.mipMaps.Count; t++)
422 {
423 if (image.mipMaps[t].origWidth <= Mips[0].width &&
424 image.mipMaps[t].origHeight <= Mips[0].height &&
425 Mips.Count > 1)
426 {
427 if (!Mips.Exists(m => m.width == image.mipMaps[t].origWidth && m.height == image.mipMaps[t].origHeight))
428 {
429 image.mipMaps.RemoveAt(t--);
430 }
431 }
432 }
433
434 // put empty mips if missing
435 for (int t = 0; t < Mips.Count; t++)
436 {
437 if (Mips[t].width <= image.mipMaps[0].origWidth &&
438 Mips[t].height <= image.mipMaps[0].origHeight)
439 {
440 if (!image.mipMaps.Exists(m => m.origWidth == Mips[t].width && m.origHeight == Mips[t].height))
441 {
442 var mipmap = new MipMap(Mips[t].width, Mips[t].height, pixelFormat);
443 image.mipMaps.Add(mipmap);
444 }
445 }
446 }
447 }
448
449 var compressedMips = new List<byte[]>();
450
451 for (int m = 0; m < image.mipMaps.Count; m++)
452 {
453 // Mips go big to small
454
455 if (m > image.mipMaps.Count - 6) // 0 indexed
456 {
457 // Lower 6 mips are never stored compressed so don't bother wasting time compressing the data
458 compressedMips.Add(null);
459 continue;
460 }
461
462 if (Export.Game.IsOTGame() && Export.Game < MEGame.ME3)
463 {
464 compressedMips.Add(TextureCompression.CompressTexture(image.mipMaps[m].data, isPackageStored ? StorageTypes.pccLZO : StorageTypes.extLZO)); //LZO
465 }
466 else if (Export.Game == MEGame.ME3)
467 {
468 compressedMips.Add(TextureCompression.CompressTexture(image.mipMaps[m].data, isPackageStored ? StorageTypes.pccZlib : StorageTypes.extZlib)); //ZLib
469 }
470 else if (Export.Game.IsLEGame())
471 {
472 compressedMips.Add(TextureCompression.CompressTexture(image.mipMaps[m].data, isPackageStored ? StorageTypes.pccOodle : StorageTypes.extOodle)); //Oodle

477 for (int m = 0; m < image.mipMaps.Count; m++)
478 {
479
480 var mipmap = new Texture2DMipInfo
481 {
482 Export = Export,
483 width = image.mipMaps[m].origWidth,
484 height = image.mipMaps[m].origHeight,
485 TextureCacheName = textureCache
486 };
487 bool mipShouldBePackageStored = isPackageStored || m >= image.mipMaps.Count - 6 || textureCache == null;

489 if (Mips.Exists(x => x.width == mipmap.width && x.height == mipmap.height))
490 {
491 var oldMip = Mips.First(x => x.width == mipmap.width && x.height == mipmap.height);

493 }
494 else
495 {
496 // New mipmaps
497 mipmap.storageType = CalculateStorageType(Mips[0].storageType, Export.Game, mipShouldBePackageStored);
498 }

500
501 //Investigate. this has something to do with archive storage types
502 //if (mod.arcTexture != null)
503 //{
504 // if (mod.arcTexture[m].storageType != mipmap.storageType)
505 // {
506 // mod.arcTexture = null;
507 // }
508 //}

511 mipmap.height = image.mipMaps[m].height;
512 mipmaps.Add(mipmap);

521 Texture2DMipInfo x = mipmaps[m];

556 else
557 {

564 mipmap.compressedSize = mipmap.Mip.Length;
565 }
566
567
568 if (mipmap.storageType is StorageTypes.pccUnc or StorageTypes.extUnc)
569 {
570 mipmap.compressedSize = mipmap.uncompressedSize;
571 mipmap.Mip = image.mipMaps[m].data;
572 }
573
574 if (mipmap.storageType is StorageTypes.pccLZO or StorageTypes.pccZlib or StorageTypes.pccOodle)
575 {
576 mipmap.Mip = compressedMips[m];
577 mipmap.compressedSize = mipmap.Mip.Length;
578 }
579
580
581 if (mipmap.storageType is StorageTypes.extZlib or StorageTypes.extLZO or StorageTypes.extUnc or StorageTypes.extOodle)
582 {
583 if (!string.IsNullOrEmpty(mipmap.TextureCacheName) && mipmap.Export.Game != MEGame.ME1)
584 {
585 //Check local dir
586 string tfcarchive = mipmap.TextureCacheName + ".tfc";
587 var localDirectoryTFCPath = forcedTFCPath ?? Path.Combine(Path.GetDirectoryName(mipmap.Export.FileRef.FilePath), tfcarchive);
588 if (File.Exists(localDirectoryTFCPath))
589 {
590 try
591 {
592 using var fs = new FileStream(localDirectoryTFCPath, FileMode.Open, FileAccess.ReadWrite);
593 tfcGuid = fs.ReadGuid();
594 fs.Seek(0, SeekOrigin.End);
595 mipmap.externalOffset = (int)fs.Position;
596 fs.Write(mipmap.Mip, 0, mipmap.compressedSize);
597 }
598 catch (Exception e)
599 {
600 throw new Exception("Problem appending to TFC file " + tfcarchive + ": " + e.Message);
601 }
602 continue;
603 }
604
605 //Check game
606 var gameFiles = MELoadedFiles.GetFilesLoadedInGame(mipmap.Export.Game, includeTFCs: true);
607 if (gameFiles.TryGetValue(tfcarchive, out string archiveFile))
608 {
609 try
610 {
611 using var fs = new FileStream(archiveFile, FileMode.Open, FileAccess.ReadWrite);
612 tfcGuid = fs.ReadGuid();
613 fs.Seek(0, SeekOrigin.End);
614 mipmap.externalOffset = (int)fs.Position;
615 fs.Write(mipmap.Mip, 0, mipmap.compressedSize);
616 }
617 catch (Exception e)
618 {
619 throw new Exception("Problem appending to TFC file " + archiveFile + ": " + e.Message);
620 }
621 continue;
622 }
623
624
625 //Cache not found. Make new TFC
626 try
627 {
628 using var fs = new FileStream(forcedTFCPath ?? localDirectoryTFCPath, FileMode.OpenOrCreate, FileAccess.Write);
629 fs.WriteGuid(tfcGuid);
630 mipmap.externalOffset = (int)fs.Position;
631 fs.Write(mipmap.Mip, 0, mipmap.compressedSize);
632 }
633 catch (Exception e)
634 {
635 throw new Exception("Problem creating new TFC file " + tfcarchive + ": " + e.Message);
636 }
637 continue;
638 }
639 }
640 }
641 mipmaps[m] = mipmap;
642 if (Mips.Count == 1)
643 break;
644 }

650
651 // The bottom 6 mips are apparently always pcc stored. If there is less than 6 mips, set neverstream to true, which tells game
652 // and toolset to never look into archives for mips.
653 if (locallyStored)
654 {
655 // Rules for default neverstream
656 // 1. Must be Package Stored
657 // 2. Must have at least 6 not empty mips
658
659 // Never stream forces all textures in package to be loaded and not streamed
660 // which is required since streaming only works for external textures

779 public static StorageTypes CalculateStorageType(StorageTypes prevType, MEGame game, bool isPackageStored)
780 {
781 StorageTypes type = prevType;
782 //ME2,ME3: Force compression type (not implemented yet)
783 if (game is MEGame.ME3)
784 {
785 if (type is StorageTypes.extLZO) //ME3 LZO -> ZLIB
786 type = StorageTypes.extZlib;
787 if (type is StorageTypes.pccLZO) //ME3 PCC LZO -> PCCZLIB
788 type = StorageTypes.pccZlib;
789 if (type is StorageTypes.extUnc) //ME3 Uncomp -> ZLib
790 type = StorageTypes.extZlib;
791 //Leave here for future. WE might need this after dealing with double compression
792 //if (type == StorageTypes.pccUnc && mipmap.width > 32) //ME3 Uncomp -> ZLib
793 // type = StorageTypes.pccZlib;
794 if (type is StorageTypes.pccUnc && !isPackageStored) //Moving texture to store externally.
795 type = StorageTypes.extZlib;
796 }
797 else if (game is MEGame.ME2)
798 {
799 if (type is StorageTypes.extZlib) //ME2 ZLib -> LZO
800 type = StorageTypes.extLZO;
801 if (type is StorageTypes.pccZlib) //ME2 PCC ZLib -> LZO
802 type = StorageTypes.pccLZO;
803 if (type is StorageTypes.extUnc) //ME2 Uncomp -> LZO
804 type = StorageTypes.extLZO;
805 //Leave here for future. We might neable this after dealing with double compression
806 //if (type == StorageTypes.pccUnc && mipmap.width > 32) //ME2 Uncomp -> LZO
807 // type = StorageTypes.pccLZO;
808 if (type is StorageTypes.pccUnc && !isPackageStored) //Moving texture to store externally. make sure bottom 6 are pcc stored
809 type = StorageTypes.extLZO;
810
811 // TEXTURE WORK BRANCH TOOLING ONLY!!
812 //if (type == StorageTypes.extLZO)
813 // type = StorageTypes.pccLZO;
814 }
815 else if (game.IsLEGame())
816 {
817 if (type is StorageTypes.extUnc)
818 type = StorageTypes.extOodle; // Compress external unc to Oodle
819 if (type is StorageTypes.pccUnc && !isPackageStored) //Moving texture to store externally. make sure bottom 6 are pcc stored
820 type = StorageTypes.extOodle;
821 }
822
823 // Force storage type to either ext or pcc. Does not handle LZMA or empty
824 if (isPackageStored)
825 {
826 type = type switch
827 {
828 StorageTypes.extOodle => StorageTypes.pccOodle,
829 StorageTypes.extLZO => StorageTypes.pccLZO,
830 StorageTypes.extUnc => StorageTypes.pccUnc,
831 StorageTypes.extZlib => StorageTypes.pccZlib,
832 _ => type
833 };
834 }
835 else
836 {
837 type = type switch
838 {
839 StorageTypes.pccOodle => StorageTypes.extOodle,
840 StorageTypes.pccLZO => StorageTypes.extLZO,
841 StorageTypes.pccUnc => StorageTypes.extUnc,
842 StorageTypes.pccZlib => StorageTypes.extZlib,
843 _ => type
844 };
845 }
846
847 return type;
