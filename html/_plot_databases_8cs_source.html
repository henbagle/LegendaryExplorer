252 public static PlotDatabaseBase GetDatabaseContainingElement(PlotElement el, MEGame game)
253 {
254 PlotElement e = el;
255 while (e != null && e.Parent != null)
256 {
257 if (e is PlotModElement pme)

259 return GetModPlotContainerForGame(game).Mods.FirstOrDefault(m => m.ModRoot == pme);

261 e = e.Parent;
262 }
263 return GetBasegamePlotDatabaseForGame(game);

273 public static PlotElement BridgeBasegameAndModDatabases(MEGame game, string appdataFolder)
274 {
275 var mpc = GetModPlotContainerForGame(game);
276 if(mpc.Mods.IsEmpty()) mpc.LoadModsFromDisk(appdataFolder);
277
278 var roots = new List<PlotElement> { GetBasegamePlotDatabaseForGame(game).Root, mpc.GameHeader };
279
280 // Do we already have a bridge?
281 if (roots.Count > 0 && roots.TrueForAll(r => r.Parent != null && r.Parent == roots[0].Parent))
282 {
283 return roots[0].Parent;
284 }
285 else
286 {
287 var gameRootPlot = GetNewRootPlotElement(game, roots, true);
288 return gameRootPlot;
289 }
290 }

300 public static PlotElement GetNewRootPlotElement(MEGame game, List<PlotElement> children, bool assignParents = false)
301 {
302 var gameNumber = game switch
303 {
304 MEGame.LE3 => 3,
305 MEGame.LE2 => 2,
306 MEGame.LE1 => 1,
307 _ => throw new ArgumentException("Cannot create root plot element for non-LE game", nameof(game))
308 };
309 var plotParent = new PlotElement(0, 0, $"Legendary Edition - Mass Effect {gameNumber} Plots",
310 PlotElementType.None, null, children);
311
312 if (assignParents)
313 {
314 foreach (var el in children) el.AssignParent(plotParent);
315 }
316
317 return plotParent;
318 }
