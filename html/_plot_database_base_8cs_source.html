103 public void AddElement(PlotElement element, PlotElement parent)
104 {
105 // This method does not check if the parent is already in the database!
106 if (parent != null)
107 {
108 element.AssignParent(parent);
109 }
110 else if (element.Parent is null)
111 throw new Exception("Element must already have parent, or parent element must be supplied!");
112
113 switch (element.Type)
114 {
115 case PlotElementType.Integer:
116 Ints.Add(element.PlotId, element);
117 return;
118 case PlotElementType.Float:
119 Floats.Add(element.PlotId, element);
120 return;
121 case PlotElementType.Conditional when element is PlotConditional pc:
122 Conditionals.Add(pc.PlotId, pc);
123 return;
124 case PlotElementType.Transition when element is PlotTransition pe:
125 Transitions.Add(pe.PlotId, pe);
126 return;
127 }
128
129 if (element is PlotBool pb)
130 {
131 Bools.Add(pb.PlotId, pb);
132 }
133 else
134 {
135 Organizational.Add(element.ElementId, element);
136 }
137 }

145 public void RemoveElement(PlotElement element, bool removeAllChildren = false)
146 {
147 if (element == Root) throw new ArgumentException("Cannot remove root element from database!");
148 if (element.Children.Any())
149 {
150 if (removeAllChildren)
151 {
152 var children = element.Children.ToList();
153 foreach (var child in children)
154 {
155 RemoveElement(child, true);
156 }
157 }
158 else
159 {
160 throw new ArgumentException("Cannot remove an element with children.");
161 }
162 }
163
164 element.RemoveFromParent();
165 switch (element.Type)
166 {
167 case PlotElementType.Integer:
168 TryRemoveFromDictionary(element, Ints);
169 return;
170 case PlotElementType.Float:
171 TryRemoveFromDictionary(element, Floats);
172 return;
173 case PlotElementType.Conditional when element is PlotConditional pc:
174 TryRemoveFromDictionary(pc, Conditionals);
175 return;
176 case PlotElementType.Transition when element is PlotTransition pe:
177 TryRemoveFromDictionary(pe, Transitions);
178 return;
179 }
180
181 if (element is PlotBool pb)
182 {
183 TryRemoveFromDictionary(pb, Bools);
184 }
185 else
186 {
187 TryRemoveFromDictionary(element, Organizational);
188 }

190 void TryRemoveFromDictionary<T>(T el, Dictionary<int, T> dict) where T : PlotElement
191 {
192 if (dict.TryGetValue(el.RelevantId, out var value) && el == value)
193 {
194 dict.Remove(el.RelevantId);
195 }
196 }
