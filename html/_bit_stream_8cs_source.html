117 protected void InternalFlushBuffer(bool flushAllBits)
118 {
119 if (flushAllBits)
120 {
121 // Add bits until we can write out all bits during the flush.
122 var bitsToFlush = (8 - (BitBuffer.LengthBits % 8)) % 8;
123
124 for (var i = 0; i < bitsToFlush; ++i)
125 {
126 BitBuffer.Write(false);
127 }
128 }
129
130 var toTransfer = Math.Min(TemporaryBuffer.Length, BitBuffer.LengthBytes);
131
132 BitBuffer.ReadBytes(TemporaryBuffer, 0, toTransfer);
