87 private void Compile(Function func)
88 {
89 if (Target is UFunction uFunction)
90 {
91 CompilationUnit = func;
92 var nextItem = uFunction.Children;
93 UProperty returnValue = null;
94 while (uFunction.Export.FileRef.TryGetUExport(nextItem, out ExportEntry nextChild))
95 {
96 var objBin = ObjectBinary.From(nextChild);
97 switch (objBin)
98 {
99 case UProperty uProperty:
100 if (uProperty.PropertyFlags.Has(EPropertyFlags.ReturnParm))
101 {
102 returnValue = uProperty;
103 }
104 else if (uProperty.PropertyFlags.Has(EPropertyFlags.Parm))
105 {
106 parameters.Add(nextChild.ObjectName.Instanced, uProperty);
107 }
108 else
109 {
110 locals.Add(nextChild.ObjectName.Instanced, uProperty);
111 }
112 nextItem = uProperty.Next;
113 break;
114 default:
115 nextItem = 0;
116 break;
117 }
118 }
119
120
121 if (func.IsNative)
122 {
123 foreach (FunctionParameter functionParameter in func.Parameters)
124 {
125 WriteOpCode(OpCodes.NativeParm);
126 WriteObjectRef(ResolveSymbol(functionParameter));
127 }
128 WriteOpCode(OpCodes.Nothing);
129 }
130 else
131 {
132 foreach (FunctionParameter parameter in func.Parameters.Where(param => param.IsOptional))
133 {
134 if (parameter.DefaultParameter is Expression expr)
135 {
136 WriteOpCode(OpCodes.DefaultParmValue);
137
138 using (WriteSkipPlaceholder())
139 {
140 Emit(AddConversion(parameter.VarType, expr));
141 WriteOpCode(OpCodes.EndParmValue);
142 }
143 }
144 else
145 {
146 WriteOpCode(OpCodes.Nothing);
147 }
148 }
149
150 Emit(func.Body);
151
152 WriteOpCode(OpCodes.Return);
153 if (returnValue != null)
154 {
155 WriteOpCode(OpCodes.ReturnNullValue);
156 WriteObjectRef(returnValue.Export);
157 }
158 else
159 {
160 WriteOpCode(OpCodes.Nothing);
161 }
162 }
163
164 WriteOpCode(OpCodes.EndOfScript);
165
166 foreach ((Label label, List<JumpPlaceholder> jumpPlaceholders) in LabelJumps)
167 {
168 foreach (JumpPlaceholder jumpPlaceholder in jumpPlaceholders)
169 {
170 jumpPlaceholder.End(label.StartOffset);
171 }
172 }
173
174 Target.ScriptBytecodeSize = GetMemLength();
175 Target.ScriptBytes = GetByteCode();
176 }
177 else
178 {
179 throw new Exception("Cannot compile a function to a state!");
180 }
181 }

190 private void Compile(State state)

438 private JumpPlaceholder EmitConditionalJump(Expression condition)
439 {
440 JumpPlaceholder jump;
441 if (Game.IsGame3() && condition is InOpReference inOp && inOp.LeftOperand.ResolveType()?.PropertyType == EPropertyType.Object
442 && inOp.LeftOperand.GetType() == typeof(SymbolReference) && inOp.RightOperand is NoneLiteral
443 && (inOp.Operator.OperatorKeyword == "==" || inOp.Operator.OperatorKeyword == "!="))
444 {
445 SymbolReference symRef = (SymbolReference)inOp.LeftOperand;
446 WriteOpCode(symRef.Node.Outer is Function ? OpCodes.OptIfLocal : OpCodes.OptIfInstance);
447 WriteObjectRef(ResolveSymbol(symRef.Node));
448 WriteByte((byte)(inOp.Operator.OperatorKeyword == "!=" ? 1 : 0));
449 jump = WriteJumpPlaceholder(JumpType.Conditional);
450 }
451 else if (Game.IsGame3() && condition is PreOpReference preOp && preOp.Operator.OperatorKeyword == "!"
452 && preOp.Operand.GetType() == typeof(SymbolReference) && preOp.Operand.ResolveType() == SymbolTable.BoolType)
453 {
454 SymbolReference symRef = (SymbolReference)preOp.Operand;
455 WriteOpCode(symRef.Node.Outer is Function ? OpCodes.OptIfLocal : OpCodes.OptIfInstance);
456 WriteObjectRef(ResolveSymbol(symRef.Node));
457 WriteByte(0);
458 jump = WriteJumpPlaceholder(JumpType.Conditional);
459 }
460 else if (Game.IsGame3() && condition is SymbolReference sr && sr.GetType() == typeof(SymbolReference) && sr.ResolveType() == SymbolTable.BoolType)
461 {
462 WriteOpCode(sr.Node.Outer is Function ? OpCodes.OptIfLocal : OpCodes.OptIfInstance);
463 WriteObjectRef(ResolveSymbol(sr.Node));
464 WriteByte(1);
465 jump = WriteJumpPlaceholder(JumpType.Conditional);
466 }
467 else
468 {
469 WriteOpCode(OpCodes.JumpIfNot);
470 jump = WriteJumpPlaceholder(JumpType.Conditional);
471 Emit(condition);
472 }
473
474 return jump;
475 }
