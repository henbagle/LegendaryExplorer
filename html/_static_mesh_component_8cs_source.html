28 public override List<(UIndex, string)> GetUIndexes(MEGame game)
29 {
30 var uIndexes = new List<(UIndex, string)>();
31 for (int i = 0; i < LODData.Length; i++)
32 {
33 StaticMeshComponentLODInfo lodInfo = LODData[i];
34 uIndexes.AddRange(lodInfo.ShadowMaps.Select((uIndex, j) => (uIndex, $"LODData[{i}].ShadowMaps[{j}]")));
35 uIndexes.AddRange(lodInfo.ShadowVertexBuffers.Select((uIndex, j) => (uIndex, $"LODData[{i}].ShadowVertexBuffers[{j}]")));

37 }

69 {
70 var uIndexes = new List<(UIndex, string)>();
71 switch (this)
72 {
73 case LightMap_1D lightMap_1D:
74 uIndexes.Add((lightMap_1D.Owner, $"{prefix}LightMap.Owner"));
75 break;
76 case LightMap_2D lightMap2D:
77 uIndexes.Add((lightMap2D.Texture1, $"{prefix}LightMap.Texture1"));
78 uIndexes.Add((lightMap2D.Texture2, $"{prefix}LightMap.Texture2"));
79 uIndexes.Add((lightMap2D.Texture3, $"{prefix}LightMap.Texture3"));
80 if (game < MEGame.ME3)
81 {
82 uIndexes.Add((lightMap2D.Texture4, $"{prefix}LightMap.Texture4"));
83 }
84 break;
85 case LightMap_4or6 lightMap4Or6:
86 uIndexes.Add((lightMap4Or6.Texture1, $"{prefix}LightMap.Texture1"));
87 uIndexes.Add((lightMap4Or6.Texture2, $"{prefix}LightMap.Texture2"));
88 uIndexes.Add((lightMap4Or6.Texture3, $"{prefix}LightMap.Texture3"));
89 break;
90 }
91
92 return uIndexes;

190 public static void Serialize(this SerializingContainer2 sc, ref LightMap lmap)
191 {
192 if (sc.IsLoading)
193 {
194 var type = (ELightMapType)sc.ms.ReadInt32();
195 switch (type)
196 {
197 case ELightMapType.LMT_None:
198 lmap = new LightMap();
199 break;
200 case ELightMapType.LMT_1D:
201 lmap = new LightMap_1D();
202 break;
203 case ELightMapType.LMT_2D:
204 lmap = new LightMap_2D();
205 break;
206 case ELightMapType.LMT_3:
207 lmap = new LightMap_3();
208 break;
209 case ELightMapType.LMT_4:
210 case ELightMapType.LMT_6:
211 lmap = new LightMap_4or6();
212 break;
213 case ELightMapType.LMT_5:
214 lmap = new LightMap_5();
215 break;
216 default:
217 throw new ArgumentOutOfRangeException();
218 }
219
220 lmap.LightMapType = type;
221 }
222 else
223 {
224 if (!sc.Game.IsGame3() && lmap.LightMapType > ELightMapType.LMT_2D)
225 {
226 lmap = new LightMap();
227 }
228 sc.ms.Writer.WriteInt32((int)lmap.LightMapType);
229 }
230
231 switch (lmap.LightMapType)
232 {
233 case ELightMapType.LMT_None:
234 break;
235 case ELightMapType.LMT_1D:
236 sc.Serialize((LightMap_1D)lmap);
237 break;
238 case ELightMapType.LMT_2D:
239 sc.Serialize((LightMap_2D)lmap);
240 break;
241 case ELightMapType.LMT_3:
242 sc.Serialize((LightMap_3)lmap);
243 break;
244 case ELightMapType.LMT_4:
245 case ELightMapType.LMT_6:
246 sc.Serialize((LightMap_4or6)lmap);
247 break;
248 case ELightMapType.LMT_5:
249 sc.Serialize((LightMap_5)lmap);
250 break;
251 default:
252 throw new ArgumentOutOfRangeException();
253 }
