87 public static byte[] CompressTexture(byte[] inputData, StorageTypes type)
88 {
89 int maxBlockSize = type is StorageTypes.extOodle or StorageTypes.pccOodle ? maxBlockSizeOodle : maxBlockSizeOT;
90 using MemoryStream ouputStream = MemoryManager.GetMemoryStream(inputData.Length);
91 uint compressedSize = 0;
92 int dataBlockLeft = inputData.Length;
93 int newNumBlocks = (inputData.Length + maxBlockSize - 1) / maxBlockSize;
94 var blocks = new ChunkBlock[newNumBlocks];
95 // skip blocks header and table - filled later
96 ouputStream.Seek(SizeOfChunk + SizeOfChunkBlock * newNumBlocks, SeekOrigin.Begin);
97
98 for (int b = 0; b < newNumBlocks; b++)
99 {
100 var block = new ChunkBlock
101 {
102 uncomprSize = Math.Min(maxBlockSize, dataBlockLeft),
103 uncompressedOffset = inputData.Length - dataBlockLeft
104 };
105 dataBlockLeft -= block.uncomprSize;
106 blocks[b] = block;
107 }
108
109 Parallel.For(0, blocks.Length, b =>
110 {
111 ref ChunkBlock block = ref blocks[b];
112 ReadOnlySpan<byte> uncompressedSpan = inputData.AsSpan(block.uncompressedOffset, block.uncomprSize);
113 switch (type)
114 {
115 case StorageTypes.extLZO:
116 case StorageTypes.pccLZO:
117 block.compressedBuffer = LZO2.Compress(uncompressedSpan);
118 break;
119 case StorageTypes.extZlib:
120 case StorageTypes.pccZlib:
121 block.compressedBuffer = Zlib.Compress(uncompressedSpan);
122 break;
123 case StorageTypes.extOodle:
124 case StorageTypes.pccOodle:
125 block.compressedBuffer = OodleHelper.Compress(uncompressedSpan, uncompressedSpan.Length, OodleHelper.OodleFormat.Leviathan, OodleHelper.OodleCompressionLevel.Normal);
126 break;
127 default:
128 throw new Exception("Compression type not expected!");
129 }
130 if (block.compressedBuffer.Length == 0)
131 throw new Exception("Compression failed!");
132 block.comprSize = (uint)block.compressedBuffer.Length;
133 });
134
135 for (int b = 0; b < blocks.Length; b++)
136 {
137 ChunkBlock block = blocks[b];
138 ouputStream.Write(block.compressedBuffer, 0, (int)block.comprSize);
139 compressedSize += block.comprSize;
140 }
141
142 ouputStream.SeekBegin();
143 ouputStream.WriteUInt32(textureTag);
144 ouputStream.WriteInt32(maxBlockSize);
145 ouputStream.WriteUInt32(compressedSize);
146 ouputStream.WriteInt32(inputData.Length);
147 foreach (ChunkBlock block in blocks)
148 {
149 ouputStream.WriteUInt32(block.comprSize);
150 ouputStream.WriteInt32(block.uncomprSize);
151 }
152
153 return ouputStream.ToArray();
154 }

156 public static void DecompressTexture(byte[] DecompressedBuffer, Stream stream, StorageTypes type, int uncompressedSize, int compressedSize)
157 {
158 int maxBlockSize = type is StorageTypes.extOodle or StorageTypes.pccOodle ? maxBlockSizeOodle : maxBlockSizeOT;
159 uint blockTag = stream.ReadUInt32();
160 if (blockTag != textureTag)
161 throw new Exception("Texture magic wrong");
162 int blockSize = stream.ReadInt32();
163 if (blockSize != maxBlockSize)
164 throw new Exception("Texture header broken");
165 int compressedChunkSize = stream.ReadInt32();
166 int uncompressedChunkSize = stream.ReadInt32();
167 if (uncompressedChunkSize != uncompressedSize)
168 throw new Exception("Texture header broken");
169
170 int blocksCount = (uncompressedChunkSize + maxBlockSize - 1) / maxBlockSize;
171 if ((compressedChunkSize + SizeOfChunk + SizeOfChunkBlock * blocksCount) != compressedSize)
172 throw new Exception("Texture header broken");
173
174 var blocks = new List<ChunkBlock>();
175 int uncompressedOffset = 0;
176 for (uint b = 0; b < blocksCount; b++)
177 {
178 var block = new ChunkBlock
179 {
180 comprSize = stream.ReadUInt32(),
181 uncomprSize = stream.ReadInt32(),
182 uncompressedOffset = uncompressedOffset
183 };
184 blocks.Add(block);
185 uncompressedOffset += block.uncomprSize;
186 }
187
188 for (int b = 0; b < blocks.Count; b++)
189 {
190 ChunkBlock block = blocks[b];
191 block.compressedBuffer = stream.ReadToBuffer(blocks[b].comprSize);
192 blocks[b] = block;
193 }
194
195 Parallel.For(0, blocks.Count, b =>
196 {
197 uint dstLen;
198 ChunkBlock block = blocks[b];
199 Span<byte> uncompressedBuff = DecompressedBuffer.AsSpan(block.uncompressedOffset, block.uncomprSize);
200 switch (type)
201 {
202 case StorageTypes.extLZO:
203 case StorageTypes.pccLZO:
204 dstLen = LZO2.Decompress(block.compressedBuffer, uncompressedBuff);
205 break;
206 case StorageTypes.extZlib:
207 case StorageTypes.pccZlib:
208 dstLen = Zlib.Decompress(block.compressedBuffer, uncompressedBuff);
209 break;
210 case StorageTypes.extOodle:
211 case StorageTypes.pccOodle:
212 dstLen = (uint)OodleHelper.Decompress(block.compressedBuffer, uncompressedBuff);
213 break;
214 case StorageTypes.extLZMA:
215 dstLen = (uint)LZMA.Decompress(block.compressedBuffer, uncompressedBuff);
216 break;
217 default:
218 throw new Exception("Compression type not expected!");
219 }
220 if (dstLen != block.uncomprSize)
221 throw new Exception("Decompressed data size not expected!");
222 });
223 }
224 }
