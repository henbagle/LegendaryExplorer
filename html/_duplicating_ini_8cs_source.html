77 public static DuplicatingIni ParseIni(string iniText)
78 {
79 DuplicatingIni di = new DuplicatingIni();
80 var splits = iniText.Split('\n');
81 Section currentSection = null;
82 foreach (var line in splits)
83 {
84 string trimmed = line.Trim();
85 if (string.IsNullOrWhiteSpace(trimmed)) continue; //blank line
86 if (trimmed.StartsWith("[") && trimmed.EndsWith("]"))
87 {
88 //New section
89 currentSection = new Section()
90 {
91 Header = trimmed.Trim('[', ']')
92 };
93 di.Sections.Add(currentSection);
94 }
95 else if (currentSection == null)
96 {
97 continue; //this parser only supports section items
98 }
99 else
100 {
101 currentSection.Entries.Add(new IniEntry(trimmed));
102 }
103 }
104 return di;
105 }
