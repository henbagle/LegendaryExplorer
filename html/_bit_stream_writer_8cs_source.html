234 public override void Write(byte[] buffer, int offset, int count)
235 {
236 var transferred = 0;
237
238 while (transferred < count)
239 {
240 if (BitBuffer.AvailableBytes == 0)
241 {
242 InternalFlushBuffer(false);
243 }
244
245 var toTransfer = Math.Min(count, BitBuffer.AvailableBytes);
246
247 BitBuffer.Write(buffer, offset, toTransfer);
248
249 offset += toTransfer;
250 transferred += toTransfer;
251 }
