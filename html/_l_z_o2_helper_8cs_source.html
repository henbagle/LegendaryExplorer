50 public static uint Decompress(ReadOnlySpan<byte> src, Span<byte> dst)
51 {
52 uint srcLen = (uint)src.Length;
53 uint dstLen = (uint)dst.Length;
54 int status;
55 unsafe
56 {
57 fixed (byte* inPtr = &MemoryMarshal.GetReference(src))
58 fixed (byte* outPtr = &MemoryMarshal.GetReference(dst))
59 {
60 status = LZODecompress(inPtr, srcLen, outPtr, ref dstLen);
61 }
62 }
63 if (status != 0)
64 return 0;
65
66 return dstLen;
67 }

69 public static byte[] Compress(ReadOnlySpan<byte> src)
70 {
71 var bufLen = GetCompressionBound(src.Length);
72 byte[] tmpBuf = MemoryManager.GetByteArray(bufLen);
73 uint dstLen = 0;
74 unsafe
75 {
76 fixed (byte* inPtr = &MemoryMarshal.GetReference(src))
77 {
78 int status = LZOCompress(inPtr, (uint)src.Length, tmpBuf, ref dstLen);
79 if (status != 0)
80 return new byte[0];
81 }
82 }
83
84 byte[] dst = new byte[dstLen];
85 Array.Copy(tmpBuf, dst, dstLen);
86
87 MemoryManager.ReturnByteArray(tmpBuf);
88 return dst;
89 }

91 public static int Compress(ReadOnlySpan<byte> inputBuffer, Span<byte> outputBuffer)
92 {
93 uint compressedCount = (uint)outputBuffer.Length;
94 unsafe
95 {
96 fixed (byte* inPtr = &MemoryMarshal.GetReference(inputBuffer))
97 fixed (byte* outPtr = &MemoryMarshal.GetReference(outputBuffer))
98 {
99 int status = LZOCompress(Unsafe.AsRef<byte>(inPtr), (uint)inputBuffer.Length, Unsafe.AsRef<byte>(outPtr), ref compressedCount);
100 if (status != 0)
101 {
102 return 0;
103 }
104 }
105 }
106 return (int)compressedCount;
107 }
