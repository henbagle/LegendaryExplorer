153 public static IEnumerable<Type> GetParentTypes(this Type type)
154 {
155 // is there any base type?
156 if (type?.GetBaseTypeEx() == null)
157 {
158 yield break;
159 }
160
161 // return all implemented or inherited interfaces
162 foreach (var i in type.GetInterfacesEx())
163 {
164 yield return i;
165 }
166
167 // return all inherited types
168 var currentBaseType = type.GetBaseTypeEx();
169 while (currentBaseType != null)
170 {
171 yield return currentBaseType;
172 currentBaseType = currentBaseType.GetBaseTypeEx();
173 }
174 }

706 public static bool IsInstanceOfTypeEx(this Type type, object objectToCheck)
707 {
708 Argument.IsNotNull(nameof(type), type);
709 Argument.IsNotNull(nameof(objectToCheck), objectToCheck);
710
711 var instanceType = objectToCheck.GetType();
712
713 if (ConvertableDictionary.ContainsKey(type) && ConvertableDictionary[type].Contains(instanceType))
714 {
715 return true;
716 }
717
718 if (type.IsAssignableFromEx(instanceType))
719 {
720 return true;
721 }
722
723 var castable = (from method in type.GetMethodsEx(BindingFlags.Public | BindingFlags.Static)
724 where method.ReturnType == instanceType &&
725 method.Name.Equals("op_Implicit", StringComparison.Ordinal) ||
726 method.Name.Equals("op_Explicit", StringComparison.Ordinal)
727 select method).Any();
728
729 return castable;
730 }
