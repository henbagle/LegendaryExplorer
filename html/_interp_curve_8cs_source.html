147 //ONLY WORKS WHEN T is float, System.Numerics.Vector3 or System.Numerics.Vector2
148 public class InterpCurve<T>
149 {
150 public List<InterpCurvePoint<T>> Points = new();
151 public EInterpMethodType InterpMethod = EInterpMethodType.IMT_UseFixedTangentEvalAndNewAutoTangents;
152
153 public int AddPoint(float inVal, T outVal)
154 {
155 int i = 0;
156 for (; i < Points.Count && Points[i].InVal < inVal; i++) { }
157
158 Points.Insert(i, new InterpCurvePoint<T>(inVal, outVal));
159 return i;
160 }
161 public int AddPoint(float inVal, T outVal, T arriveTangent, T leaveTangent, EInterpCurveMode curveMode)
162 {
163 int i = 0;
164 for (; i < Points.Count && Points[i].InVal < inVal; i++) { }
165
166 Points.Insert(i, new InterpCurvePoint<T>(inVal, outVal, arriveTangent, leaveTangent, curveMode));
167 return i;
168 }
169 public T Eval(float inVal, T defaultVal)
170 {
171 int len = Points.Count;
172
173 if (len == 0)
174 {
175 return defaultVal;
176 }
177
178 if (len == 1 || inVal < Points[0].InVal)
179 {
180 return Points[0].OutVal;
181 }
182
183 if (inVal >= Points[len - 1].InVal)
184 {
185 return Points[len - 1].OutVal;
186 }
187
188 for (int i = 1; i < len; i++)
189 {
190 if (inVal >= Points[i].InVal) continue;
191
192 float diff = Points[i].InVal - Points[i - 1].InVal;
193
194 if (diff == 0f || Points[i - 1].InterpMode == EInterpCurveMode.CIM_Constant)
195 {
196 return Points[i - 1].OutVal;
197 }
198
199 float amount = (inVal - Points[i - 1].InVal) / diff;
200
201 if (Points[i - 1].InterpMode == EInterpCurveMode.CIM_Linear)
202 {
203 return Lerp(Points[i - 1].OutVal, Points[i].OutVal, amount);
204 }
205
206 if (InterpMethod == EInterpMethodType.IMT_UseBrokenTangentEval)
207 {
208 return CubicInterp(Points[i - 1].OutVal, Points[i - 1].LeaveTangent, Points[i].OutVal, Points[i].ArriveTangent, amount);
209 }
210
211 return CubicInterp(Points[i - 1].OutVal, Mul(Points[i - 1].LeaveTangent, diff), Points[i].OutVal, Mul(Points[i].ArriveTangent, diff), amount);
212 }
213
214 return Points[len - 1].OutVal;
215 }
