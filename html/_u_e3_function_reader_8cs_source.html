64 public static UnFunction ReadFunction(ExportEntry export, byte[] dataOverride = null)
65 {
66 MemoryStream memoryStream = dataOverride == null ? export.GetReadOnlyBinaryStream() : new MemoryStream(dataOverride);
67
68 using var reader = new EndianReader(memoryStream) { Endian = export.FileRef.Endian };
69 if (dataOverride is not null)
70 {
71 reader.ReadBytes(12); //netindex?, none
72 }
73 int super = reader.ReadInt32();
74 int nextCompilingChainItem = reader.ReadInt32();
75 if (!export.Game.IsLEGame())
76 {
77 reader.ReadBytes(12);
78 int line = reader.ReadInt32(); // UE3 Debugger leftovers
79 int textPos = reader.ReadInt32(); // UE3 Debugger leftovers
80 }
81 else
82 {
83 reader.Skip(8); // ChildProbe, Line
84 }
85
86 int scriptSize = reader.ReadInt32();
87 byte[] bytecode = reader.ReadBytes(scriptSize);
88 int nativeIndex = reader.ReadInt16();
89 int operatorPrecedence = reader.ReadByte();
90 int functionFlags = reader.ReadInt32();
91 if (export.Game.IsOTGame() && (functionFlags & _flagSet.GetMask("Net")) != 0)
92 {
93 reader.ReadInt16(); // repOffset
94 }
95
96 int friendlyNameIndex = reader.ReadInt32();
97 reader.ReadInt32();
98 return new UnFunction(export, export.FileRef.GetNameEntry(friendlyNameIndex),
99 new FlagValues(functionFlags, _flagSet), bytecode, nativeIndex, operatorPrecedence);
100 }
101 }
