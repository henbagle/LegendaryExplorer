37 {
38 var names = base.GetNames(game);
39
40 for (int i = 0; i < Quests.Count; i++)
41 {
42 var q = Quests[i];
43 for (int j = 0; j < q.Tasks.Count; j++)
44 {
45 var task = q.Tasks[j];

47 }
