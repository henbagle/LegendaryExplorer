15 private static List<EntryStringPair> ExportExportToPackageInternal(ExportEntry sourceExport, IMEPackage targetPackage, out IEntry portedEntry, PackageCache globalCache = null, PackageCache pc = null)
16 {
17 List<EntryStringPair> issues = new List<EntryStringPair>();
18
19 // We will want to cache files in memory to greatly speed this up
20 var newCache = pc == null;
21 pc ??= new PackageCache();
22 Dictionary<ImportEntry, ExportEntry> impToExpMap = new Dictionary<ImportEntry, ExportEntry>();
23
24 // Check and resolve all imports upstream in the level
25 var unresolvableImports = RecursiveGetAllLevelImportsAsExports(sourceExport, impToExpMap, globalCache, pc, true);
26 issues.AddRange(unresolvableImports);
27
28 // Imports are resolvable. We should port in level imports then port in the rest
29 var impToExpMapList = impToExpMap.ToList().OrderBy(x => x.Key.InstancedFullPath.Length); // Shorter names go first... should help ensure parents are generated first... maybe.....
30
31 foreach (var mapping in impToExpMapList)
32 {
33 if (targetPackage.FindEntry(mapping.Key.InstancedFullPath) == null)
34 {
35 // port it in
36 //Debug.WriteLine($"Porting in: {mapping.Key.InstancedFullPath}");
37 var parent = PortParents(mapping.Value, targetPackage);
38 var relinkResults1 = EntryImporter.ImportAndRelinkEntries(EntryImporter.PortingOption.CloneAllDependencies, mapping.Value, targetPackage, parent, true, out _);
39 issues.AddRange(relinkResults1);
40 }
41 else
42 {
43 //Debug.WriteLine($"Already exists due to other porting in: {mapping.Key.InstancedFullPath}");
44 }
45 }
46
47 // Import the original item now
48 var lParent = PortParents(sourceExport, targetPackage);
49
50 // Test the entry was not ported in already, such as from a Parent reference
51 var newEntry = targetPackage.FindEntry(sourceExport.InstancedFullPath);
52 if (newEntry == null)
53 {
54 var relinkResults2 = EntryImporter.ImportAndRelinkEntries(EntryImporter.PortingOption.CloneAllDependencies, sourceExport, targetPackage, lParent, true, out newEntry);
55 issues.AddRange(relinkResults2);
56 }
57
58 if (newCache)
59 {
60 pc.ReleasePackages();
61 }
62
63 portedEntry = newEntry;
64
65 return issues;
66 }

111 public static IEntry PortParents(IEntry source, IMEPackage target, bool importAsImport = false)
112 {
113 var packagename = Path.GetFileNameWithoutExtension(source.FileRef.FilePath);
114 if (packagename != null && IsGlobalNonStartupFile(packagename))
115 {
116 PrepareGlobalFileForPorting(source.FileRef, packagename);
117 }
118
119
120 Stack<IEntry> parentStack = new Stack<IEntry>();
121 IEntry entry = source;
122 while (entry.Parent != null)
123 {
124 parentStack.Push(entry.Parent);
125 entry = entry.Parent;
126 }
127
128 var parentCount = parentStack.Count;
129
130 // Create parents first
131 IEntry parent = null;
132 foreach (var pEntry in parentStack)
133 {
134 var existingEntry = target.FindEntry(pEntry.InstancedFullPath);
135 if (existingEntry == null)
136 {
137 var entriesBC = target.ExportCount;
138 if (pEntry.ClassName == "Package")
139 {
140 // Port in package
141 if (importAsImport)
142 {
143 var newImport = new ImportEntry(target, parent, pEntry.ObjectName)
144 {
145 ClassName = "Package",
146 PackageFile = "Core"
147 };
148 target.AddImport(newImport);
149 parent = newImport;
150 }
151 else
152 {
153 EntryImporter.ImportAndRelinkEntries(EntryImporter.PortingOption.AddSingularAsChild, pEntry, target, parent, false, out parent);
154 }
155 }
156 else
157 {
158 // Port in with relink... this could get really ugly performance wise
159 EntryImporter.ImportAndRelinkEntries(EntryImporter.PortingOption.AddSingularAsChild, pEntry, target, parent, true, out parent);
160 }
161 var entriesAC = target.ExportCount;
162 if (entriesAC - entriesBC > parentCount)
163 {
164 // We ported in too many things!!
165 Debugger.Break();
166 }
167 }
168 else
169 {
170 parent = existingEntry;
171 }
172 }
173
174 return parent;
175 }

182 public static void PrepareGlobalFileForPorting(IMEPackage sourcePackage, string packagename)
183 {
184 // SirC if you ever see this i'm really sorry lol
185 // This is an absolute hackjob cause changing relinker to do this would be hell
186 // Mgamerz jan 2021
187 //
188 // Modify the global package by adding a same-named top level package,
189 // and repoint everything at the root of the package to have this as a link
190 // this way relinker doesn't need changed to import from these files properly
191 if (sourcePackage.FindExport(packagename) == null)
192 {
193 ExportCreator.CreatePackageExport(sourcePackage, packagename);
194 }
195
196 var fileRefExp = sourcePackage.FindExport(packagename);
197 var objReferencer = sourcePackage.Exports.FirstOrDefault(x => x.idxLink == 0 && x.ObjectName == "ObjectReferencer" && x.ClassName == "ObjectReferencer");
198 if (objReferencer != null)
199 {
200 // We need to enumerate all referenced objects. They are the items unique in this 'package' that are not sourced from others.
201 // We only care about the roots as the rest will cascade the change
202 var refed = objReferencer.GetProperty<ArrayProperty<ObjectProperty>>("ReferencedObjects").Select(x => x.ResolveToEntry(sourcePackage) as ExportEntry).Where(x => x != null && x.idxLink == 0).ToList();
203
204 foreach (var refX in refed)
205 {
206 // Point under package so it matches how it would be
207 // if it was cooked into a non-master file
208 refX.idxLink = fileRefExp.UIndex;
209 }
210 }
211
212 sourcePackage.InvalidateLookupTable();
213
214
215 }

225 private static List<EntryStringPair> RecursiveGetAllLevelImportsAsExports(ExportEntry sourceExport, Dictionary<ImportEntry, ExportEntry> resolutionMap, PackageCache globalCache, PackageCache cache, bool clipRootLevelPackgesOnImports = true)
226 {
227 List<EntryStringPair> unresolvableImports = new List<EntryStringPair>();
228 var references = EntryImporter.GetAllReferencesOfExport(sourceExport);
229 var importReferences = references.OfType<ImportEntry>().OrderBy(x => x.InstancedFullPath).ToList();
230 foreach (var import in importReferences)
231 {
232 // Populate imports to next file up
233 var instancedFullPath = import.InstancedFullPath;
234 if (instancedFullPath.StartsWith("Core.") || instancedFullPath.StartsWith("Engine."))
235 continue; // A lot of these are not resolvable cause they're native
236 if (import.Game == MEGame.ME2 && instancedFullPath == "BioVFX_Z_TEXTURES.Generic.Glass_Shards_Norm")
237 continue; // This texture for some reason is not stored in package files... not sure where, or how it is loaded into memory
238 if (import.IsAKnownNativeClass())
239 continue; // Known native items can never be imported
240 var resolved = EntryImporter.ResolveImport(import, globalCache, cache, clipRootLevelPackage: clipRootLevelPackgesOnImports);
241 if (resolved == null)
242 {
243 unresolvableImports.Add(new EntryStringPair(import, $"Import {import.InstancedFullPath} could not be resolved - cannot be safely used"));
244 continue;
245 }
246
247 // see if this is level resolved item
248 var sourcePath = Path.GetFileNameWithoutExtension(resolved.FileRef.FilePath);
249 if (IsLevelFile(sourcePath) || IsGlobalNonStartupFile(sourcePath))
250 {
251
252 resolutionMap[import] = resolved;
253 RecursiveGetAllLevelImportsAsExports(resolved, resolutionMap, globalCache, cache, clipRootLevelPackgesOnImports);
254 }
255 }
256
257 return unresolvableImports;
258 }

273 public static bool IsLevelFile(string packageNameWithoutExtension)
274 {
275 if (packageNameWithoutExtension.StartsWith("BioA_", StringComparison.InvariantCultureIgnoreCase)) return true; //Art
276 if (packageNameWithoutExtension.StartsWith("BioD_", StringComparison.InvariantCultureIgnoreCase)) return true; //Design
277 if (packageNameWithoutExtension.StartsWith("BioP_", StringComparison.InvariantCultureIgnoreCase)) return true; //Persistent
278 if (packageNameWithoutExtension.StartsWith("BioS_", StringComparison.InvariantCultureIgnoreCase)) return true; // Sound (ME2)
279 if (packageNameWithoutExtension.StartsWith("BioSnd_", StringComparison.InvariantCultureIgnoreCase)) return true; // Sound (ME2,ME3)
280 return false;
281 }
