96 private static string _gamePath;
97 public static string DefaultGamePath
98 {
99 get
100 {
101 if (string.IsNullOrEmpty(_gamePath))
102 {
103 if (string.IsNullOrEmpty(LegendaryExplorerCoreLibSettings.Instance?.LEDirectory))
104 {
105 return null;
106 }
107 _gamePath = Path.Combine(LegendaryExplorerCoreLibSettings.Instance.LEDirectory, "Game", "ME2");
108 }
109 return Path.GetFullPath(_gamePath); //normalize
110 }
111 set
112 {
113 if (value != null)
114 {
115 if (value.Contains("BioGame", StringComparison.OrdinalIgnoreCase))
116 value = value.Substring(0, value.LastIndexOf("BioGame", StringComparison.OrdinalIgnoreCase));
117 }
118 _gamePath = value;
119 }
120 }

132 {
133 DefaultGamePath = Path.Join(LegendaryExplorerCoreLibSettings.Instance.LEDirectory, "Game", "ME2");
134 }
135 else
136 {

138 if (LEDirectory.LookupDefaultPath()) ReloadDefaultGamePath(false);
