464 {
465 UProperty childProp = null;

467 {

469 if (current.Export.ClassName == ByteCodeCompilerVisitor.PropertyTypeName(member.VarType))

475 }
476 propertyCompletions.Add(CreatePropertyStub(member, funcObj.Export, ref childProp));
477 properties.Add(childProp);
478 }
479 while (existingEnumerator.MoveNext())
480 {
481 EntryPruner.TrashEntryAndDescendants(existingEnumerator.Current.Export);

693 uMapProperty.ValueType = 0;
