22 public static List<EntryStringPair> RelinkAll(IDictionary<IEntry, IEntry> crossPccObjectMap, bool importExportDependencies = false)
23 {
24 var relinkReport = new List<EntryStringPair>();
25 //relink each modified export
26
27 //We must convert this to a list, as this list will be updated as imports are cross mapped during relinking.
28 //This process speeds up same-relinks later.
29 //This is a list because otherwise we would get a concurrent modification exception.
30 //Since we only enumerate exports and append imports to this list we will not need to worry about recursive links
31 //I am sure this won't come back to be a pain for me.
32
33 // Used for quick mapping lookups
34 var crossPackageMap = new ListenableDictionary<IEntry, IEntry>(crossPccObjectMap);
35
36 // Used to perform a full relink
37 var mappingList = crossPackageMap.ToList();
38
39 crossPackageMap.OnDictionaryChanged += (sender, args) =>
40 {
41 if (args.Type == DictChangeType.AddItem)
42 {
43 mappingList.Add(new KeyValuePair<IEntry, IEntry>(args.Key, args.Value));
44 //Debug.WriteLine($"Adding relink mapping {args.Key.ObjectName} {args.Key.UIndex} -> {args.Value.UIndex}");
45 }
46 };
47 //can't be a foreach since we might append things to the list
48 // ReSharper disable once ForCanBeConvertedToForeach
49
50 // Used for forcing further relinks
51 int i = 0;
52 while (i < mappingList.Count)
53 {
54 var entryMap = mappingList[i];
55 if (entryMap.Key is ExportEntry sourceExport && entryMap.Value is ExportEntry relinkingExport)
56 {
57 Relink(sourceExport, relinkingExport, crossPackageMap, relinkReport, importExportDependencies);
58 }
59 i++;
60
61 // Potential way to work around OrderedMultiValueDictionary performance issues - if the concat on listenabledictionary doesn't work
62 // Have we reached the end of the current pass? If so, recreate the list. This means we only have to generate the list
63 // a few times instead of thousands
64 // If we are at end pass this won't make a difference
65 //if (i == mappingList.Count && crossPackageMap.Count != mappingList.Count)
66 //{
67 // mappingList = crossPackageMap.ToList();
68 //}
69 }
70 crossPccObjectMap.ReplaceAll(crossPccObjectMap);
71 return relinkReport;
72 }

74 public static void Relink(ExportEntry sourceExport, ExportEntry relinkingExport, IDictionary<IEntry, IEntry> crossPCCObjectMappingList,
75 List<EntryStringPair> relinkReport, bool importExportDependencies = false)
76 {
77 IMEPackage sourcePcc = sourceExport.FileRef;
78
79 // Relink header (component map)
80 if (relinkingExport.HasComponentMap && relinkingExport.ComponentMap.Count > 0)
81 {
82 OrderedMultiValueDictionary<NameReference, int> newComponentMap = new OrderedMultiValueDictionary<NameReference, int>();
83 foreach (var cmk in sourceExport.ComponentMap)
84 {
85 // This code makes a lot of assumptions, like how components are always directly below the current export
86 var nameIndex = relinkingExport.FileRef.FindNameOrAdd(cmk.Key.Name);
87 EntryImporter.ImportAndRelinkEntries(EntryImporter.PortingOption.CloneAllDependencies, sourceExport.FileRef.GetUExport(cmk.Value + 1), relinkingExport.FileRef, relinkingExport, true, out var newComponent);
88
89 newComponentMap.Add(new KeyValuePair<NameReference, int>(cmk.Key, newComponent.UIndex - 1)); // TODO: Relink the
90 }
91
92 relinkingExport.ComponentMap = newComponentMap;
93 }
94
95
96
97 byte[] prePropBinary = relinkingExport.GetPrePropBinary();
98 //Relink stack
99 if (relinkingExport.HasStack)
100 {
101
102 int uIndex = BitConverter.ToInt32(prePropBinary, 0);
103 var relinkResult = relinkUIndex(sourceExport.FileRef, relinkingExport, ref uIndex, "Stack: Node",
104 crossPCCObjectMappingList, "", importExportDependencies);
105 if (relinkResult is null)
106 {
107 prePropBinary.OverwriteRange(0, BitConverter.GetBytes(uIndex));
108 }
109 else
110 {
111 relinkReport.Add(relinkResult);
112 }
113
114 uIndex = BitConverter.ToInt32(prePropBinary, 4);
115 relinkResult = relinkUIndex(sourceExport.FileRef, relinkingExport, ref uIndex, "Stack: StateNode",
116 crossPCCObjectMappingList, "", importExportDependencies);
117 if (relinkResult is null)
118 {
119 prePropBinary.OverwriteRange(4, BitConverter.GetBytes(uIndex));
120 }
121 else
122 {
123 relinkReport.Add(relinkResult);
124 }
125 }
126 //Relink Component's TemplateOwnerClass
127 else if (relinkingExport.TemplateOwnerClassIdx is var toci and >= 0)
128 {
129
130 int uIndex = BitConverter.ToInt32(prePropBinary, toci);
131 var relinkResult = relinkUIndex(sourceExport.FileRef, relinkingExport, ref uIndex, "TemplateOwnerClass",
132 crossPCCObjectMappingList, "", importExportDependencies);
133 if (relinkResult is null)
134 {
135 prePropBinary.OverwriteRange(toci, BitConverter.GetBytes(uIndex));
136 }
137 else
138 {
139 relinkReport.Add(relinkResult);
140 }
141 }
142
143 //Relink Properties
144 // NOTES: this used to be relinkingExport, not source, Changed near end of jan 2021 - Mgamerz - Due to ported items possibly not having way to reference original items
145 PropertyCollection props = sourceExport.GetProperties();
146 bool removedProperties = false;
147 if (sourcePcc.Game != relinkingExport.Game && props.Count > 0)
148 {
149 props = EntryPruner.RemoveIncompatibleProperties(sourcePcc, props, sourceExport.ClassName, relinkingExport.Game, ref removedProperties);
150 if (removedProperties)
151 {
152 relinkReport.Add(new EntryStringPair(relinkingExport, $"{relinkingExport.UIndex} {relinkingExport.InstancedFullPath}: Some properties were removed from this object because they do not exist in {relinkingExport.Game}!"));
153 }
154 }
155 relinkPropertiesRecursive(sourcePcc, relinkingExport, props, crossPCCObjectMappingList, "", relinkReport, importExportDependencies);
156
157 //Relink Binary
158 try
159 {
160 if (relinkingExport.Game != sourcePcc.Game && (relinkingExport.IsClass || relinkingExport.ClassName is "State" or "Function"))
161 {
162 relinkReport.Add(new EntryStringPair(relinkingExport, $"{relinkingExport.UIndex} {relinkingExport.InstancedFullPath} binary relinking failed. Cannot port {relinkingExport.ClassName} between games!"));
163 }
164 else if (ObjectBinary.From(relinkingExport) is ObjectBinary objBin)
165 {
166
167 // This doesn't work on functions! Finding the children through the probe doesn't work
168
169 if (objBin.Export is {ClassName: "State"})
170 {
171 // We can't relink labeltable as it depends on none
172 // Use the source export instead
173 objBin = ObjectBinary.From(sourceExport);
174 }
175
176
177 List<(UIndex, string)> indices = objBin.GetUIndexes(relinkingExport.FileRef.Game);
178
179 foreach ((UIndex uIndex, string propName) in indices)
180 {
181 var result = relinkUIndex(sourcePcc, relinkingExport, ref uIndex.value, $"(Binary Property: {propName})",
182 crossPCCObjectMappingList, "", importExportDependencies);
183 if (result != null)
184 {
185 relinkReport.Add(result);
186 }
187 }
188
189 //UStruct is abstract baseclass for Class, State, and Function, and can have script in it
190 if (objBin is UStruct uStructBinary && uStructBinary.ScriptBytes.Length > 0)
191 {
192 if (relinkingExport.Game == MEGame.ME3 || relinkingExport.Game.IsLEGame())
193 {
194 (List<Token> tokens, _) = Bytecode.ParseBytecode(uStructBinary.ScriptBytes, sourceExport);
195 foreach (Token token in tokens)
196 {
197 RelinkToken(token, uStructBinary.ScriptBytes, sourceExport, relinkingExport,
198 crossPCCObjectMappingList, relinkReport, importExportDependencies);
199 }
200 }
201 else
202 {
203 var func = sourceExport.ClassName == "State" ? UE3FunctionReader.ReadState(sourceExport) : UE3FunctionReader.ReadFunction(sourceExport);
204 func.Decompile(new TextBuilder(), false, false); //parse bytecode
205 var nameRefs = func.NameReferences;
206 var entryRefs = func.EntryReferences;
207 foreach ((long position, NameReference nameRef) in nameRefs)
208 {
209 if (position < uStructBinary.ScriptBytes.Length)
210 {
211 RelinkNameReference(nameRef.Name, position, uStructBinary.ScriptBytes, relinkingExport);
212 }
213 }
214
215 foreach ((long position, IEntry entry) in entryRefs)
216 {
217 if (position < uStructBinary.ScriptBytes.Length)
218 {
219 RelinkUnhoodEntryReference(entry, position, uStructBinary.ScriptBytes, sourceExport, relinkingExport,
220 crossPCCObjectMappingList, relinkReport, importExportDependencies);
221 }
222 }
223 }
224 }
225 relinkingExport.WritePrePropsAndPropertiesAndBinary(prePropBinary, props, objBin);
226 return;
227 }
228 }
229 catch (Exception e) when (!LegendaryExplorerCoreLib.IsDebug)
230 {
231 relinkReport.Add(new EntryStringPair(relinkingExport, $"{relinkingExport.UIndex} {relinkingExport.InstancedFullPath} binary relinking failed due to exception: {e.Message}"));
232 }
233
234 relinkingExport.WritePrePropsAndProperties(prePropBinary, props, removedProperties ? relinkingExport.propsEnd() : sourceExport.propsEnd());
235 }

237 private static void relinkPropertiesRecursive(IMEPackage importingPCC, ExportEntry relinkingExport, PropertyCollection transplantProps,
238 IDictionary<IEntry, IEntry> crossPCCObjectMappingList, string prefix, List<EntryStringPair> relinkResults,
239 bool importExportDependencies = false)
240 {
241 foreach (Property prop in transplantProps)
242 {
243 //Debug.WriteLine($"{prefix} Relink recursive on {prop.Name}");
244 if (prop is StructProperty structProperty)
245 {
246 relinkPropertiesRecursive(importingPCC, relinkingExport, structProperty.Properties, crossPCCObjectMappingList,
247 $"{prefix}{structProperty.Name}.", relinkResults, importExportDependencies);
248 }
249 else if (prop is ArrayProperty<StructProperty> structArrayProp)
250 {
251 for (int i = 0; i < structArrayProp.Count; i++)
252 {
253 StructProperty arrayStructProperty = structArrayProp[i];
254 relinkPropertiesRecursive(importingPCC, relinkingExport, arrayStructProperty.Properties, crossPCCObjectMappingList,
255 $"{prefix}{arrayStructProperty.Name}[{i}].", relinkResults, importExportDependencies);
256 }
257 }
258 else if (prop is ArrayProperty<ObjectProperty> objArrayProp)
259 {
260 foreach (ObjectProperty objProperty in objArrayProp)
261 {
262 int uIndex = objProperty.Value;
263 var result = relinkUIndex(importingPCC, relinkingExport, ref uIndex, objProperty.Name, crossPCCObjectMappingList, prefix, importExportDependencies);
264 objProperty.Value = uIndex;
265 if (result != null)
266 {
267 relinkResults.Add(result);
268 }
269 }
270 }
271 else if (prop is ObjectProperty objectProperty)
272 {
273 int uIndex = objectProperty.Value;
274 var result = relinkUIndex(importingPCC, relinkingExport, ref uIndex, objectProperty.Name, crossPCCObjectMappingList, prefix, importExportDependencies);
275 objectProperty.Value = uIndex;
276 if (result != null)
277 {
278 relinkResults.Add(result);
279 }
280 }
281 else if (prop is DelegateProperty delegateProp)
282 {
283 int uIndex = delegateProp.Value.Object;
284 var result = relinkUIndex(importingPCC, relinkingExport, ref uIndex, delegateProp.Name, crossPCCObjectMappingList, prefix, importExportDependencies);
285 delegateProp.Value = new ScriptDelegate(uIndex, delegateProp.Value.FunctionName);
286 if (result != null)
287 {
288 relinkResults.Add(result);
289 }
290 }
291 }

294 private static EntryStringPair relinkUIndex(IMEPackage importingPCC, ExportEntry relinkingExport, ref int uIndex, string propertyName,
295 IDictionary<IEntry, IEntry> crossPCCObjectMappingList, string prefix, bool importExportDependencies = false)
296 {
297 if (uIndex == 0)
298 {
299 return null; //do not relink 0
300 }
301
302 IMEPackage destinationPcc = relinkingExport.FileRef;
303 if (importingPCC == destinationPcc && uIndex < 0)
304 {
305 return null; //do not relink same-pcc imports.
306 }
307 int sourceObjReference = uIndex;
308 //if (sourceObjReference == -1009)
309 // Debugger.Break();
310 //Debug.WriteLine($"{prefix} Relinking:{propertyName}");
311 if (crossPCCObjectMappingList.TryGetValue(importingPCC.GetEntry(uIndex), out IEntry targetEntry))
312 {
313 //relink
314 uIndex = targetEntry.UIndex;
315
316 //Debug.WriteLine($"{prefix} Relink hit: {sourceObjReference}{propertyName} : {targetEntry.InstancedFullPath}");
317 }
318 else if (uIndex < 0) //It's an unmapped import
319 {
320 //objProperty is currently pointing to importingPCC as that is where we read the properties from
321 int n = uIndex;
322 int origvalue = n;
323 //Debug.WriteLine("Relink miss, attempting JIT relink on " + n + " " + rootNode.Text);
324 if (importingPCC.IsImport(n))
325 {
326 //Get the original import
327 ImportEntry origImport = importingPCC.GetImport(n);
328 string origImportFullName = origImport.InstancedFullPath; //used to be just FullPath - but some imports are indexed!
329 //Debug.WriteLine("We should import " + origImport.GetFullPath);
330
331 IEntry crossImport = null;
332 string linkFailedDueToError = null;
333 try
334 {
335 crossImport = EntryImporter.GetOrAddCrossImportOrPackage(origImportFullName, importingPCC, destinationPcc);
336 }
337 catch (Exception e)
338 {
339 //Error during relink
340 linkFailedDueToError = e.Message;
341 }
342
343 if (crossImport != null)
344 {
345 crossPCCObjectMappingList.Add(origImport, crossImport); //add to mapping to speed up future relinks
346 uIndex = crossImport.UIndex;
347 // Debug.WriteLine($"Relink hit: Dynamic CrossImport for {origvalue} {importingPCC.GetEntry(origvalue).InstancedFullPath} -> {uIndex}");
348
349 }
350 else
351 {
352 string path = importingPCC.GetEntry(uIndex) != null ? importingPCC.GetEntry(uIndex).InstancedFullPath : "Entry not found: " + uIndex;
353 if (linkFailedDueToError != null)
354 {
355 Debug.WriteLine($"Relink failed: CrossImport porting failed for {relinkingExport.ObjectName.Instanced} {relinkingExport.UIndex}: {propertyName} ({uIndex}): {importingPCC.GetEntry(origvalue).InstancedFullPath}");
356 return new EntryStringPair(relinkingExport, $"Relink failed for {prefix}{propertyName} {uIndex} in export {path}({relinkingExport.UIndex}): {linkFailedDueToError}");
357 }
358
359 if (destinationPcc.GetEntry(uIndex) != null)
360 {
361 Debug.WriteLine($"Relink failed: CrossImport porting failed for {relinkingExport.ObjectName.Instanced} {relinkingExport.UIndex}: {propertyName} ({uIndex}): {importingPCC.GetEntry(origvalue).InstancedFullPath}");
362 return new EntryStringPair(relinkingExport, $"Relink failed: CrossImport porting failed for {prefix}{propertyName} {uIndex} {destinationPcc.GetEntry(uIndex).InstancedFullPath} in export {relinkingExport.InstancedFullPath}({relinkingExport.UIndex})");
363 }
364
365 return new EntryStringPair(relinkingExport, $"Relink failed: New export does not exist - this is probably a bug in cross import code for {prefix}{propertyName} {uIndex} in export {relinkingExport.InstancedFullPath}({relinkingExport.UIndex})");
366 }
367 }
368 }
369 else
370 {
371 bool importingFromGlobalFile = false;
372 //It's an export
373 //Attempt lookup
374 ExportEntry sourceExport = importingPCC.GetUExport(uIndex);
375 string instancedFullPath = sourceExport.InstancedFullPath;
376 string sourceFilePath = sourceExport.FileRef.FilePath;
377 if (EntryImporter.IsSafeToImportFrom(sourceFilePath, destinationPcc.Game))
378 {
379 importingFromGlobalFile = true;
380 instancedFullPath = $"{Path.GetFileNameWithoutExtension(sourceFilePath)}.{instancedFullPath}";
381 }
382
383 IEntry existingEntry = destinationPcc.FindEntry(instancedFullPath);
384
385 if (existingEntry != null)
386 {
387#if DEBUG
388 if (existingEntry.InstancedFullPath.StartsWith(UnrealPackageFile.TrashPackageName))
389 {
390 // RELINKED TO TRASH!
391 Debugger.Break();
392 }
393#endif
394 //Debug.WriteLine($"Relink hit [EXPERIMENTAL]: Existing entry in file was found, linking to it: {uIndex} {sourceExport.InstancedFullPath} -> {existingEntry.InstancedFullPath}");
395 uIndex = existingEntry.UIndex;
396
397 }
398 else if (importExportDependencies)
399 {
400 if (importingFromGlobalFile)
401 {
402 uIndex = EntryImporter.GetOrAddCrossImportOrPackageFromGlobalFile(sourceExport.InstancedFullPath, importingPCC, destinationPcc, crossPCCObjectMappingList).UIndex;
403 }
404 else
405 {
406 IEntry parent = null;
407 if (sourceExport.Parent != null && !crossPCCObjectMappingList.TryGetValue(sourceExport.Parent, out parent))
408 {
409 parent = EntryImporter.GetOrAddCrossImportOrPackage(sourceExport.ParentInstancedFullPath, importingPCC, destinationPcc, true, crossPCCObjectMappingList);
410 }
411 ExportEntry importedExport = EntryImporter.ImportExport(destinationPcc, sourceExport, parent?.UIndex ?? 0, true, crossPCCObjectMappingList);
412 uIndex = importedExport.UIndex;
413 }
414 }
415 else
416 {
417 string path = importingPCC.GetEntry(uIndex)?.InstancedFullPath ?? $"Entry not found: {uIndex}";
418 Debug.WriteLine($"Relink failed in {relinkingExport.ObjectName.Instanced} {relinkingExport.UIndex}: {propertyName} {uIndex} {path}");
419 return new EntryStringPair(relinkingExport, $"Relink failed: {prefix}{propertyName} {uIndex} in export {relinkingExport.InstancedFullPath}({relinkingExport.UIndex})");
420 }
421 }
422
423 return null;
424 }

445 private static void RelinkToken(Token t, byte[] script, ExportEntry sourceExport, ExportEntry destinationExport,
446 IDictionary<IEntry, IEntry> crossFileRefObjectMap, List<EntryStringPair> relinkFailedReport,
447 bool importExportDependencies = false)
448 {
449 //Debug.WriteLine($"Attempting function relink on token at position {t.pos}. Number of listed relinkable items {t.inPackageReferences.Count}");
450
451 foreach ((int pos, int type, int value) in t.inPackageReferences)
452 {
453 switch (type)
454 {
455 case Token.INPACKAGEREFTYPE_NAME:
456 int newValue = destinationExport.FileRef.FindNameOrAdd(sourceExport.FileRef.GetNameEntry(value));
457 Debug.WriteLine($"Function relink hit @ 0x{t.pos + pos:X6}, cross ported a name: {sourceExport.FileRef.GetNameEntry(value)}");
458 script.OverwriteRange(pos, BitConverter.GetBytes(newValue));
459 break;
460 case Token.INPACKAGEREFTYPE_ENTRY:
461 relinkAtPosition(pos, value, $"(Script at @ 0x{t.pos + pos:X6}: {t.text})");
462 break;
463 }
464 }
