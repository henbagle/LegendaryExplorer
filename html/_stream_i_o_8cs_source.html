37 public static MemoryStream ReadToMemoryStream(this Stream stream, long size)
38 {
39 var memory = MemoryManager.GetMemoryStream((int)size);
40
41 var left = size;
42 var data = new byte[4096];
43 while (left > 0)
44 {
45 var block = (int)(Math.Min(left, 4096));
46 stream.Read(data, 0, block);
47 memory.Write(data, 0, block);
48 left -= block;
49 }
50
51 memory.Seek(0, SeekOrigin.Begin);
52 return memory;
53 }

139 public static void WriteFromStream(this Stream stream, Stream inputStream, int count)
140 {
141 var buffer = new byte[0x10000];
142 do
143 {
144 int readed = inputStream.Read(buffer, 0, Math.Min(buffer.Length, count));
145 if (readed > 0)
146 stream.Write(buffer, 0, readed);
147 else
148 break;
149 count -= readed;
150 } while (count != 0);
151 }

200 public static string ReadStringUnicodeNull(this Stream stream)
201 {
202 long startPos = stream.Position;
203 while (stream.Position < stream.Length)
204 {
205 if (stream.ReadInt16() == 0)
206 {
207 // Found double null terminator
208 var len = (int)(stream.Position - startPos);
209 stream.Position = startPos;
210 return ReadStringUnicodeNull(stream, len);
211 }
212 }
213
214 return null; // never found

404 public static float ReadFloat16(this Stream stream)
405 {
406 Span<byte> buffer = stackalloc byte[sizeof(ushort)];
407 if (stream.Read(buffer) != sizeof(ushort))
408 ThrowEndOfStreamException();
409 ushort u = BitConverter.ToUInt16(buffer);
410 int sign = (u >> 15) & 0x00000001;
411 int exp = (u >> 10) & 0x0000001F;
412 int mant = u & 0x000003FF;
413 switch (exp)
414 {
415 case 0:
416 return 0f;
417 case 31:
418 return 65504f;
419 }
420 exp += (127 - 15);
421 int i = (sign << 31) | (exp << 23) | (mant << 13);
422 return BitConverter.ToSingle(BitConverter.GetBytes(i), 0);
423 }

507 public static byte[] ReadFully(this Stream input, byte[] buffer)
508 {
509 if (buffer == null)
510 {
511 throw new ArgumentNullException(nameof(buffer));
512 }
513 if (input == null)
514 {
515 throw new ArgumentNullException(nameof(input));
516 }
517 if (buffer.Length == 0)
518 {
519 throw new ArgumentException("Buffer has length of 0");
520 }
521 // We could do all our own work here, but using MemoryStream is easier
522 // and likely to be just as efficient.
523 using (var tempStream = new MemoryStream()) // do not use memory manager as GetBuffer() should not be used with it
524 {
525 Copy(input, tempStream, buffer);
526 // No need to copy the buffer if it's the right size
527 if (tempStream.Length == tempStream.GetBuffer().Length)
528 {
529 return tempStream.GetBuffer();
530 }
531 // Okay, make a copy that's the right size
532 return tempStream.ToArray();
533 }
534 }

549 public static void Copy(this Stream input, Stream output, byte[] buffer)
550 {
551 if (buffer == null)
552 {
553 throw new ArgumentNullException(nameof(buffer));
554 }
555
556 if (input == null)
557 {
558 throw new ArgumentNullException(nameof(input));
559 }
560
561 if (output == null)
562 {
563 throw new ArgumentNullException(nameof(output));
564 }
565
566 if (buffer.Length == 0)
567 {
568 throw new ArgumentException("Buffer has length of 0");
569 }
570
571 int read;
572
573 while ((read = input.Read(buffer, 0, buffer.Length)) > 0)
574 {
575 output.Write(buffer, 0, read);
576 }
577 }
