14 public Statement DecompileStatement(ushort? startPosition = null)
15 {
16 StartPositions.Push(startPosition ?? (ushort)Position);
17 var token = PeekByte;
18
19 switch (token)
20 {
21 // return [expression];
22 case (byte)OpCodes.Return:
23 return DecompileReturn();
24
25 // switch (expression)
26 case (byte)OpCodes.Switch:
27 return DecompileSwitch();
28
29 // case expression :
30 case (byte)OpCodes.Case:
31 return DecompileCase();
32
33 // if (expression) // while / for / do until
34 case (byte)OpCodes.JumpIfNot:
35 {
36 PopByte();
37 var jump = new IfNotJump(ReadUInt16(), DecompileExpression(), Position - StartPositions.Peek());
38 StatementLocations.Add(StartPositions.Pop(), jump);
39 return jump;
40 }
41
42 case (byte)OpCodes.Jump:
43 {
44 PopByte();
45 ushort jumpLoc = ReadUInt16();
46 if (ForEachScopes.Count > 0 && jumpLoc == ForEachScopes.Peek())
47 {
48 var brk = new BreakStatement();
49 StatementLocations.Add(StartPositions.Pop(), brk);
50 return brk;
51 }
52 var jump = new UnconditionalJump(jumpLoc);
53 StatementLocations.Add(StartPositions.Pop(), jump);
54 return jump;
55 }
56 // continue (iterator)
57 case (byte)OpCodes.IteratorNext:
58 {
59 PopByte();
60 var itNext = new ContinueStatement();
61 StatementLocations.Add(StartPositions.Pop(), itNext);
62 if (PopByte() != (byte)OpCodes.Jump)
63 {
64 return null; //there should always be a jump after an iteratornext that's not at the end of the loop
65 }
66 //skip the jump address
67 PopByte();
68 PopByte();
69 return itNext;
70 }
71 // break;
72 case (byte)OpCodes.IteratorPop:
73 {
74 PopByte();
75 return DecompileStatement(StartPositions.Pop());
76 }
77 // stop;
78 case (byte)OpCodes.Stop:
79 PopByte();
80 var stopStatement = new StopStatement(null, null);
81 StatementLocations.Add(StartPositions.Pop(), stopStatement);
82 return stopStatement;
83
84 // Goto label
85 case (byte)OpCodes.GotoLabel:
86 PopByte();
87 var gotoLabel = new StateGoto(DecompileExpression());
88 StatementLocations.Add(StartPositions.Pop(), gotoLabel);
89 return gotoLabel;
90
91 // assignable expression = expression;
92 case (byte)OpCodes.Let:
93 case (byte)OpCodes.LetBool:
94 case (byte)OpCodes.LetDelegate:
95 return DecompileAssign();
96
97 // [skip x bytes]
98 case (byte)OpCodes.Skip: // TODO: this should never occur as statement, possibly remove?
99 PopByte();
100 ReadUInt16();
101 StartPositions.Pop();
102 return DecompileStatement();
103
104 case (byte)OpCodes.Nothing:
105 PopByte();
106 StartPositions.Pop();
107 return DecompileStatement(); // TODO, should probably have a nothing expression or statement, this is ugly
108
109 // foreach IteratorFunction(...)
110 case (byte)OpCodes.Iterator:
111 return DecompileForEach();
112
113 // foreach arrayName(valuevariable[, indexvariable])
114 case (byte)OpCodes.DynArrayIterator:
115 return DecompileForEach(true);
116
117 case (byte)OpCodes.LabelTable:
118 DecompileLabelTable();
119 StartPositions.Pop();
120 return DecompileStatement();
121
122 case (byte)OpCodes.OptIfLocal:
123 case (byte)OpCodes.OptIfInstance:
124 {
125 PopByte();
126 IEntry obj = ReadObject();
127 var condition = new SymbolReference(null, obj.ObjectName.Instanced);
128 bool not = Convert.ToBoolean((byte) ReadByte());
129 if (obj.ClassName == "BoolProperty")
130 {
131 var ifJump = new IfNotJump(
132 ReadUInt16(), not ? (Expression)condition : new PreOpReference(new PreOpDeclaration("!", SymbolTable.BoolType, 0, null), condition),
133 Position - StartPositions.Peek());
134 StatementLocations.Add(StartPositions.Pop(), ifJump);
135 return ifJump;
136 }
137
138 var nullJump = new NullJump(ReadUInt16(), condition, not)
139 {
140 SizeOfExpression = Position - StartPositions.Peek()
141 };
142 StatementLocations.Add(StartPositions.Pop(), nullJump);
143 return nullJump;
144 }
145 case (byte)OpCodes.FilterEditorOnly:
146 {
147 PopByte();
148 var edFilter = new InEditorJump(ReadUInt16());
149 StatementLocations.Add(StartPositions.Pop(), edFilter);
150 return edFilter;
151 }
152 case (byte)OpCodes.EatReturnValue:
153 PopByte();
154 ReadObject();
155 return DecompileStatement(StartPositions.Pop());
156
157 case (byte)OpCodes.Assert:
158 return DecompileAssert();
159 default:
160 var expr = DecompileExpression();
161 if (expr != null)
162 {
163 var statement = new ExpressionOnlyStatement(expr);
164 StatementLocations.Add(StartPositions.Pop(), statement);
165 return statement;
166 }
167
168 // ERROR!
169 return null;
170 }

173 private void DecompileLabelTable()
174 {
175 PopByte();
176 var name = ReadNameReference();
177 var ofs = ReadUInt32();
178 while (ofs != 0x0000FFFF) // ends with non-ref + max-offset
179 {
180 LabelTable.Add(new LabelTableEntry
181 {
182 NameRef = name,
183 Offset = ofs
184 });
185
186 name = ReadNameReference();
187 ofs = ReadUInt32();
188 }

205

226 public Statement DecompileForEach(bool isDynArray = false)
227 {
228 PopByte();
229 var scopeStatements = new List<Statement>();
230
231 var iteratorFunc = DecompileExpression();
232 if (iteratorFunc == null)
233 return null;
234
235 if (isDynArray)
236 {
237 Expression dynArrVar = DecompileExpression();
238 bool hasIndex = Convert.ToBoolean(ReadByte());
239 Expression dynArrIndex = DecompileExpression();
240 iteratorFunc = new DynArrayIterator(iteratorFunc, dynArrVar, dynArrIndex);
241 }
242
243 var scopeEnd = ReadUInt16(); // MemOff
244 ForEachScopes.Push(scopeEnd);
245
246 Scopes.Add(scopeStatements);
247 CurrentScope.Push(Scopes.Count - 1);
248 IteratorNext finalIteratorNext = null;
249 while (Position < Size)
250 {
251 if (CurrentIs(OpCodes.IteratorNext))
252 {
253 PopByte(); // IteratorNext
254 if (PeekByte == (byte)OpCodes.IteratorPop)
255 {
256 StatementLocations[(ushort)(Position - 1)] = finalIteratorNext = new IteratorNext();
257 StatementLocations[(ushort)Position] = new IteratorPop();
258 PopByte(); // IteratorPop
259 break;
260 }
261 Position--;
262 }
263
264 var current = DecompileStatement();
265 if (current == null)
266 return null; // ERROR ?
267
268 scopeStatements.Add(current);
269 }
270 CurrentScope.Pop();
271 ForEachScopes.Pop();
272
273 var statement = new ForEachLoop(iteratorFunc, new CodeBody(scopeStatements))
274 {
275 iteratorPopPos = Position - 1
276 };
277 if (finalIteratorNext is not null) finalIteratorNext.Outer = statement;
278 StatementLocations.Add(StartPositions.Pop(), statement);
279 return statement;
280 }

328 public Statement DecompileCase()
329 {
330 PopByte();
331 var offs = ReadUInt16(); // MemOff
332 Statement statement;
333
334 if (offs == (ushort)0xFFFF)
335 {
336 statement = new DefaultCaseStatement(null, null);
337 }

368 public Statement DecompileJump()
369 {
370 PopByte();
371 var jumpOffs = ReadUInt16(); // discard jump destination
372 Statement statement;
373
374 if (ForEachScopes.Count != 0 && jumpOffs == ForEachScopes.Peek()) // A jump to the IteratorPop of a ForEach means break afaik.
375 statement = new BreakStatement(null, null);
