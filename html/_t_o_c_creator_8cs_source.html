39 private static List<string> GetFiles(string baseFolder, bool isLE2LE3)
40 {
41 var res = new List<string>();
42 string directoryName = Path.GetFileName(baseFolder);
43 // Do not include the directory's existing PCConsoleTOC.bin
44 res.AddRange(GetTocableFiles(baseFolder).Except(new[] { Path.Combine(baseFolder, "PCConsoleTOC.bin") }, StringComparer.InvariantCultureIgnoreCase));
45 DirectoryInfo folder = new DirectoryInfo(baseFolder);
46 var folders = folder.GetDirectories();
47 if (folders.Length != 0)
48 {
49 if (!directoryName.Equals("BioGame", StringComparison.InvariantCultureIgnoreCase))
50 {
51 //treat as dlc and include all folders.
52 foreach (DirectoryInfo f in folders)
53 res.AddRange(GetFiles(Path.Combine(baseFolder, f.Name), isLE2LE3));
54 }
55 else
56 {
57 //BioGame, only do certain folders
58 foreach (DirectoryInfo f in folders)
59 {
60 if (f.Name == "CookedPCConsole" || f.Name == "Movies")
61 res.AddRange(GetFiles(Path.Combine(baseFolder, f.Name), isLE2LE3));
62 // LE2 and LE3 have the DLC folders included in TOC
63 else if (isLE2LE3 && f.Name == "DLC")
64 res.AddRange(GetFiles(Path.Combine(baseFolder, f.Name), isLE2LE3));
65 // LE1 has the Content/Packages/ISACT folder included in TOC
66 else if (f.Name == "Content")
67 res.AddRange(GetFiles(Path.Combine(baseFolder, f.Name, "Packages", "ISACT"), isLE2LE3));
68 }
69 }
70 }
71
72 return res;
73 }

244 public static List<string> GetLE1Files(List<string> basegameFiles, string biogameDirectory)
245 {
246 if (!Directory.Exists(Path.Combine(biogameDirectory, "DLC"))) return basegameFiles;
247
248 // Build dictionary of DLCs in mount priority
249 Dictionary<int, string> dlcMounts = new Dictionary<int, string>();
250 string[] dlcList = Directory.GetDirectories(Path.Combine(biogameDirectory, "DLC"), "*.*", SearchOption.TopDirectoryOnly);
251 foreach (var dlcFolder in dlcList)
252 {
253 if (!(new DirectoryInfo(dlcFolder).Name).StartsWith("DLC_", StringComparison.OrdinalIgnoreCase))
254 continue;
255
256 string autoLoadPath = Path.Combine(dlcFolder, "autoload.ini"); //CHECK IF FILE EXISTS?
257 if (File.Exists(autoLoadPath))
258 {
259 DuplicatingIni dlcAutoload = DuplicatingIni.LoadIni(autoLoadPath);
260 int mount = Convert.ToInt32(dlcAutoload["ME1DLCMOUNT"]["ModMount"].Value);
261 dlcMounts.Add(mount, dlcFolder);
262 }
263 }
264
265 // filename, filepath
266 Dictionary<string, string> outFiles = new Dictionary<string, string>();
267 foreach (var dlc in dlcMounts.OrderByDescending(t => t.Key))
268 {
269 var files = GetFiles(dlc.Value, false);
270 foreach (var file in files)
271 {
272 var name = new FileInfo(file).Name.ToUpper();
273 if(!outFiles.ContainsKey(name)) outFiles.Add(name, file);
274 }
275 }
276
277 // Add in basegame files
278 foreach (var file in basegameFiles)

280 var name = new FileInfo(file).Name.ToUpper();
281 if(!outFiles.ContainsKey(name)) outFiles.Add(name, file);

292 public static MemoryStream CreateTOCForEntries(List<(string relativeFilename, int size)> filesystemInfo)
293 {
294 if (filesystemInfo.Count != 0)
295 {
296 var tbf = new TOCBinFile();
297
298 // Generate hashes for all names
299 Dictionary<(string relativeFilename, int size), uint> fullHashMap = new Dictionary<(string relativeFilename, int size), uint>(); // Unbounded hash table size
300 foreach (var f in filesystemInfo)
301 {
302 fullHashMap[f] = GetStringFullHash(Path.GetFileName(f.relativeFilename));
303 }
304
305 // Calculate optimal hash table size for performance
306 var hashTableSize = filesystemInfo.Count; // Initial size is 100% 1:1
307 List<TOCBinFile.TOCHashTableEntry> hashBuckets;
308
309 while (true)
310 {
311 hashBuckets = new List<TOCBinFile.TOCHashTableEntry>(hashTableSize);
312 for (int i = 0; i < hashTableSize; i++) hashBuckets.Add(new TOCBinFile.TOCHashTableEntry()); // Initial population
313
314 // Populate the buckets with file entries
315 foreach (var hashPair in fullHashMap)
316 {
317 var bucketIdx = GetBoundedHashValue(hashPair.Value, hashTableSize);
318 hashBuckets[(int)bucketIdx].TOCEntries.Add(new TOCBinFile.Entry
319 {
320 flags = 0, // Not sure how we handle this or if it matters
321 name = hashPair.Key.relativeFilename,
322 size = hashPair.Key.size
323 });
324 }
325
326
327 // Check fill rate. We must have over 75% fill rate or we will lower the hash table size by 25% and try again (down to 50% of file table size)
328 var emptyHashBucketsCount = hashBuckets.Count(x => x.TOCEntries.Count == 0);
329 Debug.WriteLine($@"Hash fill rate: {100 - (emptyHashBucketsCount * 100.0f / hashTableSize)}%");
330 if (emptyHashBucketsCount > hashTableSize / 4)
331 {
332 // Resize the table and generate it all again.
333 int shrunkTableSize = Math.Max(filesystemInfo.Count / 2, hashTableSize - (hashTableSize / 4)); // we will never go below 50%. We will decrement by 25% each time.
334 if (shrunkTableSize == hashTableSize) // We can't go any lower than 50%
335 {
336 // WARNING: This will be suboptimal hash table size, but we are going to use it anyways to prevent crash
337 Debug.WriteLine(@"WARN: Hash table fill rate is low; even at 50% of the file size");
338 break;
339 }
340
341 // Update size.
342 hashTableSize = shrunkTableSize;
343 continue; // try again
344 }
345 else
346 {
347 // This is a satisfactory TOC.
348 break;
349 }
350 }
351
352
353
354 tbf.HashBuckets = hashBuckets;
355 Debug.WriteLine($@"Hash table stats: File count: {tbf.HashBuckets.Sum(x=>x.TOCEntries.Count)} Bucket count: {tbf.HashBuckets.Count}");
356
357 return tbf.Save();
358 }
359
360 return null;
361 }

368 private static uint GetStringFullHash(string strToHash)
369 {
370 initCRCTable();
371
372 uint hash = 0;
373 var upperCaseStr = strToHash.ToUpper();
374 for (var i = 0; i < upperCaseStr.Length; ++i)
375 {
376 char upperChar = upperCaseStr[i];
377 hash = ((hash >> 8) & 0x00FFFFFF) ^ crcTable[(hash ^ ((byte)upperChar)) & 0x000000FF]; // ASCII
378 hash = ((hash >> 8) & 0x00FFFFFF) ^ crcTable[(hash) & 0x000000FF]; // This is for unicode as each character is two bytes.
379 }
380 return hash;
381 }

410 private const uint CRC_POLYNOMIAL = 0x04C11DB7;

415 private static void initCRCTable()
416 {
417 if (crcTable != null) return;
418 crcTable = new uint[256];
419 // Table has 256 entries.
420 for (uint idx = 0; idx < 256; idx++)
421 {
422 // Generate CRCs based on the polynomial
423 for (uint crc = idx << 24, bitIdx = 8; bitIdx != 0; bitIdx--)
424 {
425 crc = ((crc & 0x80000000) == 0x80000000) ? (crc << 1) ^ CRC_POLYNOMIAL : crc << 1;
426 crcTable[idx] = crc;
427 }
428 }
