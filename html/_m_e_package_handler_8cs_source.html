139 public static IMEPackage OpenMEPackage(string pathToFile, IPackageUser user = null, bool forceLoadFromDisk = false, bool quickLoad = false, object diskIOSyncLock = null)
140 {
141 Debug.WriteLine($"Opening package {pathToFile}");
142 if (File.Exists(pathToFile))
143 {
144 pathToFile = Path.GetFullPath(pathToFile); //STANDARDIZE INPUT IF FILE EXISTS (it might be a memory file!)
145 }
146
147 IMEPackage package;
148 if (forceLoadFromDisk || !GlobalSharedCacheEnabled || quickLoad) //Quick loaded packages cannot be cached
149 {
150 if (quickLoad)
151 {
152 // Quickload: Don't read entire file.
153 if (diskIOSyncLock != null)
154 {
155 lock (diskIOSyncLock)
156 {
157 using (FileStream fs = new FileStream(pathToFile, FileMode.Open, FileAccess.Read))
158 {
159 package = LoadPackage(fs, pathToFile, false, quickLoad);
160 }
161 }
162 }
163 else
164 {
165 using (FileStream fs = new FileStream(pathToFile, FileMode.Open, FileAccess.Read))
166 {
167 package = LoadPackage(fs, pathToFile, false, quickLoad);
168 }
169 }
170
171 }
172 else
173 {
174 // Reading and operating on memory is faster than seeking on disk
175 if (diskIOSyncLock != null)
176 {
177 MemoryStream ms;
178 lock (diskIOSyncLock)
179 {
180 ms = ReadAllFileBytesIntoMemoryStream(pathToFile);
181 }
182 var p = LoadPackage(ms, pathToFile, true);
183 ms.Dispose();
184 return p;
185 }
186 else
187 {
188 return LoadPackage(ReadAllFileBytesIntoMemoryStream(pathToFile), pathToFile, true);
189 }
190
191 }
192
193 }
194 else
195 {
196 package = openPackages.GetOrAdd(pathToFile, fpath =>
197 {
198 // Reading and operating on memory is faster than seeking on disk
199 if (diskIOSyncLock != null)
200 {
201 MemoryStream ms;
202 lock (diskIOSyncLock)
203 {
204 ms = ReadAllFileBytesIntoMemoryStream(fpath);
205 }
206 var p = LoadPackage(ms, fpath, true);
207 ms.Dispose();
208 return p;
209 }
210 else
211 {
212 return LoadPackage(ReadAllFileBytesIntoMemoryStream(fpath), fpath, true);
213 }
214 });
215 }

219 if (user != null)
220 {
221 package.RegisterTool(user);
222 addToPackagesInTools(package);

259 private static IMEPackage LoadPackage(Stream stream, string filePath = null, bool useSharedCache = false, bool quickLoad = false)
260 {
261#if DEBUG && !AZURE
262 // This is only for net5-packagecache branch to trace package opening
263 //Debug.WriteLine($"Loading package {filePath}");
264 //if (filePath != null && filePath.EndsWith("Core.pcc"))
265 // Debug.WriteLine("hi");
266#endif
267 ushort version = 0;
268 ushort licenseVersion = 0;
269 bool fullyCompressed = false;
270
271 var er = new EndianReader(stream);
272 if (stream.ReadUInt32() == UnrealPackageFile.packageTagBigEndian) er.Endian = Endian.Big;
273
274 // This is stored as integer by cooker as it is flipped by size word in big endian
275 uint versionLicenseePacked = er.ReadUInt32();
276 if (versionLicenseePacked is 0x00020000 or 0x00010000 && er.Endian == Endian.Little && filePath != null) //can only load fully compressed packages from disk since we won't know what the .us file has
277 {
278 //block size - this is a fully compressed file. we must decompress it
279 //for some reason fully compressed files use a little endian package tag
280 var usfile = filePath + ".us";
281 if (File.Exists(usfile))
282 {
283 fullyCompressed = true;
284 }
285 else if (File.Exists(filePath + ".UNCOMPRESSED_SIZE"))
286 {
287 fullyCompressed = true;
288 }
289 }
290
291 if (!fullyCompressed)
292 {
293 version = (ushort)(versionLicenseePacked & 0xFFFF);
294 licenseVersion = (ushort)(versionLicenseePacked >> 16);
295 }
296
297
298 IMEPackage pkg;
299 if (fullyCompressed ||
300 (version == MEPackage.ME3UnrealVersion && (licenseVersion == MEPackage.ME3LicenseeVersion || licenseVersion == MEPackage.ME3Xenon2011DemoLicenseeVersion)) ||
301 version == MEPackage.ME3WiiUUnrealVersion && licenseVersion == MEPackage.ME3LicenseeVersion ||
302 version == MEPackage.ME2UnrealVersion && licenseVersion == MEPackage.ME2LicenseeVersion || //PC and Xbox share this
303 version == MEPackage.ME2PS3UnrealVersion && licenseVersion == MEPackage.ME2PS3LicenseeVersion ||
304 version == MEPackage.ME2DemoUnrealVersion && licenseVersion == MEPackage.ME2LicenseeVersion ||
305 version == MEPackage.ME1UnrealVersion && licenseVersion == MEPackage.ME1LicenseeVersion ||
306 version == MEPackage.ME1PS3UnrealVersion && licenseVersion == MEPackage.ME1PS3LicenseeVersion ||
307 version == MEPackage.ME1XboxUnrealVersion && licenseVersion == MEPackage.ME1XboxLicenseeVersion ||
308
309 // LEGENDARY
310 version == MEPackage.LE1UnrealVersion && licenseVersion == MEPackage.LE1LicenseeVersion ||
311 version == MEPackage.LE2UnrealVersion && licenseVersion == MEPackage.LE2LicenseeVersion ||
312 version == MEPackage.LE3UnrealVersion && licenseVersion == MEPackage.LE3LicenseeVersion
313
314
315
316 )
317 {
318 stream.Position -= 8; //reset to start
319 pkg = quickLoad ? MEConstructorQuickStreamDelegate(stream, filePath) : MEStreamConstructorDelegate(stream, filePath);
320 MemoryAnalyzer.AddTrackedMemoryItem($"MEPackage {Path.GetFileName(filePath)}", new WeakReference(pkg));
321 }
322 else if (version == UDKPackage.UDKUnrealVersion || version == 867 && licenseVersion == 0)
323 {
324 //UDK
325 stream.Position -= 8; //reset to start
326 pkg = UDKStreamConstructorDelegate(stream, filePath);
327 MemoryAnalyzer.AddTrackedMemoryItem($"UDKPackage {Path.GetFileName(filePath)}", new WeakReference(pkg));
328 }
329 else
330 {
331 throw new FormatException("Not an ME1, ME2, ME3, LE1, LE2, LE3,or UDK (2015) package file.");
332 }
333
334 if (useSharedCache)
335 {
336 pkg.noLongerUsed += Package_noLongerUsed;
337 }
338
339 return pkg;
340 }

347 public static void CreateAndSavePackage(string path, MEGame game)
348 {
349 switch (game)
350 {
351 case MEGame.UDK:
352 UDKConstructorDelegate(path, true).Save();
353 break;
354 case MEGame.LELauncher:
355 throw new ArgumentException("Cannot create a package for LELauncher, it doesn't use packages");
356 case MEGame.Unknown:
357 throw new ArgumentException("Cannot create a package file for an Unknown game!", nameof(game));
358 default:
359 var package = MEConstructorDelegate(path, game);
360 package.setPlatform(MEPackage.GamePlatform.PC); //Platform must be set or saving code will throw exception (cannot save non-PC platforms)
361 package.Save();
362 break;
363 }
364 }
