101 private (int structSize, int structAlign) GetSizeAndAlign(MEGame game)
102 {
103 int structSize = 0;
104 int structAlign = 4;
105 VariableType prev = null;
106 int bitfieldPos = 0;
107 foreach (VariableDeclaration varDecl in VariableDeclarations)
108 {
109 VariableType cur = varDecl.VarType;
110 int varSize = cur.Size(game);
111 int varAlign = 4;
112 if (cur is StaticArrayType staticArrayType)
113 {
114 cur = staticArrayType.ElementType;
115 }
116 if (cur.PropertyType == EPropertyType.Bool)
117 {
118 if (prev?.PropertyType == EPropertyType.Bool)
119 {
120 varSize = 0;
121 bitfieldPos++;
122 if (bitfieldPos > 32)
123 {
124 //cannot pack more than 32 bools into a single bitfield (not that this will ever come up...)
125 cur = null;
126 }
127 }
128 else
129 {
130 bitfieldPos = 0;
131 }
132 }
133 else if (cur.PropertyType == EPropertyType.Byte)
134 {
135 varAlign = 1;
136 }
137 else if (cur.PropertyType == EPropertyType.String)
138 {
139 if (game.IsLEGame())
140 {
141 varSize = 16 * varDecl.ArrayLength;
142 varAlign = 8;
143 }
144 else
145 {
146 varSize = 12 * varDecl.ArrayLength; //TODO: verify this
147 }
148 }
149 else if (cur is DynamicArrayType)
150 {
151 if (game.IsLEGame())
152 {
153 varSize = 16;
154 varAlign = 8;
155 }
156 else
157 {
158 varSize = 12;
159 }
160 }
161 else if (cur is Struct curStruct)
162 {
163 (varSize, varAlign) = curStruct.GetSizeAndAlign(game);
164 varSize *= varDecl.ArrayLength;
165 }
166 else if (cur.PropertyType is EPropertyType.Object or EPropertyType.Delegate && game.IsLEGame())
167 {
168 varAlign = 8;
169 }
170
171 structSize = structSize.Align(varAlign) + varSize;
172
173 structAlign = Math.Max(structAlign, varAlign);
174 prev = cur;
175 }
176
177 structSize = structSize.Align(structAlign);
178
179 return (structSize, structAlign);
180 }

243 public PropertyCollection MakeBaseProps(IMEPackage pcc, PackageCache packageCache, bool useStructDefaultsForStructProperties = true)
244 {

250 continue;

253 VariableType targetType = varDeclAST.VarType is StaticArrayType staticArrayType ? staticArrayType.ElementType : varDeclAST.VarType;
254 int staticArrayLength = varDeclAST.IsStaticArray ? varDeclAST.ArrayLength : 1;
255 for (int i = 0; i < staticArrayLength; i++)
256 {
257 Property prop;
258 switch (targetType)

371 {
372 var props = new PropertyCollection();
