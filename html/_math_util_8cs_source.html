88 public static unsafe bool NearEqual(float a, float b)
89 {
90 // Check if the numbers are really close -- needed
91 // when comparing numbers near zero.
92 if (IsZero(a - b))
93 return true;
94
95 // Original from Bruce Dawson: http://randomascii.wordpress.com/2012/02/25/comparing-floating-point-numbers-2012-edition/
96 int aInt = *(int*)&a;
97 int bInt = *(int*)&b;
98
99 // Different signs means they do not match.
100 if (aInt < 0 != bInt < 0)
101 return false;
102
103 // Find the difference in ULPs.
104 int ulp = Math.Abs(aInt - bInt);
105
106 // Choose of maxUlp = 4
107 // according to http://code.google.com/p/googletest/source/browse/trunk/include/gtest/internal/gtest-internal.h
108 const int maxUlp = 4;
109 return ulp <= maxUlp;
110 }

389 public static int Wrap(int value, int min, int max)
390 {
391 if (min > max)
392 throw new ArgumentException(string.Format("min {0} should be less than or equal to max {1}", min, max), "min");
393
394 // Code from http://stackoverflow.com/a/707426/1356325
395 int range_size = max - min + 1;
396
397 if (value < min)
398 value += range_size * ((min - value) / range_size + 1);
399
400 return min + (value - min) % range_size;
401 }

411 public static float Wrap(float value, float min, float max)
412 {
413 if (NearEqual(min, max)) return min;
414
415 double mind = min;
416 double maxd = max;
417 double valued = value;
418
419 if (mind > maxd)
420 throw new ArgumentException(string.Format("min {0} should be less than or equal to max {1}", min, max), "min");
421
422 var range_size = maxd - mind;
423 return (float)(mind + (valued - mind) - range_size * Math.Floor((valued - mind) / range_size));
424 }

455 public static double Gauss(double amplitude, double x, double y, double centerX, double centerY, double sigmaX, double sigmaY)
456 {
457 var cx = x - centerX;
458 var cy = y - centerY;
459
460 var componentX = cx * cx / (2 * sigmaX * sigmaX);
461 var componentY = cy * cy / (2 * sigmaY * sigmaY);
462
463 return amplitude * Math.Exp(-(componentX + componentY));
464 }
