21 protected override void Serialize(SerializingContainer2 sc)
22 {
23 if (sc.Pcc.Platform != MEPackage.GamePlatform.PC) return; //We do not support non-PC shader cache
24 byte platform = sc.Game.IsLEGame() ? (byte)5 : (byte)0;
25 sc.Serialize(ref platform);
26 sc.Serialize(ref ShaderTypeCRCMap, SCExt.Serialize, SCExt.Serialize);
27 if (sc.Game is MEGame.ME3 or MEGame.LE1 or MEGame.LE2 or MEGame.LE3 && sc.IsLoading)
28 {
29 int nameMapCount = sc.ms.ReadInt32();
30 sc.ms.Skip(nameMapCount * 12);
31 }
32 else if (sc.Game is MEGame.ME3 or MEGame.LE1 or MEGame.LE2 or MEGame.LE3 && sc.IsSaving)
33 {
34 sc.ms.Writer.WriteInt32(0);
35 }
36
37 if (sc.Game == MEGame.ME1)
38 {
39 sc.Serialize(ref VertexFactoryTypeCRCMap, SCExt.Serialize, SCExt.Serialize);
40 }
41
42 if (sc.IsLoading)
43 {
44 int shaderCount = sc.ms.ReadInt32();
45 Shaders = new OrderedMultiValueDictionary<Guid, Shader>(shaderCount);
46 for (int i = 0; i < shaderCount; i++)
47 {
48 Shader shader = null;
49 sc.Serialize(ref shader);
50 Shaders.Add(shader.Guid, shader);
51 }
52 }
53 else
54 {
55 sc.ms.Writer.WriteInt32(Shaders.Count);
56 foreach ((_, Shader shader) in Shaders)
57 {
58 var temp = shader;
59 sc.Serialize(ref temp);
60 }
61 }
62
63 if (sc.Game != MEGame.ME1)
64 {
65 sc.Serialize(ref VertexFactoryTypeCRCMap, SCExt.Serialize, SCExt.Serialize);
66 }
67 sc.Serialize(ref MaterialShaderMaps, SCExt.Serialize, SCExt.Serialize);
68 if (sc.Game is not (MEGame.ME2 or MEGame.LE2 or MEGame.LE1))
69 {
70 int dummy = 0;
71 sc.Serialize(ref dummy);
72 }

170 public List<(NameReference, string)> GetNames(MEGame game)
171 {
172 var names = new List<(NameReference, string)>();
173
174 names.AddRange(Shaders.Select((kvp, i) => (kvp.Key, $"Shaders[{i}].ShaderType")));
175
176 int j = 0;
177 foreach (var msm in MeshShaderMaps)
178 {
179 names.Add((msm.VertexFactoryType, $"MeshShaderMaps[{j}].VertexFactoryType"));
180 names.AddRange(msm.Shaders.Select((kvp, i) => (kvp.Key, $"MeshShaderMaps[{j}].Shaders[{i}].ShaderType")));
181 ++j;
182 }
183 names.AddRange(StaticParameters.GetNames(game).Select(tuple => (tuple.Item1, $"StaticParameters.{tuple.Item2}")));
184
185 if (game >= MEGame.ME3)
186 {
187 var uniformExpressionArrays = new List<(string, MaterialUniformExpression[])>
188 {
189 (nameof(UniformPixelVectorExpressions), UniformPixelVectorExpressions),
190 (nameof(UniformPixelScalarExpressions), UniformPixelScalarExpressions),
191 (nameof(Uniform2DTextureExpressions), Uniform2DTextureExpressions),
192 (nameof(UniformCubeTextureExpressions), UniformCubeTextureExpressions),
193 (nameof(UniformCubeTextureExpressions), UniformVertexVectorExpressions),
194 (nameof(UniformCubeTextureExpressions), UniformVertexScalarExpressions),
195 };
196
197 foreach ((string prefix, MaterialUniformExpression[] expressions) in uniformExpressionArrays)
198 {
199 for (int i = 0; i < expressions.Length; i++)
200 {
201 MaterialUniformExpression expression = expressions[i];
202 names.Add((expression.ExpressionType, $"{prefix}[{i}].ExpressionType"));
203 switch (expression)
204 {
205 case MaterialUniformExpressionTextureParameter texParamExpression:
206 names.Add((texParamExpression.ParameterName, $"{prefix}[{i}].ParameterName"));
207 break;
208 case MaterialUniformExpressionScalarParameter scalarParameterExpression:
209 names.Add((scalarParameterExpression.ParameterName, $"{prefix}[{i}].ParameterName"));
210 break;
211 case MaterialUniformExpressionVectorParameter vecParameterExpression:
212 names.Add((vecParameterExpression.ParameterName, $"{prefix}[{i}].ParameterName"));
213 break;
214 }
215 }
216 }
217 }
218
219 return names;
220 }
