78 public bool SameAsOrSubClassOf(string name)
79 {
80 string inputName = name.ToLower();
81 if (inputName == "object")
82 {
83 return true;
84 }
85 string nodeName = Name.ToLower();
86 if (nodeName == inputName)
87 return true;
88 Class current = this;
89 while (current.Parent != null && current.Parent.Name.ToLower() != "object")
90 {
91 if (current.Parent.Name.ToLower() == inputName)
92 return true;
93 current = (Class)current.Parent;
94 }
95 return false;
96 }
