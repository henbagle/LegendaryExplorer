418 public BytecodeToken ReadNext(bool dontAddToken = false)
419 {
420 ME1OpCodes b = (ME1OpCodes)_reader.ReadByte();
421 _reader.BaseStream.Position--;
422
423 if (OpCodesThatReturnNextToken.Contains(b))
424 {
425 BytecodeToken t = new BytecodeToken("", (int)_reader.BaseStream.Position);
426
427 t.OpCode = b;
428 ReadTokens.Add(t);
429 }
430
431 //if (b == ME1OpCodes.EX_Skip)
432 //{
433 // Debugger.Break();
434 //}
435
436 //Debug.WriteLine("Read bytecode " + ((byte)b).ToString("X2") + " " + b + " at " + _reader.BaseStream.Position.ToString("X8"));
437 BytecodeToken bct = ReadNextInternal();
438 //Debug.WriteLine("Bytecode finished: " + ((byte)b).ToString("X2") + " " + b + " from " + _reader.BaseStream.Position.ToString("X8"));
439
440 if (!OpCodesThatReturnNextToken.Contains(b))
441 {
442 bct.OpCode = b;
443 ReadTokens.Add(bct);
444 }
445 return bct;

448 private BytecodeToken ReadNextInternal()
449 {
450 int readerpos = (int)_reader.BaseStream.Position;
451 ME1OpCodes b = (ME1OpCodes)_reader.ReadByte();
452 switch (b)
453 {
454 case ME1OpCodes.EX_LocalVariable:
455 case ME1OpCodes.EX_InstanceVariable:
456 case ME1OpCodes.EX_NativeParm:
457 return ReadRef(r => r != null ? r.ObjectName.Instanced : "Unresolved");
458
459 case ME1OpCodes.EX_DefaultVariable:
460 return ReadRef(r => $"Default.{r.ObjectName.Instanced}");
461
462 case ME1OpCodes.EX_Return:
463 {
464 BytecodeToken returnValue = ReadNext();
465 return new ReturnToken(returnValue, readerpos);
466 }
467 case ME1OpCodes.EX_Assert:
468 {
469 _reader.ReadInt16();
470 _reader.ReadByte();
471 return WrapNextBytecode(c => new BytecodeToken($"assert({c})", readerpos));
472 }
473
474 case ME1OpCodes.EX_Switch:
475 {
476 //if (_package.Game == MEGame.ME1 && _package.Platform != MEPackage.GamePlatform.PS3)
477 // if (_package.Platform == MEPackage.GamePlatform.Xenon || _package.Game != MEGame.ME1)
478 //{
479 //ME1 Xbox this is occurs before the EX_Switch token...
480 _reader.ReadByte(); //Property type or something
481 //This is a workaround for EX_Switch, maybe
482 //}
483 BytecodeToken switchExpr = ReadNext();
484 return new SwitchToken(switchExpr.ToString(), switchExpr, readerpos);
485 }
486 case ME1OpCodes.EX_Case:
487 {
488 short offset = _reader.ReadInt16();
489 if (offset == -1) return new DefaultToken(readerpos);
490 BytecodeToken caseExpr = ReadNext();
491 return new CaseToken(caseExpr.ToString(), readerpos);
492 }
493
494 case ME1OpCodes.EX_Jump:
495 {
496 int offset = _reader.ReadInt16();
497 return new UncondJumpToken(offset, readerpos);
498 }
499
500 case ME1OpCodes.EX_JumpIfNot:
501 {
502 short offset = _reader.ReadInt16();
503 BytecodeToken condition = ReadNext();
504 if (IsInvalid(condition)) return WrapErrToken("if (!" + condition, condition);
505 return new JumpIfNotToken(offset, condition, readerpos);
506 }
507
508 case ME1OpCodes.EX_LabelTable:
509 {
510 var token = new LabelTableToken(readerpos);
511
512 var labelName = ReadName();
513 var offset = _reader.ReadInt32();
514 while (offset != 0xFFFF)
515 {
516 token.AddLabel(labelName, offset);
517 labelName = ReadName();
518 offset = _reader.ReadInt32();
519 }
520
521 token.updateUIText();
522 return token;
523 }
524
525 case ME1OpCodes.EX_GotoLabel:
526 return WrapNextBytecode(op => Token("goto " + op, readerpos));
527
528 case ME1OpCodes.EX_Self:
529 return Token("self", readerpos);
530
531 case ME1OpCodes.EX_Skip:
532 _reader.ReadInt16();
533 //Returning readnext causes a new token to be read
534 return ReadNext();
535
536 case ME1OpCodes.EX_EatReturnValue:
537 _reader.ReadInt32();
538 return ReadNext();
539
540 case ME1OpCodes.EX_Nothing:
541 return new NothingToken(readerpos, "EX_Nothing");
542
543 case ME1OpCodes.EX_Stop:
544 //_reader.ReadInt16(); Stop seems to be only 1 byte. Not 3
545 return new StopToken(readerpos);
546
547 case ME1OpCodes.EX_IntZero:
548 return Token("0", readerpos);
549
550 case ME1OpCodes.EX_IntOne:
551 return Token("1", readerpos);
552
553 case ME1OpCodes.EX_True:
554 return Token("true", readerpos);
555
556 case ME1OpCodes.EX_False:
557 return Token("false", readerpos);
558
559 case ME1OpCodes.EX_NoObject:
560 case ME1OpCodes.EX_EmptyDelegate:
561 return Token("None", readerpos);
562
563 case ME1OpCodes.EX_Let:
564 case ME1OpCodes.EX_LetBool:
565 case ME1OpCodes.EX_LetDelegate:
566 BytecodeToken lhs = ReadNext();
567 if (IsInvalid(lhs)) return lhs;
568 BytecodeToken rhs = ReadNext();
569 if (IsInvalid(rhs)) return WrapErrToken(lhs + " = " + rhs, rhs);
570 return Token(lhs + " = " + rhs, readerpos);
571
572 case ME1OpCodes.EX_IntConst:
573 return Token(_reader.ReadInt32().ToString(), readerpos);
574
575 case ME1OpCodes.EX_FloatConst:
576 return Token(_reader.ReadSingle().ToString(), readerpos);
577
578 case ME1OpCodes.EX_StringConst:
579 {
580 var s = ReadAsciiz().Replace("\n", "\\n").Replace("\t", "\\t");
581 return Token($"\"{s}\"", readerpos);
582 }
583
584 case ME1OpCodes.EX_ByteConst:
585 case ME1OpCodes.EX_IntConstByte:
586 return Token(_reader.ReadByte().ToString(), readerpos);
587
588 case ME1OpCodes.EX_ObjectConst:
589 {
590 var item = ReadEntryRef(out var objectIndex);
591 if (item == null) return ErrToken("Unresolved class item " + objectIndex);
592 return Token($"{item.ClassName}'{item.ObjectName.Instanced}'", readerpos);
593 }
594
595 case ME1OpCodes.EX_NameConst:
596 return Token($"'{ReadName()}'", readerpos);
597 case ME1OpCodes.EX_EndFunctionParms:
598 return new EndParmsToken(")", readerpos);
599
600 case ME1OpCodes.EX_ClassContext:
601 case ME1OpCodes.EX_Context:
602 {
603 var contextId = _reader.BaseStream.Position - 1;
604 //Debug.WriteLine($"Reading EX_Context START at 0x{(contextId):X8} (ID {contextId})");
605 var context = ReadNext();
606 if (IsInvalid(context)) return context;
607
608 //if (_package.Platform != MEPackage.GamePlatform.PC && _reader.BaseStream.Position % 2 != 0)
609 //{
610 // _reader.ReadByte(); //Byte align
611 //}
612
613 int exprSize = _reader.ReadInt16();
614 //Debug.WriteLine($" >> {contextId}: ExprSize {exprSize} at 0x{(_reader.BaseStream.Position - 2):X8}");
615
616
617 int bSize = _reader.ReadByte();
618 //Debug.WriteLine($" >> {contextId}: bSize {bSize} at 0x{(_reader.BaseStream.Position - 1):X8}");
619
620 //Debug.WriteLine($" >> {contextId}: Value at 0x{(_reader.BaseStream.Position):X8}");
621 var value = ReadNext();
622
623 //Debug.WriteLine($" >> {contextId}: END OF EX_Context 0x{(_reader.BaseStream.Position):X8}");
624
625 if (IsInvalid(value)) return WrapErrToken($"{context}.{value}", value);
626 return Token($"{context}.{value}", readerpos);
627 }
628
629 case ME1OpCodes.EX_InterfaceContext:
630 return ReadNext();
631
632 case ME1OpCodes.EX_FinalFunction:
633 {
634 var item = ReadEntryRef(out var functionIndex);
635 if (item == null) return ErrToken("Unresolved function item " + functionIndex);
636 string functionName = item.ObjectName.Instanced;
637 return ReadCall(readerpos, functionName);
638 }
639
640 case ME1OpCodes.EX_PrimitiveCast:
641 {
642 ECastToken conversionType = (ECastToken)_reader.ReadByte();
643 var v = ReadNext();
644 string castStr;
645 if (Enum.IsDefined(typeof(ECastToken), conversionType))
646 {
647 castStr = conversionType.ToString();
648 }
649 else
650 {
651 castStr = "UNKNOWN_CAST";
652 }
653 return Token($"{castStr}({v})", readerpos);
654 }
655
656 case ME1OpCodes.EX_VirtualFunction:
657 return ReadCall(readerpos, ReadName());
658
659 case ME1OpCodes.EX_GlobalFunction:
660 return ReadCall(readerpos, "Global." + ReadName());
661
662 case ME1OpCodes.EX_BoolVariable:
663 return ReadNext();
664 case ME1OpCodes.EX_ReturnNothing:
665 ReadEntryRef(out var objectRefIdx);
666 if (_package.IsEntry(objectRefIdx))
667 {
668 return Token($"ReturnNothing({_package.getObjectName(objectRefIdx)})", readerpos);
669 }
670 else
671 {
672 return Token($"ReturnNothing(Unknown reference {objectRefIdx})", readerpos);
673 }
674 case ME1OpCodes.EX_DynamicCast:
675 {
676 var item = ReadEntryRef(out var typeIndex);
677 return WrapNextBytecode(op => Token($"{item.ObjectName.Instanced}({op})", readerpos));
678 }
679
680 case ME1OpCodes.EX_Metacast:
681 {
682 var item = ReadEntryRef(out var typeIndex);
683 if (item == null) return ErrToken("Unresolved class item " + typeIndex);
684 return WrapNextBytecode(op => Token($"Class<{item.ObjectName.Instanced}>({op})", readerpos));
685 }
686
687 case ME1OpCodes.EX_StructMember:
688 {
689 var field = ReadEntryRef(out var _1);
690 var structType = ReadEntryRef(out var _2);
691 int wSkip = field.FileRef.Platform == MEPackage.GamePlatform.Xenon && field.FileRef.Game == MEGame.ME1 ? _reader.ReadByte() : _reader.ReadInt16(); //ME1 Xenon seems to only use 1 byte?
692 var token = ReadNext();
693 if (IsInvalid(token)) return token;
694 return Token($"{token}.{field.ObjectName.Instanced}", readerpos);
695 }
696
697 case ME1OpCodes.EX_ArrayElement:
698 case ME1OpCodes.EX_DynArrayElement:
699 {
700 var index = ReadNext();
701 if (IsInvalid(index)) return index;
702 var array = ReadNext();
703 if (IsInvalid(array)) return array;
704 return Token($"{array}[{index}]", readerpos);
705 }
706
707 case ME1OpCodes.EX_DynArrayLength:
708 return WrapNextBytecode(op => Token($"{op}.Length", readerpos));
709
710 case ME1OpCodes.EX_StructCmpEq:
711 return CompareStructs("==");
712
713 case ME1OpCodes.EX_StructCmpNe:
714 return CompareStructs("!=");
715
716 case ME1OpCodes.EX_EndOfScript:
717 return new EndOfScriptToken(readerpos);
718
719 case ME1OpCodes.EX_EmptyParmValue when _package.Platform != MEPackage.GamePlatform.Xenon || _package.Game != MEGame.ME1:
720 case ME1OpCodes.EX_GoW_DefaultValue:
721 return new DefaultValueToken("", readerpos);
722
723 case ME1OpCodes.EX_DefaultParmValue:
724 {
725 var size = _reader.ReadInt16();
726 var offset = _reader.BaseStream.Position;
727 var defaultValueExpr = ReadNext();
728 _reader.BaseStream.Position = offset + size;
729 return new DefaultParamValueToken(defaultValueExpr.ToString(), readerpos);
730 }
731
732 case ME1OpCodes.EX_LocalOutVariable:
733 var packageItem = ReadEntryRef(out var valueIndex);
734 if (packageItem == null) return ErrToken("Unresolved package item " + valueIndex);
735 return Token(packageItem.ObjectName.Instanced, readerpos);
736
737 case ME1OpCodes.EX_Iterator:
738 var expr = ReadNext();
739 int loopEnd = _reader.ReadInt16();
740 if (IsInvalid(expr)) return WrapErrToken("foreach " + expr, expr);
741 return new ForeachToken(loopEnd, expr, readerpos);
742
743 case ME1OpCodes.EX_IteratorPop:
744 return new IteratorPopToken(readerpos);
745
746 case ME1OpCodes.EX_IteratorNext:
747 return new IteratorNextToken(readerpos);
748
749 case ME1OpCodes.EX_New:
750 var outer = ReadNext();
751 if (IsInvalid(outer)) return outer;
752 var name = ReadNext();
753 if (IsInvalid(name)) return name;
754 var flags = ReadNext();
755 if (IsInvalid(flags)) return flags;
756 var cls = ReadNext();
757 if (IsInvalid(cls)) return cls;
758 return Token($"new({JoinTokens(outer, name, flags, cls)})", readerpos);
759
760 case ME1OpCodes.EX_VectorConst:

1191 private BytecodeToken ReadNativeCall(byte b)
1192 {
1193 int pos = (int)_reader.BaseStream.Position - 1;
1194 int nativeIndex;
1195 if ((b & 0xF0) == 0x60) //Extended Native
1196 {
1197 byte b2 = _reader.ReadByte();
1198 nativeIndex = ((b - 0x60) << 8) + b2;
1199 }
1200 else
1201 {
1202 nativeIndex = b; //Native
1203 }
1204
1205 var function = CachedNativeFunctionInfo.GetNativeFunction(nativeIndex); //have to figure out how to do this, it's looking up name of native function
1206 if (function == null)
1207 return ErrToken("// invalid native function " + nativeIndex);
1208 if (function.PreOperator || function.PostOperator)
1209 {
1210 var p = ReadNext();
1211 if (IsInvalid(p)) return p;
1212 ReadNext(); // end of parms
1213 if (function.PreOperator)
1214 return Token((function.HumanReadableControlToken ?? function.Name) + p, pos, nativeIndex);
1215 return Token(p + (function.HumanReadableControlToken ?? function.Name), pos, nativeIndex);
1216 }
1217 if (function.Operator)
1218 {
1219 var p1 = ReadNext();
1220 if (IsInvalid(p1)) return WrapErrToken(function.Name + "(" + p1, p1);
1221 var p2 = ReadNext();
1222 if (IsInvalid(p2)) return WrapErrToken(p1 + " " + function.Name + " " + p2, p2);
1223 ReadNext(); // end of parms
1224 return Token(p1 + " " + (function.HumanReadableControlToken ?? function.Name) + " " + p2, pos, nativeIndex);
1225 }
1226 var totalToken = ReadCall(pos, function.Name);
1227 totalToken.NativeIndex = nativeIndex;
1228 return totalToken;
1229 }
