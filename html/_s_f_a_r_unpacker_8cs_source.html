102 public SFARUnpacker(string path)

113 private void LoadHeader(Stream stream)
114 {
115 uint tag = stream.ReadUInt32();
116 if (tag != SfarTag)
117 throw new Exception("Wrong SFAR tag");
118 uint sfarVersion = stream.ReadUInt32();
119 if (sfarVersion != SfarVersion)
120 throw new Exception("Wrong SFAR version");
121
122 uint dataOffset = stream.ReadUInt32();
123 uint entriesOffset = stream.ReadUInt32();
124 TotalFilesInDLC = stream.ReadUInt32();
125 uint sizesArrayOffset = stream.ReadUInt32();
126 maxBlockSize = stream.ReadUInt32();
127 uint compressionTag = stream.ReadUInt32();
128 if (compressionTag != LZMATag)
129 throw new Exception("Not LZMA compression for SFAR file");
130
131 uint numBlockSizes = 0;
132 stream.JumpTo(entriesOffset);
133 filesList = new List<DLCEntry>();
134 for (int i = 0; i < TotalFilesInDLC; i++)
135 {
136 DLCEntry file = new DLCEntry
137 {
138 filenameHash = stream.ReadToBuffer(16),
139 compressedBlockSizesIndex = stream.ReadInt32(),
140 uncomprSize = stream.ReadUInt32()
141 };
142 file.uncomprSize |= (long)stream.ReadByte() << 32;
143 file.dataOffset = stream.ReadUInt32();
144 file.dataOffset |= (long)stream.ReadByte() << 32;
145 file.numBlocks = (uint)((file.uncomprSize + maxBlockSize - 1) / maxBlockSize);
146 filesList.Add(file);
147 numBlockSizes += file.numBlocks;
148 UncompressedSize += file.uncomprSize;
149 }
150
151 stream.JumpTo(sizesArrayOffset);
152 blockSizes = new List<ushort>();
153 for (int i = 0; i < numBlockSizes; i++)
154 {
155 blockSizes.Add(stream.ReadUInt16());
156 }
157
158 filenamesIndex = -1;
159 for (int i = 0; i < TotalFilesInDLC; i++)
160 {
161 if (StructuralComparisons.StructuralEqualityComparer.Equals(filesList[i].filenameHash, FileListHash))
162 {
163 stream.JumpTo(filesList[i].dataOffset);
164 int compressedBlockSize = blockSizes[filesList[i].compressedBlockSizesIndex];
165 byte[] inBuf = stream.ReadToBuffer(compressedBlockSize);
166 byte[] outBuf = LZMA.Decompress(inBuf, (uint)filesList[i].uncomprSize);
167 if (outBuf.Length == 0)
168 throw new Exception();
169 StreamReader filenamesStream = new StreamReader(new MemoryStream(outBuf));
170 while (filenamesStream.EndOfStream == false)
171 {
172 string name = filenamesStream.ReadLine();
173 byte[] hash = MD5.Create().ComputeHash(Encoding.ASCII.GetBytes(name.ToLowerInvariant()));
174 for (int l = 0; l < TotalFilesInDLC; l++)
175 {
176 if (StructuralComparisons.StructuralEqualityComparer.Equals(filesList[l].filenameHash, hash))
177 {
178 DLCEntry f = filesList[l];
179 f.filenamePath = name.Replace('/', '\\');
180 filesList[l] = f;
181 }
182 }
183 }
184 filenamesIndex = i;
185 break;
186 }
187 }
188 }

190 public void ExtractEntry(DLCEntry entry, Stream input, Stream output)
191 {
192 input.JumpTo(entry.dataOffset);
193 if (entry.compressedBlockSizesIndex == -1)
194 {
195 output.WriteFromStream(input, entry.uncomprSize);
196 }
197 else
198 {
199 var uncompressedBlockBuffers = new List<byte[]>();
200 var compressedBlockBuffers = new List<byte[]>();
201 var blockBytesLeft = new List<long>();
202 long bytesLeft = entry.uncomprSize;
203 for (int j = 0; j < entry.numBlocks; j++)
204 {
205 blockBytesLeft.Add(bytesLeft);
206 int compressedBlockSize = blockSizes[entry.compressedBlockSizesIndex + j];
207 int uncompressedBlockSize = (int)Math.Min(bytesLeft, maxBlockSize);
208 if (compressedBlockSize == 0)
209 {
210 compressedBlockSize = (int)maxBlockSize;
211 }
212 compressedBlockBuffers.Add(input.ReadToBuffer(compressedBlockSize));
213 uncompressedBlockBuffers.Add(null);
214 bytesLeft -= uncompressedBlockSize;
215 }
216
217 Parallel.For(0, entry.numBlocks, j =>
218 {
219 int compressedBlockSize = blockSizes[entry.compressedBlockSizesIndex + (int)j];
220 int uncompressedBlockSize = (int)Math.Min(blockBytesLeft[(int)j], maxBlockSize);
221 if (compressedBlockSize == 0 || compressedBlockSize == blockBytesLeft[(int)j])
222 {
223 uncompressedBlockBuffers[(int)j] = compressedBlockBuffers[(int)j];
224 }
225 else
226 {
227 uncompressedBlockBuffers[(int)j] = LZMA.Decompress(compressedBlockBuffers[(int)j], (uint)uncompressedBlockSize);
228 if (uncompressedBlockBuffers[(int)j].Length == 0)
229 throw new Exception();
230 }
231 });
232
233 for (int j = 0; j < entry.numBlocks; j++)
234 {
235 output.WriteFromBuffer(uncompressedBlockBuffers[j]);
236 }
237 }

247 public void Extract(string outPath)
248 {
249 if (!File.Exists(filePath))
250 throw new Exception("filename missing");
251
252 UnpackCanceled = false;
253 IndeterminateState = true;
254 CurrentOverallStatus = $"Extracting {GetPrettyDLCNameFromPath(filePath)}";
255 CurrentStatus = $"Loading {GetPrettyDLCNameFromPath(filePath)} into memory ({FileSize.FormatSize(new FileInfo(filePath).Length)})";
256 byte[] buffer = File.ReadAllBytes(filePath);
257 CurrentFilesProcessed = 0;
258 IndeterminateState = false;
259
260 using (MemoryStream stream = new MemoryStream(buffer))
261 {
262 for (int i = 0; i < TotalFilesInDLC; i++, CurrentFilesProcessed++)
263 {
264 if (UnpackCanceled)
265 break;
266
267 if (filenamesIndex == i)
268 continue;
269 if (filesList[i].filenamePath == null)
270 throw new Exception("filename missing");
271
272 CurrentStatus = "File " + (i + 1) + " of " + filesList.Count + " - " + Path.GetFileName(filesList[i].filenamePath);
273 CurrentProgress = (int)(100.0 * CurrentFilesProcessed) / (int)TotalFilesInDLC;
274
275 int pos = filesList[i].filenamePath.IndexOf("\\BIOGame\\DLC\\", StringComparison.OrdinalIgnoreCase);
276 string filename = filesList[i].filenamePath.Substring(pos + ("\\BIOGame\\DLC\\").Length);
277 string dir = Path.GetDirectoryName(outPath);
278 Directory.CreateDirectory(Path.GetDirectoryName(Path.Combine(dir, filename)));
279 using (FileStream outputFile = new FileStream(Path.Combine(dir, filename), FileMode.Create, FileAccess.Write))
280 {
281 ExtractEntry(filesList[i], stream, outputFile);
282 }
283 }
284 }
285
286 if (UnpackCanceled)
287 {
288 CurrentStatus = "Canceling, cleaning up...";
289 IndeterminateState = true;
290 string dir = Path.GetDirectoryName(outPath);
291 for (int i = 0; i < TotalFilesInDLC; i++)
292 {
293 if (filenamesIndex == i)
294 continue;
295 int pos = filesList[i].filenamePath.IndexOf("\\BIOGame\\DLC\\", StringComparison.OrdinalIgnoreCase);
296 string filename = filesList[i].filenamePath.Substring(pos + ("\\BIOGame\\DLC\\").Length);
297 if (File.Exists(Path.Combine(dir, filename)))
298 File.Delete(Path.Combine(dir, filename));
299 }
300 IndeterminateState = false;
301 return;
302 }
303
304 File.Delete(filePath);
305 using (FileStream outputFile = new FileStream(filePath, FileMode.Create, FileAccess.Write))
306 {
307 outputFile.WriteUInt32(SfarTag);
308 outputFile.WriteUInt32(SfarVersion);
309 outputFile.WriteUInt32(HeaderSize);
310 outputFile.WriteUInt32(HeaderSize);
311 outputFile.WriteUInt32(0);
312 outputFile.WriteUInt32(HeaderSize);
313 outputFile.WriteUInt32((uint)MaxBlockSize);
314 outputFile.WriteUInt32(LZMATag);
315 }
