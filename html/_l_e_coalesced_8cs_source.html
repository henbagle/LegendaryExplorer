20 public static string ReadCoalescedString(this BinaryReader reader)
21 {
22 var length = reader.ReadInt32();
23 var bytes = reader.ReadBytes(length * -2);
24 if (bytes.Length != 0)
25 {
26 var str = Encoding.Unicode.GetString(bytes, 0, bytes.Length - 2);
27 return str;
28 }
29
30 return string.Empty;
31 }
32
33 public static void WriteCoalescedString(this BinaryWriter writer, string str)
34 {
35 if (string.IsNullOrEmpty(str))
36 {
37 writer.Write((Int32)0);
38 return;
39 }
40
41 var length = str.Length;
42 var wideString = Encoding.Unicode.GetBytes(str + '\0');
43
44 writer.Write((Int32)(length + 1) * -1);
45 writer.Write(wideString);
46 }
47 }

120 private static LECoalescedBundle ReadFromBinaryReader(BinaryReader reader, string name)
121 {
122 LECoalescedBundle bundle = new(name);
123
124 var fileCount = reader.ReadInt32();
125 //Debug.WriteLine($"Bundle {bundle.Name}, {fileCount} files");
126
127 for (int i = 0; i < fileCount; i++)
128 {
129 var iniFullName = reader.ReadCoalescedString();
130 var sectionCount = reader.ReadInt32();
131 bundle.Files[Path.GetFileName(iniFullName)] = ReadCoalescedIni(reader, sectionCount);
132 }
133
134 return bundle;
135 }

137 private static DuplicatingIni ReadCoalescedIni(BinaryReader reader, int sectionCount)
138 {
139 DuplicatingIni ini = new DuplicatingIni();
140 DuplicatingIni.Section s = null;
141 for (int i = 0; i < sectionCount; i++)
142 {
143 if (s != null)
144 {
145 ini.Sections.Add(s);
146 }
147 s = new DuplicatingIni.Section() { Header = reader.ReadCoalescedString() };
148
149 var pairCount = reader.ReadInt32();
150 //Debug.WriteLine($"Section {s.Header}, {pairCount} pairs");
151 for (int j = 0; j < pairCount; j++)
152 {
153 var key = reader.ReadCoalescedString();
154 var val = reader.ReadCoalescedString();
155
156 s.Entries.Add(new DuplicatingIni.IniEntry(key, val));
157 }
158 }
159
160 if (s != null) // in case file has zero sections we must check if it is null.
161 {
162 ini.Sections.Add(s);
163 }
164
165 return ini;
166 }

168 public static LECoalescedBundle ReadFromDirectory(string name, string path)
169 {
170 var manifestPath = Path.Combine(path, "mele.extractedbin");
171 if (!File.Exists(manifestPath))
172 {
173 throw new Exception("Didn't find a manifest in path");
174 }
175
176 LECoalescedManifestInfo manifest = new(manifestPath);
177 LECoalescedBundle bundle = new(manifest.DestinationFilename);
178
179 DuplicatingIni ini = null;
180 foreach (var relativePath in manifest.ParticipatingRelativeFilePaths)
181 {
182 var filePath = Path.Combine(path, relativePath);
183 if (!File.Exists(filePath)) { throw new Exceptions.CBundleException("Failed to find a file according to manifest, either the file was removed or the manifest was changed"); }
184 StreamReader reader = new(filePath);
185
186 ini = new DuplicatingIni();
187
188 DuplicatingIni.Section currentSection = null;
189 string line = null;
190 while ((line = reader.ReadLine()) is not null)
191 {
192 // Empty line
193 if (string.IsNullOrWhiteSpace(line)) { continue; }
194
195 // Section header
196 if (line.StartsWith('[') && line.EndsWith(']'))
197 {
198 var header = line.Substring(1, line.Length - 2);
199 if (header.Length < 1 || string.IsNullOrWhiteSpace(header)) { throw new Exceptions.CBundleException("Expected to have a header with text"); }
200
201 if (currentSection is not null)
202 {
203 ini.Sections.Add(currentSection);
204 }
205 currentSection = new DuplicatingIni.Section() { Header = header };
206
207 continue;
208 }
209
210 // Pair
211 var chunks = line.Split('=', 2);
212 if (chunks.Length != 2) { throw new Exceptions.CBundleException("Expected to have exactly two chunks after splitting the line by ="); }
213
214 if (chunks[0].EndsWith("||")) // It's a multiline value UGH
215 {
216 var strippedKey = chunks[0].Substring(0, chunks[0].Length - 2);
217
218 if (currentSection.Entries.Count > 0 && currentSection.Entries[^1].Key == strippedKey) // It's a second or further line in multiline value
219 {
220 var last = currentSection.Entries[^1];
221 currentSection.Entries[^1] = new(last.Key, last.Value + "\r\n" + chunks[1]);
222 }
223 else
224 {
225 currentSection.Entries.Add(new DuplicatingIni.IniEntry(strippedKey, chunks[1]));
226 }
227 }
228 else
229 {
230 currentSection.Entries.Add(new DuplicatingIni.IniEntry(chunks[0], chunks[1]));
231 }
232
233 }
234
235 if (currentSection is not null)
236 {
237 ini.Sections.Add(currentSection);
238 }
239 bundle.Files[Path.GetFileName(relativePath)] = ini;
240 }
241
242 return bundle;
243 }

245 public void WriteToDirectory(string destinationPath)
246 {
247 Directory.CreateDirectory(destinationPath);
248
249 foreach (var file in Files)
250 {
251 var outPath = Path.Combine(destinationPath, file.Key);
252 using var writerStream = new StreamWriter(outPath);
253 foreach (var section in file.Value.Sections)
254 {
255 writerStream.WriteLine($"[{section.Header}]");
256 foreach (var pair in section.Entries)
257 {
258 var lines = splitValue(pair.Value);
259 if (lines is null || lines.Count() == 1)
260 {
261 writerStream.WriteLine($"{pair.Key}={pair.Value}");
262 continue;
263 }
264
265 // multiline
266 foreach (var line in lines)
267 {
268 writerStream.WriteLine($"{pair.Key}||={line}");
269 }
270 }
271 }
272 }
273
274 // Write out a manifest to rebuild from.
275 var manifestPath = Path.Combine(destinationPath, "mele.extractedbin");
276 using var manifestWriter = new StreamWriter(manifestPath);
277 manifestWriter.WriteLine($"{Name}");
278 foreach (var file in Files)
279 {
280 manifestWriter.WriteLine(file.Key);
281 }

291 internal List<string> splitValue(string val)
292 {
293 List<string> splitVal = null;
294
295 if (val.Contains("\r\n"))
296 {
297 splitVal = val.Split("\r\n").ToList();
298 }
299 else if (val.Contains('\r') && !val.Contains('\n'))
300 {
301 splitVal = val.Split('\r').ToList();
302 }
303 else if (!val.Contains('\r') && val.Contains('\n'))
304 {
305 splitVal = val.Split('\n').ToList();
306 }
307 else if (val.Contains('\r') && val.Contains('\n'))
308 {
309 throw new Exception("Value contains both CR and LF but not in a CRLF sequence!");
310 }
311
312 return splitVal;
313 }

315 public void WriteToStream(Stream ms)
316 {
317 var writer = new BinaryWriter(ms);
318
319 writer.Write(Files.Count);
320 foreach (var file in Files)
321 {
322 writer.WriteCoalescedString(GetIniFullPath(file.Key));
323 writer.Write(file.Value.Sections.Count);
324
325 foreach (var section in file.Value.Sections)
326 {
327 if (section is null)
328 {
329 continue;
330 }
331
332 writer.WriteCoalescedString(section.Header);
333 writer.Write(section.Entries.Count);
334
335 foreach (var pair in section.Entries)
336 {
337 writer.WriteCoalescedString(pair.Key);
338 writer.WriteCoalescedString(pair.Value);
339 }
340 }
341 }

344 private string GetIniFullPath(string filename)
345 {
346 var extension = Path.GetExtension(filename).ToLower();
347 switch (extension)
348 {
349 case ".ini":
350 return $@"..\..\BIOGame\Config\{filename}";
351 case ".int":
352 case ".ita":
353 case ".deu":
354 case ".pol":
355 case ".fra":
356 return $@"..\..\Localization\{extension.Substring(1).ToUpper()}\{filename}";
357 }
358 throw new Exception($"Filename '{filename}' has invalid file extension for LE1/LE2 Coalesced filename");
