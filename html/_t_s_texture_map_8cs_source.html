60 private bool IsSameOffsets(ExportEntry exportEntry)
61 {
62 Texture2D utd = new Texture2D(exportEntry);
63 var topMip = utd.GetTopMip();
64 if (topMip.externalOffset == 0 || TopMipOffset != topMip.externalOffset)
65 {
66 // Force calculation
67 return false;
68 }
69
70 Guid tfcGuid = default;
71 var guid = exportEntry.GetProperty<StructProperty>("TFCFileGuid");
72 if (guid != null)
73 tfcGuid = new FGuid(guid).ToGuid();
74
75 if (TFCGuid != tfcGuid)
76 {
77 // Different TFC
78 return false;
79 }
80
81 if (topMip.TextureCacheName != TFCName)
82 {
83 // Different TFC
84 return false;
85 }
86

96 private static TextureMapMemoryEntry EnsureParent(ExportEntry exportEntry, string selectedFolder, Dictionary<string, TextureMapMemoryEntry> textureMapMemoryEntries, List<string> additionalTFCs, Func<IEntry, TextureMapMemoryEntry> generatorDelegate, Action<TextureMapMemoryEntry> addRootNodeDelegate, Dictionary<string, uint> crcCache = null)
97 {
98 IEntry parentT = exportEntry;
99 List<IEntry> parents = new List<IEntry>();
100 while (parentT.HasParent)
101 {
102 parents.Insert(0, parentT.Parent);
103 parentT = parentT.Parent;
104 }
105
106
107 TextureMapMemoryEntry lastParent = null;
108 for (int i = 0; i < parents.Count; i++)
109 {
110 var p = parents[i];
111 if (!textureMapMemoryEntries.TryGetValue(p.InstancedFullPath, out lastParent))
112 {
113 if (p.IsTexture() && p is ExportEntry pe)
114 {
115 // Parent is texture. Normally this doesn't occur but devs be devs
116 lastParent = ParseTexture(pe, selectedFolder, textureMapMemoryEntries, crcCache, additionalTFCs, generatorDelegate, addRootNodeDelegate);
117 }
118 else
119 {
120 // Parent doesn't exist, create
121 lastParent = generatorDelegate(p);
122 lastParent.Parent = i > 0 ? textureMapMemoryEntries[parents[i - 1].InstancedFullPath] : null;
123 // Set the parent child
124 lastParent.Parent?.Children.Add(lastParent);
125 if (lastParent.Parent == null)
126 {
127 addRootNodeDelegate?.Invoke(lastParent);
128 }
129 }
130
131 textureMapMemoryEntries[p.InstancedFullPath] = lastParent;
132 }
133
134 }
135
136 return lastParent;
137 }

248 public TextureMapPackageEntry(string basePath, ExportEntry exportEntry, List<string> additionalTFCs = null, Dictionary<string, uint> crcCache = null)
249 {
250 RelativePackagePath = exportEntry.FileRef.FilePath.Substring(basePath.Length).TrimStart('\\', '/');
251 PackageName = Path.GetFileName(RelativePackagePath);
252 ExportPath = exportEntry.InstancedFullPath;
253
254 var tex2D = ObjectBinary.From<UTexture2D>(exportEntry);
255 NumMips = tex2D.Mips.Count;
256 NumEmptyMips = tex2D.Mips.Count(x => x.StorageType == StorageTypes.empty);
257
258 var topMip = tex2D.Mips[NumEmptyMips]; // Skip the number of empty mips and use first non-empty
259 if (NumMips > 0)
260 {
261 Width = (short)topMip.SizeX;
262 Height = (short)topMip.SizeY;
263 }
264
265 HasExternalReferences = tex2D.Mips.Any(x => !x.IsLocallyStored);
266 if (HasExternalReferences)
267 {
268 ExternalStorageSize = tex2D.Mips.Where(x => !x.IsLocallyStored).Sum(x => x.CompressedSize);
269 foreach (var em in tex2D.Mips.Where(x => !x.IsLocallyStored && x.StorageType != StorageTypes.empty))
270 {
271 CompressedMipInfos.Add(new MEMTextureMap.CompressedMipInfo() { Offset = em.DataOffset, CompressedSize = em.CompressedSize, UncompressedSize = em.UncompressedSize, StorageType = em.StorageType });
272 }
273
274 }
275
276 var props = exportEntry.GetProperties();
277 var format = props.GetProp<EnumProperty>(@"Format");
278 if (format != null)
279 {
280 PixelFormat = Image.getPixelFormatType(format.Value);
281 }
282
283 var cache = props.GetProp<NameProperty>(@"TextureFileCacheName");
284 if (cache != null)
285 {
286 TFCName = cache.Value.Name;
287 }
288
289 var guid = props.GetProp<StructProperty>(@"TFCFileGuid");
290 if (guid != null)
291 {
292 TFCGuid = new FGuid(guid).ToGuid();
293 }
294
295 if (exportEntry.Game == MEGame.ME1)
296 {
297 IEntry pEntry = exportEntry;
298 while (pEntry.Parent != null)
299 {
300 pEntry = pEntry.Parent;
301 }
302
303 if (pEntry.ClassName == @"Package")
304 {
305 MasterPackageName = pEntry.ObjectName;
306 }
307 }
308
309 // This needs some optimization once it's working
310 Texture2D t2d = new Texture2D(exportEntry);
311 var canCache = t2d.GetTopMip().storageType != StorageTypes.empty && crcCache != null;
312 if (canCache && crcCache.TryGetValue($"{TFCName}_{t2d.GetTopMip().externalOffset}", out var crc))
313 {
314 CRC = crc;
315 }
316 try
317 {
318 CRC = Texture2D.GetTextureCRC(exportEntry, additionalTFCs);
319 if (canCache)
320 {
321 crcCache[$"{TFCName}_{t2d.GetTopMip().externalOffset}"] = CRC;
322 }
323
324 }
325 catch (Exception)
326 {
327 // CRC could not be calculated
328 }
329 }

450 public static TextureMap GenerateMapForFolder(string rootDirectory,
451 Func<IEntry, TextureMapMemoryEntry> nodeGeneratorDelegate,
452 Action<TextureMapMemoryEntry> addRootElementDelegate,
453 Action<string, int, int> progressDelegate = null,
454 CancellationToken cts = default)
455 {
456
457 var rootNodes = new List<TextureMapMemoryEntry>();
458 MEGame game = MEGame.Unknown;
459
460 void addRootItem(TextureMapMemoryEntry entry)
461 {
462 rootNodes.Add(entry);
463 addRootElementDelegate?.Invoke(entry);
464 }
465
466 // Mapping of full paths to their entries
467 progressDelegate?.Invoke(@"Calculating texture map", -1, -1);
468 Dictionary<string, TextureMapMemoryEntry> entries = new Dictionary<string, TextureMapMemoryEntry>();
469 var packageFiles = Directory.GetFiles(rootDirectory, "*.*", SearchOption.AllDirectories).Where(x => x.RepresentsPackageFilePath()).ToList();
470 var tfcs = Directory.GetFiles(rootDirectory, "*.tfc", SearchOption.AllDirectories).ToList();
471 progressDelegate?.Invoke(@"Calculating texture map", 0, packageFiles.Count);
472
473
474 // Pass 1: Find all unique memory texture paths
475 int numDone = 0;
476 Dictionary<uint, MEMTextureMap.TextureMapEntry> vanillaMap = null;
477 Dictionary<string, uint> crcCache = new();
478 foreach (var p in packageFiles)
479 {
480 var filename = Path.GetFileName(p);
481 progressDelegate?.Invoke($@"Scanning {filename}", numDone, packageFiles.Count);
482
483 if (cts.IsCancellationRequested)
484 break;
485 using var package = MEPackageHandler.OpenMEPackage(p);
486
487 if (game != MEGame.Unknown && game != package.Game)
488 {
489 // This workspace has files from multiple games!
490 throw new Exception("A directory being scanned cannot have packages from different games in it");
491 }
492 else
493 {
494 game = package.Game;
495 vanillaMap = MEMTextureMap.LoadTextureMap(game);
496 }
497
498 var textures = package.Exports.Where(x => !x.IsDefaultObject && x.IsTexture());
499 foreach (var t in textures)
500 {
501 if (cts.IsCancellationRequested) break;
502 TextureMapMemoryEntry.ParseTexture(t, rootDirectory, entries, crcCache, tfcs, nodeGeneratorDelegate, addRootItem);
503 }
504
505 // Todo: Actually finish someday
506 //if (filename.StartsWith(ME1_MOD_MASTER_TEXTURE_PACKAGE_PREFIX))
507 //{
508 // ME1MasterTexturePackages.Add(p);
509 // Texture2D.AdditionalME1MasterTexturePackages.Add(p); // TODO: THIS NEEDS CLEANED UP AND MANAGED IN TEXTURE2D.CS
510 //}
511
512 numDone++;
513 progressDelegate?.Invoke(null, numDone, packageFiles.Count);
514 }
515
516 // Pass 2: Find any unique items among the unique paths (e.g. CRC not equal to other members of same entry)
517 var allTextures = rootNodes.SelectMany(x => x.GetAllTextureEntries());
518
519 // Pass 3: Find items that have matching CRCs across memory entries
520 Dictionary<uint, List<TextureMapMemoryEntry>> crcMap = new Dictionary<uint, List<TextureMapMemoryEntry>>();
521 foreach (var t in allTextures)
522 {
523 if (t.Instances.Any())
524 {
525 var firstCRC = t.Instances[0].CRC;
526
527 var areAllEqualCRC = t.Instances.All(x => x.CRC == firstCRC);
528 if (!areAllEqualCRC)
529 {
530 // Some textures are not the same across the same entry!
531 // This will lead to weird engine behavior as memory is dumped and newly loaded data is different
532 Debug.WriteLine(@"UNMATCHED CRCSSSSSSSSSSSSSSSSSSSSSSSSSSS");
533 SetUnmatchedCRC(t, true);
534 }
535 else
536 {
537 if (!crcMap.TryGetValue(firstCRC, out var list))
538 {
539 list = new List<TextureMapMemoryEntry>();
540 crcMap[firstCRC] = list;
541 }
542
543 list.Add(t);
544 }
545 }
546 }
547
548 return new TextureMap()
549 {
550 VanillaMap = vanillaMap,
551 CalculatedMap = rootNodes,
552 Game = game,
553 };
554
