25 public override List<(NameReference, string)> GetNames(MEGame game)
26 {
27 var names = base.GetNames(game);
28
29 for (int i = 0; i < StateEvents.Count; i++)
30 {
31 BioStateEvent stateEvent = StateEvents[i];
32 for (int j = 0; j < stateEvent.Elements.Count; j++)
33 {
34 BioStateEventElement stateEventElement = stateEvent.Elements[j];
35 switch (stateEventElement)
36 {
37 case BioStateEventElementFunction bioStateEventElementFunction:
38 names.Add(bioStateEventElementFunction.PackageName, $"[{i}] State Transition: [{j}] Transition : Package Name");
39 names.Add(bioStateEventElementFunction.ClassName, $"[{i}] State Transition: [{j}] Transition : Class Name");
40 names.Add(bioStateEventElementFunction.FunctionName, $"[{i}] State Transition: [{j}] Transition : Function Name");
41 break;
42 case BioStateEventElementLocalBool bioStateEventElementLocalBool:
43 names.Add(bioStateEventElementLocalBool.ObjectTag, $"[{i}] State Transition: [{j}] Transition : Object Tag");
44 names.Add(bioStateEventElementLocalBool.FunctionName, $"[{i}] State Transition: [{j}] Transition : Function Name");
45 break;
46 case BioStateEventElementLocalFloat bioStateEventElementLocalFloat:
47 names.Add(bioStateEventElementLocalFloat.ObjectTag, $"[{i}] State Transition: [{j}] Transition : Object Tag");
48 names.Add(bioStateEventElementLocalFloat.FunctionName, $"[{i}] State Transition: [{j}] Transition : Function Name");
49 break;
