99 public static string DefaultGamePath
100 {
101 get
102 {
103 if (string.IsNullOrEmpty(_DefaultGamePath))
104 {
105 if (string.IsNullOrEmpty(LegendaryExplorerCoreLibSettings.Instance?.LEDirectory))
106 {
107 return null;
108 }
109 _DefaultGamePath = Path.Combine(LegendaryExplorerCoreLibSettings.Instance.LEDirectory, "Game", "ME3");
110 }
111 return Path.GetFullPath(_DefaultGamePath); //normalize
112 }
113 set
114 {
115 if (value != null)
116 {
117 if (value.Contains("BIOGame", StringComparison.OrdinalIgnoreCase))
118 value = value.Substring(0, value.LastIndexOf("BIOGame", StringComparison.OrdinalIgnoreCase));
119 }
120 _DefaultGamePath = value;
121 }
122 }
