66 private List<Statement> Parse(bool requireBrackets = true)
67 {
68 if (requireBrackets && Consume(TokenType.LeftBracket) == null) throw ParseError("Expected '{'!", CurrentPosition);
69
70 var statements = new List<Statement>();
71 var subObjects = new List<Subobject>();
72 Symbols.PushScope("DefaultProperties");
73 try
74 {
75 var current = ParseTopLevelStatement();
76 while (current != null)
77 {
78 if (current is Subobject subObj)
79 {
80 subObjects.Add(subObj);
81 }
82 statements.Add(current);
83 current = ParseTopLevelStatement();
84 }
85 InSubOject = true;
86 ParseSubObjectBodys(subObjects);
87 }
88 finally
89 {
90 Symbols.PopScope();
91 }
92
93 if (requireBrackets && Consume(TokenType.RightBracket) == null) throw ParseError("Expected '}'!", CurrentPosition);
94 return statements;
95 }

97 private Statement ParseTopLevelStatement()
98 {
99 if (CurrentIs("BEGIN") && (NextIs("Object") || NextIs("Template")))
100 {
101 if (IsStructDefaults)
102 {
103 throw ParseError($"SubObjects are not allowed in {STRUCTDEFAULTPROPERTIES}!", CurrentPosition);
104 }
105 return ParseSubobjectDeclaration();
106 }
107
108 return ParseNonStructAssignment();
109 }
110 private void ParseSubObjectBodys(List<Subobject> subObjects)
111 {
112 foreach(Subobject currentSubObj in subObjects)
113 {
114 var subSubObjects = new List<Subobject>();
115 var statements = currentSubObj.Statements;
116 var objectClass = currentSubObj.Class;
117 var objectName = currentSubObj.NameDeclaration.Name;
118 ExpressionScopes.Push(objectClass.GetInheritanceString());
119 SubObjectClasses.Push(objectClass);
120 Symbols.PushScope(objectName);
121 Tokens = currentSubObj.Tokens;
122 try
123 {
124 var current = ParseTopLevelStatement();
125 while (current != null)
126 {
127 if (current is Subobject subObj)
128 {
129 subSubObjects.Add(subObj);
130 }
131 statements.Add(current);
132 current = ParseTopLevelStatement();
133 }
134 ParseSubObjectBodys(subSubObjects);
135 }
136 finally
137 {
138 Symbols.PopScope();
139 ExpressionScopes.Pop();
140 SubObjectClasses.Pop();
141 }

145 private Subobject ParseSubobjectDeclaration()
146 {
147 var startPos = CurrentPosition;
148
149 //BEGIN
150 CurrentToken.SyntaxType = EF.Keyword;
151 Tokens.Advance();
152 var objOrTemplateToken = CurrentToken;
153 objOrTemplateToken.SyntaxType = EF.Keyword;
154 Tokens.Advance();
155 bool isTemplate = objOrTemplateToken.Value.CaseInsensitiveEquals("Template");
156
157 if (Consume("Class") is not {} classKeywordToken || Consume(TokenType.Assign) is not {} classAssignToken)
158 {
159 throw ParseError($"Expected 'Class=' after 'Begin {objOrTemplateToken.Value}'!", CurrentPosition);
160 }
161 classKeywordToken.SyntaxType = EF.Keyword;
162 classAssignToken.SyntaxType = EF.Operator;
163
164 var classNameToken = Consume(TokenType.Word);
165 if (classNameToken is null)
166 {
167 throw ParseError("Expected name of class!", CurrentPosition);
168 }
169 classNameToken.SyntaxType = EF.TypeName;
170
171 if (!Symbols.TryGetType(classNameToken.Value, out Class objectClass))
172 {
173 throw ParseError($"{classNameToken} is not the name of a class!", classNameToken);
174 }
175
176 if (objectClass.OuterClass is Class outerClass && SubObjectClasses.Any() && !SubObjectClasses.Peek().SameAsOrSubClassOf(outerClass))
177 {
178 TypeError($"A '{objectClass.Name}' must be declared within a '{outerClass.Name}', not a '{SubObjectClasses.Peek().Name}'!", classNameToken);
179 }
180
181 if (Consume("Name") is not {} nameKeywordToken || Consume(TokenType.Assign) is not {} nameAssignToken)
182 {
183 throw ParseError("Expected 'Name=' after Class reference!", CurrentPosition);
184 }
185 nameKeywordToken.SyntaxType = EF.Keyword;
186 nameAssignToken.SyntaxType = EF.Operator;
187
188 var nameToken = Consume(TokenType.Word);
189 if (nameToken is null)
190 {
191 throw ParseError($"Expected name of {objOrTemplateToken.Value}!", CurrentPosition);
192 }
193 string objectName = nameToken.Value;
194
195 var bodyStartPos = CurrentToken.StartPos;
196 int nestedLevel = 1;
197 var tokens = new List<ScriptToken>();
198 while (true)
199 {
200 if (CurrentTokenType == TokenType.EOF)
201 {
202 throw ParseError("SubObject declaration has no end!", startPos);
203 }
204 if (CurrentIs("BEGIN") && (NextIs("Object") || NextIs("Template")))
205 nestedLevel++;
206 else if (CurrentIs("END") && (NextIs("Object") || NextIs("Template")))
207 nestedLevel--;
208 if (nestedLevel <= 0)
209 {
210 break;
211 }
212 tokens.Add(CurrentToken);
213 Tokens.Advance();
214 }
215 var bodyEndPos = PrevToken.EndPos;
216 //END
217 CurrentToken.SyntaxType = EF.Keyword;
218 Tokens.Advance();
219 //Object or Template
220 CurrentToken.SyntaxType = EF.Keyword;
221 Tokens.Advance();
222
223 var subObj = new Subobject(new VariableDeclaration(objectClass, default, objectName), objectClass, new List<Statement>(), isTemplate, startPos, PrevToken.EndPos)
224 {
225 Tokens = new TokenStream(tokens)
226 };
227 if (!Symbols.TryAddSymbol(objectName, subObj))
228 {
229 throw ParseError($"'{objectName}' has already been defined in this scope!", nameToken);
230 }
231 return subObj;
232 }

250 private AssignStatement ParseAssignment(bool inStruct)
251 {
252 if (Consume(TokenType.Word) is ScriptToken propName)
253 {
254 SymbolReference target = ParsePropName(propName, inStruct);
255 if (InSubOject && target.Node is VariableDeclaration {IsTransient: true})
256 {
257 TypeError("Cannot assign to a transient property in a SubObject!", propName);
258 }
259 VariableType targetType = target.ResolveType();
260 if (Matches(TokenType.LeftSqrBracket))
261 {
262 Expression expression = ParseLiteral();
263 if (expression is not IntegerLiteral intLit)
264 {
265 throw ParseError("Expected an integer index!", expression?.StartPos ?? CurrentPosition, expression?.EndPos);
266 }
267
268 if (targetType is not StaticArrayType arrType)
269 {
270 TypeError("Cannot index a property that is not a static array!", intLit);
271 }
272 else if (intLit.Value >= arrType.Length)
273 {
274 TypeError($"'{propName}' only has {arrType.Length} elements!", intLit);
275 }
276 else if (intLit.Value < 0)
277 {
278 TypeError("Index cannot be a negative number!", intLit);
279 }
280 else
281 {
282 targetType = arrType.ElementType;
283 }
284
285 if (Consume(TokenType.RightSqrBracket) is not {} closeBracket)
286 {
287 throw ParseError("Expected a ']'!", CurrentPosition);
288 }
289 target = new ArraySymbolRef(target, intLit, target.StartPos, closeBracket.EndPos);
290 }
291 else if (targetType is StaticArrayType)
292 {
293 throw ParseError($"Cannot assign directly to a static array! You must assign to each index individually, (eg. {propName.Value}[0] = ...)", propName);
294 }
295 if (Matches(TokenType.Assign, EF.Operator))
296 {
297 Expression literal = ParseValue(targetType);
298 return new AssignStatement(target, literal, propName.StartPos, literal.EndPos);
299 }
300
301 throw ParseError("Expected '=' in assignment statement!", CurrentPosition);
302 }
303
304 throw ParseError("Expected name of property!", CurrentPosition);
305 }

307 private Expression ParseValue(VariableType targetType)
308 {
309 Expression literal;
310 if (Matches(TokenType.LeftBracket))
311 {
312 if (targetType is not Struct targetStruct)
313 {
314 throw ParseError($"A '{{' is used to start a struct. Expected a {targetType.FullTypeName()} literal!", CurrentPosition);
315 }
316 literal = FinishStructLiteral(targetStruct);
317 }
318 else if (Matches(TokenType.LeftParenth))
319 {
320 switch (targetType)
321 {
322 case DynamicArrayType dynamicArrayType:
323 literal = FinishDynamicArrayLiteral(dynamicArrayType);
324 break;
325 case Struct:
326 ParseError("Use '{' for struct literals, not '('.", CurrentPosition);
327 goto default;
328 default:
329 throw ParseError($"A '(' is used to start a dynamic array literal. Expected a {targetType.FullTypeName()} literal!", CurrentPosition);
330 }
331 }
332 else
333 {
334 bool isNegative = Matches(TokenType.MinusSign, EF.Operator);
335
336 literal = ParseLiteral();
337 if (literal is not null)
338 {
339 if (isNegative)
340 {
341 switch (literal)
342 {
343 case FloatLiteral floatLiteral:
344 floatLiteral.Value *= -1;
345 break;
346 case IntegerLiteral integerLiteral:
347 integerLiteral.Value *= -1;
348 break;
349 default:
350 throw ParseError("Malformed constant value!", CurrentPosition);
351 }
352 }
353 }
354 else
355 {
356 if (isNegative)
357 {
358 throw ParseError("Unexpected '-' !", CurrentPosition);
359 }
360
361 if (Consume(TokenType.Word) is { } token)
362 {
363 if (Consume(TokenType.NameLiteral) is { } objName)
364 {
365 literal = ParseObjectLiteral(token, objName, false);
366 }
367 else
368 {
369 literal = ParseBasicRef(token);
370 if (literal is SymbolReference {Node: Const cnst})
371 {
372 literal = cnst.Literal;
373 }
374 }
375 }
376 else
377 {
378 throw ParseError("Expected a value!", CurrentPosition);
379 }
380 }
381 }
382
383 VerifyLiteral(targetType, ref literal);
384 return literal;
385 }

387 private StructLiteral FinishStructLiteral(Struct targetStruct)
388 {
389 ScriptToken openingBracket = PrevToken;
390 var statements = new List<Statement>();
391
392 if (!Matches(TokenType.RightBracket))
393 {
394 ExpressionScopes.Push(targetStruct.GetScope());
395 try
396 {
397 var statement = ParseAssignment(true);
398 statements.Add(statement);
399 while (Matches(TokenType.Comma))
400 {
401 statement = ParseAssignment(true);
402 statements.Add(statement);
403 }
404 if (!Matches(TokenType.RightBracket))
405 {
406 throw ParseError("Expected struct literal to end with a '}'!", openingBracket.StartPos, CurrentPosition);
407 }
408 }
409 finally
410 {
411 ExpressionScopes.Pop();
412 }
413 }
414
415 return new StructLiteral(targetStruct, statements, PrevToken.StartPos, PrevToken.EndPos);
416 }

419 {
420 ScriptToken openingParen = PrevToken;

422
423 if (!Matches(TokenType.RightParenth))
424 {

426 var value = ParseValue(targetType);
427 values.Add(value);
428 while (Matches(TokenType.Comma))

431 values.Add(value);

443 private void VerifyLiteral(VariableType targetType, ref Expression literal)
444 {
445 switch (targetType)
446 {
447 case Class targetClass:
448 if (literal is not NoneLiteral)
449 {
450 VariableType valueClass;
451 if (literal is ObjectLiteral objectLiteral)
452 {
453 if (Pcc.FindEntry(objectLiteral.Name.Value) is not IEntry)
454 {
455 TypeError($"Could not find '{objectLiteral.Name.Value}' in this file!");
456 }
457 valueClass = objectLiteral.Class;
458 }
459 else if (literal is SymbolReference {Node: Subobject {Class: Class subObjClass}})
460 {
461 valueClass = subObjClass;
462 }
463 else
464 {
465 TypeError($"Expected an {OBJECT} literal or sub-object name!", literal);
466 break;
467 }
468
469 if (valueClass is not (Class or ClassType)
470 || valueClass is Class literalClass && !literalClass.SameAsOrSubClassOf(targetClass)
471 || valueClass is ClassType && targetClass.Name is not ("Class" or "Object"))
472 {
473 TypeError($"Expected an object of class {targetClass.Name} or a subclass!", literal);
474 }
475 }
476
477 break;
478 case ClassType targetClassLimiter:
479 if (literal is not NoneLiteral)
480 {
481 if (literal is not ObjectLiteral {Class: ClassType literalClassType})
482 {
483 TypeError($"Expected a class literal!", literal);
484 }
485 else if (targetClassLimiter.ClassLimiter != literalClassType.ClassLimiter && !((Class)literalClassType.ClassLimiter).SameAsOrSubClassOf(targetClassLimiter.ClassLimiter.Name))
486 {
487 TypeError($"Cannot assign a value of type '{literalClassType.FullTypeName()}' to a variable of type '{targetClassLimiter.FullTypeName()}'.", literal);
488 }
489 }
490
491 break;
492 case DelegateType delegateType:
493 if (literal is not NoneLiteral)
494 {
495 if (literal is not SymbolReference {Node: Function func})
496 {
497 if (literal is ObjectLiteral {Class: ClassType {ClassLimiter: Class containingclass}} && Matches(TokenType.Dot) && Consume(TokenType.Word) is ScriptToken funcNameToken
498 && Symbols.TryGetSymbolInScopeStack(funcNameToken.Value, out func, containingclass.GetScope()))
499 {
500 literal = new CompositeSymbolRef(literal, NewSymbolReference(func, funcNameToken, false), true, literal.StartPos, funcNameToken.EndPos);
501 }
502 else
503 {
504 TypeError("Expected a function reference!", literal);
505 break;
506 }
507 }
508 if (!func.SignatureEquals(delegateType.DefaultFunction))
509 {
510 TypeError($"Expected a function with the same signature as {(delegateType.DefaultFunction.Outer as Class)?.Name}.{delegateType.DefaultFunction.Name}!", literal);
511 }
512 }
513 break;
514 case DynamicArrayType:
515 if (literal is not DynamicArrayLiteral)
516 {
517 TypeError($"Expected a dynamic array literal!", literal);
518 }
519 break;
520 case Enumeration enumeration:
521 if (literal is not NoneLiteral)
522 {
523 if (literal is not SymbolReference {Node: EnumValue enumVal})
524 {
525 var prevToken = PrevToken;
526 //this handles the case where a property has the same name as the Enumeration, and reparses it as an enumvalue
527 if (Symbols.TryGetType(prevToken.Value, out Enumeration enum2) && enum2 == enumeration
528 && Matches(TokenType.Dot) && Consume(TokenType.Word) is ScriptToken enumValueToken)
529 {
530 prevToken.SyntaxType = EF.Enum;
531 prevToken.AssociatedNode = enum2;
532 if (enumeration.Values.FirstOrDefault(val => val.Name.CaseInsensitiveEquals(enumValueToken.Value)) is EnumValue enumValue)
533 {
534 literal = NewSymbolReference(enumValue, enumValueToken, false);
535 break;
536 }
537 throw ParseError("Expected valid enum value!", CurrentPosition);
538 }
539 TypeError($"Expected an enum value!", literal);
540 }
541 else if (enumeration != enumVal.Enum)
542 {
543 TypeError($"Expected an {enumeration.Name} value, not an {enumVal.Enum.Name} value!", literal);
544 }
545 }
546 break;
547 case Struct:
548 if (literal is not StructLiteral)
549 {
550 TypeError($"Expected a {STRUCT} literal!", literal);
551 }
552 break;
553 default:
554 switch (targetType.PropertyType)
555 {
556 case EPropertyType.Byte:
557 if (literal is not IntegerLiteral byteLiteral)
558 {
559 TypeError($"Expected a {BYTE}!", literal);
560 }
561 else if (byteLiteral.Value is < 0 or > 255)
562 {
563 TypeError($"{byteLiteral.Value} is not in the range of valid byte values: [0, 255]", literal);
564 }
565
566 break;
567 case EPropertyType.Int:
568 if (literal is not IntegerLiteral)
569 {
570 TypeError($"Expected an integer!", literal);
571 }
572
573 break;
574 case EPropertyType.Bool:
575 if (literal is not BooleanLiteral)
576 {
577 TypeError($"Expected {TRUE} or {FALSE}!", literal);
578 }
579
580 break;
581 case EPropertyType.Float:
582 if (literal is IntegerLiteral intLit)
583 {
584 literal = new FloatLiteral(intLit.Value, intLit.StartPos, intLit.EndPos);
585 }
586 else if (literal is not FloatLiteral)
587 {
588 TypeError($"Expected a floating point number!", literal);
589 }
590
591 break;
592 case EPropertyType.Name:
593 if (literal is not NameLiteral)
594 {
595 TypeError($"Expected a {NAME} literal!", literal);
596 }
597
598 break;
599 case EPropertyType.String:
600 if (literal is not StringLiteral)
601 {
602 TypeError($"Expected a {STRING} literal!", literal);
603 }
604
605 break;
606 case EPropertyType.StringRef:
607 if (literal is not StringRefLiteral)
608 {
609 TypeError($"Expected a {STRINGREF} literal!", literal);
610 }
611
612 break;
613 default:
614 throw new ArgumentOutOfRangeException();
615 }
616
617 break;
618 }

621 private SymbolReference ParsePropName(ScriptToken token, bool inStruct)
622 {
623 string specificScope = ExpressionScopes.Peek();
624 if (!Symbols.TryGetSymbolInScopeStack(token.Value, out ASTNode symbol, specificScope)
625 || inStruct && symbol.Outer is not Struct)
626 {
627 //TODO: better error message
628 TypeError($"{specificScope} has no member named '{token.Value}'!", token);
629 symbol = new VariableType("ERROR");
630 }
631
632 if (!inStruct && token.Value.CaseInsensitiveEquals("Name") || token.Value.CaseInsensitiveEquals("ObjectArchetype "))
633 {
634 TypeError($"Cannot set '{token.Value}' property!", token);
635 }
636
637 return NewSymbolReference(symbol, token, false);

640 private SymbolReference ParseBasicRef(ScriptToken token)
641 {
642 string specificScope = Symbols.CurrentScopeName;
643 if (!Symbols.TryGetSymbolInScopeStack(token.Value, out ASTNode symbol, specificScope))
644 {
645 //const, or enum
646 if (Symbols.TryGetType(token.Value, out VariableType destType))
647 {
648 token.AssociatedNode = destType;
649 if (destType is Enumeration enm && Matches(TokenType.Dot))
650 {
651 token.SyntaxType = EF.Enum;
652 if (Consume(TokenType.Word) is { } enumValName
653 && enm.Values.FirstOrDefault(val => val.Name.CaseInsensitiveEquals(enumValName.Value)) is EnumValue enumValue)
654 {
655 enumValName.AssociatedNode = enm;
656 return NewSymbolReference(enumValue, enumValName, false);
657 }
658 throw ParseError("Expected valid enum value!", CurrentPosition);
659 }
660 if (destType is Const cnst)
661 {
662 return NewSymbolReference(cnst, token, false);
663 }
664 }
665 //TODO: better error message
666 TypeError($"{specificScope} has no member named '{token.Value}'!", token);
667 symbol = new VariableType("ERROR");
668 }
669
670 return NewSymbolReference(symbol, token, false);
671 }
