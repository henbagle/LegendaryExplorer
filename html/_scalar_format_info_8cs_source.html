84 internal override bool Format(StringBuilder sb, string format, object arg, IFormatProvider formatProvider, ref int maxLength)
85 {
86 var success = true;
87 var originalLength = sb.Length;
88
89 if (PushCurrentObject(arg))
90 {
91 if (ShowDataType)
92 {
93 success = success && AppendInstanceTypeName(arg, sb, ref maxLength);
94 }
95
96 if (!string.IsNullOrEmpty(InstanceName))
97 {
98 var name = string.Format("{0}{1}{2}={2}", sb.Length > originalLength ? " " : "", InstanceName, IsMultiLinedFormat ? " " : "");
99 success = FormatInfoUtils.TryAppendString(this, sb, name, success, ref maxLength);
100 }
101 else if (sb.Length > originalLength)
102 {
103 success = FormatInfoUtils.TryAppendChar(this, sb, ' ', success, ref maxLength);
104 }
105
106 var argType = GetInstanceType(arg);
107
108 if (arg == null)
109 {
110 success = FormatInfoUtils.TryAppendString(this, sb, FormatInfoUtils.DefaultNullFormatted, success, ref maxLength);
111 }
112 else if (arg is char)
113 {
114 success = success && FormatChar(sb, (char) arg, true, false, ref maxLength);
115 }
116 else if (arg is string)
117 {
118 success = success && FormatString(sb, (string) arg, true, ref maxLength);
119 }
120 else if (arg is Boolean)
121 {
122 success = FormatInfoUtils.TryAppendString(this, sb, ((bool) arg) ? "true" : "false", success, ref maxLength);
123 }
124 else if (argType.IsEnum)
125 {
126 var plainName = Enum.GetName(argType, arg);
127
128 if (plainName != null)
129 {
130 success = FormatInfoUtils.TryAppendString(this, sb, plainName, success, ref maxLength);
131 }
132 else
133 {
134 success = success && AppendEnumFlags((Enum) arg, sb, ref maxLength); // arg is an OR-ed combination of values
135 }
136 }
137 else
138 {
139 success = FormatInfoUtils.TryAppendString(this, sb, arg.ToString(), success, ref maxLength);
140 }
141
142 while (success && sb.Length - originalLength < _fieldLength)
143 {
144 success = FormatInfoUtils.TryAppendChar(this, sb, ' ', success, ref maxLength);
145 }
146
147 PopCurrentObject();
148 }
149 else
150 {
151 success = success && FormatInfoUtils.ReportInfiniteLoop(sb, arg, InstanceName, ref maxLength);
152 }
153
154 if (!success)
155 {
156 sb.Length = originalLength;
157 }
158
159 return success;
160 }

181 private bool FormatChar(StringBuilder sb, char c, bool quote, bool stringEscaping, ref int maxLength)
182 {
183 var success = true;
184 var originalLength = sb.Length;
185
186 var quotationMark = (quote ? "'" : string.Empty);
187
188 if (c == '\n')
189 {
190 success = FormatInfoUtils.TryAppendString(this, sb, string.Format("{0}\\n{0}", quotationMark), success, ref maxLength);
191 }
192 else if (c == '\r')
193 {
194 success = FormatInfoUtils.TryAppendString(this, sb, string.Format("{0}\\r{0}", quotationMark), success, ref maxLength);
195 }
196 else if (c == '\t')
197 {
198 success = FormatInfoUtils.TryAppendString(this, sb, string.Format("{0}\\t{0}", quotationMark), success, ref maxLength);
199 }
200 else if (c < ' ')
201 {
202 success = FormatInfoUtils.TryAppendString(this, sb, string.Format("0x{0:X2}", (int) c), success, ref maxLength);
203 }
204 else if (!stringEscaping && c == '\'')
205 {
206 success = FormatInfoUtils.TryAppendString(this, sb, string.Format("{0}\\'{0}", quotationMark), success, ref maxLength);
207 }
208 else if (stringEscaping && c == '\\')
209 {
210 success = FormatInfoUtils.TryAppendString(this, sb, string.Format("{0}\\\\{0}", quotationMark), success, ref maxLength);
211 }
212 else if (stringEscaping && c == '"')
213 {
214 success = FormatInfoUtils.TryAppendString(this, sb, string.Format("{0}\\\"{0}", quotationMark), success, ref maxLength);
215 }
216 else
217 {
218 success = FormatInfoUtils.TryAppendString(this, sb, string.Format("{1}{0}{1}", c, quotationMark), success, ref maxLength);
219 }
220
221 if (!success)
222 {
223 sb.Length = originalLength;
224 }
225
226 return success;

286 private bool AppendEnumFlags(Enum value, StringBuilder sb, ref int maxLength)
287 {
288 var success = true;
289 var originalLength = sb.Length;
290
291 var enumType = value.GetType();
292 var underlyingType = Enum.GetUnderlyingType(enumType);
293
294 // The following code tries to recover flags in a flagged enum value
295 // in .NET 4.0 this can be done much easier using Enum.HasFlag instance-level method, but with some unpredictable outcomes
296 // e.g. if enumeration defines Left, Right and Both=Left | Right; then Both might be shown as Left | Right | Both which is avoided by this function
297
298 var isInt64 = false;
299 var isUint64 = false;
300 Int64 int64Value = 0;
301 UInt64 uint64Value = 0;
302
303 if (underlyingType == typeof (sbyte) || underlyingType == typeof (Int16) || underlyingType == typeof (Int32) ||
304 underlyingType == typeof (Int64))
305 {
306 isInt64 = true;
307 int64Value = ((IConvertible) value).ToInt64(null);
308 }
309 else if (underlyingType == typeof (byte) || underlyingType == typeof (UInt16) || underlyingType == typeof (UInt32) ||
310 underlyingType == typeof (UInt64))
311 {
312 isUint64 = true;
313 uint64Value = ((IConvertible) value).ToUInt64(null);
314 }
315
316 Int64 constructedInt64Value = 0;
317 UInt64 constructedUint64Value = 0;
318
319 var values = Enum.GetValues(enumType);
320 var constructionComplete = false;
321 var flags = new List<object>(); // Enumeration values that are incorporated in OR-ed value
322
323 foreach (var simpleValue in values)
324 {
325 if (isInt64)
326 {
327 var curInt64Value = ((IConvertible) simpleValue).ToInt64(null);
328
329 if ((int64Value & curInt64Value) == curInt64Value && (constructedInt64Value & curInt64Value) != curInt64Value)
330 {
331 // simpleValue is part of the resulting OR-ed value and is not included in currently constructed value (or is not completely included)
332
333 constructedInt64Value = constructedInt64Value | curInt64Value;
334 flags.Add(simpleValue);
335
336 if (constructedInt64Value == int64Value)
337 {
338 constructionComplete = true;
339 break;
340 }
341 }
342 }
343 else if (isUint64)
344 {
345 var curUint64Value = ((IConvertible) simpleValue).ToUInt64(null);
346
347 if ((uint64Value & curUint64Value) == curUint64Value && (constructedUint64Value & curUint64Value) != curUint64Value)
348 {
349 // simpleValue is part of the resulting OR-ed value and is not included in currently constructed value (or is not completely included)
350
351 constructedUint64Value = constructedUint64Value | curUint64Value;
352 flags.Add(simpleValue);
353
354 if (constructedUint64Value == uint64Value)
355 {
356 constructionComplete = true;
357 break;
358 }
359 }
360 }
361 }
362
363 if (!constructionComplete)
364 {
365 success = FormatInfoUtils.TryAppendString(this, sb, value.ToString(), success, ref maxLength);
366 // Append it as number or whatever Enum.ToString() makes out of this value
367 }
368 else
369 {
370 var isFirst = true;
371
372 foreach (var simpleValue in flags)
373 {
374 if (!isFirst)
375 {
376 success = FormatInfoUtils.TryAppendChar(this, sb, '+', success, ref maxLength);
377 }
378 isFirst = false;
379
380 success = FormatInfoUtils.TryAppendString(this, sb, Enum.GetName(enumType, simpleValue), success, ref maxLength);
381
382 if (!success)
383 {
384 break;
385 }
386 }
387 }
388
389 if (!success)
390 {
391 sb.Length = originalLength;
392 }
393
394 return success;
395 }
