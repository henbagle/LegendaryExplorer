44 static ParallelCRC()
45 {
46 unchecked
47 {
48 Table = new uint[256 * CRC_NUM_TABLES];
49 int i;
50 for (i = 0; i < 256; i++)
51 {
52 uint r = (uint)i;
53 for (int j = 0; j < 8; j++)
54 r = r >> 1 ^ kCrcPoly & ~((r & 1) - 1);
55 Table[i] = r;
56 }
57 for (; i < 256 * CRC_NUM_TABLES; i++)
58 {
59 uint r = Table[i - 256];
60 Table[i] = Table[r & 0xFF] ^ r >> 8;
61 }
62 }
63 }

85 public void Update(byte[] data, int offset, int count)
86 {
87 new ArraySegment<byte>(data, offset, count); // check arguments
88
89 if (count <= ThreadCost || ProcessorCount <= 1)
90 {
91 value = ProcessBlock(value, data, offset, count);
92 return;
93 }
94
95 // choose optimal number of threads to use
96
97 int threadCount = ProcessorCount;
98 L0:
99 int bytesPerThread = (count + threadCount - 1) / threadCount;
100 if (bytesPerThread < ThreadCost >> 1)
101 {
102 threadCount--;
103 goto L0;
104 }
105
106 // threadCount >= 2
107
108 Job lastJob = null;
109 while (count > bytesPerThread)
110 {
111 var job = new Job(new ArraySegment<byte>(data, offset, bytesPerThread), this, lastJob);
112 ThreadPool.QueueUserWorkItem(job.Start);
113 offset += bytesPerThread;
114 count -= bytesPerThread;
115 lastJob = job;
116 }
117
118 // lastJob != null
119 var lastBlockCRC = ProcessBlock(kInitial, data, offset, count);
120 lastJob.WaitAndDispose();
121 value = Combine(value, lastBlockCRC, count);
122 }

124 private static uint ProcessBlock(uint crc, byte[] data, int offset, int count)
125 {
126 /*
127 * A copy of Optimized implementation.
128 */
129
130 if (count < 0) throw new ArgumentOutOfRangeException("count");
131 if (count == 0) return crc;
132
133 var table = Table;
134
135 for (; (offset & 7) != 0 && count != 0; count--)
136 crc = crc >> 8 ^ table[(byte)crc ^ data[offset++]];
137
138 if (count >= 8)
139 {
140 int end = count - 8 & ~7;
141 count -= end;
142 end += offset;
143
144 while (offset != end)
145 {
146 crc ^= (uint)(data[offset] + (data[offset + 1] << 8) + (data[offset + 2] << 16) + (data[offset + 3] << 24));
147 uint high = (uint)(data[offset + 4] + (data[offset + 5] << 8) + (data[offset + 6] << 16) + (data[offset + 7] << 24));
148 offset += 8;
149
150 crc = table[(byte)crc + 0x700]
151 ^ table[(byte)(crc >>= 8) + 0x600]
152 ^ table[(byte)(crc >>= 8) + 0x500]
153 ^ table[/*(byte)*/(crc >> 8) + 0x400]
154 ^ table[(byte)high + 0x300]
155 ^ table[(byte)(high >>= 8) + 0x200]
156 ^ table[(byte)(high >>= 8) + 0x100]
157 ^ table[/*(byte)*/(high >> 8) + 0x000];
158 }
159 }
160
161 while (count-- != 0)
162 crc = crc >> 8 ^ table[(byte)crc ^ data[offset++]];
163
164 return crc;
165 }

186 /*
187 * CRC values combining algorithm.
188 * Taken from DotNetZip project sources (http://dotnetzip.codeplex.com/)
189 */
190
195 private static uint Combine(uint crc1, uint crc2, int length2)
196 {
197 if (length2 <= 0) return crc1;
198 if (crc1 == kInitial) return crc2;
199
200 if (even_cache == null)
201 {
202 Prepare_even_odd_Cache();
203 }
204
205 uint[] even = CopyArray(even_cache);
206 uint[] odd = CopyArray(odd_cache);
207
208 crc1 = ~crc1;
209 crc2 = ~crc2;
210
211 uint len2 = (uint)length2;
212
213 // apply len2 zeros to crc1 (first square will put the operator for one
214 // zero byte, eight zero bits, in even)
215 do
216 {
217 // apply zeros operator for this bit of len2
218 gf2_matrix_square(even, odd);
219
220 if ((len2 & 1) != 0) crc1 = gf2_matrix_times(even, crc1);
221 len2 >>= 1;
222
223 if (len2 == 0) break;
224
225 // another iteration of the loop with odd and even swapped
226 gf2_matrix_square(odd, even);
227 if ((len2 & 1) != 0) crc1 = gf2_matrix_times(odd, crc1);
228 len2 >>= 1;
229 } while (len2 != 0);
230
231 crc1 ^= crc2;
232 return ~crc1;
233 }
234
235 private static uint[] even_cache = null;
236 private static uint[] odd_cache;
237
238 private static void Prepare_even_odd_Cache()
239 {
240 var even = new uint[32]; // even-power-of-two zeros operator
241 var odd = new uint[32]; // odd-power-of-two zeros operator
242
243 // put operator for one zero bit in odd
244 odd[0] = kCrcPoly; // the CRC-32 polynomial
245 for (int i = 1; i < 32; i++) odd[i] = 1U << i - 1;
246
247 // put operator for two zero bits in even
248 gf2_matrix_square(even, odd);
249
250 // put operator for four zero bits in odd
251 gf2_matrix_square(odd, even);
252
253 odd_cache = odd;
254 even_cache = even;
255 }
256
258 private static uint gf2_matrix_times(uint[] matrix, uint vec)
259 {
260 uint sum = 0;
261 int i = 0;
262 while (vec != 0)
263 {
264 if ((vec & 1) != 0) sum ^= matrix[i];
265 vec >>= 1;
266 i++;
267 }
268 return sum;
269 }
270
273 private static void gf2_matrix_square(uint[] square, uint[] mat)
274 {
275 for (int i = 0; i < 32; i++)
276 square[i] = gf2_matrix_times(mat, mat[i]);
277 }

288 class Job : IDisposable
289 {
290 private ArraySegment<byte> data;
291 private Job previousJob;
292 private ParallelCRC accumulator;
293
294 private ManualResetEventSlim finished; // replace with ManualResetEvent if necessary
295
296 public Job(ArraySegment<byte> data, ParallelCRC accumulator, Job previousJob)
297 {
298 this.data = data;
299 this.accumulator = accumulator;
300 this.previousJob = previousJob;
301 finished = new ManualResetEventSlim(false);
302 }
303
304 public void Start(object arg)
305 {
306 var crc = ProcessBlock(kInitial, data.Array, data.Offset, data.Count);
307 if (previousJob != null) previousJob.WaitAndDispose();
308 accumulator.value = Combine(accumulator.value, crc, data.Count);
309 finished.Set();
310 }
311
312 public void WaitAndDispose()
313 {
314 finished.Wait();
315 Dispose();
316 }
317
318 public void Dispose()
319 {
320 if (finished != null) finished.Dispose();
321 finished = null;
322 }
323 }
