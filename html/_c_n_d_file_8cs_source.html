22 private void Read(Stream stream)
23 {
24 if (stream.ReadInt32() != Magic)
25 {
26 throw new Exception("This is not a conditional file!");
27 }
28
29 int version = stream.ReadInt32();
30 if (version != Version)
31 {
32 throw new Exception($"Wrong file version! Expected '{Version}', got '{version}'");
33 }
34
35 //count of serialized conditional bodies. Conditionals with identical bytecode can share the same serialized bytecode,
36 //so this can be less than the entryCount. We don't need to know it for deserializing
37 stream.SkipInt16();
38
39 int entryCount = stream.ReadInt16();
40
41 ConditionalEntries = new List<ConditionalEntry>(entryCount);
42 for (int i = 0; i < entryCount; i++)
43 {
44 ConditionalEntries.Add(new ConditionalEntry
45 {
46 ID = stream.ReadInt32(),
47 Offset = stream.ReadInt32()
48 });
49 }
50
51 int streamLength = (int)stream.Length;
52 List<ConditionalEntry> sortedEntries = ConditionalEntries.OrderBy(entry => entry.Offset).ToList();
53 for (int i = 0; i < entryCount; i++)
54 {
55 ConditionalEntry entry = sortedEntries[i];
56 int nextOffset = streamLength;
57 //we have to scan ahead because multiple entries can share the same offset
58 for (int j = i; j < entryCount; j++)
59 {
60 if (sortedEntries[j].Offset > entry.Offset)
61 {
62 nextOffset = sortedEntries[j].Offset;
63 break;
64 }
65 }
66 int size = nextOffset - entry.Offset;
67 stream.JumpTo(entry.Offset);
68 entry.Data = stream.ReadToBuffer(size);
69 }
70 }
71
72 private void Write(Stream stream)
73 {
74 stream.WriteInt32(Magic);
75 stream.WriteInt32(Version);
76 stream.WriteUInt16((ushort)ConditionalEntries.Count); // Serialized count. We don't save like BW, so this will always be the same as the amount of conditionals.
77 stream.WriteUInt16((ushort)ConditionalEntries.Count);
78
79 //This works, but is not the saving method bioware used.
80 //to replicate that, we would need to sort by Data size and combine conditions with the same Data
81
82 int sumOffset = (int)(stream.Position) + ConditionalEntries.Count * 8;
83 //DO NOT CONVERT TO FOREACH! breaks it for some unknown reason
84 for (int i = 0; i < ConditionalEntries.Count; i++)
85 {
86 ConditionalEntry entry = ConditionalEntries[i];
87 stream.WriteInt32(entry.ID);
88 stream.WriteInt32(sumOffset);
89 sumOffset += entry.Data.Length;
90 }
91
92 foreach (ConditionalEntry entry in ConditionalEntries)
93 {
94 stream.WriteFromBuffer(entry.Data);
95 }
96 }
