129 public static byte[] ConvertTexture2D(ExportEntry export, MEGame newGame, List<int> offsets = null, StorageTypes newStorageType = StorageTypes.empty)
130 {
131 MemoryStream bin = export.GetReadOnlyBinaryStream();
132 if (bin.Length == 0)
133 {
134 return bin.ToArray();
135 }
136 using var os = MemoryManager.GetMemoryStream();
137
138 if (!export.Game.IsGame3())
139 {
140 bin.Skip(16);
141 }
142 if (!newGame.IsGame3())
143 {
144 os.WriteZeros(16);//includes fileOffset, but that will be set during save
145 }
146
147 int mipCount = bin.ReadInt32();
148 long mipCountPosition = os.Position;
149 os.WriteInt32(mipCount);
150 List<Texture2DMipInfo> mips = Texture2D.GetTexture2DMipInfos(export, export.GetProperty<NameProperty>("TextureFileCacheName")?.Value);
151 int offsetIdx = 0;
152 int trueMipCount = 0;
153 for (int i = 0; i < mipCount; i++)
154 {
155 var storageType = (StorageTypes)bin.ReadInt32();
156 int uncompressedSize = bin.ReadInt32();
157 int compressedSize = bin.ReadInt32();
158 int fileOffset = bin.ReadInt32();
159 byte[] texture;
160 switch (storageType)
161 {
162 case StorageTypes.pccUnc:
163 texture = bin.ReadToBuffer(uncompressedSize);
164 break;
165 case StorageTypes.pccLZO:
166 case StorageTypes.pccZlib:
167 case StorageTypes.pccOodle:
168 texture = bin.ReadToBuffer(compressedSize);
169 if (offsets != null)
170 {
171 //the caller has transferred this mip, so use the provided data
172 texture = Array.Empty<byte>();
173 fileOffset = offsets[offsetIdx++];
174 compressedSize = offsets[offsetIdx] - fileOffset;
175 if (newStorageType != StorageTypes.empty)
176 {
177 storageType = newStorageType;
178 }
179 }
180 break;
181 case StorageTypes.empty:
182 texture = new byte[0];
183 break;
184 default:
185 if (export.Game != newGame)
186 {
187 storageType &= (StorageTypes)~StorageFlags.externalFile;
188 texture = Texture2D.GetTextureData(mips[i], export.Game, MEDirectories.GetDefaultGamePath(export.Game),false); //copy in external textures
189 }
190 else
191 {
192 texture = Array.Empty<byte>();
193 }
194 break;
195
196 }
197
198 int width = bin.ReadInt32();
199 int height = bin.ReadInt32();
200 if (newGame == MEGame.UDK && storageType == StorageTypes.empty)
201 {
202 continue;
203 }
204 trueMipCount++;
205 os.WriteInt32((int)storageType);
206 os.WriteInt32(uncompressedSize);
207 os.WriteInt32(compressedSize);
208 os.WriteInt32(fileOffset);//fileOffset will be fixed during save
209 os.WriteFromBuffer(texture);
210 os.WriteInt32(width);
211 os.WriteInt32(height);
212
213 }
214
215 long postMipPosition = os.Position;
216 os.JumpTo(mipCountPosition);
217 os.WriteInt32(trueMipCount);
218 os.JumpTo(postMipPosition);
219
220 int unk1 = 0;
221 if (export.Game != MEGame.UDK)
222 {
223 unk1 = bin.ReadInt32();
224 }
225 if (newGame != MEGame.UDK)
226 {
227 os.WriteInt32(unk1);
228 }
229
230 Guid textureGuid = export.Game != MEGame.ME1 ? bin.ReadGuid() : Guid.NewGuid();
231
232 if (newGame != MEGame.ME1)
233 {
234 os.WriteGuid(textureGuid);
235 }
236
237 if (export.Game == MEGame.UDK)
238 {
239 bin.Skip(32);
240 }
241 if (newGame == MEGame.UDK)
242 {
243 os.WriteZeros(4 * 8);
244 }
245

247 {
248 bin.Skip(4);
249 }
250 if (newGame == MEGame.ME3 || newGame.IsLEGame())
251 {
252 os.WriteInt32(0);
253 }
254 if (export.ClassName == "LightMapTexture2D")
255 {
