93 public static PropertyCollection ReadProps(ExportEntry export, Stream rawStream, string typeName, bool includeNoneProperty = false, bool requireNoneAtEnd = true, IEntry entry = null, PackageCache packageCache = null)
94 {
95 var stream = new EndianReader(rawStream) { Endian = export.FileRef.Endian };
96 var props = new PropertyCollection();
97 long startPosition = stream.Position;
98 IMEPackage pcc = export.FileRef;
99 try
100 {
101 while (stream.Position + 8 <= stream.Length)
102 {
103 int propertyStartPosition = (int)stream.Position;
104 int nameIdx = stream.ReadInt32();
105 if (!pcc.IsName(nameIdx))
106 {
107 stream.Seek(-4, SeekOrigin.Current);
108 break;
109 }
110 string name = pcc.GetNameEntry(nameIdx);
111 if (name == "None")
112 {
113 props.Items.Add(new NoneProperty(stream) { StartOffset = propertyStartPosition, ValueOffset = propertyStartPosition });
114 stream.Seek(4, SeekOrigin.Current);
115 break;
116 }
117 var nameRef = new NameReference(name, stream.ReadInt32());
118 int typeIdx = stream.ReadInt32();
119 stream.Seek(4, SeekOrigin.Current);
120 int size = stream.ReadInt32();
121 if (!pcc.IsName(typeIdx) || size < 0 || size > stream.Length - stream.Position)
122 {
123 stream.Seek(-16, SeekOrigin.Current);
124 LECLog.Warning($@"Found broken properties, could not resolve as name. Export: {export.InstancedFullPath}");
125 break;
126 }
127 int staticArrayIndex = stream.ReadInt32();
128 string namev = pcc.GetNameEntry(typeIdx);
129 //Debug.WriteLine("Reading " + nameRef.Instanced + " (" + namev + ") at 0x" + (stream.Position - 24).ToString("X8"));
130 if (!Enum.TryParse(namev, out PropertyType type))
131 {
132 type = PropertyType.Unknown;
133 }
134
135 Property prop = null;
136 switch (type)
137 {
138 case PropertyType.StructProperty:
139 string structType = pcc.GetNameEntry(stream.ReadInt32());
140 stream.Seek(4, SeekOrigin.Current);
141 int valOffset = (int)stream.Position;
142 if (GlobalUnrealObjectInfo.IsImmutable(structType, pcc.Game))
143 {
144 PropertyCollection structProps = ReadImmutableStruct(export, stream, structType, size, packageCache, entry);
145 prop = new StructProperty(structType, structProps, nameRef, true) { ValueOffset = valOffset };
146 }
147 else
148 {
149 PropertyCollection structProps = ReadProps(export, stream.BaseStream, structType, includeNoneProperty, entry: entry, packageCache: packageCache);
150 prop = new StructProperty(structType, structProps, nameRef) { ValueOffset = valOffset };
151 }
152 break;
153 case PropertyType.IntProperty:
154 prop = new IntProperty(stream, nameRef);
155 break;
156 case PropertyType.FloatProperty:
157 prop = new FloatProperty(stream, nameRef);
158 break;
159 case PropertyType.ObjectProperty:
160 case PropertyType.InterfaceProperty:
161 case PropertyType.ComponentProperty:
162 prop = new ObjectProperty(stream, nameRef)
163 {
164 InternalPropType = type
165 };
166 break;
167 case PropertyType.NameProperty:
168 prop = new NameProperty(stream, pcc, nameRef);
169 break;
170 case PropertyType.BoolProperty:
171 prop = new BoolProperty(stream, pcc, nameRef);
172 break;
173 case PropertyType.BioMask4Property:
174 prop = new BioMask4Property(stream, nameRef);
175 break;
176 case PropertyType.ByteProperty:
177 {
178 if (size != 1)
179 {
180 NameReference enumType;
181 if (pcc.Game is MEGame.ME3 or MEGame.LE1 or MEGame.LE2 or MEGame.LE3 or MEGame.UDK || pcc.Platform == MEPackage.GamePlatform.PS3)
182 {
183 enumType = new NameReference(pcc.GetNameEntry(stream.ReadInt32()), stream.ReadInt32());
184 }
185 else
186 {
187 //Debug.WriteLine("Reading enum for ME1/ME2 at 0x" + propertyStartPosition.ToString("X6"));
188
189 //Attempt to get info without lookup first
190 var enumname = GlobalUnrealObjectInfo.GetEnumType(pcc.Game, nameRef, typeName);
191 ClassInfo classInfo = null;
192 if (enumname == null && entry is ExportEntry exp)
193 {
194 classInfo = GlobalUnrealObjectInfo.generateClassInfo(exp);
195 }
196
197 //Use DB info or attempt lookup
198 enumType = new NameReference(enumname ?? GlobalUnrealObjectInfo.GetEnumType(pcc.Game, nameRef, typeName == @"ScriptStruct" ? entry.ObjectName : typeName, classInfo));
199 }
200 try
201 {
202 prop = new EnumProperty(stream, pcc, enumType, nameRef);
203 }
204 catch (Exception ex)
205 {
206 //ERROR
207 //Debugger.Break();
208 LECLog.Warning($"Error parsing ByteProperty: {ex.Message}");
209 prop = new UnknownProperty(stream, 0, enumType, nameRef);
210 }
211 }
212 else
213 {
214 if (pcc.Game >= MEGame.ME3 || pcc.Platform == MEPackage.GamePlatform.PS3)
215 {
216 stream.Seek(8, SeekOrigin.Current);
217 }
218 prop = new ByteProperty(stream, nameRef);
219 }
220 }
221 break;
222 case PropertyType.ArrayProperty:
223 {
224 //Debug.WriteLine("Reading array properties, starting at 0x" + stream.Position.ToString("X5"));
225 var valStart = stream.Position;
226 prop = ReadArrayProperty(stream, export, typeName, nameRef, IncludeNoneProperties: includeNoneProperty, parsingEntry: entry, packageCache: packageCache);
227 //this can happen with m_aObjComments that were hex edited with old versions of the toolset
228 //technically valid, so we should support reading them
229 if (stream.Position < valStart + size)
230 {
231 //Old code: JumpTo
232 stream.Seek(valStart + size, SeekOrigin.Begin);
233 }
234 }
235 break;
236 case PropertyType.StrProperty:
237 {
238 prop = new StrProperty(stream, nameRef);
239 }
240 break;
241 case PropertyType.StringRefProperty:
242 prop = new StringRefProperty(stream, nameRef);
243 break;
244 case PropertyType.DelegateProperty:
245 prop = new DelegateProperty(stream, pcc, nameRef);
246 break;
247 case PropertyType.Unknown:
248 {
249 // Debugger.Break();
250 prop = new UnknownProperty(stream, size, pcc.GetNameEntry(typeIdx), nameRef);
251 }
252 break;
253 case PropertyType.None:
254 prop = new NoneProperty(stream);
255 break;
256 }
257
258 if (prop != null)
259 {
260 prop.StaticArrayIndex = staticArrayIndex;
261 prop.StartOffset = propertyStartPosition;
262 props.Items.Add(prop);
263 }
264 }
265 }
266 catch (Exception e)
267 {
268 Debug.WriteLine("Exception: " + e.Message);
269 }
270 if (props.Count > 0)
271 {
272 //error reading props.
273 if (props[^1].PropType != PropertyType.None && requireNoneAtEnd)
274 {
275 if (entry != null)
276 {
277 Debug.WriteLine(entry.FileRef.FilePath);
278 Debug.WriteLine($"#{entry.UIndex} {entry.ObjectName.Instanced} - Invalid properties: Does not end with None");
279 }
280#if DEBUG
281 props.endOffset = (int)stream.Position;
282 return props;
283#else
284 stream.Seek(startPosition, SeekOrigin.Begin);
285 return new PropertyCollection { endOffset = (int)stream.Position };
286#endif
287 }
288 //remove None Property
289 if (props[^1].PropType == PropertyType.None && !includeNoneProperty)
290 {
291 props.Items.RemoveAt(props.Count - 1);
292 }
293 }
294 props.endOffset = (int)stream.Position;
295 return props;
296 }

298 public static PropertyCollection ReadImmutableStruct(ExportEntry export, EndianReader stream, string structType, int size, PackageCache packageCache, IEntry parsingEntry = null)
299 {
300 IMEPackage pcc = export.FileRef;
301 //strip transients unless this is a class definition
302 bool stripTransients = parsingEntry is not {ClassName: "Class" or "ScriptStruct"};
303
304 MEGame structValueLookupGame = pcc.Game;
305 var props = new PropertyCollection();
306 switch (pcc.Game)
307 {
308 case MEGame.ME1 when parsingEntry != null && parsingEntry.FileRef.Platform == MEPackage.GamePlatform.PS3 && ME3UnrealObjectInfo.Structs.ContainsKey(structType):
309 case MEGame.ME2 when parsingEntry != null && parsingEntry.FileRef.Platform == MEPackage.GamePlatform.PS3 && ME3UnrealObjectInfo.Structs.ContainsKey(structType):
310 structValueLookupGame = MEGame.ME3;
311 break;
312 case MEGame.ME3 when ME3UnrealObjectInfo.Structs.ContainsKey(structType):
313 case MEGame.UDK when ME3UnrealObjectInfo.Structs.ContainsKey(structType):
314 case MEGame.ME2 when ME2UnrealObjectInfo.Structs.ContainsKey(structType):
315 case MEGame.ME1 when ME1UnrealObjectInfo.Structs.ContainsKey(structType):
316 case MEGame.LE3 when ME3UnrealObjectInfo.Structs.ContainsKey(structType):
317 case MEGame.LE2 when ME3UnrealObjectInfo.Structs.ContainsKey(structType):
318 case MEGame.LE1 when ME3UnrealObjectInfo.Structs.ContainsKey(structType):
319 break;
320 default:
321 Debug.WriteLine("Unknown struct type: " + structType);
322 props.Add(new UnknownProperty(stream, size) { StartOffset = (int)stream.Position });
323 return props;
324 }
325
326 PropertyCollection defaultProps = GlobalUnrealObjectInfo.getDefaultStructValue(structValueLookupGame, structType, stripTransients, packageCache, false);
327 if (defaultProps == null)
328 {
329 int startPos = (int)stream.Position;
330 props.Items.Add(new UnknownProperty(stream, size) { StartOffset = startPos });
331 return props;
332 }
333
334 foreach (var prop in defaultProps)
335 {
336 Property property;
337 if (prop is StructProperty defaultStructProperty)
338 {
339 //Set correct struct type
340 property = ReadImmutableStructProp(export, stream, prop, structType, defaultStructProperty.StructType, packageCache);
341 }
342 else
343 {
344 property = ReadImmutableStructProp(export, stream, prop, structType, packageCache: packageCache);
345 }
346
347 if (property.PropType != PropertyType.None)
348 {
349 props.Items.Add(property);
350 }
351 }
352 return props;
353 }

357 {
358 IMEPackage pcc = export.FileRef;
359 if (stream.Position + 1 >= stream.Length)
360 {
361 throw new EndOfStreamException("tried to read past bounds of Export Data");
362 }
363 int startPos = (int)stream.Position;
364
365 switch (template.PropType)
366 {
367 case PropertyType.FloatProperty:
368 return new FloatProperty(stream, template.Name) { StartOffset = startPos };
369 case PropertyType.IntProperty:
370 return new IntProperty(stream, template.Name) { StartOffset = startPos };
371 case PropertyType.ObjectProperty:
372 case PropertyType.InterfaceProperty:
373 case PropertyType.ComponentProperty:
374 return new ObjectProperty(stream, template.Name) { StartOffset = startPos, InternalPropType = template.PropType };
375 case PropertyType.StringRefProperty:
376 return new StringRefProperty(stream, template.Name) { StartOffset = startPos };
377 case PropertyType.NameProperty:
378 return new NameProperty(stream, pcc, template.Name) { StartOffset = startPos };
379 case PropertyType.BoolProperty:
380 //always say it's ME3 so that bools get read as 1 byte
381 return new BoolProperty(stream, pcc, template.Name, true) { StartOffset = startPos };
382 case PropertyType.ByteProperty:
383 if (template is EnumProperty)
384 {
385 string enumType = GlobalUnrealObjectInfo.GetEnumType(pcc.Game, template.Name, structType);
386 return new EnumProperty(stream, pcc, enumType, template.Name) { StartOffset = startPos };
387 }
388 return new ByteProperty(stream, template.Name) { StartOffset = startPos };
389 case PropertyType.BioMask4Property:
390 return new BioMask4Property(stream, template.Name) { StartOffset = startPos };
391 case PropertyType.StrProperty:
392 return new StrProperty(stream, template.Name) { StartOffset = startPos };
393 case PropertyType.ArrayProperty:

395 arrayProperty.StartOffset = startPos;
396 return arrayProperty;//this implementation needs checked, as I am not 100% sure of it's validity.
397 case PropertyType.StructProperty:
398 int valuePos = (int)stream.Position;
399 PropertyCollection structProps = ReadImmutableStruct(export, stream, GlobalUnrealObjectInfo.GetPropertyInfo(pcc.Game, template.Name, structType, containingExport: export).Reference, 0, packageCache, export);
400 var structProp = new StructProperty(nestedStructType ?? structType, structProps, template.Name, true)
401 {
402 StartOffset = startPos,
403 ValueOffset = valuePos
404 };
405 return structProp;//this implementation needs checked, as I am not 100% sure of it's validity.
406 case PropertyType.None:
407 return new NoneProperty { StartOffset = startPos };
408 case PropertyType.DelegateProperty:
409 return new DelegateProperty(stream, pcc, template.Name) { StartOffset = startPos };
410 case PropertyType.Unknown:
411 default:
412 throw new NotImplementedException("cannot read Unknown property of Immutable struct");
413 }

648 var diff = new PropertyCollection();

654 }

656 {

659 diff.Add(new StructProperty(thisStruct.StructType, thisStruct.Properties.Diff(otherStruct.Properties), thisStruct.Name, thisStruct.IsImmutable));
660 }
661 else
662 {
663 diff.Add(thisProp.DeepClone());
664 }
665 }
666 }
