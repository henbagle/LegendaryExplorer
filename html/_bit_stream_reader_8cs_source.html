184 public override int Read(byte[] buffer, int offset, int count)
185 {
186 var transferred = 0;
187 int toTransfer;
188
189 if (BitBuffer.LengthBytes > 0)
190 {
191 toTransfer = Math.Min(BitBuffer.LengthBytes, count);
192 BitBuffer.ReadBytes(buffer, offset, toTransfer);
193 offset += toTransfer;
194 count -= toTransfer;
195 transferred += toTransfer;
196 }
197
198 toTransfer = Math.Min(count, Math.Min(BitBuffer.AvailableBytes, TemporaryBuffer.Length));
199
200 var read = BaseStream.Read(TemporaryBuffer, 0, toTransfer);
201
202 BitBuffer.Write(TemporaryBuffer, 0, read);
203
204 toTransfer = Math.Min(BitBuffer.LengthBytes, count);
205
206 BitBuffer.ReadBytes(buffer, offset, toTransfer);
207
208 transferred += toTransfer;
209
210 return transferred;
211 }

216 public override bool ReadBoolean()

226 throw new EndOfStreamException();

239 public override int ReadByte()

243 return BitBuffer.ReadByte();

249 ? result
250 : BitBuffer.ReadByte();
