54 private static int MaterialShaderMapsOffset(MEGame game)
55 {
56 if (game.IsLEGame())
57 {
58 var expectedSize = game switch
59 {
60 MEGame.LE3 => LE3RefShaderCacheSize,
61 MEGame.LE2 => LE2RefShaderCacheSize,
62 MEGame.LE1 => LE1RefShaderCacheSize,
63 _ => 0
64 };
65 var actualsize = new FileInfo(shaderfilePath(game)).Length;
66 if (expectedSize != actualsize)

74 case MEGame.LE2:
75 LE2RefShaderCacheSize = actualsize;
76 break;

78 LE1RefShaderCacheSize = actualsize;

95 private static void populateOffsets(MEGame game, int offsetOfShaderCacheOffset)
96 {
97 string filePath = shaderfilePath(game);
98 if (File.Exists(filePath))
99 {
100 Dictionary<Guid, int> offsetDict = game switch
101 {
102 MEGame.ME3 => ME3ShaderOffsets ??= new Dictionary<Guid, int>(),
103 MEGame.ME2 => ME2ShaderOffsets ??= new Dictionary<Guid, int>(),
104 MEGame.ME1 => ME1ShaderOffsets ??= new Dictionary<Guid, int>(),
105 MEGame.LE3 => LE3ShaderOffsets ??= new Dictionary<Guid, int>(),
106 MEGame.LE2 => LE2ShaderOffsets ??= new Dictionary<Guid, int>(),
107 MEGame.LE1 => LE1ShaderOffsets ??= new Dictionary<Guid, int>(),
108 _ => null
109 };
110 if (offsetDict == null || offsetDict.Count > 0) return;
111
112 using FileStream fs = File.OpenRead(filePath);
113 fs.JumpTo(offsetOfShaderCacheOffset);
114 int binaryOffset = fs.ReadInt32() + 12;
115 fs.JumpTo(binaryOffset);
116 fs.Skip(1);
117 int nameCount = fs.ReadInt32();
118 fs.Skip(nameCount * 12);
119 if (game is not MEGame.ME2)
120 {
121 nameCount = fs.ReadInt32();
122 fs.Skip(nameCount * 12);
123 }
124
125 int shaderCount = fs.ReadInt32();
126 for (int i = 0; i < shaderCount; i++)
127 {
128 fs.Skip(8);
129 Guid shaderGuid = fs.ReadGuid();
130 int shaderEndOffset = fs.ReadInt32();
131 offsetDict.Add(shaderGuid, (int)fs.Position + 2);
132 fs.Skip(shaderEndOffset - fs.Position);
133 }
134
135 if (game != MEGame.ME1)
136 {
137 nameCount = fs.ReadInt32();
138 fs.Skip(nameCount * 12);
139 }
140 switch (game)
141 {
142 case MEGame.ME3:
143 ME3MaterialShaderMapsOffset = (int)fs.Position;
144 break;
145 case MEGame.ME2:
146 ME2MaterialShaderMapsOffset = (int)fs.Position;
147 break;
148 case MEGame.ME1:
149 ME1MaterialShaderMapsOffset = (int)fs.Position;
150 break;
151 case MEGame.LE3:
152 LE3MaterialShaderMapsOffset = (int)fs.Position;
153 //Debug.WriteLine($"{nameof(LE3MaterialShaderMapsOffset)}: {LE3MaterialShaderMapsOffset}");
154 break;
155 case MEGame.LE2:
156 LE2MaterialShaderMapsOffset = (int)fs.Position;
157 //Debug.WriteLine($"{nameof(LE2MaterialShaderMapsOffset)}: {LE2MaterialShaderMapsOffset}");
158 break;
159 case MEGame.LE1:
160 LE1MaterialShaderMapsOffset = (int)fs.Position;
161 //Debug.WriteLine($"{nameof(LE1MaterialShaderMapsOffset)}: {LE1MaterialShaderMapsOffset}");
162 break;
163 }
164 }

167 public static MaterialShaderMap GetMaterialShaderMap(MEGame game, StaticParameterSet staticParameterSet)
168 {
169 string filePath = shaderfilePath(game);
170 if (File.Exists(filePath))
171 {
172 using FileStream fs = File.OpenRead(filePath);
173 using IMEPackage shaderCachePackage = MEPackageHandler.OpenMEPackageFromStream(fs, quickLoad:true);
174 ReadNames(fs, shaderCachePackage);
175
176 int offsetOfShaderCacheOffset = shaderCachePackage.ExportOffset + 36;
177 populateOffsets(game, offsetOfShaderCacheOffset);
178 if (staticParameterSet is null)
179 {
180 return null;
181 }
182
183 var sc = new SerializingContainer2(fs, shaderCachePackage, true);
184 sc.ms.JumpTo(MaterialShaderMapsOffset(game));
185
186 int count = fs.ReadInt32();
187 for (int i = 0; i < count; i++)
188 {
189 StaticParameterSet sps = null;
190 sc.Serialize(ref sps);
191 if (sps == staticParameterSet)
192 {
193 MaterialShaderMap msm = null;
194 sc.Serialize(ref msm);
195 return msm;
196 }
197
198 if (game >= MEGame.ME3)
199 {
200 sc.ms.Skip(8);
201 }
202
203 int nextMSMOffset = sc.ms.ReadInt32();
204 sc.ms.Skip(nextMSMOffset - sc.ms.Position);
205 }
206 }
207
208 return null;
209 }

240 public static void RemoveStaticParameterSetsThatAreInTheGlobalCache(HashSet<StaticParameterSet> paramSets, MEGame game)
241 {
242 string filePath = shaderfilePath(game);
243 if (File.Exists(filePath))
244 {
245 using FileStream fs = File.OpenRead(filePath);
246 //read just the header of the package, then read the name list
247 using IMEPackage shaderCachePackage = MEPackageHandler.OpenMEPackageFromStream(fs, quickLoad: true);
248 ReadNames(fs, shaderCachePackage);
249 var sc = new SerializingContainer2(fs, shaderCachePackage, true);
250 sc.ms.JumpTo(MaterialShaderMapsOffset(game));
251
252 int count = fs.ReadInt32();
253 for (int i = 0; i < count && paramSets.Count > 0; i++)
254 {
255 StaticParameterSet sps = null;
256 sc.Serialize(ref sps);
257 if (paramSets.Contains(sps))
258 {
259 paramSets.Remove(sps);
260 }
261
262 if (game >= MEGame.ME3)
263 {
264 sc.ms.Skip(8);
265 }
266
267 int nextMSMOffset = sc.ms.ReadInt32();
268 sc.ms.Skip(nextMSMOffset - sc.ms.Position);
269 }
270 }

273 private static void ReadNames(FileStream fs, IMEPackage shaderCachePackage)
274 {
275 fs.JumpTo(shaderCachePackage.NameOffset);
276 var names = new List<string>(shaderCachePackage.NameCount);
277 for (int i = 0; i < shaderCachePackage.NameCount; i++)
278 {
279 var name = fs.ReadUnrealString();
280 names.Add(name);
281 if (shaderCachePackage.Game == MEGame.ME1 && shaderCachePackage.Platform != MEPackage.GamePlatform.PS3)
282 fs.Skip(8);
283 else if (shaderCachePackage.Game == MEGame.ME2 && shaderCachePackage.Platform != MEPackage.GamePlatform.PS3)
284 fs.Skip(4);
285 }
286
287 shaderCachePackage.restoreNames(names);
288 }
