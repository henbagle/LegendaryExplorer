387 public float GetHue()
388 {
389 if (R == G && G == B)
390 return 0; // 0 makes as good an UNDEFINED value as any
391
392 float r = R / 255.0f;
393 float g = G / 255.0f;
394 float b = B / 255.0f;
395
396 float max, min;
397 float delta;
398 float hue = 0.0f;
399
400 max = r; min = r;
401
402 if (g > max) max = g;
403 if (b > max) max = b;
404
405 if (g < min) min = g;
406 if (b < min) min = b;
407
408 delta = max - min;
409
410 if (r == max)
411 {
412 hue = (g - b) / delta;
413 }
414 else if (g == max)
415 {
416 hue = 2 + (b - r) / delta;
417 }
418 else if (b == max)
419 {
420 hue = 4 + (r - g) / delta;
421 }
422 hue *= 60;
423
424 if (hue < 0.0f)
425 {
426 hue += 360.0f;
427 }
428 return hue;
429 }

435 public float GetSaturation()
436 {
437 float r = R / 255.0f;
438 float g = G / 255.0f;
439 float b = B / 255.0f;
440
441 float max, min;
442 float l, s = 0;
443
444 max = r; min = r;
445
446 if (g > max) max = g;
447 if (b > max) max = b;
448
449 if (g < min) min = g;
450 if (b < min) min = b;
451
452 // if max == min, then there is no color and
453 // the saturation is zero.
454 //
455 if (max != min)
456 {
457 l = (max + min) / 2;
458
459 if (l <= .5)
460 {
461 s = (max - min) / (max + min);
462 }
463 else
464 {
465 s = (max - min) / (2 - max - min);
466 }
467 }
468 return s;
469 }
