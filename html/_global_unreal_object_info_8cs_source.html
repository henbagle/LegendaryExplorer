41 public static bool IsA(string className, string baseClass, MEGame game, Dictionary<string, ClassInfo> customClassInfos = null, string knownSuperClass = null)
42 {
43 if (className == baseClass) return true;
44 if (baseClass == @"Object") return true; //Everything inherits from Object
45 if (knownSuperClass != null && baseClass == knownSuperClass) return true; // We already know it's a direct descendant
46 var classes = game switch
47 {
48 MEGame.ME1 => ME1UnrealObjectInfo.Classes,
49 MEGame.ME2 => ME2UnrealObjectInfo.Classes,
50 MEGame.ME3 => ME3UnrealObjectInfo.Classes,
51 MEGame.UDK => ME3UnrealObjectInfo.Classes,
52 MEGame.LE1 => LE1UnrealObjectInfo.Classes,
53 MEGame.LE2 => LE2UnrealObjectInfo.Classes,
54 MEGame.LE3 => LE3UnrealObjectInfo.Classes,
55 _ => throw new ArgumentOutOfRangeException(nameof(game), game, null)
56 };
57 while (true)
58 {
59 if (className == baseClass)
60 {
61 return true;
62 }
63
64 if (customClassInfos != null && customClassInfos.TryGetValue(className, out ClassInfo info))
65 {
66 className = info.baseClass;
67 }
68 else if (classes.TryGetValue(className, out info))
69 {
70 className = info.baseClass;
71 }
72 else if (knownSuperClass != null && classes.TryGetValue(knownSuperClass, out info))
73 {
74 // We don't have this class in DB but we have super class (e.g. this is custom class without custom class info generated).
75 // We will just ignore this class and jump to our known super class
76 className = info.baseClass;
77 knownSuperClass = null; // Don't use it again
78 }
79 else
80 {
81 break;
82 }
83 }
84 return false;
85 }

174 public static PropertyCollection getClassDefaults(MEGame game, string className, string notIncludingClass = null)
175 {
176 Dictionary<string, ClassInfo> classes = GetClasses(game);
177 if (classes.TryGetValue(className, out ClassInfo info))
178 {
179 try
180 {
181 PropertyCollection props = new();
182 while (info != null && info.ClassName != notIncludingClass)
183 {
184 Stream loadStream = null;
185 string filepathTL = Path.Combine(MEDirectories.GetBioGamePath(game), info.pccPath);
186 if (File.Exists(info.pccPath))
187 {
188 loadStream = MEPackageHandler.ReadAllFileBytesIntoMemoryStream(info.pccPath);
189 }
190 else if (info.pccPath == Me3ExplorerCustomNativeAdditionsName)
191 {
192 loadStream = LegendaryExplorerCoreUtilities.GetCustomAppResourceStream(game);
193 }
194 else if (File.Exists(filepathTL))
195 {
196 loadStream = MEPackageHandler.ReadAllFileBytesIntoMemoryStream(filepathTL);
197 }
198
199 if (game == MEGame.ME1 && !File.Exists(filepathTL))
200 {
201 filepathTL = Path.Combine(ME1Directory.DefaultGamePath, info.pccPath); //for files from ME1 DLC
202 if (File.Exists(filepathTL))
203 {
204 loadStream = MEPackageHandler.ReadAllFileBytesIntoMemoryStream(filepathTL);
205 }
206 }
207 if (loadStream != null)
208 {
209 using IMEPackage importPCC = MEPackageHandler.OpenMEPackageFromStream(loadStream);
210 ExportEntry classExport = importPCC.GetUExport(info.exportIndex);
211 UClass classBin = ObjectBinary.From<UClass>(classExport);
212 ExportEntry defaults = importPCC.GetUExport(classBin.Defaults);
213
214 foreach (var prop in defaults.GetProperties())
215 {
216 if (!props.ContainsNamedProp(prop.Name))
217 {
218 props.Add(prop);
219 }
220 }
221 }
222
223 classes.TryGetValue(info.baseClass, out info);
224 }
225 return props;
226 }
227 catch
228 {
229 return null;
230 }
231 }
232 return null;
233 }

384 public static PropertyCollection getDefaultStructValue(MEGame game, string structName, bool stripTransients, PackageCache packageCache = null, bool shouldReturnClone = true)
385 {
386 if (game == MEGame.UDK)
387 {
388 game = MEGame.ME3;
389 }
390 var defaultStructValues = game switch
391 {
392 MEGame.ME1 => stripTransients ? ME1UnrealObjectInfo.defaultStructValuesME1 : DefaultStructValuesWithTransientsME1,
393 MEGame.ME2 => stripTransients ? ME2UnrealObjectInfo.defaultStructValuesME2 : DefaultStructValuesWithTransientsME2,
394 MEGame.ME3 => stripTransients ? ME3UnrealObjectInfo.defaultStructValuesME3 : DefaultStructValuesWithTransientsME3,
395 MEGame.LE1 => stripTransients ? LE1UnrealObjectInfo.defaultStructValuesLE1 : DefaultStructValuesWithTransientsLE1,
396 MEGame.LE2 => stripTransients ? LE2UnrealObjectInfo.defaultStructValuesLE2 : DefaultStructValuesWithTransientsLE2,
397 MEGame.LE3 => stripTransients ? LE3UnrealObjectInfo.defaultStructValuesLE3 : DefaultStructValuesWithTransientsLE3,
398 _ => throw new ArgumentOutOfRangeException(nameof(game), game, null)
399 };
400 if (defaultStructValues.TryGetValue(structName, out var cachedProps))
401 {
402 return shouldReturnClone ? cachedProps.DeepClone() : cachedProps;
403 }
404 var structs = game switch
405 {
406 MEGame.ME1 => ME1UnrealObjectInfo.Structs,
407 MEGame.ME2 => ME2UnrealObjectInfo.Structs,
408 MEGame.ME3 => ME3UnrealObjectInfo.Structs,
409 MEGame.LE1 => LE1UnrealObjectInfo.Structs,
410 MEGame.LE2 => LE2UnrealObjectInfo.Structs,
411 MEGame.LE3 => LE3UnrealObjectInfo.Structs,
412 _ => throw new ArgumentOutOfRangeException(nameof(game), game, null)
413 };
414 bool isImmutable = IsImmutable(structName, game);
415 if (structs.TryGetValue(structName, out ClassInfo info))
416 {
417 try
418 {
419 PropertyCollection props = new();
420 var infoStack = new Stack<ClassInfo>();
421 while (info is not null)
422 {
423 foreach ((NameReference propName, PropertyInfo propInfo) in info.properties)
424 {
425 if (stripTransients && propInfo.Transient)
426 {
427 continue;
428 }
429
430 if (getDefaultProperty(game, propName, propInfo, packageCache, stripTransients, isImmutable) is Property prop)
431 {
432 props.Add(prop);
433 if (propInfo.IsStaticArray())
434 {
435 for (int i = 1; i < propInfo.StaticArrayLength; i++)
436 {
437 prop = getDefaultProperty(game, propName, propInfo, packageCache, stripTransients, isImmutable);
438 prop.StaticArrayIndex = i;
439 props.Add(prop);
440 }
441 }
442 }
443 }
444 string filepath = null;
445 if (MEDirectories.GetBioGamePath(game) is string bioGamePath)
446 {
447 filepath = Path.Combine(bioGamePath, info.pccPath);
448 }
449
450 Stream loadStream = null;
451 IMEPackage cachedPackage = null;
452 if (packageCache != null)
453 {
454 packageCache.TryGetCachedPackage(filepath, true, out cachedPackage);
455 if (cachedPackage == null)
456 packageCache.TryGetCachedPackage(info.pccPath, true, out cachedPackage); // some cache types may have different behavior (such as relative package cache)
457
458 if (cachedPackage != null)
459 {
460 // Use this one
461 readDefaultProps(cachedPackage, props, packageCache: packageCache);
462 }
463 }
464 else if (filepath != null && MEPackageHandler.TryGetPackageFromCache(filepath, out cachedPackage))
465 {
466 readDefaultProps(cachedPackage, props, packageCache: packageCache);
467 }
468 else if (File.Exists(info.pccPath))
469 {
470 filepath = info.pccPath;
471 loadStream = MEPackageHandler.ReadAllFileBytesIntoMemoryStream(info.pccPath);
472 }
473 else if (info.pccPath == GlobalUnrealObjectInfo.Me3ExplorerCustomNativeAdditionsName)
474 {
475 filepath = game switch
476 {
477 MEGame.ME1 => "GAMERESOURCES_ME1",
478 MEGame.ME2 => "GAMERESOURCES_ME2",
479 MEGame.ME3 => "GAMERESOURCES_ME3",
480 MEGame.LE1 => "GAMERESOURCES_LE1",
481 MEGame.LE2 => "GAMERESOURCES_LE2",
482 MEGame.LE3 => "GAMERESOURCES_LE3",
483 _ => throw new ArgumentOutOfRangeException(nameof(game), game, null)
484 };
485 loadStream = LegendaryExplorerCoreUtilities.LoadFileFromCompressedResource("GameResources.zip", LegendaryExplorerCoreLib.CustomResourceFileName(game));
486 }
487 else if (filepath != null && File.Exists(filepath))
488 {
489 loadStream = MEPackageHandler.ReadAllFileBytesIntoMemoryStream(filepath);
490 }
491
492 if (cachedPackage == null && loadStream != null)
493 {
494 using IMEPackage importPcc = MEPackageHandler.OpenMEPackageFromStream(loadStream, filepath, useSharedPackageCache: true);
495 readDefaultProps(importPcc, props, packageCache);
496 }
497 structs.TryGetValue(info.baseClass, out info);
498 }
499 props.Add(new NoneProperty());
500
501 defaultStructValues.TryAdd(structName, props);
502 return shouldReturnClone ? props.DeepClone() : props;
503 }
504 catch (Exception e)
505 {
506 LECLog.Warning($@"Exception getting default {game} struct property for {structName}: {e.Message}");
507 return null;
508 }
509 }
510 return null;
